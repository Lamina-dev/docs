18/**
19 * @brief 扩展欧几里得算法，求 gcd(a, b) 及 Bezout 系数
20 * @param a 第一个整数
21 * @param b 第二个整数
22 * @param s 输出 Bezout 系数 s，满足 a*s + b*t = gcd(a,b)
23 * @param t 输出 Bezout 系数 t
24 * @return gcd(a, b)
25 */
26inline int64_t extended_gcd(int64_t a, int64_t b, int64_t& s, int64_t& t) {
27 int64_t old_r = a, r = b;
28 int64_t old_s = 1, ss = 0;
29 int64_t old_t = 0, tt = 1;
30
31 while (r != 0) {
32 int64_t q = old_r / r;
33 int64_t tmp = r;
34 r = old_r - q * r;
35 old_r = tmp;
36
37 tmp = ss;
38 ss = old_s - q * ss;
39 old_s = tmp;
40
41 tmp = tt;
42 tt = old_t - q * tt;
43 old_t = tmp;
44 }
45
46 s = old_s;
47 t = old_t;
48 return old_r;
49}

81/** @brief 模整数类，封装模 p 下的算术运算 */
82class ModInt {
83 int64_t val_;
84 int64_t mod_;
85
86public:
87 /** @brief 默认构造，值为 0，模为 2 */
88 ModInt() : val_(0), mod_(2) {}
89
90 /**
91 * @brief 构造模整数
92 * @param v 整数值（自动取模归约到 [0, p)）
93 * @param p 模数
94 */
95 ModInt(int64_t v, int64_t p) : mod_(p) {
96 val_ = v % p;
97 if (val_ < 0) val_ += p;
98 }
99
100 /** @brief 获取当前值 */
101 int64_t value() const { return val_; }
102
103 /** @brief 获取模数 */
104 int64_t modulus() const { return mod_; }
105
106 /** @brief 模加法 */
107 ModInt operator+(const ModInt& other) const {
108 return ModInt(val_ + other.val_, mod_);
109 }
110
111 /** @brief 模减法 */
112 ModInt operator-(const ModInt& other) const {
113 return ModInt(val_ - other.val_ + mod_, mod_);
114 }
115
116 /** @brief 模乘法 */
117 ModInt operator*(const ModInt& other) const {
118
119 uint64_t q;
120 uint64_t result = lmmp_mulmod_ulong_(
121 static_cast<uint64_t>(val_),
122 static_cast<uint64_t>(other.val_),
123 static_cast<uint64_t>(mod_), &q);
124 return ModInt(static_cast<int64_t>(result), mod_);
125 }
126
127 /** @brief 模除法（乘以逆元） */
128 ModInt operator/(const ModInt& other) const {
129 return *this * other.inverse();
130 }
131
132 /** @brief 取负 */
133 ModInt operator-() const {
134 return ModInt(val_ == 0 ? 0 : mod_ - val_, mod_);
135 }
136
137 /** @brief 判等 */
138 bool operator==(const ModInt& other) const {
139 return val_ == other.val_ && mod_ == other.mod_;
140 }
141
142 /** @brief 判不等 */
143 bool operator!=(const ModInt& other) const {
144 return !(*this == other);
145 }
146
147 /** @brief 小于比较（先比模数，再比值） */
148 bool operator<(const ModInt& other) const {
149 if (mod_ != other.mod_) return mod_ < other.mod_;
150 return val_ < other.val_;
151 }
152
153 /** @brief 大于比较 */
154 bool operator>(const ModInt& other) const {
155 return other < *this;
156 }
157
158 /** @brief 判断是否为零 */
159 bool is_zero() const { return val_ == 0; }
160
161 /**
162 * @brief 求模逆元
163 * @return 当前值在模 mod_ 下的乘法逆元
164 * @throw std::domain_error 若值为 0 或不可逆
165 */
166 ModInt inverse() const {
167 if (val_ == 0) {
168 throw std::domain_error("ModInt::inverse(): zero is not invertible");
169 }
170 int64_t s, t;
171 int64_t g = extended_gcd(val_, mod_, s, t);
172 if (g != 1) {
173 throw std::domain_error(
174 "ModInt::inverse(): element is not invertible (gcd != 1)");
175 }
176 return ModInt(s, mod_);
177 }
178
179 /**
180 * @brief 模幂运算
181 * @param base 底数
182 * @param exp 指数（可为负，负指数先求逆）
183 * @return base^exp mod p
184 */
185 static ModInt pow(ModInt base, int64_t exp) {
186 if (exp < 0) {
187 base = base.inverse();
188 exp = -exp;
189 }
190 if (exp == 0) return ModInt(1, base.modulus());
191
192 uint64_t result = lmmp_powmod_ulong_(
193 static_cast<uint64_t>(base.val_),
194 static_cast<uint64_t>(exp),
195 static_cast<uint64_t>(base.mod_));
196 return ModInt(static_cast<int64_t>(result), base.mod_);
197 }
198};

200/**
201 * @brief 中国剩余定理（两模数）
202 * @param r1 第一个余数
203 * @param m1 第一个模数
204 * @param r2 第二个余数
205 * @param m2 第二个模数
206 * @return pair(合并余数, 合并模数 m1*m2)
207 * @throw std::domain_error 若模数不互素
208 */
209inline std::pair<int64_t, int64_t> crt(int64_t r1, int64_t m1,
210 int64_t r2, int64_t m2) {
211 int64_t s, t;
212 int64_t g = extended_gcd(m1, m2, s, t);
213 if (g != 1) {
214 throw std::domain_error("crt(): moduli must be coprime");
215 }
216
217 int64_t M = m1 * m2;
218 int64_t diff = ((r2 - r1) % m2 + m2) % m2;
219
220 int64_t s_mod = ((s % m2) + m2) % m2;
221
222 uint64_t q_unused;
223 uint64_t factor = lmmp_mulmod_ulong_(
224 static_cast<uint64_t>(s_mod),
225 static_cast<uint64_t>(diff),
226 static_cast<uint64_t>(m2), &q_unused);
227
228#if defined(__GNUC__) || defined(__clang__)
229 __int128 x = static_cast<__int128>(r1) + static_cast<__int128>(m1) * factor;
230 __int128 bigM = static_cast<__int128>(m1) * m2;
231 x = ((x % bigM) + bigM) % bigM;
232 return {static_cast<int64_t>(x), static_cast<int64_t>(bigM)};
233#else
234 int64_t x = r1 + m1 * static_cast<int64_t>(factor);
235 x = ((x % M) + M) % M;
236 return {x, M};
237#endif
238}

240/**
241 * @brief 多模数中国剩余定理
242 * @param residues 余数列表
243 * @param primes 对应的模数列表（两两互素）
244 * @return pair(合并余数, 合并模数)
245 * @throw std::invalid_argument 若输入为空或大小不匹配
246 */
247inline std::pair<int64_t, int64_t> multi_crt(const std::vector<int64_t>& residues,
248 const std::vector<int64_t>& primes) {
249 if (residues.empty() || residues.size() != primes.size()) {
250 throw std::invalid_argument("multi_crt(): residues and primes must be non-empty and same size");
251 }
252
253 int64_t combined = residues[0];
254 int64_t modulus = primes[0];
255
256 for (size_t i = 1; i < residues.size(); ++i) {
257 auto [x, m] = crt(combined, modulus, residues[i], primes[i]);
258 combined = x;
259 modulus = m;
260 }
261
262 return {combined, modulus};
263}

265/**
266 * @brief 有理数重构：从模像 x mod m 恢复有理数 a/b
267 * @param x 模像值
268 * @param m 模数
269 * @return pair(分子 a, 分母 b)；失败时返回 (0, 0)
270 */
271inline std::pair<int64_t, int64_t> rational_reconstruction(int64_t x, int64_t m) {
272 if (m <= 0) {
273 return {0, 0};
274 }
275
276 x = ((x % m) + m) % m;
277
278 int64_t bound = static_cast<int64_t>(std::floor(std::sqrt(static_cast<double>(m) / 2.0)));
279 if (bound == 0) bound = 1;
280
281 int64_t r0 = m, r1 = x;
282 int64_t s0 = 0, s1 = 1;
283
284 while (r1 > bound) {
285 int64_t q = r0 / r1;
286 int64_t r_new = r0 - q * r1;
287 int64_t s_new = s0 - q * s1;
288
289 r0 = r1;
290 r1 = r_new;
291 s0 = s1;
292 s1 = s_new;
293 }
294
295 int64_t a = r1;
296 int64_t b = s1;
297
298 if (b < 0) {
299 a = -a;
300 b = -b;
301 }
302
303 if (std::abs(a) >= bound || b == 0 || b >= bound) {
304
305 return {0, 0};
306 }
307
308 int64_t s_unused, t_unused;
309 int64_t g = extended_gcd(std::abs(a), b, s_unused, t_unused);
310 if (g != 1) {
311 return {0, 0};
312 }
313
314 return {a, b};
315}

327/**
328 * @brief 判断素数 p 是否为"好素数"（不整除任何首项系数）
329 * @param p 待检测素数
330 * @param leading_coeffs 首项系数列表
331 * @return 若 p 不整除任何系数则返回 true
332 */
333inline bool is_good_prime(int64_t p, const std::vector<int64_t>& leading_coeffs) {
334 return std::none_of(leading_coeffs.begin(), leading_coeffs.end(),
335 [p](int64_t c) { return (c % p) == 0; });
336}

338/**
339 * @brief 生成指定数量的好素数
340 * @param count 需要的素数个数
341 * @param leading_coeffs 首项系数列表（用于过滤）
342 * @param start_from 搜索起始值
343 * @return 好素数列表
344 */
345inline std::vector<int64_t> generate_good_primes(int count,
346 const std::vector<int64_t>& leading_coeffs,
347 uint64_t start_from = 1000000000ULL) {
348 std::vector<int64_t> result;
349 result.reserve(count);
350 uint64_t candidate = start_from;
351 while ((int)result.size() < count) {
352 candidate = lmmp_next_prime_ulong_(candidate);
353 if (is_good_prime(static_cast<int64_t>(candidate), leading_coeffs)) {
354 result.push_back(static_cast<int64_t>(candidate));
355 }
356 }
357 return result;
358}
