11void RelationStore::add_relation(const SymbolicExpr& lhs, const SymbolicExpr& rhs,
12 RelationalNode::Op op, PropertyStore& prop_store) {
13 // Store the relation regardless of pattern
14 relations_.push_back(Relation{lhs, rhs, op});
15
16 // Detect simple "variable op 0" pattern for sign property derivation.
17 // LHS must be a single VariableNode and RHS must be a NumberNode with value 0.
18 if (!lhs.root || !rhs.root) {
19 return;
20 }
21
22 auto var_node = std::dynamic_pointer_cast<VariableNode>(lhs.root);
23 if (!var_node) {
24 return; // LHS is not a single variable — composite relation, store without decomposition
25 }
26
27 auto num_node = std::dynamic_pointer_cast<NumberNode>(rhs.root);
28 if (!num_node || !num_node->is_zero()) {
29 return; // RHS is not zero — store without decomposition
30 }
31
32 // Map operator to sign property
33 // GT → Positive
34 // GEQ → NonNegative
35 // LT → Negative
36 // LEQ → NonPositive
37 // NEQ → NonZero
38 switch (op) {
39 case RelationalNode::Op::GT:
40 prop_store.declare_sign(var_node->name, Sign::Positive);
41 break;
42 case RelationalNode::Op::GEQ:
43 prop_store.declare_sign(var_node->name, Sign::NonNegative);
44 break;
45 case RelationalNode::Op::LT:
46 prop_store.declare_sign(var_node->name, Sign::Negative);
47 break;
48 case RelationalNode::Op::LEQ:
49 prop_store.declare_sign(var_node->name, Sign::NonPositive);
50 break;
51 case RelationalNode::Op::NEQ:
52 prop_store.declare_sign(var_node->name, Sign::NonZero);
53 break;
54 case RelationalNode::Op::EQ:
55 // EQ with zero could imply Zero sign, but the design only specifies
56 // GT, GEQ, LT, LEQ, NEQ mappings. Store without sign derivation.
57 break;
58 }
59}

65bool RelationStore::has_relation(const SymbolicExpr& lhs, const SymbolicExpr& rhs,
66 RelationalNode::Op op) const {
67 for (const auto& rel : relations_) {
68 if (rel.op != op) {
69 continue;
70 }
71 // Compare LHS structurally
72 if (!rel.lhs.root && !lhs.root) {
73 // Both null — match on LHS
74 } else if (!rel.lhs.root || !lhs.root) {
75 continue; // One null, one not — no match
76 } else if (!rel.lhs.root->equals(*lhs.root)) {
77 continue;
78 }
79 // Compare RHS structurally
80 if (!rel.rhs.root && !rhs.root) {
81 return true; // Both null — full match
82 } else if (!rel.rhs.root || !rhs.root) {
83 continue; // One null, one not — no match
84 } else if (rel.rhs.root->equals(*rhs.root)) {
85 return true;
86 }
87 }
88 return false;
89}
