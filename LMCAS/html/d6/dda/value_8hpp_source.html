109 bool is_numeric() const {
110 if (type == Type::Int || type == Type::Float ||
111 type == Type::BigInt || type == Type::Rational ||
112 type == Type::Irrational) {
113 return true;
114 }
115 // Symbolic 表达式只要能化简为数值节点，就视为可走 numeric 路径；否则不视为
116 // numeric，避免上层算子把无法求值的符号悄悄塞进数值路径。
117 if (type == Type::Symbolic) {
118 const auto& sp = std::get<std::shared_ptr<SymbolicExpr>>(data);
119 if (!sp) return false;
120 try {
121 auto simp = sp->simplify();
122 return simp && simp->is_number();
123 } catch (...) {
124 return false;
125 }
126 }
127 return false;

130 /**
131 * @brief 将值转换为浮点数
132 * @return 浮点数近似值，非数值类型返回 0.0
133 */
134 lmmc_real_t as_number() const {
135 if (type == Type::Infinity) {
136 lmmc_real_t inf;
137 lmmc_inf(&inf);
138 auto sign = std::get_if<int>(&data);
139 return (sign && *sign > 0) ? inf : -inf;
140 }
141 if (type == Type::Int) return static_cast<lmmc_real_t>(std::get<int>(data));
142 if (type == Type::Float) return std::get<lmmc_real_t>(data);
143 if (type == Type::BigInt) {
144 // 直接用 BigInt::to_double() 即可，避免先 to_int() 后再转 double 在大数上失真。
145 return std::get<::BigInt>(data).to_double();
146 }
147 if (type == Type::Rational) {
148 return std::get<::Rational>(data).to_double();
149 }
150 if (type == Type::Irrational) {
151 return std::get<::Irrational>(data).to_double();
152 }
153 if (type == Type::Symbolic) {
154 return std::get<std::shared_ptr<SymbolicExpr>>(data)->to_numeric();
155 }
156 return 0.0;
157 }

193 /**
194 * @brief 将值转换为符号表达式
195 * @return 符号表达式智能指针，不可转换时返回数值 0
196 */
197 std::shared_ptr<SymbolicExpr> as_symbolic() const {
198 if (type == Type::Infinity) {
199 auto sign = std::get_if<int>(&data);
200 return SymbolicExpr::infinity(sign ? *sign : 1);
201 }
202 if (type == Type::Symbolic) return std::get<std::shared_ptr<SymbolicExpr>>(data);
203 if (type == Type::Int || type == Type::Float || type == Type::Rational || type == Type::BigInt) {
204 return SymbolicExpr::number(as_rational());
205 }
206 if (type == Type::Irrational) {
207 return as_irrational().to_symbolic();
208 }
209 if (type == Type::Matrix) {
210 const auto& mat = std::get<std::vector<std::vector<Value>>>(data);
211 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> sym_mat;
212 for (const auto& row : mat) {
213 std::vector<std::shared_ptr<SymbolicExpr>> sym_row;
214 for (const auto& val : row) {
215 sym_row.push_back(val.as_symbolic());
216 }
217 sym_mat.push_back(sym_row);
218 }
219 return SymbolicExpr::matrix(sym_mat);
220 }
221 return SymbolicExpr::number(0);
222 }

236 /**
237 * @brief 将值转换为可读字符串
238 * @return 格式化的字符串表示
239 */
240 std::string to_string() const {
241 if (type == Type::String) return _str_cache;
242 if (!_str_cache.empty()) return _str_cache;
243 switch (type) {
244 case Type::Infinity: {
245 auto sign = std::get_if<int>(&data);
246 return (sign && *sign > 0) ? "inf" : "-inf";
247 }
248 case Type::Null:
249 return "null";
250 case Type::Int:
251 return std::to_string(std::get<int>(data));
252 case Type::Float: {
253 lmmc_real_t val = std::get<lmmc_real_t>(data);
254 std::string str = std::to_string(val);
255 str.erase(str.find_last_not_of('0') + 1, std::string::npos);
256 str.erase(str.find_last_not_of('.') + 1, std::string::npos);
257 return str;
258 }
259 case Type::Array: {
260 std::string res = "[";
261 const auto& arr = std::get<std::vector<Value>>(data);
262 for (size_t i = 0; i < arr.size(); ++i) {
263 if (i) res += ", ";
264 res += arr[i].to_string();
265 }
266 res += "]";
267 return res;
268 }
269 case Type::Matrix: {
270 std::string res = "[";
271 const auto& mat = std::get<std::vector<std::vector<Value>>>(data);
272 for (size_t i = 0; i < mat.size(); ++i) {
273 if (i) res += ", ";
274 res += "[";
275 for (size_t j = 0; j < mat[i].size(); ++j) {
276 if (j) res += ", ";
277 res += mat[i][j].to_string();
278 }
279 res += "]";
280 }
281 res += "]";
282 return res;
283 }
284 case Type::BigInt:
285 return std::get<::BigInt>(data).to_string();
286 case Type::Rational:
287 return std::get<::Rational>(data).to_string();
288 case Type::Irrational:
289 return std::get<::Irrational>(data).to_string();
290 case Type::Symbolic:
291 return std::get<std::shared_ptr<SymbolicExpr>>(data)->to_string();
292 default:
293 return "<unknown>";
294 }

297 bool operator==(const Value& other) const {
298 if (type != other.type) return false;
299 if (type == Type::String) {
300 return _str_cache == other._str_cache;
301 }
302 if (data.index() != other.data.index()) return false;
303 return std::visit([&other](const auto& val1) -> bool {
304 using T = std::decay_t<decltype(val1)>;
305 if constexpr (std::is_same_v<T, std::nullptr_t>) {
306 return true;
307 } else if constexpr (std::is_same_v<T, std::shared_ptr<SymbolicExpr>>) {
308 if (!val1 && !std::get<T>(other.data)) return true;
309 if (!val1 || !std::get<T>(other.data)) return false;
310 return val1->compare(std::get<T>(other.data)) == 0;
311 } else {
312 return val1 == std::get<T>(other.data);
313 }
314 }, data);
315 }
316
317 bool operator<(const Value& other) const {
318 if (type != other.type) return static_cast<int>(type) < static_cast<int>(other.type);
319 if (type == Type::String) {
320 return _str_cache < other._str_cache;
321 }
322 if (data.index() != other.data.index()) return data.index() < other.data.index();
323 return std::visit([&other](const auto& val1) -> bool {
324 using T = std::decay_t<decltype(val1)>;
325 if constexpr (std::is_same_v<T, std::nullptr_t>) {
326 return false;
327 } else if constexpr (std::is_same_v<T, std::shared_ptr<SymbolicExpr>>) {
328 if (!val1 && !std::get<T>(other.data)) return false;
329 if (!val1) return true;
330 if (!std::get<T>(other.data)) return false;
331 return val1->compare(std::get<T>(other.data)) < 0;
332 } else {
333 return val1 < std::get<T>(other.data);
334 }
335 }, data);
336 }

343 Value vector_add(const Value& other) const {

353 return Value();

355 }

380 /**
381 * @brief 矩阵乘法
382 * @param other 右矩阵
383 * @return 乘积矩阵
384 */
385 Value matrix_multiply(const Value& other) const {
386 if (!is_matrix() || !other.is_matrix()) return Value();
387 const auto& a = std::get<std::vector<std::vector<Value>>>(data);
388 const auto& b = std::get<std::vector<std::vector<Value>>>(other.data);
389 if (a.empty() || b.empty() || a[0].size() != b.size()) return Value();
390 size_t rows = a.size();
391 size_t cols = b[0].size();
392 size_t inner = a[0].size();
393 std::vector<std::vector<Value>> result(rows, std::vector<Value>(cols, Value(0.0)));
394 for (size_t i = 0; i < rows; ++i) {
395 for (size_t j = 0; j < cols; ++j) {
396 lmmc_real_t sum = 0.0;
397 for (size_t k = 0; k < inner; ++k) {
398 if (!a[i][k].is_numeric() || !b[k][j].is_numeric()) return Value();
399 sum += a[i][k].as_number() * b[k][j].as_number();
400 }
401 result[i][j] = Value(sum);
402 }
403 }
404 return Value(result);
405 }
