33struct SymbolicPolyCoeff {
34 std::shared_ptr<SymbolicExpr> val; ///< 内部符号表达式
35
36 /** @brief 默认构造，值为 0 */
37 SymbolicPolyCoeff() : val(SymbolicExpr::number(0)) {}
38
39 /**
40 * @brief 从整数构造
41 * @param v 整数值
42 */
43 explicit SymbolicPolyCoeff(int v) : val(SymbolicExpr::number(v)) {}
44
45 /**
46 * @brief 从符号表达式构造
47 * @param v 符号表达式
48 */
49 SymbolicPolyCoeff(std::shared_ptr<SymbolicExpr> v) : val(std::move(v)) {}
50
51 /** @brief 判等（通过化简差为零判断） */
52 bool operator==(const SymbolicPolyCoeff& other) const {
53
54 if (!val || !other.val) return false;
55
56 if (val == other.val) return true;
57
58 auto diff = SymbolicExpr::add(val, SymbolicExpr::multiply(other.val, SymbolicExpr::number(-1)));
59 return diff->simplify()->is_zero();
60 }
61
62 bool operator!=(const SymbolicPolyCoeff& other) const {
63 return !(*this == other);
64 }
65
66 /** @brief 加法 */
67 SymbolicPolyCoeff operator+(const SymbolicPolyCoeff& other) const {
68 return SymbolicPolyCoeff(SymbolicExpr::add(val, other.val));
69 }
70
71 /** @brief 减法 */
72 SymbolicPolyCoeff operator-(const SymbolicPolyCoeff& other) const {
73 return SymbolicPolyCoeff(

76 }

122template<>
123inline BigInt extract_coeff_value<BigInt>(const std::shared_ptr<SymbolicExpr>& c) {
124 auto simp = c->simplify();
125 if (auto n = std::dynamic_pointer_cast<NumberNode>(simp->root)) {
126 if (std::holds_alternative<BigInt>(n->value)) return std::get<BigInt>(n->value);
127 if (std::holds_alternative<Rational>(n->value)) {
128
129 Rational r = std::get<Rational>(n->value);
130 if (r.is_integer()) return r.to_BigInt();
131 // 非整数 Rational：保留为 0，避免悄悄截断。
132 return BigInt(0);
133 }
134 if (std::holds_alternative<lmmc_real_t>(n->value)) {
135 lmmc_real_t d = std::get<lmmc_real_t>(n->value);
136 // 只接受确确实实是整数级别的 double，否则返回 0 而不是悄悄截断。
137 if (std::isfinite(d) && d == std::floor(d) &&
138 d >= static_cast<lmmc_real_t>(std::numeric_limits<long long>::min()) &&
139 d <= static_cast<lmmc_real_t>(std::numeric_limits<long long>::max())) {
140 return BigInt(static_cast<long long>(d));
141 }
142 return BigInt(0);
143 }
144 }
145
146 if (simp->is_zero()) return BigInt(0);
147
148 if (simp->is_one()) return BigInt(1);
149
150 return BigInt(0);
151}

166/**
167 * @brief 判断符号节点是否依赖指定变量
168 * @param node 符号节点
169 * @param var 变量名
170 * @return 包含该变量返回 true
171 */
172inline bool depends_on_var(const std::shared_ptr<SymbolicNode>& node, const std::string& var) {
173 if (!node) return false;
174
175 struct DepVisitor : public SymbolicVisitor {
176 std::string v;
177 bool found = false;
178 void visit(NumberNode&) override {}
179 void visit(VariableNode& n) override { if (n.name == v) found = true; }
180 void visit(AddNode& n) override { for(auto& op : n.operands) { if(found) return; op->accept(*this); } }
181 void visit(MultiplyNode& n) override { for(auto& op : n.operands) { if(found) return; op->accept(*this); } }
182 void visit(PowerNode& n) override { n.base->accept(*this); if(!found) n.exponent->accept(*this); }
183 void visit(FunctionNode& n) override { for(auto& arg : n.arguments) { if(found) return; arg->accept(*this); } }
184 void visit(MatrixNode& n) override {
185 if (std::holds_alternative<MatrixNode::DenseStorage>(n.storage)) {
186 for (auto& item : std::get<MatrixNode::DenseStorage>(n.storage)) {
187 if (item) { item->accept(*this); if (found) return; }
188 }
189 } else {
190 for (auto& [idx, item] : std::get<MatrixNode::SparseStorage>(n.storage)) {
191 item->accept(*this); if (found) return;
192 }
193 }
194 }
195 void visit(RelationalNode& n) override {
196 if (n.left) n.left->accept(*this);
197 if (!found && n.right) n.right->accept(*this);
198 }
199 void visit(LogicalNode& n) override {
200 if (n.left) n.left->accept(*this);
201 if (!found && n.right) n.right->accept(*this);
202 }
203 } visitor;
204 visitor.v = var;
205 node->accept(visitor);
206 return visitor.found;
207}

219/**
220 * @brief 递归地将符号节点转换为一元多项式
221 * @tparam T 系数类型
222 * @param node 符号节点
223 * @param var 主变量名
224 * @return 对应的一元多项式
225 */
226template <typename T>
227Polynomial<T> symbolic_to_poly_recursive(const std::shared_ptr<SymbolicNode>& node, const std::string& var) {
228 if (!node) return Polynomial<T>(var);
229
230 if (!depends_on_var(node, var)) {
231 return Polynomial<T>({extract_coeff_value<T>(std::make_shared<SymbolicExpr>(node))}, var);
232 }
233
234 if (auto v = std::dynamic_pointer_cast<VariableNode>(node)) {
235 if (v->name == var) {
236 return Polynomial<T>({T(0), T(1)}, var);
237 }
238 }
239
240 if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
241 Polynomial<T> res(var);
242 for (auto& op : add->operands) {
243 res = res + symbolic_to_poly_recursive<T>(op, var);
244 }
245 return res;
246 }
247
248 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
249 Polynomial<T> res({T(1)}, var);
250 for (auto& op : mul->operands) {
251 res = res * symbolic_to_poly_recursive<T>(op, var);
252 }
253 return res;
254 }
255
256 if (auto pow = std::dynamic_pointer_cast<PowerNode>(node)) {
257 if (auto exp_num = std::dynamic_pointer_cast<NumberNode>(pow->exponent)) {
258 // 只有指数能确认为非负整数时才走幂展开；否则保持表达式不展开，
259 // 避免 x^(1/2) / x^(-1) 这种被强转 (int) 后被错误展开成 x^0=1 之类。
260 bool is_nonneg_int = false;
261 int e_val = 0;
262
263 if (std::holds_alternative<BigInt>(exp_num->value)) {
264 const auto& bi = std::get<BigInt>(exp_num->value);
265 if (!bi.IsNegative()) {
266 int v = bi.to_int();
267 // 多项式幂展开会按指数次方做矩阵幂运算，过大时直接放弃。
268 if (v >= 0 && v < 1000) {
269 is_nonneg_int = true;
270 e_val = v;
271 }
272 }
273 } else if (std::holds_alternative<Rational>(exp_num->value)) {
274 const auto& r = std::get<Rational>(exp_num->value);
275 if (r.is_integer()) {
276 BigInt bi = r.to_BigInt();
277 if (!bi.IsNegative()) {
278 int v = bi.to_int();
279 if (v >= 0 && v < 1000) {
280 is_nonneg_int = true;
281 e_val = v;
282 }
283 }
284 }
285 } else if (std::holds_alternative<lmmc_real_t>(exp_num->value)) {
286 lmmc_real_t d = std::get<lmmc_real_t>(exp_num->value);
287 // 只有当 double 严格等于其取整值且为非负时，才视为整数指数。
288 if (std::isfinite(d) && d >= 0 && d < 1000.0 &&
289 d == std::floor(d)) {
290 is_nonneg_int = true;
291 e_val = static_cast<int>(d);
292 }
293 }
294
295 if (is_nonneg_int) {
296 if (e_val == 0) return Polynomial<T>({T(1)}, var);
297 auto base_poly = symbolic_to_poly_recursive<T>(pow->base, var);
298 Polynomial<T> res({T(1)}, var);
299 for (int i = 0; i < e_val; ++i) res = res * base_poly;
300 return res;
301 }
302 }
303 }
304
305 return Polynomial<T>(var);
306}

314template <typename T>
315std::shared_ptr<SymbolicExpr> poly_to_symbolic(const Polynomial<T>& poly) {
316 if (poly.is_zero()) return SymbolicExpr::number(0);
317
318 std::vector<std::shared_ptr<SymbolicExpr>> terms;
319
320 for (size_t i = 0; i < poly.coeffs.size(); ++i) {
321 if (poly.coeffs[i] == T(0)) continue;
322
323 auto coeff_node = SymbolicExpr::number(0);
324
325 if constexpr (std::is_same_v<T, BigInt>) {
326 coeff_node = SymbolicExpr::number(poly.coeffs[i]);
327 } else if constexpr (std::is_same_v<T, Rational>) {
328 coeff_node = SymbolicExpr::number(poly.coeffs[i]);
329 } else {
330
331 coeff_node = SymbolicExpr::number(poly.coeffs[i]);
332 }
333
334 if (i == 0) {
335 terms.push_back(coeff_node);
336 } else {
337
338 auto var_node = std::make_shared<SymbolicExpr>(std::make_shared<VariableNode>(poly.variable_name));
339 std::shared_ptr<SymbolicExpr> var_part;
340 if (i == 1) {
341 var_part = var_node;
342 } else {
343 var_part = SymbolicExpr::power(var_node, SymbolicExpr::number((int)i));
344 }
345
346 if (poly.coeffs[i] == T(1)) {
347 terms.push_back(var_part);
348 } else if (poly.coeffs[i] == T(-1)) {
349 terms.push_back(SymbolicExpr::multiply(SymbolicExpr::number(-1), var_part));
350 } else {
351 terms.push_back(SymbolicExpr::multiply(coeff_node, var_part));
352 }
353 }
354 }
355
356 std::reverse(terms.begin(), terms.end());
357
358 if (terms.empty()) return SymbolicExpr::number(0);
359 if (terms.size() == 1) return terms[0];
360
361 auto res = terms[0];
362 for (size_t i = 1; i < terms.size(); ++i) {
363 res = SymbolicExpr::add(res, terms[i]);
364 }
365 return res;
366}
