41/**
42 * @brief 符号表达式 AST 节点基类。
43 *
44 * 所有具体节点（数值、变量、运算符、函数等）均继承此类。
45 * 提供哈希、比较、克隆等通用接口，支持 Visitor 模式遍历。
46 */
47class SymbolicNode {
48protected:
49 mutable std::atomic<std::size_t> cached_hash{0};
50 mutable std::atomic<bool> hash_computed{false};
51
52 SymbolicNode() = default;
53 SymbolicNode(const SymbolicNode&) : cached_hash(0), hash_computed(false) {}
54 SymbolicNode& operator=(const SymbolicNode&) { return *this; }
55
56 /** @brief 计算当前节点的哈希值（由子类实现）。 */
57 virtual std::size_t compute_hash() const = 0;
58
59 /** @brief 与同类型节点进行比较（由子类实现）。 */
60 virtual int compare_same_type(const SymbolicNode& other) const = 0;
61
62public:
63 virtual ~SymbolicNode() = default;
64
65 /** @brief 接受 Visitor 访问。 */
66 virtual void accept(SymbolicVisitor& visitor) = 0;
67
68 /** @brief 深拷贝当前节点及其子树。 */
69 virtual std::shared_ptr<SymbolicNode> clone() const = 0;
70
71 /** @brief 返回节点类型的排序优先级，用于规范化排序。 */
72 virtual int type_priority() const = 0;
73
74 /**
75 * @brief 获取节点哈希值（带缓存）。
76 * @return 哈希值
77 */
78 std::size_t hash() const {
79 if (!hash_computed.load(std::memory_order_acquire)) {
80 std::size_t h = compute_hash();
81 cached_hash.store(h, std::memory_order_relaxed);
82 hash_computed.store(true, std::memory_order_release);
83 return h;
84 }
85 return cached_hash.load(std::memory_order_relaxed);
86 }
87
88 /**
89 * @brief 与另一个节点进行全序比较。
90 * @param other 待比较的节点
91 * @return 小于返回 -1，等于返回 0，大于返回 1
92 */
93 int compare(const SymbolicNode& other) const {
94 if (type_priority() != other.type_priority()) {
95 return type_priority() < other.type_priority() ? -1 : 1;
96 }
97 return compare_same_type(other);
98 }
99
100 /**
101 * @brief 判断两个节点是否结构相等。
102 * @param other 待比较的节点
103 * @return 相等返回 true
104 */
105 bool equals(const SymbolicNode& other) const {
106 if (this == &other) return true;
107 if (hash() != other.hash()) return false;
108 if (type_priority() != other.type_priority()) return false;
109 return compare_same_type(other) == 0;
110 }
111
112 /** @brief 判断节点是否为数值节点。 */
113 virtual bool is_number() const { return false; }
114 /** @brief 判断节点数值是否为 1。 */
115 virtual bool is_one() const { return false; }
116 /** @brief 判断节点数值是否为 0。 */
117 virtual bool is_zero() const { return false; }
118 /** @brief 判断节点是否可证明为严格正数（非数值节点默认返回 false）。 */
119 virtual bool is_positive() const { return false; }
120};

218/**
219 * @brief 数值节点，存储 BigInt、Rational 或浮点数。
220 */
221class NumberNode : public SymbolicNode {
222public:
223 std::variant<BigInt, Rational, lmmc_real_t> value; ///< 数值，支持大整数、有理数、浮点数
224
225 explicit NumberNode(const BigInt& v) : value(v) {}
226 explicit NumberNode(const Rational& v) : value(v) {}
227 explicit NumberNode(lmmc_real_t v) : value(v) {}
228 explicit NumberNode(std::variant<BigInt, Rational, lmmc_real_t> v) : value(std::move(v)) {}
229
230 int type_priority() const override { return -10; }
231
232protected:
233 std::size_t compute_hash() const override {
234 // 与 compare_same_type 对齐：相同数值（如 1、Rational(1,1)、1.0）必须哈希一致，
235 // 否则 NodeMap/NodeSet 在 equals() 的 hash() 短路里会误判不等。
236 // compare_same_type 在任一为 double 时统一转 double 比较，所以这里也按 double 哈希。
237 auto to_real = [](const std::variant<BigInt, Rational, lmmc_real_t>& v) -> lmmc_real_t {
238 if (std::holds_alternative<lmmc_real_t>(v)) return std::get<lmmc_real_t>(v);
239 if (std::holds_alternative<Rational>(v)) return (lmmc_real_t)std::get<Rational>(v).to_double();
240 return (lmmc_real_t)std::get<BigInt>(v).to_double();
241 };
242 lmmc_real_t d = to_real(value);
243 // NaN 一律映射到一个统一桶，避免不同 NaN 位模式造成 hash 不稳。
244 if (d != d) return std::hash<int>{}(0);
245 // 0.0 与 -0.0 在 == 比较里相等，统一规约为 0.0。
246 if (d == 0.0) d = 0.0;
247 return std::hash<lmmc_real_t>{}(d);
248 }
249
250 int compare_same_type(const SymbolicNode& other) const override {
251 const auto& o = static_cast<const NumberNode&>(other);
252
253 bool is_l_real = std::holds_alternative<lmmc_real_t>(value);
254 bool is_r_real = std::holds_alternative<lmmc_real_t>(o.value);
255
256 if (is_l_real || is_r_real) {
257 auto to_lmmc_real = [](const std::variant<BigInt, Rational, lmmc_real_t>& v) {
258 if (std::holds_alternative<lmmc_real_t>(v)) return std::get<lmmc_real_t>(v);
259 if (std::holds_alternative<Rational>(v)) return (lmmc_real_t)std::get<Rational>(v).to_double();
260 return (lmmc_real_t)std::get<BigInt>(v).to_double();
261 };
262
263 lmmc_real_t v1 = to_lmmc_real(value);
264 lmmc_real_t v2 = to_lmmc_real(o.value);
265 return LMMC_REAL_CMP(&v1, &v2);
266 }
267
268 bool is_l_rat = std::holds_alternative<Rational>(value);
269 bool is_r_rat = std::holds_alternative<Rational>(o.value);
270
271 if (is_l_rat || is_r_rat) {
272 Rational r1 = is_l_rat ? std::get<Rational>(value) : Rational(std::get<BigInt>(value));
273 Rational r2 = is_r_rat ? std::get<Rational>(o.value) : Rational(std::get<BigInt>(o.value));
274 if (r1 < r2) return -1;
275 if (r1 == r2) return 0;
276 return 1;
277 }
278
279 const auto& b1 = std::get<BigInt>(value);
280 const auto& b2 = std::get<BigInt>(o.value);
281 if (b1 < b2) return -1;
282 if (b1 == b2) return 0;
283 return 1;
284 }
285
286public:
287 void accept(SymbolicVisitor& visitor) override { SymbolicVisitor::DepthGuard guard(visitor); visitor.visit(*this); }
288 std::shared_ptr<SymbolicNode> clone() const override {
289
290 if (std::holds_alternative<BigInt>(value)) return std::make_shared<NumberNode>(std::get<BigInt>(value));
291 if (std::holds_alternative<Rational>(value)) return std::make_shared<NumberNode>(std::get<Rational>(value));
292 return std::make_shared<NumberNode>(std::get<lmmc_real_t>(value));
293 }
294
295 bool is_number() const override { return true; }
296 bool is_zero() const override {
297
298 if (std::holds_alternative<BigInt>(value)) return std::get<BigInt>(value) == BigInt(0);
299 if (std::holds_alternative<Rational>(value)) return std::get<Rational>(value) == Rational(0);
300 lmmc_real_t v = std::get<lmmc_real_t>(value);
301 int eq;
302 lmmc_double_nearly_equal(v, 0.0, &eq);
303 return eq != 0;
304 }
305 bool is_one() const override {
306 if (std::holds_alternative<BigInt>(value)) return std::get<BigInt>(value) == BigInt(1);
307 if (std::holds_alternative<Rational>(value)) return std::get<Rational>(value) == Rational(1);
308 lmmc_real_t v = std::get<lmmc_real_t>(value);
309 int eq;
310 lmmc_double_nearly_equal(v, 1.0, &eq);
311 return eq != 0;
312 }
313 bool is_positive() const override {
314 if (std::holds_alternative<BigInt>(value)) {
315 const auto& b = std::get<BigInt>(value);
316 return !b.IsNegative() && !(b == BigInt(0));
317 }
318 if (std::holds_alternative<Rational>(value)) {
319 const auto& r = std::get<Rational>(value);
320 return r > Rational(0);
321 }
322 lmmc_real_t v = std::get<lmmc_real_t>(value);
323 return std::isfinite(v) && v > 0.0;
324 }

353/**
354 * @brief 加法节点，表示多个操作数的求和。
355 *
356 * 构造时自动扁平化嵌套的 AddNode，并按规范顺序排序操作数。
357 */
358class AddNode : public SymbolicNode {
359public:
360 std::vector<std::shared_ptr<SymbolicNode>> operands; ///< 加法操作数列表
361
362 explicit AddNode(std::vector<std::shared_ptr<SymbolicNode>> ops) {
363
364 for (const auto& op : ops) {
365 if (auto add = std::dynamic_pointer_cast<AddNode>(op)) {
366 operands.insert(operands.end(), add->operands.begin(), add->operands.end());
367 } else {
368 operands.push_back(op);
369 }
370 }
371
372 std::sort(operands.begin(), operands.end(), [](const auto& a, const auto& b) {
373 bool a_num = a->is_number();
374 bool b_num = b->is_number();
375 if (a_num != b_num) return !a_num;
376 return a->compare(*b) < 0;
377 });
378 }
379
380 int type_priority() const override { return 5; }
381
382protected:
383 std::size_t compute_hash() const override {
384 std::size_t seed = 0;
385 hash_combine(seed, type_priority());
386 for (const auto& op : operands) {
387 hash_combine(seed, op->hash());
388 }
389 return seed;
390 }
391
392 int compare_same_type(const SymbolicNode& other) const override {
393 const auto& o = static_cast<const AddNode&>(other);
394 if (operands.size() != o.operands.size()) {
395 return operands.size() < o.operands.size() ? -1 : 1;
396 }
397 for (size_t i = 0; i < operands.size(); ++i) {
398 int cmp = operands[i]->compare(*o.operands[i]);
399 if (cmp != 0) return cmp;
400 }
401 return 0;
402 }
403
404public:
405 void accept(SymbolicVisitor& visitor) override { SymbolicVisitor::DepthGuard guard(visitor); visitor.visit(*this); }
406 std::shared_ptr<SymbolicNode> clone() const override {
407 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
408 new_ops.reserve(operands.size());
409 for (const auto& op : operands) new_ops.push_back(op->clone());
410 return std::make_shared<AddNode>(std::move(new_ops));
411 }

581 * @brief 矩阵节点，支持稠密和稀疏两种存储方式。
582 *
583 * 当非零元素占比低于 20% 时自动使用稀疏存储，否则使用稠密存储。
584 */
585class MatrixNode : public SymbolicNode {
586public:
587 using DenseStorage = std::vector<std::shared_ptr<SymbolicNode>>; ///< 稠密存储（按行优先展开）
588 using SparseStorage = std::map<size_t, std::shared_ptr<SymbolicNode>>; ///< 稀疏存储（索引 -> 节点）

590 /**
591 * @brief 验证并返回网格的列数（要求各行列数一致）。
592 * @param grid 二维节点网格
593 * @return 列数
594 * @throw std::invalid_argument 各行列数不一致时抛出
595 */
596 static size_t validate_grid_columns(const std::vector<std::vector<std::shared_ptr<SymbolicNode>>>& grid) {
597 if (grid.empty()) return 0;
598 size_t ncols = grid[0].size();
599 for (size_t i = 1; i < grid.size(); ++i) {
600 if (grid[i].size() != ncols) {
601 throw std::invalid_argument("MatrixNode: all rows must have the same number of columns");
602 }
603 }
604 return ncols;
605 }

607 /**
608 * @brief 根据稀疏度选择存储方式，从二维网格创建存储。
609 * @param grid 二维节点网格
610 * @param total_elements 总元素数
611 * @param ncols 列数
612 * @return 稠密或稀疏存储
613 */
614 static std::variant<DenseStorage, SparseStorage> create_storage_from_grid(
615 const std::vector<std::vector<std::shared_ptr<SymbolicNode>>>& grid, size_t total_elements, size_t ncols);
616
617 const size_t rows; ///< 行数
618 const size_t cols; ///< 列数
619 const std::variant<DenseStorage, SparseStorage> storage; ///< 实际存储
620
621 MatrixNode(const std::vector<std::vector<std::shared_ptr<SymbolicNode>>>& grid)
622 : rows(grid.size()),
623 cols(grid.empty() ? 0 : validate_grid_columns(grid)),
624 storage(create_storage_from_grid(grid, rows * cols, cols)) {}
625
626 MatrixNode(size_t r, size_t c, DenseStorage dense)
627 : storage(std::move(dense)), rows(r), cols(c) {}
628
629 MatrixNode(size_t r, size_t c, SparseStorage sparse)
630 : storage(std::move(sparse)), rows(r), cols(c) {}

632 int type_priority() const override { return 6; }
633
634protected:
635 std::size_t compute_hash() const override {
636 std::size_t seed = 0;
637 hash_combine(seed, type_priority());
638 hash_combine(seed, rows);
639 hash_combine(seed, cols);
640
641 if (std::holds_alternative<DenseStorage>(storage)) {
642 const auto& dense = std::get<DenseStorage>(storage);
643 for (size_t i = 0; i < dense.size(); ++i) {
644 if (dense[i] && !dense[i]->is_zero()) {
645 hash_combine(seed, i);
646 hash_combine(seed, dense[i]->hash());
647 }
648 }
649 } else {
650 const auto& sparse = std::get<SparseStorage>(storage);
651 for (const auto& [idx, val] : sparse) {
652
653 if (val && !val->is_zero()) {
654 hash_combine(seed, idx);
655 hash_combine(seed, val->hash());
656 }
657 }
658 }
659 return seed;
660 }

662 int compare_same_type(const SymbolicNode& other) const override {
663 const auto& o = static_cast<const MatrixNode&>(other);
664 if (rows != o.rows) return rows < o.rows ? -1 : 1;
665 if (cols != o.cols) return cols < o.cols ? -1 : 1;
666
667 for (size_t r = 0; r < rows; ++r) {
668 for (size_t c = 0; c < cols; ++c) {
669 auto v1 = this->get(r, c);
670 auto v2 = o.get(r, c);
671
672 bool z1 = !v1 || v1->is_zero();
673 bool z2 = !v2 || v2->is_zero();
674
675 if (z1 && z2) continue;
676 if (z1) return -1;
677 if (z2) return 1;
678
679 int cmp = v1->compare(*v2);
680 if (cmp != 0) return cmp;
681 }
682 }
683 return 0;
684 }

686public:
687 void accept(SymbolicVisitor& visitor) override { SymbolicVisitor::DepthGuard guard(visitor); visitor.visit(*this); }

689 std::shared_ptr<SymbolicNode> clone() const override {
690 if (std::holds_alternative<DenseStorage>(storage)) {
691 const auto& dense = std::get<DenseStorage>(storage);
692 DenseStorage new_dense;
693 new_dense.reserve(dense.size());
694 // dense 存储允许保留 nullptr 表示零元素，clone 时必须保留这种空槽，
695 // 不能直接调用 e->clone() 触发空指针解引用。
696 for (const auto& e : dense) {
697 new_dense.push_back(e ? e->clone() : nullptr);
698 }
699 return std::make_shared<MatrixNode>(rows, cols, std::move(new_dense));
700 } else {
701 const auto& sparse = std::get<SparseStorage>(storage);
702 SparseStorage new_sparse;
703 for(const auto& [idx, node] : sparse) {
704 new_sparse[idx] = node ? node->clone() : nullptr;
705 }
706 return std::make_shared<MatrixNode>(rows, cols, std::move(new_sparse));
707 }
708 }

710 /**
711 * @brief 获取指定位置的矩阵元素。
712 * @param r 行索引（从 0 开始）
713 * @param c 列索引（从 0 开始）
714 * @return 对应节点，越界返回 nullptr，稀疏缺失返回零节点
715 */
716 std::shared_ptr<SymbolicNode> get(size_t r, size_t c) const {
717 if (r >= rows || c >= cols) return nullptr;
718
719 size_t idx = r * cols + c;
720 if (std::holds_alternative<DenseStorage>(storage)) {
721 return std::get<DenseStorage>(storage)[idx];
722 } else {
723 const auto& sparse = std::get<SparseStorage>(storage);
724 auto it = sparse.find(idx);
725 if (it != sparse.end()) return it->second;
726 return std::make_shared<NumberNode>(0.0);
727 }
728 }

735};

737inline std::variant<MatrixNode::DenseStorage, MatrixNode::SparseStorage> MatrixNode::create_storage_from_grid(
738 const std::vector<std::vector<std::shared_ptr<SymbolicNode>>>& grid, size_t total_elements, size_t ncols) {
739
740 size_t non_zeros = 0;
741 for (const auto& row : grid) {
742 for (const auto& item : row) {
743 if (item && !item->is_zero()) non_zeros++;
744 }
745 }
746
747 bool use_sparse = total_elements > 0 && ((double)non_zeros / total_elements < 0.2);
748
749 if (use_sparse) {
750 SparseStorage s;
751 size_t r = 0;
752 for (const auto& row : grid) {
753 size_t c = 0;
754 for (const auto& item : row) {
755 if (item && !item->is_zero()) {
756 s[r * ncols + c] = item;
757 }
758 c++;
759 }
760 r++;
761 }
762 return s;
763 } else {
764 DenseStorage d;
765 d.reserve(total_elements);
766 for (const auto& row : grid) {
767 d.insert(d.end(), row.begin(), row.end());
768 }
769 return d;
770 }
