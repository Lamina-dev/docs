2 * @file query_interface.hpp
3 * @brief QueryInterface class — unified tri-state query API for property questions on arbitrary expressions.
4 *
5 * The QueryInterface is the single public entry point for property queries on SymbolicExpr trees.
6 * It handles special cases (null root, NaN, Infinity, Matrix, Relational, Logical nodes) and
7 * delegates to the InferenceEngine for NumberNode, VariableNode, and composite node queries.
8 */
9#pragma once

15namespace lamina {
16
17// Forward declaration

21/**
22 * @brief Unified query API for property questions on arbitrary SymbolicExpr trees.
23 *
24 * Provides query_positive, query_negative, query_nonnegative, query_real,
25 * query_integer, and query_nonzero methods that return Tribool results.
26 *
27 * Dispatch logic:
28 * - Null root node → Unknown
29 * - MatrixNode, RelationalNode, LogicalNode → Unknown
30 * - NumberNode with NaN → False for integer, Unknown for sign queries
31 * - FunctionNode::Infinity → derive sign from infinity's sign, False for integer
32 * - All other nodes → delegate to InferenceEngine
33 */
34class LAMINA_API QueryInterface {
35public:
36 /**
37 * @brief Construct a QueryInterface bound to an AssumptionContext.
38 * @param ctx The AssumptionContext used for property lookups and inference
39 */
40 explicit QueryInterface(const AssumptionContext& ctx);
41
42 /// @name Public query methods
43 /// @{
44
45 /// Query whether the expression is positive (> 0).
46 Tribool query_positive(const SymbolicExpr& expr) const;
47
48 /// Query whether the expression is negative (< 0).
49 Tribool query_negative(const SymbolicExpr& expr) const;
50
51 /// Query whether the expression is non-negative (>= 0).
52 Tribool query_nonnegative(const SymbolicExpr& expr) const;
53
54 /// Query whether the expression is real.
55 Tribool query_real(const SymbolicExpr& expr) const;
56
57 /// Query whether the expression is an integer.
58 Tribool query_integer(const SymbolicExpr& expr) const;
59
60 /// Query whether the expression is non-zero (!= 0).
61 Tribool query_nonzero(const SymbolicExpr& expr) const;
62
63 /// @}
64
65private:
66 const AssumptionContext& ctx_;
67
68 /// Check if the root node is a special case that should return Unknown immediately.
69 /// Returns true if the node is null, MatrixNode, RelationalNode, or LogicalNode.
70 bool is_unhandled_type(const std::shared_ptr<SymbolicNode>& node) const;
71
72 /// Check if a NumberNode holds NaN.
73 bool is_nan_number(const NumberNode& node) const;
74
75 /// Check if a FunctionNode represents infinity (FuncType::Infinity).
76 bool is_infinity_node(const std::shared_ptr<SymbolicNode>& node) const;
77
78 /// Determine the sign of an infinity expression.
79 /// Positive infinity: FunctionNode::Infinity directly.
80 /// Negative infinity: MultiplyNode(-1, FunctionNode::Infinity).
81 /// Returns +1 for positive infinity, -1 for negative infinity, 0 if indeterminate.
82 int get_infinity_sign(const std::shared_ptr<SymbolicNode>& node) const;
83};
