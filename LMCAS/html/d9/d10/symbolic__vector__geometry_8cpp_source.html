37double vector_angle(
38 const std::vector<std::shared_ptr<SymbolicExpr>>& a,
39 const std::vector<std::shared_ptr<SymbolicExpr>>& b
40) {
41
42 if (a.size() != b.size() || a.empty()) {
43 return std::numeric_limits<double>::quiet_NaN();
44 }
45
46 double norm_a = 0, norm_b = 0, dot = 0;
47 bool numeric = true;
48 for (size_t i = 0; i < a.size(); ++i) {
49 if (!a[i]->is_number() || !b[i]->is_number()) {
50 numeric = false;
51 break;
52 }
53 auto na_var = a[i]->get_number_value();
54 auto nb_var = b[i]->get_number_value();
55
56 auto to_double = [](const auto& v) -> double {
57 using T = std::decay_t<decltype(v)>;
58 if constexpr (std::is_same_v<T, int>) return static_cast<double>(v);
59 else if constexpr (std::is_same_v<T, BigInt>) return v.to_double();
60 else if constexpr (std::is_same_v<T, Rational>) return v.to_double();
61 else return 0.0;
62 };
63 double na = std::visit(to_double, na_var);
64 double nb = std::visit(to_double, nb_var);
65 norm_a += na * na;
66 norm_b += nb * nb;
67 dot += na * nb;
68 }
69 if (!numeric) return std::numeric_limits<double>::quiet_NaN();
70 if (norm_a == 0.0 || norm_b == 0.0) {
71 return std::numeric_limits<double>::quiet_NaN();
72 }
73 double cosv = dot / (std::sqrt(norm_a) * std::sqrt(norm_b));
74 // Clamp into [-1, 1] to avoid std::acos domain errors caused by rounding.
75 if (cosv > 1.0) cosv = 1.0;
76 else if (cosv < -1.0) cosv = -1.0;
77 return std::acos(cosv);
78}

81 const LineSymbolic& line,
82 const PlaneSymbolic& plane
83) {
84
85 auto n_dot_a = vector_dot(plane.normal, line.point);
86 auto n_dot_b = vector_dot(plane.normal, line.direction);
87 // If n . direction == 0 the line is parallel to the plane: either lies in
88 // the plane (infinite intersections) or has no intersection. In both cases
89 // we cannot return a single point, so signal "no unique intersection" via
90 // an empty vector instead of dividing by zero.
91 auto n_dot_b_simpl = n_dot_b ? n_dot_b->simplify() : nullptr;
92 if (!n_dot_b_simpl || n_dot_b_simpl->is_zero()) {
93 return {};
94 }
95 auto t = SymbolicExpr::divide(SymbolicExpr::add(plane.d, SymbolicExpr::multiply(SymbolicExpr::number(-1), n_dot_a)), n_dot_b);
96 std::vector<std::shared_ptr<SymbolicExpr>> intersection;
97 for (size_t i = 0; i < line.point.size(); ++i) {
98 intersection.push_back(SymbolicExpr::add(line.point[i], SymbolicExpr::multiply(t, line.direction[i])));
99 }
100 return intersection;
