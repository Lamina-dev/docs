108 // Condition: wildcard `_u` is bound to the integration variable itself.
109 auto u_is_var = [](const std::string& wc) {
110 return [wc](const MatchMap& m, const std::string& var) -> bool {
111 auto it = m.find(wc);
112 if (it == m.end()) return false;
113 auto v = std::dynamic_pointer_cast<VariableNode>(it->second.root);
114 return v && v->name == var;
115 };
116 };

118 // Condition: `_u` is the integration variable AND `_a` does not depend on it.
119 auto u_is_var_a_indep = [](const std::string& u_wc, const std::string& a_wc) {
120 return [u_wc, a_wc](const MatchMap& m, const std::string& var) -> bool {
121 auto it_u = m.find(u_wc);
122 if (it_u == m.end()) return false;
123 auto v = std::dynamic_pointer_cast<VariableNode>(it_u->second.root);
124 if (!v || v->name != var) return false;
125 auto it_a = m.find(a_wc);
126 if (it_a == m.end()) return false;
127 return !depends_on_var(it_a->second.root, var);
