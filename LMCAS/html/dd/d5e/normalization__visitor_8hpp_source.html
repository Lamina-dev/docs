17 * @brief 计算 AST 节点的多项式次数
18 * @param node 输入节点
19 * @return 节点对应的多项式次数
20 */
21inline int get_node_degree_helper(const std::shared_ptr<SymbolicNode>& node) {
22 if (!node) return 0;
23 if (std::dynamic_pointer_cast<VariableNode>(node)) return 1;
24 if (auto p = std::dynamic_pointer_cast<PowerNode>(node)) {
25 if (auto e = std::dynamic_pointer_cast<NumberNode>(p->exponent)) {
26 if (std::holds_alternative<BigInt>(e->value)) return (int)std::get<BigInt>(e->value).to_int();
27 if (std::holds_alternative<lmmc_real_t>(e->value)) return (int)std::get<lmmc_real_t>(e->value);
28 }
29 return 1;
30 }
31 if (auto m = std::dynamic_pointer_cast<MultiplyNode>(node)) {
32 int d = 0;
33 for (auto& op : m->operands) d += get_node_degree_helper(op);
34 return d;
35 }
36 return 0;
37}

39/** @brief 节点比较器，按多项式次数降序排列，用于同类项合并 */
40struct NodeCompare {
41 bool operator()(const std::shared_ptr<SymbolicNode>& lhs, const std::shared_ptr<SymbolicNode>& rhs) const {
42 if (!lhs && !rhs) return false;
43 if (!lhs) return true;
44 if (!rhs) return false;
45
46 int d1 = get_node_degree_helper(lhs);
47 int d2 = get_node_degree_helper(rhs);
48 if (d1 != d2) return d1 > d2;
49
50 bool isNum1 = std::dynamic_pointer_cast<NumberNode>(lhs) != nullptr;
51 bool isNum2 = std::dynamic_pointer_cast<NumberNode>(rhs) != nullptr;
52 if (isNum1 != isNum2) return isNum1;
53
54 if (lhs->type_priority() != rhs->type_priority()) {
55 return lhs->type_priority() < rhs->type_priority();
56 }
57
58 return lhs->compare(*rhs) < 0;
59 }
60};

126inline bool check_negative_arg(const std::shared_ptr<SymbolicNode>& arg, std::shared_ptr<SymbolicNode>& out_positive) {
127 if (auto num = std::dynamic_pointer_cast<NumberNode>(arg)) {
128 if (std::holds_alternative<lmmc_real_t>(num->value) && std::get<lmmc_real_t>(num->value) < 0) {
129 out_positive = std::make_shared<NumberNode>(std::abs(std::get<lmmc_real_t>(num->value)));
130 return true;
131 }
132 if (std::holds_alternative<BigInt>(num->value) && std::get<BigInt>(num->value).to_double() < 0) {
133 out_positive = std::make_shared<NumberNode>(std::get<BigInt>(num->value) * BigInt(-1));
134 return true;
135 }
136 if (std::holds_alternative<Rational>(num->value) && std::get<Rational>(num->value).to_double() < 0) {
137 out_positive = std::make_shared<NumberNode>(std::get<Rational>(num->value) * Rational(-1));
138 return true;
139 }
140 return false;
141 }
142 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(arg)) {
143 if (!mul->operands.empty()) {
144 if (auto num = std::dynamic_pointer_cast<NumberNode>(mul->operands[0])) {
145 bool is_neg = false;
146 std::shared_ptr<NumberNode> pos_num = nullptr;
147 if (std::holds_alternative<lmmc_real_t>(num->value) && std::get<lmmc_real_t>(num->value) < 0) {
148 is_neg = true;
149 pos_num = std::make_shared<NumberNode>(std::abs(std::get<lmmc_real_t>(num->value)));
150 } else if (std::holds_alternative<BigInt>(num->value) && std::get<BigInt>(num->value).to_double() < 0) {
151 is_neg = true;
152 pos_num = std::make_shared<NumberNode>(std::get<BigInt>(num->value) * BigInt(-1));
153 } else if (std::holds_alternative<Rational>(num->value) && std::get<Rational>(num->value).to_double() < 0) {
154 is_neg = true;
155 pos_num = std::make_shared<NumberNode>(std::get<Rational>(num->value) * Rational(-1));
156 }
157
158 if (is_neg) {
159 std::vector<std::shared_ptr<SymbolicNode>> new_ops = mul->operands;
160
161 bool is_minus_one = false;
162 if (std::holds_alternative<lmmc_real_t>(num->value)) {
163 int eq_1;
164 lmmc_double_nearly_equal_tol(std::get<lmmc_real_t>(num->value), -1.0, 1e-9, 1e-9, &eq_1);
165 is_minus_one = (eq_1 != 0);
166 }
167 else if (std::holds_alternative<BigInt>(num->value)) is_minus_one = (std::get<BigInt>(num->value) == BigInt(-1));
168 else if (std::holds_alternative<Rational>(num->value)) is_minus_one = (std::get<Rational>(num->value) == Rational(-1));
169
170 if (is_minus_one) {
171 new_ops.erase(new_ops.begin());
172 if (new_ops.size() == 1) out_positive = new_ops[0];
173 else out_positive = std::make_shared<MultiplyNode>(new_ops);
174 } else {
175
176 new_ops[0] = pos_num;
177 out_positive = std::make_shared<MultiplyNode>(new_ops);
178 }
179 return true;
180 }
181 }
182 }
183 }
184 return false;
185}

187/**
188 * @brief 提取节点中 π 的有理系数（如 2π/3 中的 2/3）
189 * @param node 待检查的节点
190 * @param k 输出参数，存放 π 的有理系数
191 * @return 若节点为 k*π 形式则返回 true
192 */
193inline bool get_pi_coeff(const std::shared_ptr<SymbolicNode>& node, Rational& k) {
194 if (auto v = std::dynamic_pointer_cast<VariableNode>(node)) {
195 if (v->name == "pi") {
196 k = Rational(1);
197 return true;
198 }
199 }
200 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
201 bool has_pi = false;
202 k = Rational(1);
203
204 for (const auto& op : mul->operands) {
205 if (auto v = std::dynamic_pointer_cast<VariableNode>(op)) {
206 if (v->name == "pi") {
207 if (has_pi) return false;
208 has_pi = true;
209 } else return false;
210 } else if (auto n = std::dynamic_pointer_cast<NumberNode>(op)) {
211 if (std::holds_alternative<Rational>(n->value)) k = k * std::get<Rational>(n->value);
212 else if (std::holds_alternative<BigInt>(n->value)) k = k * Rational(std::get<BigInt>(n->value));
213 else return false;
214 } else return false;
215 }
216 return has_pi;
217 }
218 return false;

242 * @brief 展开两个节点的乘积（分配律）
243 * @param lhs 左操作数
244 * @param rhs 右操作数
245 * @return 展开后的 AST 节点
246 */
247 std::shared_ptr<SymbolicNode> expand_product(const std::shared_ptr<SymbolicNode>& lhs, const std::shared_ptr<SymbolicNode>& rhs) {
248
249 auto add_lhs = std::dynamic_pointer_cast<AddNode>(lhs);
250 auto add_rhs = std::dynamic_pointer_cast<AddNode>(rhs);
251
252 if (add_lhs && add_rhs) {
253
254 std::vector<std::shared_ptr<SymbolicNode>> new_terms;
255 for (const auto& op1 : add_lhs->operands) {
256 for (const auto& op2 : add_rhs->operands) {
257 auto prod = expand_product(op1, op2);
258 new_terms.push_back(prod);
259 }
260 }
261
262 return std::make_shared<AddNode>(new_terms);
263 } else if (add_lhs) {
264
265 std::vector<std::shared_ptr<SymbolicNode>> new_terms;
266 for (const auto& op : add_lhs->operands) {
267 auto prod = expand_product(op, rhs);
268 new_terms.push_back(prod);
269 }
270 return std::make_shared<AddNode>(new_terms);
271 } else if (add_rhs) {
272
273 std::vector<std::shared_ptr<SymbolicNode>> new_terms;
274 for (const auto& op : add_rhs->operands) {
275 auto prod = expand_product(lhs, op);
276 new_terms.push_back(prod);
277 }
278 return std::make_shared<AddNode>(new_terms);
279 }

558 if (std::dynamic_pointer_cast<MatrixNode>(p->base)) { has_matrix = true; break; }
559 }
560 }
561
562 if (has_matrix) {
563 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
564 std::shared_ptr<NumberNode> scalar_part = std::make_shared<NumberNode>(BigInt(1));
565
566 for(const auto& op : sc) {
567 if (auto num = std::dynamic_pointer_cast<NumberNode>(op)) {
568 scalar_part = multiply_numbers(scalar_part, num);
569 } else {
570 new_ops.push_back(op);
571 }
572 }
573
574 std::vector<std::shared_ptr<SymbolicNode>> fused_ops;
575 if (!new_ops.empty()) fused_ops.push_back(new_ops[0]);
576
577 for(size_t i=1; i<new_ops.size(); ++i) {
578 auto left = fused_ops.back();
579 auto right = new_ops[i];
580
581 auto m_left = std::dynamic_pointer_cast<MatrixNode>(left);
582 auto m_right = std::dynamic_pointer_cast<MatrixNode>(right);
583
584 if (m_left && m_right) {
585
586 if (m_left->cols == m_right->rows) {
587
588 if (std::holds_alternative<MatrixNode::DenseStorage>(m_left->storage) &&
589 std::holds_alternative<MatrixNode::DenseStorage>(m_right->storage)) {
590
591 const auto& d_l = std::get<MatrixNode::DenseStorage>(m_left->storage);
592 const auto& d_r = std::get<MatrixNode::DenseStorage>(m_right->storage);
593
594 size_t R = m_left->rows;
595 size_t C = m_right->cols;
596 size_t K = m_left->cols;
597
598 MatrixNode::DenseStorage res_data;
599 res_data.reserve(R*C);
600
601 for(size_t r=0; r<R; ++r) {
602 for(size_t c=0; c<C; ++c) {
603
604 std::vector<std::shared_ptr<SymbolicNode>> sum_ops;
605 for(size_t k=0; k<K; ++k) {
606 std::vector<std::shared_ptr<SymbolicNode>> prod_ops = {
607 d_l[r*K + k], d_r[k*C + c]
608 };
609 sum_ops.push_back(std::make_shared<MultiplyNode>(prod_ops));
610 }
611
612 NormalizationVisitor elem_vis(assumptions_);
613 auto elem_node = std::make_shared<AddNode>(sum_ops);
614 elem_node->accept(elem_vis);
615
616 auto res_val = elem_vis.get_result();
617 if (auto num = std::dynamic_pointer_cast<NumberNode>(res_val)) {
618 if (std::holds_alternative<lmmc_real_t>(num->value)) {
619 lmmc_real_t v = std::get<lmmc_real_t>(num->value);
620 int eq_0, eq_1;
621 lmmc_double_nearly_equal_tol(v, 0.0, 1e-10, 1e-10, &eq_0);
622 lmmc_double_nearly_equal_tol(v, 1.0, 1e-10, 1e-10, &eq_1);
623 if (eq_0) {
624 res_val = std::make_shared<NumberNode>(BigInt(0));
625 } else if (eq_1) {
626 res_val = std::make_shared<NumberNode>(BigInt(1));
627 }
628 }
629 }
630 res_data.push_back(res_val);
631 }
632 }
633
634 fused_ops.pop_back();
635 fused_ops.push_back(std::make_shared<MatrixNode>(R, C, res_data));
636 continue;
637 }
638 }
639 }
640 fused_ops.push_back(right);
641 }
642
643 if (!scalar_part->is_one()) {
644 fused_ops.insert(fused_ops.begin(), scalar_part);
645 }
646
647 if (fused_ops.empty()) result = std::make_shared<NumberNode>(BigInt(1));
648 else if (fused_ops.size() == 1) result = fused_ops[0];
649 else result = std::make_shared<MultiplyNode>(fused_ops);
650
651 return;
652 }
653
654 std::shared_ptr<NumberNode> const_acc = std::make_shared<NumberNode>(BigInt(1));
655 std::map<std::shared_ptr<SymbolicNode>, std::shared_ptr<NumberNode>, NodeCompare> bases;
656
657 for (const auto& op : sc) {
658 if (auto num = std::dynamic_pointer_cast<NumberNode>(op)) {
659 if (num->is_zero()) {
660 result = std::make_shared<NumberNode>(BigInt(0));
661 return;
662 }
663 const_acc = multiply_numbers(const_acc, num);
664 } else {
665 std::shared_ptr<SymbolicNode> base = op;
666 std::shared_ptr<NumberNode> exp = std::make_shared<NumberNode>(BigInt(1));
667 bool is_number_power = false;
668
669 if (auto pow = std::dynamic_pointer_cast<PowerNode>(op)) {
670 // Only split base/exponent of a PowerNode if the exponent is a NumberNode.
671 // For symbolic exponents (e.g. 2^x), keep the PowerNode atomic so we
672 // don't silently drop the exponent when accumulating into `bases`.
673 auto e_num = std::dynamic_pointer_cast<NumberNode>(pow->exponent);
674 if (e_num) {
675 base = pow->base;
676 exp = e_num;
677
678 if (auto b_num = std::dynamic_pointer_cast<NumberNode>(base)) {
679 long long exp_val = 0;
680 bool exp_ok = false;
681 bool exp_is_half = false;
682
683 if (std::holds_alternative<BigInt>(e_num->value)) {
684 exp_val = (long long)std::get<BigInt>(e_num->value).to_double();
685 exp_ok = true;
686 } else if (std::holds_alternative<lmmc_real_t>(e_num->value)) {
687 lmmc_real_t d = std::get<lmmc_real_t>(e_num->value);
688 int eq_half;
689 lmmc_double_nearly_equal_tol(d, 0.5, 1e-9, 1e-9, &eq_half);
690 if (d == std::floor(d)) {
691 exp_val = (long long)d;
692 exp_ok = true;
693 } else if (eq_half) {
694 exp_is_half = true;
695 }
696 } else if (std::holds_alternative<Rational>(e_num->value)) {
697 Rational r = std::get<Rational>(e_num->value);
698 if (r.get_denominator() == BigInt(1)) {
699 exp_val = (long long)r.get_numerator().to_double();
700 exp_ok = true;
701 } else if (r.get_numerator() == BigInt(1) && r.get_denominator() == BigInt(2)) {
702 exp_is_half = true;
703 }
704 }
705
706 if (exp_ok) {
707 std::shared_ptr<NumberNode> pow_val = nullptr;
708
709 if (exp_val == -1) {
710 if (std::holds_alternative<BigInt>(b_num->value)) {
711 const auto& bi = std::get<BigInt>(b_num->value);
712 if (!bi.is_zero()) pow_val = std::make_shared<NumberNode>(Rational(BigInt(1), bi));
713 } else if (std::holds_alternative<Rational>(b_num->value)) {
714 Rational r = std::get<Rational>(b_num->value);
715 if (!r.get_numerator().is_zero()) pow_val = std::make_shared<NumberNode>(Rational(r.get_denominator(), r.get_numerator()));
716 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
717 lmmc_real_t bv = std::get<lmmc_real_t>(b_num->value);
718 if (bv != 0.0) pow_val = std::make_shared<NumberNode>(1.0 / bv);
719 }
720 } else if (exp_val == 0) {
721 pow_val = std::make_shared<NumberNode>(BigInt(1));
722 } else if (std::abs(exp_val) > 0 && std::abs(exp_val) < 64) {
723
724 long long abs_exp = std::abs(exp_val);
725 std::shared_ptr<NumberNode> base_pow_val = nullptr;
726
727 if (std::holds_alternative<BigInt>(b_num->value)) {
728 BigInt b = std::get<BigInt>(b_num->value);
729 BigInt res(1);
730 for(int k=0;k<abs_exp;++k) res = res * b;
731 base_pow_val = std::make_shared<NumberNode>(res);
732 } else if (std::holds_alternative<Rational>(b_num->value)) {
733 Rational b = std::get<Rational>(b_num->value);
734 Rational res(1);
735 for(int k=0;k<abs_exp;++k) res = res * b;
736 base_pow_val = std::make_shared<NumberNode>(res);
737 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
738 lmmc_real_t b = std::get<lmmc_real_t>(b_num->value);
739 lmmc_real_t res = 1.0;
740 for(int k=0;k<abs_exp;++k) res *= b;
741 base_pow_val = std::make_shared<NumberNode>(res);
742 }
743
744 if (base_pow_val) {
745 if (exp_val > 0) {
746 pow_val = base_pow_val;
747 } else {
748
749 if (std::holds_alternative<BigInt>(base_pow_val->value)) {
750 const auto& bi = std::get<BigInt>(base_pow_val->value);
751 if (!bi.is_zero()) pow_val = std::make_shared<NumberNode>(Rational(BigInt(1), bi));
752 } else if (std::holds_alternative<Rational>(base_pow_val->value)) {
753 Rational r = std::get<Rational>(base_pow_val->value);
754 if (!r.get_numerator().is_zero()) pow_val = std::make_shared<NumberNode>(Rational(r.get_denominator(), r.get_numerator()));
755 } else if (std::holds_alternative<lmmc_real_t>(base_pow_val->value)) {
756 lmmc_real_t bv = std::get<lmmc_real_t>(base_pow_val->value);
757 if (bv != 0.0) pow_val = std::make_shared<NumberNode>(1.0 / bv);
758 }
759 }
760 }
761 }
762
763 if (pow_val) {
764 const_acc = multiply_numbers(const_acc, pow_val);
765 is_number_power = true;
766 }
767 } else if (exp_is_half) {
768
769 std::shared_ptr<NumberNode> root_val = nullptr;
770 if (std::holds_alternative<BigInt>(b_num->value)) {
771
772 double d = std::get<BigInt>(b_num->value).to_double();
773 if (d >= 0) {
774 double r = std::sqrt(d);
775 int eq_r;
776 lmmc_double_nearly_equal_tol(r, std::round(r), 1e-9, 1e-9, &eq_r);
777 if (eq_r) {
778 BigInt bi((long long)std::round(r));
779 if (bi * bi == std::get<BigInt>(b_num->value)) {
780 root_val = std::make_shared<NumberNode>(bi);
781 }
782 }
783 }
784 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
785 double d = std::get<lmmc_real_t>(b_num->value);
786 if (d >= 0) root_val = std::make_shared<NumberNode>(std::sqrt(d));
787 }
788
789 if (root_val) {
790 const_acc = multiply_numbers(const_acc, root_val);
791 is_number_power = true;
792 }
793 }
794 }
795 }
796 }
797
798 if (!is_number_power) {
799 auto it = bases.find(base);
800 if (it == bases.end()) {
801 bases[base] = exp;
802 } else {
803 bases[base] = add_numbers(it->second, exp);
804 }
805 }
806 }
807 }
808
809 std::vector<std::shared_ptr<SymbolicNode>> final_ops;
810 if (!const_acc->is_one()) {
811 final_ops.push_back(const_acc);
812 }
813
814 std::vector<std::shared_ptr<SymbolicNode>> var_ops;
815 for (auto const& [base, exp] : bases) {
816 if (exp->is_zero()) {
817 } else if (exp->is_one()) {
818 var_ops.push_back(base);
819 } else {
820 var_ops.push_back(std::make_shared<PowerNode>(base, exp));
821 }
822 }
823
824 std::sort(var_ops.begin(), var_ops.end(), [](const std::shared_ptr<SymbolicNode>& l, const std::shared_ptr<SymbolicNode>& r) {
825 int d1 = get_node_degree_helper(l);
826 int d2 = get_node_degree_helper(r);
827 if (d1 != d2) return d1 > d2;
828 return l->compare(*r) < 0;
829 });
830
831 final_ops.insert(final_ops.end(), var_ops.begin(), var_ops.end());
832
833 if (final_ops.empty()) result = std::make_shared<NumberNode>(BigInt(1));
834 else if (final_ops.size() == 1) result = final_ops[0];
835 else result = std::make_shared<MultiplyNode>(final_ops);
836 }

838 void visit(PowerNode& node) override {
839 node.base->accept(*this);
840 auto s_base = result;
841 node.exponent->accept(*this);
842 auto s_exp = result;
843
844 if (s_exp->is_zero()) {
845 result = std::make_shared<NumberNode>(BigInt(1));
846 return;
847 }
848 if (s_exp->is_one()) {
849 result = s_base;
850 return;
851 }
852
853 if (s_base->is_zero()) {
854 // 0^x = 0 only when x is provably positive; otherwise keep the
855 // PowerNode so domain issues (e.g. 0^(-1/2)) are not silently
856 // turned into 0. This mirrors SymbolicFactory::create_power.
857 if (s_exp->is_positive()) {
858 result = std::make_shared<NumberNode>(BigInt(0));
859 return;
860 }
861 result = std::make_shared<PowerNode>(s_base, s_exp);
862 return;
863 }
864 if (s_base->is_one()) {
865 result = std::make_shared<NumberNode>(BigInt(1));
866 return;
867 }
868
869 if (auto b_num = std::dynamic_pointer_cast<NumberNode>(s_base)) {
870 if (auto e_num = std::dynamic_pointer_cast<NumberNode>(s_exp)) {
871 long long exp_val = 0;
872 bool exp_ok = false;
873 bool exp_is_half = false;
874
875 if (std::holds_alternative<BigInt>(e_num->value)) {
876 exp_val = (long long)std::get<BigInt>(e_num->value).to_double();
877 exp_ok = true;
878 } else if (std::holds_alternative<lmmc_real_t>(e_num->value)) {
879 lmmc_real_t d = std::get<lmmc_real_t>(e_num->value);
880 if (d == std::floor(d)) {
881 exp_val = (long long)d;
882 exp_ok = true;
883 } else if (std::abs(d - 0.5) < 1e-9) {
884 exp_is_half = true;
885 }
886 } else if (std::holds_alternative<Rational>(e_num->value)) {
887 Rational r = std::get<Rational>(e_num->value);
888 if (r.get_denominator() == BigInt(1)) {
889 exp_val = (long long)r.get_numerator().to_double();
890 exp_ok = true;
891 } else if (r.get_numerator() == BigInt(1) && r.get_denominator() == BigInt(2)) {
892 exp_is_half = true;
893 }
894 }
895
896 if (exp_ok) {
897 std::shared_ptr<NumberNode> pow_val = nullptr;
898 if (exp_val == -1) {
899 if (std::holds_alternative<BigInt>(b_num->value)) {
900 pow_val = std::make_shared<NumberNode>(Rational(BigInt(1), std::get<BigInt>(b_num->value)));
901 } else if (std::holds_alternative<Rational>(b_num->value)) {
902 Rational r = std::get<Rational>(b_num->value);
903 if (!r.get_numerator().is_zero()) pow_val = std::make_shared<NumberNode>(Rational(r.get_denominator(), r.get_numerator()));
904 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
905 pow_val = std::make_shared<NumberNode>(1.0 / std::get<lmmc_real_t>(b_num->value));
906 }
907 } else if (exp_val == 0) {
908 pow_val = std::make_shared<NumberNode>(BigInt(1));
909 } else if (std::abs(exp_val) > 0 && std::abs(exp_val) < 64) {
910
911 long long abs_exp = std::abs(exp_val);
912 std::shared_ptr<NumberNode> base_pow_val = nullptr;
913
914 if (std::holds_alternative<BigInt>(b_num->value)) {
915 BigInt b = std::get<BigInt>(b_num->value);
916 BigInt res(1);
917 for(int k=0;k<abs_exp;++k) res = res * b;
918 base_pow_val = std::make_shared<NumberNode>(res);
919 } else if (std::holds_alternative<Rational>(b_num->value)) {
920 Rational b = std::get<Rational>(b_num->value);
921 Rational res(1);
922 for(int k=0;k<abs_exp;++k) res = res * b;
923 base_pow_val = std::make_shared<NumberNode>(res);
924 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
925 lmmc_real_t b = std::get<lmmc_real_t>(b_num->value);
926 lmmc_real_t res = 1.0;
927 for(int k=0;k<abs_exp;++k) res *= b;
928 base_pow_val = std::make_shared<NumberNode>(res);
929 }
930
931 if (base_pow_val) {
932 if (exp_val > 0) {
933 pow_val = base_pow_val;
934 } else {
935
936 if (std::holds_alternative<BigInt>(base_pow_val->value)) {
937 pow_val = std::make_shared<NumberNode>(Rational(BigInt(1), std::get<BigInt>(base_pow_val->value)));
938 } else if (std::holds_alternative<Rational>(base_pow_val->value)) {
939 Rational r = std::get<Rational>(base_pow_val->value);
940 pow_val = std::make_shared<NumberNode>(Rational(r.get_denominator(), r.get_numerator()));
941 } else if (std::holds_alternative<lmmc_real_t>(base_pow_val->value)) {
942 pow_val = std::make_shared<NumberNode>(1.0 / std::get<lmmc_real_t>(base_pow_val->value));
943 }
944 }
945 }
946 }
947
948 if (pow_val) {
949 result = pow_val;
950 return;
951 }
952 } else if (exp_is_half) {
953 if (std::holds_alternative<BigInt>(b_num->value)) {
954 double d = std::get<BigInt>(b_num->value).to_double();
955 if (d >= 0) {
956 double r = std::sqrt(d);
957 if (std::abs(r - std::round(r)) < 1e-9) {
958 BigInt bi((long long)std::round(r));
959 if (bi * bi == std::get<BigInt>(b_num->value)) {
960 result = std::make_shared<NumberNode>(bi);
961 return;
962 }
963 }
964
965 BigInt val = std::get<BigInt>(b_num->value);
966 if (val > BigInt(0) && val < BigInt(1000000)) {
967 long long v_ll = (long long)val.to_double();
968 long long root = (long long)std::sqrt(v_ll);
969 for (long long i = root; i >= 2; --i) {
970 if (v_ll % (i*i) == 0) {
971 long long s_ll = i;
972 long long k_ll = v_ll / (i*i);
973
974 auto s_node = std::make_shared<NumberNode>(BigInt(s_ll));
975 auto k_node = std::make_shared<NumberNode>(BigInt(k_ll));
976 auto half_node = std::make_shared<NumberNode>(Rational(1, 2));
977 auto pow_node = std::make_shared<PowerNode>(k_node, half_node);
978
979 result = SymbolicFactory::create_multiply({s_node, pow_node});
980 return;
981 }
982 }
983 }
984 }
985 } else if (std::holds_alternative<lmmc_real_t>(b_num->value)) {
986 double d = std::get<lmmc_real_t>(b_num->value);
987 if (d >= 0) {
988 result = std::make_shared<NumberNode>(std::sqrt(d));
989 return;
990 }
991 }
992 }
993 }
994 }
995
996 if (auto m_base = std::dynamic_pointer_cast<MultiplyNode>(s_base)) {
997 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
998 for(auto& op : m_base->operands) {
999
1000 auto term_pow = std::make_shared<PowerNode>(op, s_exp);
1001 term_pow->accept(*this);
1002
1003 if (auto mul_res = std::dynamic_pointer_cast<MultiplyNode>(result)) {
1004 new_ops.insert(new_ops.end(), mul_res->operands.begin(), mul_res->operands.end());
1005 } else {
1006 new_ops.push_back(result);
1007 }
1008 }
1009
1010 auto final_mul = std::make_shared<MultiplyNode>(new_ops);
1011 final_mul->accept(*this);
1012 return;
1013 }
1014
1015 if (auto p_base = std::dynamic_pointer_cast<PowerNode>(s_base)) {
1016
1017 std::vector<std::shared_ptr<SymbolicNode>> exp_ops;
1018 exp_ops.push_back(p_base->exponent);
1019 exp_ops.push_back(s_exp);
1020
1021 auto mul_exp = std::make_shared<MultiplyNode>(exp_ops);
1022 mul_exp->accept(*this);
1023
1024 auto new_pow = std::make_shared<PowerNode>(p_base->base, result);
1025 new_pow->accept(*this);
1026 return;
1027 }
1028
1029 result = std::make_shared<PowerNode>(s_base, s_exp);
1030 }

1032 void visit(FunctionNode& node) override {
1033 std::vector<std::shared_ptr<SymbolicNode>> s_args;
1034 for(auto& a : node.arguments) {
1035 a->accept(*this);
1036 s_args.push_back(result);
1037 }
1038
1039 if (s_args.size() == 1) {
1040 if (auto num = std::dynamic_pointer_cast<NumberNode>(s_args[0])) {
1041 lmmc_real_t val = 0.0;
1042 if (std::holds_alternative<lmmc_real_t>(num->value)) val = std::get<lmmc_real_t>(num->value);
1043 else if (std::holds_alternative<BigInt>(num->value)) val = (lmmc_real_t)std::get<BigInt>(num->value).to_double();
1044 else if (std::holds_alternative<Rational>(num->value)) val = (lmmc_real_t)std::get<Rational>(num->value).to_double();
1045
1046 switch (node.type) {
1047 case FunctionNode::FuncType::Sin:
1048 {
1049 int eq; lmmc_double_nearly_equal_tol(val, 0.0, 1e-12, 1e-12, &eq);
1050 if (eq) { result = std::make_shared<NumberNode>(BigInt(0)); return; }
1051 break;
1052 }
1053 case FunctionNode::FuncType::Cos:
1054 {
1055 int eq; lmmc_double_nearly_equal_tol(val, 0.0, 1e-12, 1e-12, &eq);
1056 if (eq) { result = std::make_shared<NumberNode>(BigInt(1)); return; }
1057 break;
1058 }
1059 case FunctionNode::FuncType::Tan:
1060 {
1061 int eq; lmmc_double_nearly_equal_tol(val, 0.0, 1e-12, 1e-12, &eq);
1062 if (eq) { result = std::make_shared<NumberNode>(BigInt(0)); return; }
1063 break;
1064 }
1065 case FunctionNode::FuncType::Exp:
1066 {
1067 int eq; lmmc_double_nearly_equal_tol(val, 0.0, 1e-12, 1e-12, &eq);
1068 if (eq) { result = std::make_shared<NumberNode>(BigInt(1)); return; }
1069 break;
1070 }
1071 case FunctionNode::FuncType::Ln:
1072 {
1073 int eq1, eq0;
1074 lmmc_double_nearly_equal_tol(val, 1.0, 1e-12, 1e-12, &eq1);
1075 if (eq1) { result = std::make_shared<NumberNode>(BigInt(0)); return; }
1076
1077 lmmc_double_nearly_equal_tol(val, 0.0, 1e-12, 1e-12, &eq0);
1078 if (eq0) {
1079 std::vector<std::shared_ptr<SymbolicNode>> inf_args;
1080 auto inf = std::make_shared<FunctionNode>(FunctionNode::FuncType::Infinity, inf_args);
1081 std::vector<std::shared_ptr<SymbolicNode>> m_args = {std::make_shared<NumberNode>(BigInt(-1)), inf};
1082 result = std::make_shared<MultiplyNode>(m_args);
1083 return;
1084 }
1085 break;
1086 }
1087 case FunctionNode::FuncType::Log:
1088 if (s_args.size() == 2) {
1089
1090 std::vector<std::shared_ptr<SymbolicNode>> args_x = { s_args[0] };
1091 std::vector<std::shared_ptr<SymbolicNode>> args_b = { s_args[1] };
1092 auto ln_x = std::make_shared<FunctionNode>(FunctionNode::FuncType::Ln, args_x);
1093 auto ln_b = std::make_shared<FunctionNode>(FunctionNode::FuncType::Ln, args_b);
1094 auto ln_b_inv = std::make_shared<PowerNode>(ln_b, std::make_shared<NumberNode>(BigInt(-1)));
1095 std::vector<std::shared_ptr<SymbolicNode>> m_args = { ln_x, ln_b_inv };

1454 /**
1455 * @brief Attempt assumption-based simplification on a node.
1456 *
1457 * Applies the following rules when an AssumptionContext is available:
1458 * - sqrt(x²) → x when x is NonNegative
1459 * - sqrt(x²) → abs(x) when x is Real (but not NonNegative)
1460 * - abs(x) → x when x is Positive
1461 * - abs(x) → -x when x is Negative
1462 *
1463 * @param node The node to attempt simplification on
1464 * @return Simplified node if a rule applied, or nullptr if no rule matched
1465 */
1466 std::shared_ptr<SymbolicNode> try_assumption_simplify(
1467 const std::shared_ptr<SymbolicNode>& node) {
1468 if (!assumptions_) return nullptr;
1469
1470 auto func = std::dynamic_pointer_cast<FunctionNode>(node);
1471 if (!func || func->arguments.size() != 1) return nullptr;
1472
1473 const auto& arg = func->arguments[0];
1474
1475 // Rule: sqrt(x²) → x when x is NonNegative
1476 // Rule: sqrt(x²) → abs(x) when x is Real (but not NonNegative)
1477 if (func->type == FunctionNode::FuncType::Sqrt) {
1478 // Check if argument is a PowerNode with exponent 2
1479 auto pow = std::dynamic_pointer_cast<PowerNode>(arg);
1480 if (pow) {
1481 auto exp_num = std::dynamic_pointer_cast<NumberNode>(pow->exponent);
1482 if (exp_num) {
1483 bool is_exp_two = false;
1484 if (std::holds_alternative<BigInt>(exp_num->value)) {
1485 is_exp_two = (std::get<BigInt>(exp_num->value) == BigInt(2));
1486 } else if (std::holds_alternative<lmmc_real_t>(exp_num->value)) {
1487 int eq;
1488 lmmc_double_nearly_equal_tol(std::get<lmmc_real_t>(exp_num->value), 2.0, 1e-12, 1e-12, &eq);
1489 is_exp_two = (eq != 0);
1490 } else if (std::holds_alternative<Rational>(exp_num->value)) {
1491 is_exp_two = (std::get<Rational>(exp_num->value) == Rational(2));
1492 }
1493
1494 if (is_exp_two) {
1495 // We have sqrt(base²) — query the base's properties
1496 SymbolicExpr base_expr(pow->base);
1497
1498 lamina::Tribool nonneg = assumptions_->is_nonnegative(base_expr);
1499 if (nonneg == lamina::Tribool::True) {
1500 // sqrt(x²) → x when x is NonNegative
1501 return pow->base;
1502 }
1503
1504 lamina::Tribool real = assumptions_->is_real(base_expr);
1505 if (real == lamina::Tribool::True) {
1506 // sqrt(x²) → abs(x) when x is Real (but not NonNegative)
1507 std::vector<std::shared_ptr<SymbolicNode>> abs_args = { pow->base };
1508 return std::make_shared<FunctionNode>(FunctionNode::FuncType::Abs, abs_args);
1509 }
1510 }
1511 }
1512 }
1513 }
1514
1515 // Rule: abs(x) → x when x is Positive
1516 // Rule: abs(x) → -x when x is Negative
1517 if (func->type == FunctionNode::FuncType::Abs) {
1518 SymbolicExpr arg_expr(arg);
1519
1520 lamina::Tribool pos = assumptions_->is_positive(arg_expr);
1521 if (pos == lamina::Tribool::True) {
1522 // abs(x) → x when x is Positive
1523 return arg;
1524 }
1525
1526 lamina::Tribool neg = assumptions_->is_negative(arg_expr);
1527 if (neg == lamina::Tribool::True) {
1528 // abs(x) → -x when x is Negative

1530 std::make_shared<NumberNode>(BigInt(-1)), arg
1531 };
1532 return std::make_shared<MultiplyNode>(mul_ops);
1533 }
1534 }
