15int PropertyStore::domain_specificity(Domain domain) const {
16 switch (domain) {
17 case Domain::Complex: return 0;
18 case Domain::Real: return 1;
19 case Domain::Rational: return 2;
20 case Domain::Integer: return 3;
21 case Domain::Natural: return 4;
22 case Domain::PositiveInt: return 5;
23 }
24 return 0;
25}

27std::vector<Domain> PropertyStore::get_ancestor_domains(Domain domain) const {
28 std::vector<Domain> ancestors;
29 switch (domain) {
30 case Domain::PositiveInt:
31 ancestors.push_back(Domain::Natural);
32 ancestors.push_back(Domain::Integer);
33 ancestors.push_back(Domain::Rational);
34 ancestors.push_back(Domain::Real);
35 ancestors.push_back(Domain::Complex);
36 break;
37 case Domain::Natural:
38 ancestors.push_back(Domain::Integer);
39 ancestors.push_back(Domain::Rational);
40 ancestors.push_back(Domain::Real);
41 ancestors.push_back(Domain::Complex);
42 break;
43 case Domain::Integer:
44 ancestors.push_back(Domain::Rational);
45 ancestors.push_back(Domain::Real);
46 ancestors.push_back(Domain::Complex);
47 break;
48 case Domain::Rational:
49 ancestors.push_back(Domain::Real);
50 ancestors.push_back(Domain::Complex);
51 break;
52 case Domain::Real:
53 ancestors.push_back(Domain::Complex);
54 break;
55 case Domain::Complex:
56 break;
57 }
58 return ancestors;

65std::unordered_set<Sign, SignHash> PropertyStore::get_implied_signs(Sign sign) const {
66 std::unordered_set<Sign, SignHash> implied;
67 switch (sign) {
68 case Sign::Positive:
69 implied.insert(Sign::NonNegative);
70 implied.insert(Sign::NonZero);
71 break;
72 case Sign::Negative:
73 implied.insert(Sign::NonPositive);
74 implied.insert(Sign::NonZero);
75 break;
76 case Sign::Zero:
77 implied.insert(Sign::NonNegative);
78 implied.insert(Sign::NonPositive);
79 break;
80 case Sign::NonNegative:
81 case Sign::NonPositive:
82 case Sign::NonZero:
83 break;
84 }
85 return implied;
86}

88void PropertyStore::check_sign_contradiction(const std::string& symbol,
89 const std::unordered_set<Sign, SignHash>& existing,
90 Sign new_sign) {
91 auto contradicts = [](Sign a, Sign b) -> bool {
92 if (a == Sign::Positive && (b == Sign::Negative || b == Sign::Zero || b == Sign::NonPositive))
93 return true;
94 if (a == Sign::Negative && (b == Sign::Positive || b == Sign::Zero || b == Sign::NonNegative))
95 return true;
96 if (a == Sign::NonNegative && b == Sign::Negative)
97 return true;
98 if (a == Sign::NonPositive && b == Sign::Positive)
99 return true;
100 if (a == Sign::Zero && (b == Sign::Positive || b == Sign::Negative || b == Sign::NonZero))
101 return true;
102 if (a == Sign::NonZero && b == Sign::Zero)
103 return true;
104 return false;
105 };
106
107 for (Sign existing_sign : existing) {
108 if (contradicts(new_sign, existing_sign)) {
109 throw std::invalid_argument(
110 "Contradiction for symbol '" + symbol + "': sign conflict between declared signs");
111 }
112 }
113}

115void PropertyStore::check_domain_sign_consistency(const std::string& symbol,
116 const SymbolProperties& props,
117 Domain new_domain) {
118 // Natural is incompatible with Negative
119 if (new_domain == Domain::Natural || new_domain == Domain::PositiveInt) {
120 if (props.signs.count(Sign::Negative)) {
121 throw std::invalid_argument(
122 "Contradiction for symbol '" + symbol +
123 "': domain Natural/PositiveInt incompatible with Negative sign");
124 }
125 }
126
127 // PositiveInt is additionally incompatible with Zero and NonPositive
128 if (new_domain == Domain::PositiveInt) {
129 if (props.signs.count(Sign::Zero)) {
130 throw std::invalid_argument(
131 "Contradiction for symbol '" + symbol +
132 "': domain PositiveInt incompatible with Zero sign");
133 }
134 if (props.signs.count(Sign::NonPositive)) {
135 throw std::invalid_argument(
136 "Contradiction for symbol '" + symbol +
137 "': domain PositiveInt incompatible with NonPositive sign");
138 }
139 }
140}

146void PropertyStore::declare_domain(const std::string& symbol, Domain domain) {
147 auto& props = properties_[symbol];
148
149 // Idempotent: same domain already set -> no-op
150 if (props.most_specific_domain == domain) {
151 return;
152 }
153
154 // Specificity preservation: if the new domain is an ancestor (less specific)
155 // of the current most-specific domain, it's a no-op.
156 if (domain_specificity(domain) < domain_specificity(props.most_specific_domain)) {
157 return;
158 }
159
160 // Check domain-sign cross-constraints before applying
161 check_domain_sign_consistency(symbol, props, domain);
162
163 // Set the new most-specific domain
164 props.most_specific_domain = domain;
165}

171void PropertyStore::declare_sign(const std::string& symbol, Sign sign) {
172 auto& props = properties_[symbol];
173
174 // Idempotent: if sign already present, no-op
175 if (props.signs.count(sign)) {
176 return;
177 }
178
179 // Collect all signs to add (the declared sign + its implications)
180 std::unordered_set<Sign, SignHash> to_add;
181 to_add.insert(sign);
182 auto implied = get_implied_signs(sign);
183 to_add.insert(implied.begin(), implied.end());
184
185 // Check each new sign against existing signs for contradictions
186 for (Sign s : to_add) {
187 if (!props.signs.count(s)) {
188 check_sign_contradiction(symbol, props.signs, s);
189 }
190 }
191
192 // Also check domain-sign cross-constraints for the new signs
193 if (props.most_specific_domain == Domain::Natural ||
194 props.most_specific_domain == Domain::PositiveInt) {
195 for (Sign s : to_add) {
196 if (s == Sign::Negative) {
197 throw std::invalid_argument(
198 "Contradiction for symbol '" + symbol +
199 "': domain Natural/PositiveInt incompatible with Negative sign");
200 }
201 }
202 }
203 if (props.most_specific_domain == Domain::PositiveInt) {
204 for (Sign s : to_add) {
205 if (s == Sign::Zero) {
206 throw std::invalid_argument(
207 "Contradiction for symbol '" + symbol +
208 "': domain PositiveInt incompatible with Zero sign");
209 }
210 if (s == Sign::NonPositive) {
211 throw std::invalid_argument(
212 "Contradiction for symbol '" + symbol +
213 "': domain PositiveInt incompatible with NonPositive sign");
214 }
215 }
216 }
217
218 // Store all signs
219 props.signs.insert(to_add.begin(), to_add.end());
220
221 // Zero implies Integer domain
222 if (sign == Sign::Zero) {
223 if (domain_specificity(Domain::Integer) > domain_specificity(props.most_specific_domain)) {
224 props.most_specific_domain = Domain::Integer;
225 }
226 }
227}

233void PropertyStore::declare_parity(const std::string& symbol, Parity parity) {
234 auto& props = properties_[symbol];
235
236 // Idempotent: same parity already set -> no-op
237 if (props.parity == parity) {
238 return;
239 }
240
241 // Unknown can always be set
242 if (parity == Parity::Unknown) {
243 props.parity = parity;
244 return;
245 }
246
247 // Contradiction: Even vs Odd
248 if (props.parity != Parity::Unknown && props.parity != parity) {
249 throw std::invalid_argument(
250 "Contradiction for symbol '" + symbol + "': parity conflict (Even vs Odd)");
251 }
252
253 // Auto-promote to Integer domain when Even or Odd is declared
254 if (domain_specificity(Domain::Integer) > domain_specificity(props.most_specific_domain)) {
255 props.most_specific_domain = Domain::Integer;
256 }
257
258 props.parity = parity;
259}

265void PropertyStore::declare_bounded(const std::string& symbol, Boundedness bounded,
266 std::optional<Interval> bounds) {
267 auto& props = properties_[symbol];
268
269 // Idempotent: same boundedness already set -> no-op
270 if (props.boundedness == bounded) {
271 return;
272 }
273
274 // Unknown can always be set
275 if (bounded == Boundedness::Unknown) {
276 props.boundedness = bounded;
277 props.bounds = std::nullopt;
278 return;
279 }
280
281 // Contradiction: Bounded vs Unbounded
282 if (props.boundedness != Boundedness::Unknown && props.boundedness != bounded) {
283 throw std::invalid_argument(
284 "Contradiction for symbol '" + symbol +
285 "': boundedness conflict (Bounded vs Unbounded)");
286 }
287
288 props.boundedness = bounded;
289 if (bounds.has_value()) {
290 props.bounds = bounds;
291 }
292}

346bool PropertyStore::has_domain(const std::string& symbol, Domain domain) const {
347 auto it = properties_.find(symbol);
348 if (it == properties_.end()) {
349 // Default is Complex; only Complex query returns true
350 return domain == Domain::Complex;
351 }
352 // The symbol has domain D. It "has" domain X if X is an ancestor of D (or equal to D).
353 // i.e., the symbol's specificity is >= the queried domain's specificity.
354 return domain_specificity(it->second.most_specific_domain) >= domain_specificity(domain);
355}
