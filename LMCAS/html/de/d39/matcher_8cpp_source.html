65static bool match_recursive(const std::shared_ptr<SymbolicNode>& p_node,
66 const std::shared_ptr<SymbolicNode>& t_node,
67 const std::unordered_set<std::string>& wildcards,
68 MatchMap& results) {
69 if (!p_node) return !t_node;
70 if (!t_node) return false;
71
72 std::string wname;
73 if (is_wildcard(p_node, wildcards, wname)) {
74
75 if (results.count(wname)) {
76
77 return results[wname].root->equals(*t_node);
78 } else {
79
80 results[wname] = SymbolicExpr(t_node);
81 return true;
82 }
83 }
84
85 if (p_node->type_priority() != t_node->type_priority()) return false;
86
87 if (p_node->is_number() || std::dynamic_pointer_cast<VariableNode>(p_node)) {
88 return p_node->equals(*t_node);
89 }
90
91 if (auto p_add = std::dynamic_pointer_cast<AddNode>(p_node)) {
92 auto t_add = std::dynamic_pointer_cast<AddNode>(t_node);
93
94 if (p_add->operands.size() > t_add->operands.size()) return false;
95
96 std::vector<bool> used(t_add->operands.size(), false);
97 if (match_commutative_recursive(p_add->operands, t_add->operands, used, 0, wildcards, results)) {
98
99 std::vector<std::shared_ptr<SymbolicNode>> rest;
100 for (size_t i = 0; i < used.size(); ++i) {
101 if (!used[i]) {
102 rest.push_back(t_add->operands[i]);
103 }
104 }
105
106 if (!rest.empty()) {
107
108 std::shared_ptr<SymbolicNode> rest_node;
109 if (rest.size() == 1) rest_node = rest[0];
110 else rest_node = SymbolicFactory::create_add(rest);
111
112 if (results.find("__Add_REST__") != results.end()) {
113 auto existing = results["__Add_REST__"];
114
115 std::vector<std::shared_ptr<SymbolicNode>> combined_ops;
116 if (auto existing_add = std::dynamic_pointer_cast<AddNode>(existing.root)) {
117 combined_ops.insert(combined_ops.end(), existing_add->operands.begin(), existing_add->operands.end());
118 } else {
119 combined_ops.push_back(existing.root);
120 }
121 combined_ops.push_back(rest_node);
122 results["__Add_REST__"] = SymbolicExpr(SymbolicFactory::create_add(combined_ops));
123 } else {
124 results["__Add_REST__"] = SymbolicExpr(rest_node);
125 }
126 }
127 return true;
128 }
129 return false;
130 }
131
132 if (auto p_mul = std::dynamic_pointer_cast<MultiplyNode>(p_node)) {
133 auto t_mul = std::dynamic_pointer_cast<MultiplyNode>(t_node);
134
135 if (p_mul->operands.size() > t_mul->operands.size()) return false;
136
137 std::vector<bool> used(t_mul->operands.size(), false);
138 if (match_commutative_recursive(p_mul->operands, t_mul->operands, used, 0, wildcards, results)) {
139
140 std::vector<std::shared_ptr<SymbolicNode>> rest;
141 for (size_t i = 0; i < used.size(); ++i) {
142 if (!used[i]) {
143 rest.push_back(t_mul->operands[i]);
144 }
145 }
146 if (!rest.empty()) {
147 std::shared_ptr<SymbolicNode> rest_node;
148 if (rest.size() == 1) rest_node = rest[0];
149 else rest_node = SymbolicFactory::create_multiply(rest);
150

152 auto existing = results["__Mul_REST__"];
153 std::vector<std::shared_ptr<SymbolicNode>> combined_ops;
154 if (auto existing_mul = std::dynamic_pointer_cast<MultiplyNode>(existing.root)) {
155 combined_ops.insert(combined_ops.end(), existing_mul->operands.begin(), existing_mul->operands.end());
156 } else {
157 combined_ops.push_back(existing.root);
158 }
159 combined_ops.push_back(rest_node);
160 results["__Mul_REST__"] = SymbolicExpr(SymbolicFactory::create_multiply(combined_ops));
161 } else {
162 results["__Mul_REST__"] = SymbolicExpr(rest_node);
163 }
164 }
165 return true;
166 }
167 return false;
168 }
169
170 if (auto p_pow = std::dynamic_pointer_cast<PowerNode>(p_node)) {
171 auto t_pow = std::dynamic_pointer_cast<PowerNode>(t_node);
172 return match_recursive(p_pow->base, t_pow->base, wildcards, results) &&
173 match_recursive(p_pow->exponent, t_pow->exponent, wildcards, results);
174 }
175
176 if (auto p_func = std::dynamic_pointer_cast<FunctionNode>(p_node)) {
177 auto t_func = std::dynamic_pointer_cast<FunctionNode>(t_node);
178 if (p_func->type != t_func->type) return false;
179 if (p_func->arguments.size() != t_func->arguments.size()) return false;
180 for (size_t i = 0; i < p_func->arguments.size(); ++i) {
181 if (!match_recursive(p_func->arguments[i], t_func->arguments[i], wildcards, results)) {
182 return false;
183 }
184 }
185 return true;
186 }
187
188 return p_node->equals(*t_node);

501SymbolicExpr RewriteEngine::apply(const SymbolicExpr& expr, int max_iterations) {
502 SymbolicExpr current = expr;
503 for (int i = 0; i < max_iterations; ++i) {
504 SymbolicExpr next = apply_step(current);
