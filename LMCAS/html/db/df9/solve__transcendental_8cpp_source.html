36 struct VarChecker : public SymbolicVisitor {
37 bool has_var = false;
38 void visit(NumberNode&) override {}
39 void visit(VariableNode&) override { has_var = true; }
40 void visit(AddNode& n) override { for (auto& op : n.operands) { if (has_var) return; op->accept(*this); } }
41 void visit(MultiplyNode& n) override { for (auto& op : n.operands) { if (has_var) return; op->accept(*this); } }
42 void visit(PowerNode& n) override { n.base->accept(*this); if (!has_var) n.exponent->accept(*this); }
43 void visit(FunctionNode& n) override { for (auto& arg : n.arguments) { if (has_var) return; arg->accept(*this); } }
44 void visit(MatrixNode&) override {}
45 } checker;
46 expr->root->accept(checker);
47 if (checker.has_var) return false;
48
49 out = expr->to_numeric();
50 return true;
51}

188 const std::shared_ptr<SymbolicExpr>& expr,

191 if (!expr || !expr->root) return std::nullopt;
192
193 auto check_mul_is_g_exp_g = [&](const std::shared_ptr<MultiplyNode>& mul)
194 -> std::optional<std::shared_ptr<SymbolicExpr>> {
195
196 for (size_t i = 0; i < mul->operands.size(); ++i) {
197 if (auto f = std::dynamic_pointer_cast<FunctionNode>(mul->operands[i])) {
198 if (f->type == FunctionNode::FuncType::Exp && f->arguments.size() == 1) {
199
200 auto exp_arg = f->arguments[0];
201 std::vector<std::shared_ptr<SymbolicNode>> remaining;
202 for (size_t j = 0; j < mul->operands.size(); ++j) {
203 if (j != i) remaining.push_back(mul->operands[j]);
204 }
205 std::shared_ptr<SymbolicNode> g_node;
206 if (remaining.size() == 1) {
207 g_node = remaining[0];
208 } else {
209 g_node = std::make_shared<MultiplyNode>(remaining);
210 }
211
212 if (nodes_equal(g_node, exp_arg) && depends_on_var(g_node, var)) {
213 return std::make_shared<SymbolicExpr>(g_node);
214 }
215 }
216 }
217 }
218 return std::nullopt;
219 };
220
221 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(expr->root)) {
222 auto g = check_mul_is_g_exp_g(mul);
223 if (g) {
224 return LambertWPattern{*g, SymbolicExpr::number(0)};
225 }
226 }
227
228 if (auto add = std::dynamic_pointer_cast<AddNode>(expr->root)) {
229 std::optional<std::shared_ptr<SymbolicExpr>> g_found;
230 std::vector<std::shared_ptr<SymbolicNode>> const_terms;
231 bool has_other_var_terms = false;
232
233 for (auto& op : add->operands) {
234 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(op)) {
235 auto g = check_mul_is_g_exp_g(mul);
236 if (g && !g_found) {
237 g_found = g;
238 continue;
239 }
240 }
241 if (!depends_on_var(op, var)) {
242 const_terms.push_back(op);
243 } else {
244 has_other_var_terms = true;
245 }
246 }
247
248 if (g_found && !has_other_var_terms) {
249 std::shared_ptr<SymbolicExpr> const_sum;
250 if (const_terms.empty()) {
251 const_sum = SymbolicExpr::number(0);
252 } else if (const_terms.size() == 1) {
253 const_sum = std::make_shared<SymbolicExpr>(const_terms[0]);
254 } else {
255 const_sum = std::make_shared<SymbolicExpr>(std::make_shared<AddNode>(const_terms));
256 }
257 auto rhs = SymbolicExpr::multiply(const_sum, SymbolicExpr::number(-1))->simplify();

339static std::vector<std::shared_ptr<SymbolicExpr>> invert_sin(
340 const std::shared_ptr<SymbolicExpr>& c, int max_roots) {
341
342 lmmc_real_t c_val;
343 if (try_evaluate_numeric(c, c_val)) {
344 if (std::abs(c_val) > 1.0 + LMMC_REAL_EPSILON) {
345 return {};
346 }
347 }
348
349 auto pi = make_pi();
350 auto asin_c = make_arcsin(c);
351
352 int num_periods = 1;
353 if (max_roots > 0) {
354
355 num_periods = (max_roots + 1) / 2;
356 }
357
358 std::vector<std::shared_ptr<SymbolicExpr>> results;
359 for (int k = 0; k < num_periods; ++k) {
360
361 auto two_k_pi = SymbolicExpr::multiply(SymbolicExpr::number(2 * k), pi);
362 auto sol1 = SymbolicExpr::add(asin_c, two_k_pi)->simplify();
363 results.push_back(sol1);
364
365 auto pi_minus_asin = SymbolicExpr::add(pi, SymbolicExpr::multiply(asin_c, SymbolicExpr::number(-1)));
366 auto sol2 = SymbolicExpr::add(pi_minus_asin, two_k_pi)->simplify();
367 results.push_back(sol2);
368 }
369
370 return results;
371}

544static std::vector<std::shared_ptr<SymbolicExpr>> invert_substitution_h(
545 const std::shared_ptr<SymbolicExpr>& h_expr,
546 const std::shared_ptr<SymbolicExpr>& u_root,
547 const std::string& var) {
548
549 if (!h_expr || !h_expr->root) return {};
550
551 if (auto func = std::dynamic_pointer_cast<FunctionNode>(h_expr->root)) {
552 if (func->arguments.size() == 1) {
553
554 if (auto v = std::dynamic_pointer_cast<VariableNode>(func->arguments[0])) {
555 if (v->name == var) {
556 switch (func->type) {
557 case FunctionNode::FuncType::Exp:
558 return invert_h_exp(u_root, var);
559 case FunctionNode::FuncType::Sin:
560 return invert_h_sin(u_root, var);
561 case FunctionNode::FuncType::Cos:
562 return invert_h_cos(u_root, var);
563 case FunctionNode::FuncType::Tan:
564 return invert_h_tan(u_root, var);
565 default:
566 break;
567 }
568 }
569 }
570
571 }
572 }
573
574 if (auto v = std::dynamic_pointer_cast<VariableNode>(h_expr->root)) {
575 if (v->name == var) {
576
577 return {u_root};
578 }
579 }
580
581 if (auto pow = std::dynamic_pointer_cast<PowerNode>(h_expr->root)) {
582 if (auto v = std::dynamic_pointer_cast<VariableNode>(pow->base)) {
583 if (v->name == var && !depends_on_var(pow->exponent, var)) {
584 if (auto exp_num = std::dynamic_pointer_cast<NumberNode>(pow->exponent)) {
585 int n = 0;
586 if (std::holds_alternative<BigInt>(exp_num->value))
587 n = std::get<BigInt>(exp_num->value).to_int();
588 else if (std::holds_alternative<double>(exp_num->value))
589 n = (int)std::get<double>(exp_num->value);
590 else if (std::holds_alternative<Rational>(exp_num->value))
591 n = (int)std::get<Rational>(exp_num->value).to_double();
592 if (n >= 2) {
593 return invert_h_power(u_root, var, n);
594 }
595 }
596 }
597 }
598 }
599
600 auto eq = SymbolicExpr::add(h_expr,
601 SymbolicExpr::multiply(u_root, SymbolicExpr::number(-1)))->simplify();
602
603 auto poly = symbolic_to_poly<SymbolicPolyCoeff>(eq, var);
604 if (!poly.is_zero() && poly.degree() >= 1) {
605 return SymbolicExpr::solve(eq, var);
606 }
607
608 return {};
609}

618static std::vector<std::shared_ptr<SymbolicExpr>> solve_inner_equation(
619 const std::shared_ptr<SymbolicExpr>& inner,
620 const std::shared_ptr<SymbolicExpr>& value,
621 const std::string& var,
622 int depth) {
623
624 if (auto v = std::dynamic_pointer_cast<VariableNode>(inner->root)) {
625 if (v->name == var) {
626 return {value};
627 }
628 }
629
630 auto inner_eq = SymbolicExpr::add(inner,
631 SymbolicExpr::multiply(value, SymbolicExpr::number(-1)))->simplify();
632
633 auto inner_poly = symbolic_to_poly<SymbolicPolyCoeff>(inner_eq, var);
634 if (!inner_poly.is_zero() && inner_poly.degree() >= 1) {
635 auto inner_solutions = SymbolicExpr::solve(inner_eq, var);
636 if (!inner_solutions.empty()) return inner_solutions;
637 }
638

642static std::vector<std::shared_ptr<SymbolicExpr>> solve_transcendental_impl(
643 const std::shared_ptr<SymbolicExpr>& expr,
644 const std::string& var,
645 int depth) {
646
647 if (!expr || !expr->root) return {};
648
649 if (depth > MAX_TRANSCENDENTAL_DEPTH) return {};
650
651 SolveOptions opts;
652 int max_roots = opts.max_roots;
653
654 auto lambert_pat = decompose_lambert_w_pattern(expr, var);
655 if (lambert_pat) {
656 auto inverted = invert_lambert_w(lambert_pat->rhs);
657 if (inverted.empty()) return {};
658
659 std::vector<std::shared_ptr<SymbolicExpr>> results;
660 for (auto& val : inverted) {
661 auto inner_solutions = solve_inner_equation(lambert_pat->inner, val, var, depth);
662 results.insert(results.end(), inner_solutions.begin(), inner_solutions.end());
663 }
664 return results;
665 }
666
667 auto exp_base_pat = decompose_exp_base_pattern(expr, var);
668 if (exp_base_pat) {
669 auto inverted = invert_exp_base(exp_base_pat->base, exp_base_pat->rhs);
670 if (inverted.empty()) return {};
671
672 std::vector<std::shared_ptr<SymbolicExpr>> results;
673 for (auto& val : inverted) {
674 auto inner_solutions = solve_inner_equation(exp_base_pat->inner, val, var, depth);
675 results.insert(results.end(), inner_solutions.begin(), inner_solutions.end());
676 }
677 return results;
678 }
679
680 auto pattern = decompose_trig_exp_pattern(expr, var);
681 if (pattern) {
682 std::vector<std::shared_ptr<SymbolicExpr>> inverted;
683
684 switch (pattern->func_type) {
685 case FunctionNode::FuncType::Sin:
686 inverted = invert_sin(pattern->rhs, max_roots);
687 break;
688 case FunctionNode::FuncType::Cos:
689 inverted = invert_cos(pattern->rhs, max_roots);
690 break;
691 case FunctionNode::FuncType::Tan:
692 inverted = invert_tan(pattern->rhs, max_roots);
693 break;
694 case FunctionNode::FuncType::Exp:
695 inverted = invert_exp(pattern->rhs);
696 break;
697 case FunctionNode::FuncType::Ln:
698 inverted = invert_ln(pattern->rhs);
699 break;
700 default:
701 return {};
702 }
703
704 if (inverted.empty()) return {};
705
706 std::vector<std::shared_ptr<SymbolicExpr>> results;
707 for (auto& val : inverted) {
708 auto inner_solutions = solve_inner_equation(pattern->inner, val, var, depth);
709 results.insert(results.end(), inner_solutions.begin(), inner_solutions.end());
710 }
711 return results;
712 }
713
714 auto subst = detect_substitution(expr, var);
715 if (subst) {
716
717 auto u_solutions = SymbolicExpr::solve(subst->poly_in_u, subst->u_var);
718 if (u_solutions.empty()) return {};
719
720 std::vector<std::shared_ptr<SymbolicExpr>> results;
721 for (auto& u_root : u_solutions) {
722
723 auto x_values = invert_substitution_h(subst->u_expr, u_root, var);
724 results.insert(results.end(), x_values.begin(), x_values.end());
725 }
726 return results;
727 }
728
729 return {};
730}

739static void collect_transcendental_subexprs(
740 const std::shared_ptr<SymbolicNode>& node,
741 const std::string& var,
742 std::vector<std::shared_ptr<SymbolicExpr>>& candidates) {
743
744 if (!node) return;
745
746 if (auto func = std::dynamic_pointer_cast<FunctionNode>(node)) {
747 if (func->arguments.size() == 1 && depends_on_var(func->arguments[0], var)) {
748 auto ft = func->type;
749 if (ft == FunctionNode::FuncType::Exp ||
750 ft == FunctionNode::FuncType::Sin ||
751 ft == FunctionNode::FuncType::Cos ||
752 ft == FunctionNode::FuncType::Tan) {
753 candidates.push_back(std::make_shared<SymbolicExpr>(node));
754 }
755 }
756
757 for (auto& arg : func->arguments) {
758 collect_transcendental_subexprs(arg, var, candidates);
759 }
760 return;
761 }
762
763 if (auto pow = std::dynamic_pointer_cast<PowerNode>(node)) {
764 if (depends_on_var(pow->base, var) && !depends_on_var(pow->exponent, var)) {
765 if (auto exp_num = std::dynamic_pointer_cast<NumberNode>(pow->exponent)) {
766 int e_val = 0;
767 if (std::holds_alternative<BigInt>(exp_num->value))
768 e_val = std::get<BigInt>(exp_num->value).to_int();
769 else if (std::holds_alternative<double>(exp_num->value))
770 e_val = (int)std::get<double>(exp_num->value);
771 else if (std::holds_alternative<Rational>(exp_num->value))
772 e_val = (int)std::get<Rational>(exp_num->value).to_double();
773
774 if (e_val >= 2) {
775
776 auto base_expr = std::make_shared<SymbolicExpr>(pow->base);
777 candidates.push_back(base_expr);
778 }
779 }
780 }
781
782 collect_transcendental_subexprs(pow->base, var, candidates);
783 collect_transcendental_subexprs(pow->exponent, var, candidates);
784 return;
785 }
786
787 if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
788 for (auto& op : add->operands) {
789 collect_transcendental_subexprs(op, var, candidates);
790 }
791 return;
792 }
793
794 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
795 for (auto& op : mul->operands) {
796 collect_transcendental_subexprs(op, var, candidates);
797 }
798 return;
799 }
800}

802static void deduplicate_candidates(std::vector<std::shared_ptr<SymbolicExpr>>& candidates) {
803 std::vector<std::shared_ptr<SymbolicExpr>> unique;
804 std::vector<std::string> seen;
805 for (auto& c : candidates) {
806 std::string s = c->to_string();
807 bool found = false;
808 for (auto& existing : seen) {
809 if (existing == s) { found = true; break; }
810 }
811 if (!found) {
812 seen.push_back(s);
813 unique.push_back(c);
814 }
815 }
816 candidates = std::move(unique);
817}

819static int extract_exp_multiplier(const std::shared_ptr<SymbolicNode>& node, const std::string& var) {
820 auto func = std::dynamic_pointer_cast<FunctionNode>(node);
821 if (!func || func->type != FunctionNode::FuncType::Exp || func->arguments.size() != 1)
822 return 0;
823
824 auto arg = func->arguments[0];
825
826 if (auto v = std::dynamic_pointer_cast<VariableNode>(arg)) {
827 if (v->name == var) return 1;
828 }
829
830 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(arg)) {
831
832 std::shared_ptr<NumberNode> num_part = nullptr;
833 bool has_var = false;
834 bool has_other = false;
835
836 for (auto& op : mul->operands) {
837 if (auto n = std::dynamic_pointer_cast<NumberNode>(op)) {
838 if (!num_part) num_part = n;
839 else has_other = true;
840 } else if (auto v = std::dynamic_pointer_cast<VariableNode>(op)) {
841 if (v->name == var) has_var = true;
842 else has_other = true;
843 } else {
844 has_other = true;
845 }
846 }
847
848 if (has_var && num_part && !has_other) {
849 int k = 0;
850 if (std::holds_alternative<BigInt>(num_part->value))
851 k = std::get<BigInt>(num_part->value).to_int();
852 else if (std::holds_alternative<double>(num_part->value))
853 k = (int)std::get<double>(num_part->value);
854 else if (std::holds_alternative<Rational>(num_part->value))
855 k = (int)std::get<Rational>(num_part->value).to_double();
856 if (k > 0) return k;
857 }
858 }
859
860 return 0;
861}

863static std::shared_ptr<SymbolicNode> rewrite_exp_as_u_power(
864 const std::shared_ptr<SymbolicNode>& node,
865 const std::string& var,
866 const std::string& u_var) {
867
868 if (!node) return node;
869
870 int k = extract_exp_multiplier(node, var);
871 if (k > 0) {
872 auto u_node = std::make_shared<VariableNode>(u_var);
873 if (k == 1) return u_node;
874 return SymbolicFactory::create_power(u_node, std::make_shared<NumberNode>(BigInt(k)));
875 }
876
877 if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
878 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
879 new_ops.reserve(add->operands.size());
880 for (auto& op : add->operands) {
881 new_ops.push_back(rewrite_exp_as_u_power(op, var, u_var));
882 }
883 return SymbolicFactory::create_add(std::move(new_ops));
884 }
885
886 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
887 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
888 new_ops.reserve(mul->operands.size());
889 for (auto& op : mul->operands) {
890 new_ops.push_back(rewrite_exp_as_u_power(op, var, u_var));
891 }
892 return SymbolicFactory::create_multiply(std::move(new_ops));
893 }
894
895 if (auto pow = std::dynamic_pointer_cast<PowerNode>(node)) {
896 auto new_base = rewrite_exp_as_u_power(pow->base, var, u_var);
897 auto new_exp = rewrite_exp_as_u_power(pow->exponent, var, u_var);
898 return SymbolicFactory::create_power(new_base, new_exp);
899 }
900
901 if (auto func = std::dynamic_pointer_cast<FunctionNode>(node)) {
902 std::vector<std::shared_ptr<SymbolicNode>> new_args;
903 new_args.reserve(func->arguments.size());
904 for (auto& arg : func->arguments) {
905 new_args.push_back(rewrite_exp_as_u_power(arg, var, u_var));
906 }
907 return std::make_shared<FunctionNode>(func->type, std::move(new_args));
908 }
909
910 return node;
911}

913static bool has_exp_k_var_terms(const std::shared_ptr<SymbolicNode>& node, const std::string& var) {
914 if (!node) return false;
915
916 if (extract_exp_multiplier(node, var) > 0) return true;
917
918 if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
919 for (auto& op : add->operands)
920 if (has_exp_k_var_terms(op, var)) return true;
921 }
922 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
923 for (auto& op : mul->operands)
924 if (has_exp_k_var_terms(op, var)) return true;
925 }
926 if (auto pow = std::dynamic_pointer_cast<PowerNode>(node)) {
927 if (has_exp_k_var_terms(pow->base, var)) return true;
928 if (has_exp_k_var_terms(pow->exponent, var)) return true;
929 }
930 if (auto func = std::dynamic_pointer_cast<FunctionNode>(node)) {
931 for (auto& arg : func->arguments)
932 if (has_exp_k_var_terms(arg, var)) return true;

935 return false;

991 }

1000 std::vector<std::shared_ptr<SymbolicNode>> new_args;

1002 for (auto& arg : func->arguments) {
