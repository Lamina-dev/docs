59 void visit(MultiplyNode& node) override {
60 if (node.operands.empty()) {
61 result = SymbolicFactory::create_number(0.0);
62 return;
63 }
64
65 std::vector<std::shared_ptr<SymbolicNode>> sum_terms;
66
67 for (size_t i = 0; i < node.operands.size(); ++i) {
68
69 node.operands[i]->accept(*this);
70 auto d_term = result;
71
72 if (d_term->is_zero()) continue;
73
74 std::vector<std::shared_ptr<SymbolicNode>> prod_terms;
75
76 for (size_t j = 0; j < node.operands.size(); ++j) {
77 if (i == j) {
78 prod_terms.push_back(d_term);
79 } else {
80 prod_terms.push_back(node.operands[j]);
81 }
82 }
83 sum_terms.push_back(SymbolicFactory::create_multiply(prod_terms));
84 }
85
86 if (sum_terms.empty()) {
87 result = SymbolicFactory::create_number(0.0);
88 } else {
89 result = SymbolicFactory::create_add(sum_terms);
90 }
91 }

93 void visit(PowerNode& node) override {
94
95 node.base->accept(*this);
96 auto du = result;
97 node.exponent->accept(*this);
98 auto dv = result;
99
100 auto u = node.base;
101 auto v = node.exponent;
102
103 if (dv->is_zero()) {
104
105 auto n = v;
106
107 auto n_minus_1 = SymbolicFactory::create_add({n, SymbolicFactory::create_number(-1.0)});
108
109 auto u_pow = std::make_shared<PowerNode>(u, n_minus_1);
110
111 result = SymbolicFactory::create_multiply({n, u_pow, du});
112 } else {
113
114 auto ln_u = std::make_shared<FunctionNode>(FunctionNode::FuncType::Ln, std::vector<std::shared_ptr<SymbolicNode>>{u});
115
116 auto t1 = SymbolicFactory::create_multiply({dv, ln_u});
117
118 auto u_inv = std::make_shared<PowerNode>(u, SymbolicFactory::create_number(-1.0));
119 auto t2 = SymbolicFactory::create_multiply({v, du, u_inv});
120
121 auto sum = SymbolicFactory::create_add({t1, t2});
122
123 auto u_pow_v = std::make_shared<PowerNode>(u, v);
124 result = SymbolicFactory::create_multiply({u_pow_v, sum});
125 }
126 }

128 void visit(FunctionNode& node) override {
129 if (node.arguments.size() != 1) {
130 result = SymbolicFactory::create_number(0.0);
131 return;
132 }
133
134 auto& arg = node.arguments[0];
135 arg->accept(*this);
136 auto d_arg = result;
137
138 if (d_arg->is_zero()) {
139 result = SymbolicFactory::create_number(0.0);
140 return;
141 }
142
143 std::shared_ptr<SymbolicNode> d_outer;
144
145 switch (node.type) {
146 case FunctionNode::FuncType::Sin:
147 d_outer = std::make_shared<FunctionNode>(FunctionNode::FuncType::Cos, node.arguments);
148 break;
149 case FunctionNode::FuncType::Cos:
150 d_outer = SymbolicFactory::create_multiply({
151 SymbolicFactory::create_number(-1.0),
152 std::make_shared<FunctionNode>(FunctionNode::FuncType::Sin, node.arguments)
153 });
154 break;
155 case FunctionNode::FuncType::Tan:
156 {
157 auto sec = std::make_shared<FunctionNode>(FunctionNode::FuncType::Sec, node.arguments);
158 d_outer = std::make_shared<PowerNode>(sec, SymbolicFactory::create_number(2.0));
159 }
160 break;
161 case FunctionNode::FuncType::Cot:
162 {
163 // d/dx cot(x) = -csc(x)^2
164 auto csc = std::make_shared<FunctionNode>(FunctionNode::FuncType::Csc, node.arguments);
165 auto csc_sq = std::make_shared<PowerNode>(csc, SymbolicFactory::create_number(2.0));
166 d_outer = SymbolicFactory::create_multiply({
167 SymbolicFactory::create_number(-1.0),
168 csc_sq
169 });
170 }
171 break;
172 case FunctionNode::FuncType::Sec:
173 {
174 // d/dx sec(x) = sec(x) * tan(x)
175 auto sec = std::make_shared<FunctionNode>(FunctionNode::FuncType::Sec, node.arguments);
176 auto tan = std::make_shared<FunctionNode>(FunctionNode::FuncType::Tan, node.arguments);
177 d_outer = SymbolicFactory::create_multiply({sec, tan});
178 }
179 break;
180 case FunctionNode::FuncType::Csc:
181 {
182 // d/dx csc(x) = -csc(x) * cot(x)
183 auto csc = std::make_shared<FunctionNode>(FunctionNode::FuncType::Csc, node.arguments);
184 auto cot = std::make_shared<FunctionNode>(FunctionNode::FuncType::Cot, node.arguments);
185 d_outer = SymbolicFactory::create_multiply({
186 SymbolicFactory::create_number(-1.0),
187 csc, cot
188 });
189 }
190 break;
191 case FunctionNode::FuncType::ArcSin:
192 {
193 // d/dx arcsin(x) = 1/sqrt(1 - x^2) = (1 - x^2)^(-1/2)
194 auto arg_sq = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(2.0));
195 auto neg_arg_sq = SymbolicFactory::create_multiply({
196 SymbolicFactory::create_number(-1.0), arg_sq
197 });
198 auto one_minus_sq = SymbolicFactory::create_add({
199 SymbolicFactory::create_number(1.0), neg_arg_sq
200 });
201 d_outer = std::make_shared<PowerNode>(
202 one_minus_sq, SymbolicFactory::create_number(-0.5));
203 }
204 break;
205 case FunctionNode::FuncType::ArcCos:
206 {
207 // d/dx arccos(x) = -1/sqrt(1 - x^2) = -(1 - x^2)^(-1/2)
208 auto arg_sq = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(2.0));
209 auto neg_arg_sq = SymbolicFactory::create_multiply({
210 SymbolicFactory::create_number(-1.0), arg_sq
211 });
212 auto one_minus_sq = SymbolicFactory::create_add({
213 SymbolicFactory::create_number(1.0), neg_arg_sq
214 });
215 auto inv_sqrt = std::make_shared<PowerNode>(
216 one_minus_sq, SymbolicFactory::create_number(-0.5));
217 d_outer = SymbolicFactory::create_multiply({
218 SymbolicFactory::create_number(-1.0), inv_sqrt
219 });
220 }
221 break;
222 case FunctionNode::FuncType::ArcTan:
223 {
224 // d/dx arctan(x) = 1/(1 + x^2) = (1 + x^2)^(-1)
225 auto arg_sq = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(2.0));
226 auto one_plus_sq = SymbolicFactory::create_add({
227 SymbolicFactory::create_number(1.0), arg_sq
228 });
229 d_outer = std::make_shared<PowerNode>(
230 one_plus_sq, SymbolicFactory::create_number(-1.0));
231 }
232 break;
233 case FunctionNode::FuncType::Sinh:
234 d_outer = std::make_shared<FunctionNode>(
235 FunctionNode::FuncType::Cosh, node.arguments);
236 break;
237 case FunctionNode::FuncType::Cosh:
238 d_outer = std::make_shared<FunctionNode>(
239 FunctionNode::FuncType::Sinh, node.arguments);
240 break;
241 case FunctionNode::FuncType::Tanh:
242 {
243 // d/dx tanh(x) = sech(x)^2 = 1/cosh(x)^2 = cosh(x)^(-2)
244 auto cosh = std::make_shared<FunctionNode>(
245 FunctionNode::FuncType::Cosh, node.arguments);
246 d_outer = std::make_shared<PowerNode>(
247 cosh, SymbolicFactory::create_number(-2.0));
248 }
249 break;
250 case FunctionNode::FuncType::Exp:
251 d_outer = std::make_shared<FunctionNode>(FunctionNode::FuncType::Exp, node.arguments);
252 break;
253 case FunctionNode::FuncType::Ln:
254 d_outer = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(-1.0));
255 break;
256 case FunctionNode::FuncType::Sqrt:
257 d_outer = SymbolicFactory::create_multiply({
258 SymbolicFactory::create_number(0.5),
259 std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(-0.5))
260 });
261 break;
262 case FunctionNode::FuncType::LambertW:
263 {
264 auto W = std::make_shared<FunctionNode>(FunctionNode::FuncType::LambertW, node.arguments);
265 auto one = SymbolicFactory::create_number(1.0);
266 auto one_plus_W = SymbolicFactory::create_add({one, W});
267 auto denom = SymbolicFactory::create_multiply({arg, one_plus_W});
268 auto denom_inv = std::make_shared<PowerNode>(denom, SymbolicFactory::create_number(-1.0));
269 d_outer = SymbolicFactory::create_multiply({W, denom_inv});
270 }
271 break;
272 case FunctionNode::FuncType::RootOf:
273 throw std::runtime_error("RootOf differentiation is not mathematically supported yet.");
274 case FunctionNode::FuncType::Erf:
275 {
276 // d/dx erf(x) = (2/sqrt(pi)) * exp(-x^2)
277 auto pi = std::make_shared<VariableNode>("pi");
278 auto sqrt_pi = std::make_shared<FunctionNode>(
279 FunctionNode::FuncType::Sqrt,
280 std::vector<std::shared_ptr<SymbolicNode>>{pi});
281 auto sqrt_pi_inv = std::make_shared<PowerNode>(sqrt_pi, SymbolicFactory::create_number(-1.0));
282 auto two = SymbolicFactory::create_number(2.0);
283 auto neg_one = SymbolicFactory::create_number(-1.0);
284 auto arg_sq = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(2.0));
285 auto neg_arg_sq = SymbolicFactory::create_multiply({neg_one, arg_sq});
286 auto exp_term = std::make_shared<FunctionNode>(
287 FunctionNode::FuncType::Exp,
288 std::vector<std::shared_ptr<SymbolicNode>>{neg_arg_sq});
289 d_outer = SymbolicFactory::create_multiply({two, sqrt_pi_inv, exp_term});
290 }
291 break;
292 case FunctionNode::FuncType::Ei:
293 {
294 // d/dx Ei(x) = exp(x) / x
295 auto exp_arg = std::make_shared<FunctionNode>(
296 FunctionNode::FuncType::Exp, node.arguments);
297 auto arg_inv = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(-1.0));
298 d_outer = SymbolicFactory::create_multiply({exp_arg, arg_inv});
299 }
300 break;
301 case FunctionNode::FuncType::Si:
302 {
303 // d/dx Si(x) = sin(x) / x
304 auto sin_arg = std::make_shared<FunctionNode>(
305 FunctionNode::FuncType::Sin, node.arguments);
306 auto arg_inv = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(-1.0));
307 d_outer = SymbolicFactory::create_multiply({sin_arg, arg_inv});
308 }
309 break;
310 case FunctionNode::FuncType::Ci:
311 {
312 // d/dx Ci(x) = cos(x) / x
313 auto cos_arg = std::make_shared<FunctionNode>(
314 FunctionNode::FuncType::Cos, node.arguments);
315 auto arg_inv = std::make_shared<PowerNode>(arg, SymbolicFactory::create_number(-1.0));
316 d_outer = SymbolicFactory::create_multiply({cos_arg, arg_inv});
317 }
318 break;
319 case FunctionNode::FuncType::Li:
320 {
321 // d/dx Li(x) = 1 / ln(x)
322 auto ln_arg = std::make_shared<FunctionNode>(
323 FunctionNode::FuncType::Ln, node.arguments);
324 d_outer = std::make_shared<PowerNode>(ln_arg, SymbolicFactory::create_number(-1.0));
325 }
326 break;
327 default:
328 d_outer = SymbolicFactory::create_number(0.0);
329 }
330
331 result = SymbolicFactory::create_multiply({d_outer, d_arg});
332 }

334 void visit(MatrixNode& node) override {
335
336 if (std::holds_alternative<MatrixNode::DenseStorage>(node.storage)) {
337 const auto& dense = std::get<MatrixNode::DenseStorage>(node.storage);
338 MatrixNode::DenseStorage new_dense;
339 for(const auto& item : dense) {
340 if (item) {
341 item->accept(*this);
342 new_dense.push_back(result);
343 } else {
344 new_dense.push_back(nullptr);
345 }
346 }
347 result = std::make_shared<MatrixNode>(node.rows, node.cols, new_dense);
348 } else {
349 const auto& sparse = std::get<MatrixNode::SparseStorage>(node.storage);
350 MatrixNode::SparseStorage new_sparse;
351 for(const auto& [idx, item] : sparse) {
352 item->accept(*this);
353 if (!result->is_zero()) {
354 new_sparse[idx] = result;
355 }
356 }
357 result = std::make_shared<MatrixNode>(node.rows, node.cols, new_sparse);
358 }
