530 auto neg_y = negate(SymbolicExpr::sqrt(u))->simplify();
531 results.push_back(sub(pos_y, shift)->simplify());
532 results.push_back(sub(neg_y, shift)->simplify());
533 }
534 return results;
535 }
536
537 auto p2 = SymbolicExpr::power(p, num(2));
538 auto q2 = SymbolicExpr::power(q, num(2));
539 auto eight_p = SymbolicExpr::multiply(num(8), p);
540 auto two_p2_minus_8r = sub(SymbolicExpr::multiply(num(2), p2), SymbolicExpr::multiply(num(8), r))->simplify();
541 auto neg_q2 = negate(q2)->simplify();
542
543 auto cubic_roots = solve_cubic(num(8), eight_p, two_p2_minus_8r, neg_q2, var);
544

546 if (!cubic_roots.empty()) {
547 m = cubic_roots[0];
548 }
549
550 if (!m) return {};

552 auto two_m = SymbolicExpr::multiply(num(2), m);
553 auto s = SymbolicExpr::sqrt(two_m)->simplify();
554
555 auto p_half = SymbolicExpr::divide(p, num(2))->simplify();
556
557 auto m_plus_p_half = SymbolicExpr::add(m, p_half)->simplify();
558
559 auto q_over_2s = SymbolicExpr::divide(q, SymbolicExpr::multiply(num(2), s))->simplify();

561 auto quad1_c = sub(m_plus_p_half, q_over_2s)->simplify();
562
563 auto neg_s = negate(s)->simplify();
564 auto quad2_c = SymbolicExpr::add(m_plus_p_half, q_over_2s)->simplify();
565
566 auto y_roots1 = solve_quadratic_internal(num(1), s, quad1_c);
567 auto y_roots2 = solve_quadratic_internal(num(1), neg_s, quad2_c);
568
569 std::vector<std::shared_ptr<SymbolicExpr>> results;
570 for (const auto& y : y_roots1) {
571 results.push_back(sub(y, shift)->simplify());
572 }
573 for (const auto& y : y_roots2) {
574 results.push_back(sub(y, shift)->simplify());
575 }
576

580static std::vector<BigInt> positive_divisors(const BigInt& n) {
581 std::vector<BigInt> divs;
582 if (n.is_zero()) {
583
584 return divs;
585 }
586 BigInt abs_n = n.Abs();
587
588 BigInt i(1);
589 BigInt limit(1000);
590
591 while (i * i <= abs_n && i <= limit) {
592 BigInt rem = abs_n % i;
593 if (rem.is_zero()) {
594 divs.push_back(i);
595 BigInt counterpart = abs_n / i;
596 if (counterpart != i) {
597 divs.push_back(counterpart);
598 }
599 }
600 i = i + BigInt(1);
601 }
602
603 std::sort(divs.begin(), divs.end());
604 return divs;
605}

607std::vector<std::pair<Polynomial<Rational>, int>> square_free_factorization(
608 const Polynomial<Rational>& poly) {
609
610 std::vector<std::pair<Polynomial<Rational>, int>> factors;
611
612 if (poly.is_zero() || poly.degree() <= 0) {
613 return factors;
614 }
615
616 Polynomial<Rational> f = poly.make_monic();
617
618 if (f.degree() == 1) {
619 factors.push_back({f, 1});
620 return factors;
621 }
622
623 Polynomial<Rational> f_prime = f.differentiate();
624 Polynomial<Rational> g = Polynomial<Rational>::gcd(f, f_prime);
625
626 if (g.degree() <= 0) {
627 factors.push_back({f, 1});
628 return factors;
629 }
630
631 auto [w, w_rem] = f.div_mod(g);
632
633 int multiplicity = 1;
634
635 while (w.degree() > 0) {
636
637 Polynomial<Rational> y = Polynomial<Rational>::gcd(g, w);
638
639 auto [z, z_rem] = w.div_mod(y);
640
641 if (z.degree() > 0) {
642
643 factors.push_back({z.make_monic(), multiplicity});
644 }
645
646 auto [g_next, g_rem] = g.div_mod(y);
647 g = g_next;
648 w = y;
649 multiplicity++;
650 }
651
652 if (g.degree() > 0) {
653 factors.push_back({g.make_monic(), multiplicity});
654 }
655
656 return factors;
657}

659static std::vector<std::shared_ptr<SymbolicExpr>> solve_closed_form_poly(
660 const Polynomial<SymbolicPolyCoeff>& poly,
661 const std::string& var)
662{
663 int deg = poly.degree();
664 if (deg <= 0) return {};
665
666 auto get_coeff = [&](int d) -> std::shared_ptr<SymbolicExpr> {
667 if (d < 0 || d > deg) return SymbolicExpr::number(0);
668 return poly.coeffs[d].val ? poly.coeffs[d].val : SymbolicExpr::number(0);
669 };
670
671 if (deg == 1) {
672 auto a = get_coeff(1);
673 auto b = get_coeff(0);
674 auto neg_b = SymbolicExpr::multiply(b, SymbolicExpr::number(-1));
675 return { SymbolicExpr::divide(neg_b, a)->simplify() };
676 } else if (deg == 2) {
677 return solve_quadratic_internal(get_coeff(2), get_coeff(1), get_coeff(0));
678 } else if (deg == 3) {
679 return solve_cubic(get_coeff(3), get_coeff(2), get_coeff(1), get_coeff(0), var);
680 } else if (deg == 4) {
681 return solve_quartic(get_coeff(4), get_coeff(3), get_coeff(2), get_coeff(1), get_coeff(0), var);
682 }
683
684 return {};
685}

687static bool convert_to_rational_poly(
688 const Polynomial<SymbolicPolyCoeff>& sym_poly,
689 Polynomial<Rational>& out_poly)
690{
691 out_poly = Polynomial<Rational>(sym_poly.variable_name);
692 out_poly.coeffs.resize(sym_poly.coeffs.size(), Rational(0));
693
694 for (size_t i = 0; i < sym_poly.coeffs.size(); ++i) {
695 auto coeff_expr = sym_poly.coeffs[i].val;
696 if (!coeff_expr) {
697 out_poly.coeffs[i] = Rational(0);
698 continue;
699 }
700
701 auto simplified = coeff_expr->simplify();
702
703 if (simplified->root) {
704 struct VarCheck : public SymbolicVisitor {
705 bool found = false;
706 void visit(NumberNode&) override {}
707 void visit(VariableNode&) override { found = true; }
708 void visit(AddNode& n) override { for (auto& op : n.operands) { if (found) return; op->accept(*this); } }
709 void visit(MultiplyNode& n) override { for (auto& op : n.operands) { if (found) return; op->accept(*this); } }
710 void visit(PowerNode& n) override { n.base->accept(*this); if (!found) n.exponent->accept(*this); }
711 void visit(FunctionNode& n) override { for (auto& arg : n.arguments) { if (found) return; arg->accept(*this); } }
712 void visit(MatrixNode&) override {}
713 } checker;
714 simplified->root->accept(checker);
715 if (checker.found) return false;
716 }
717
718 if (auto num_node = std::dynamic_pointer_cast<NumberNode>(simplified->root)) {
719 if (std::holds_alternative<Rational>(num_node->value)) {
720 out_poly.coeffs[i] = std::get<Rational>(num_node->value);
721 } else if (std::holds_alternative<BigInt>(num_node->value)) {
722 out_poly.coeffs[i] = Rational(std::get<BigInt>(num_node->value));
723 } else {
724 out_poly.coeffs[i] = Rational::from_double(std::get<double>(num_node->value));
725 }
726 } else if (simplified->is_zero()) {
727 out_poly.coeffs[i] = Rational(0);
728 } else if (simplified->is_one()) {
729 out_poly.coeffs[i] = Rational(1);
730 } else {
731
732 double val = simplified->to_numeric();
733 if (std::isnan(val) || std::isinf(val)) {
734 return false;
735 }
736 out_poly.coeffs[i] = Rational::from_double(val);
737 }
738 }
739
740 out_poly.trim();
741 return true;
742}

755std::vector<std::shared_ptr<SymbolicExpr>> solve_by_factoring(
756 const Polynomial<SymbolicPolyCoeff>& poly,
757 const std::string& var) {
758
759 if (poly.is_zero() || poly.degree() < 1) return {};
760
761 if (poly.degree() <= 4) {
762 return solve_closed_form_poly(poly, var);
763 }
764
765 std::vector<std::shared_ptr<SymbolicExpr>> results;
766
767 Polynomial<Rational> rat_poly;
768 if (!convert_to_rational_poly(poly, rat_poly)) {
769
770 return make_rootof_solutions(poly, var);
771 }
772
773 auto rational_roots = find_rational_roots(rat_poly);
774
775 for (const auto& r : rational_roots) {
776 results.push_back(SymbolicExpr::number(r));
777 }
778
779 Polynomial<Rational> quotient = rat_poly;
780 for (const auto& r : rational_roots) {
781 Polynomial<Rational> linear_factor({-r, Rational(1)}, var);
782 auto [q, rem] = quotient.div_mod(linear_factor);
783 quotient = q;
784 }
785
786 if (quotient.degree() <= 0) {
787 return results;
788 }
789
790 if (quotient.degree() <= 4) {
791 auto sym_quotient = rational_to_symbolic_poly(quotient);
792 auto factor_roots = solve_closed_form_poly(sym_quotient, var);
793 results.insert(results.end(), factor_roots.begin(), factor_roots.end());
794 return results;
795 }
796
797 auto sqfree_factors = square_free_factorization(quotient);
798
799 for (const auto& [factor, multiplicity] : sqfree_factors) {
800 if (factor.degree() <= 0) continue;
801
802 std::vector<std::shared_ptr<SymbolicExpr>> factor_solutions;
803
804 if (factor.degree() <= 4) {
805
806 auto sym_factor = rational_to_symbolic_poly(factor);
807 factor_solutions = solve_closed_form_poly(sym_factor, var);
808 } else {
809
810 auto sym_factor = rational_to_symbolic_poly(factor);
811 factor_solutions = make_rootof_solutions(sym_factor, var);
812 }
813
814 for (int m = 0; m < multiplicity; ++m) {
815 results.insert(results.end(), factor_solutions.begin(), factor_solutions.end());
816 }
817 }
818
819 return results;
820}

822std::vector<Rational> find_rational_roots(const Polynomial<Rational>& poly) {
823 std::vector<Rational> roots;
824
825 if (poly.is_zero() || poly.degree() < 1) {
826 return roots;
827 }
828
829 Polynomial<Rational> current = poly;
830
831 while (current.degree() >= 1 && current.coeffs[0] == Rational(0)) {
832 roots.push_back(Rational(0));
833
834 std::vector<Rational> new_coeffs(current.coeffs.begin() + 1, current.coeffs.end());
835 current = Polynomial<Rational>(new_coeffs, current.variable_name);
836 }
837
838 while (current.degree() >= 1) {
839
840 Rational a0 = current.coeffs[0];
841 Rational an = current.lead_coeff();
842
843 if (a0 == Rational(0)) {
844
845 roots.push_back(Rational(0));
846 std::vector<Rational> new_coeffs(current.coeffs.begin() + 1, current.coeffs.end());
847 current = Polynomial<Rational>(new_coeffs, current.variable_name);
848 continue;
849 }
850
851 BigInt lcm_den(1);
852 for (const auto& c : current.coeffs) {
853 BigInt d = c.get_denominator();
854 lcm_den = BigInt::lcm(lcm_den, d);
855 }
856
857 BigInt int_a0 = (a0 * Rational(lcm_den)).get_numerator();
858 BigInt int_an = (an * Rational(lcm_den)).get_numerator();
859
860 std::vector<BigInt> p_divs = positive_divisors(int_a0);
861 std::vector<BigInt> q_divs = positive_divisors(int_an);
862
863 if (p_divs.empty() || q_divs.empty()) {
864 break;
865 }
866
867 bool found_root = false;
868 Rational found_r;
869
870 for (const auto& p : p_divs) {
871 for (const auto& q : q_divs) {
872
873 Rational candidate_pos(p, q);
874 Rational candidate_neg(-p, q);
875
876 if (current.eval(candidate_pos) == Rational(0)) {
877 found_root = true;
878 found_r = candidate_pos;
879 break;
880 }
881 if (current.eval(candidate_neg) == Rational(0)) {
882 found_root = true;
883 found_r = candidate_neg;
884 break;
885 }
886 }
887 if (found_root) break;
888 }
889
890 if (!found_root) {
891 break;
892 }
893
894 Polynomial<Rational> linear_factor({-found_r, Rational(1)}, current.variable_name);
895
896 while (current.degree() >= 1) {
897 auto [quotient, remainder] = current.div_mod(linear_factor);
898
899 if (!remainder.is_zero()) {
900 break;
901 }
902 roots.push_back(found_r);
903 current = quotient;
904
905 if (current.degree() < 1 || current.eval(found_r) != Rational(0)) {
906 break;
907 }
908 }
909
910 if (current.degree() <= 4) {
911 break;
912 }
913 }
914
915 return roots;
916}
