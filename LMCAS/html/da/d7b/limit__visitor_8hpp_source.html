276 bool is_inf(const std::shared_ptr<SymbolicNode>& node) {
277 if (!node) return false;
278 if (auto f = std::dynamic_pointer_cast<FunctionNode>(node)) {
279 return f->type == FunctionNode::FuncType::Infinity;
280 }
281 if (auto m = std::dynamic_pointer_cast<MultiplyNode>(node)) {
282 for (auto& op : m->operands) if (is_inf(op)) return true;
283 }
284 if (auto p = std::dynamic_pointer_cast<PowerNode>(node)) {
285 if (is_inf(p->base)) return true;
286 }
287 if (auto a = std::dynamic_pointer_cast<AddNode>(node)) {
288 for (auto& op : a->operands) if (is_inf(op)) return true;
289 }
290 return false;
