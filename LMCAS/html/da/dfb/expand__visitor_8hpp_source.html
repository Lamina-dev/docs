26 void visit(AddNode& node) override {
27 std::vector<std::shared_ptr<SymbolicNode>> new_ops;
28 for (auto& op : node.operands) {
29 op->accept(*this);
30 new_ops.push_back(result);
31 }
32 result = std::make_shared<AddNode>(new_ops);
33 }
34
35 void visit(MultiplyNode& node) override {
36 std::vector<std::shared_ptr<SymbolicNode>> expanded_ops;
37 for (auto& op : node.operands) {
38 op->accept(*this);
39 expanded_ops.push_back(result);
40 }
41
42 if (expanded_ops.empty()) {
43 result = std::make_shared<NumberNode>(BigInt(1));
44 return;
45 }
46
47 NormalizationVisitor norm;
48 std::shared_ptr<SymbolicNode> current = expanded_ops[0];
49 for (size_t i = 1; i < expanded_ops.size(); ++i) {
50 current = norm.expand_product(current, expanded_ops[i]);
51 }
52 result = current;
53 }
54
55 void visit(PowerNode& node) override {
56 node.base->accept(*this);
57 auto expanded_base = result;
58 node.exponent->accept(*this);
59 auto expanded_exp = result;
60
61 if (auto exp_num = std::dynamic_pointer_cast<NumberNode>(expanded_exp)) {
62 long long e_val = -1;
63 if (std::holds_alternative<BigInt>(exp_num->value)) e_val = std::get<BigInt>(exp_num->value).to_int();
64 else if (std::holds_alternative<lmmc_real_t>(exp_num->value)) e_val = (long long)std::get<lmmc_real_t>(exp_num->value);
65 else if (std::holds_alternative<Rational>(exp_num->value)) e_val = (long long)std::get<Rational>(exp_num->value).to_double();
66
67 if (e_val == 0) {
68 result = std::make_shared<NumberNode>(BigInt(1));
69 return;
70 }
71 if (e_val > 0 && e_val < 20) {
72 NormalizationVisitor norm;
73 std::shared_ptr<SymbolicNode> current = expanded_base;
74 for (int i = 1; i < (int)e_val; ++i) {
75 current = norm.expand_product(current, expanded_base);
76 }
77 result = current;
78 return;
79 }
80 // Exponent too large for expansion (>= 20) or negative — keep as PowerNode
81 }
82 result = std::make_shared<PowerNode>(expanded_base, expanded_exp);
83 }
84
85 void visit(FunctionNode& node) override {
86 std::vector<std::shared_ptr<SymbolicNode>> new_args;
87 for (auto& arg : node.arguments) {
88 arg->accept(*this);
89 new_args.push_back(result);
90 }
91 result = std::make_shared<FunctionNode>(node.type, new_args);
92 }
93
94 void visit(MatrixNode& node) override {

98 void visit(RelationalNode& node) override {

109 node.right->accept(*this);

111 result = std::make_shared<LogicalNode>(head, tail, node.op);
112 }
