32/**
33 * @brief Stores relational constraints and derives sign properties for simple patterns.
34 *
35 * When a relation of the form `variable op 0` is added, the RelationStore notifies
36 * the PropertyStore to mark the variable with the corresponding sign:
37 * - GT → Positive
38 * - GEQ → NonNegative
39 * - LT → Negative
40 * - LEQ → NonPositive
41 * - NEQ → NonZero
42 *
43 * Composite relations (multi-variable LHS or non-zero RHS) are stored without
44 * decomposition for later use by the InferenceEngine.
45 */
46class LAMINA_API RelationStore {
47public:
48 /**
49 * @brief Store a relation and optionally derive sign properties.
50 *
51 * If the LHS is a single VariableNode and the RHS is a NumberNode with value 0,
52 * the corresponding sign property is declared on the PropertyStore.
53 *
54 * @param lhs Left-hand side expression
55 * @param rhs Right-hand side expression
56 * @param op Relational operator
57 * @param prop_store PropertyStore to notify for simple variable > 0 patterns
58 */
59 void add_relation(const SymbolicExpr& lhs, const SymbolicExpr& rhs,
60 RelationalNode::Op op, PropertyStore& prop_store);
61
62 /**
63 * @brief Retrieve all stored relations.
64 * @return Const reference to the vector of stored relations
65 */
66 const std::vector<Relation>& get_relations() const;
67
68 /**
69 * @brief Check if a specific relation is stored.
70 *
71 * Compares LHS, RHS, and operator using structural equality of the AST nodes.
72 *
73 * @param lhs Left-hand side expression
74 * @param rhs Right-hand side expression
75 * @param op Relational operator
76 * @return true if the relation is found in the store
77 */
78 bool has_relation(const SymbolicExpr& lhs, const SymbolicExpr& rhs,
79 RelationalNode::Op op) const;
80
81 /**
82 * @brief Clear all stored relations (used during scope pop).
83 */
84 void clear();
85
86private:
87 std::vector<Relation> relations_;
88};
