24static std::vector<std::shared_ptr<SymbolicExpr>> solve_polynomial_closed(
25 const Polynomial<SymbolicPolyCoeff>& poly,
26 const std::string& var) {
27
28 int deg = poly.degree();
29 if (deg <= 0) return {};
30
31 if (deg == 1) {
32 auto a = get_coeff(poly, 1);
33 auto b = get_coeff(poly, 0);
34 auto neg_b = SymbolicExpr::multiply(b, SymbolicExpr::number(-1));
35 auto result = SymbolicExpr::divide(neg_b, a);
36 return { result->simplify() };
37 }
38
39 if (deg == 2) {
40 auto a = get_coeff(poly, 2);
41 auto b = get_coeff(poly, 1);
42 auto c = get_coeff(poly, 0);
43
44 auto b2 = SymbolicExpr::power(b, SymbolicExpr::number(2));
45 auto ac4 = SymbolicExpr::multiply(SymbolicExpr::number(4), SymbolicExpr::multiply(a, c));
46 auto delta = SymbolicExpr::add(b2, SymbolicExpr::multiply(ac4, SymbolicExpr::number(-1)));
47
48 auto sqrt_delta = SymbolicExpr::power(delta, SymbolicExpr::number(0.5));
49 auto neg_b = SymbolicExpr::multiply(b, SymbolicExpr::number(-1));
50 auto two_a = SymbolicExpr::multiply(SymbolicExpr::number(2), a);
51
52 auto numerator1 = SymbolicExpr::add(neg_b, sqrt_delta);
53 auto numerator2 = SymbolicExpr::add(neg_b, SymbolicExpr::multiply(sqrt_delta, SymbolicExpr::number(-1)));
54
55 auto x1 = SymbolicExpr::divide(numerator1, two_a);
56 auto x2 = SymbolicExpr::divide(numerator2, two_a);
57
58 return { x1->simplify(), x2->simplify() };
59 }
60
61 if (deg == 3) {
62 auto a = get_coeff(poly, 3);
63 auto b = get_coeff(poly, 2);
64 auto c = get_coeff(poly, 1);
65 auto d = get_coeff(poly, 0);
66 return solve_cubic(a, b, c, d, var);
67 }
68
69 if (deg == 4) {
70 auto a = get_coeff(poly, 4);
71 auto b = get_coeff(poly, 3);
72 auto c = get_coeff(poly, 2);
73 auto d = get_coeff(poly, 1);
74 auto e = get_coeff(poly, 0);
75 return solve_quartic(a, b, c, d, e, var);
76 }
77
78 return {};
79}

81std::vector<std::shared_ptr<SymbolicExpr>> lamina::solve_dispatch(
82 const std::shared_ptr<SymbolicExpr>& expr,
83 const std::string& var,
84 const SolveOptions& opts) {
85
86 if (!expr) return {};
87
88 auto simplified = expr->simplify();
89 if (!simplified) return {};
90
91 std::shared_ptr<SymbolicExpr> f_expr = simplified;
92 if (auto rel = std::dynamic_pointer_cast<RelationalNode>(simplified->root)) {
93 if (rel->op == RelationalNode::Op::EQ) {
94 auto L = std::make_shared<SymbolicExpr>(rel->left);
95 auto R = std::make_shared<SymbolicExpr>(rel->right);
96 f_expr = SymbolicExpr::add(L, SymbolicExpr::multiply(R, SymbolicExpr::number(-1)));
97 }
98 }
99
100 auto poly = symbolic_to_poly<SymbolicPolyCoeff>(f_expr, var);
101 if (!poly.is_zero() && poly.degree() >= 1) {
102
103 if (poly.degree() <= 4) {
104 auto results = solve_polynomial_closed(poly, var);
105 if (!results.empty()) return results;
106 }
107
108 if (poly.degree() > 4) {
109 auto results = solve_by_factoring(poly, var);
110 if (!results.empty()) return results;
111
112 if (opts.return_rootof) {
113 return make_rootof_solutions(poly, var);
114 }
115 }
116 }
117
118 auto trans_results = solve_transcendental(f_expr, var);
119 if (!trans_results.empty()) return trans_results;
120
121 if (opts.allow_numeric) {
122 auto numeric_roots = solve_numeric(f_expr, var, opts);
123 if (!numeric_roots.empty()) {
124
125 std::vector<std::shared_ptr<SymbolicExpr>> results;
126 results.reserve(numeric_roots.size());
127 for (const auto& nr : numeric_roots) {
128 results.push_back(SymbolicExpr::number(nr.value));
129 }
130 return results;
131 }
132 }
133
134 return {};
135}

137std::vector<std::shared_ptr<SymbolicExpr>> SymbolicExpr::solve(std::shared_ptr<SymbolicExpr> eq, const std::string& var_name) {
138 if (!eq) return {};
139 auto simplified_eq = eq->simplify();
140
141 if (auto rel = std::dynamic_pointer_cast<RelationalNode>(simplified_eq->root)) {
142 if (rel->op != RelationalNode::Op::EQ) {
143 auto L = std::make_shared<SymbolicExpr>(rel->left);
144 auto R = std::make_shared<SymbolicExpr>(rel->right);
145 auto diff = SymbolicExpr::add(L, SymbolicExpr::multiply(R, SymbolicExpr::number(-1)));
146
147 auto poly = symbolic_to_poly<SymbolicPolyCoeff>(diff, var_name);
148
149 if (!poly.is_zero()) {
150 int max_deg = poly.degree();
151 if (max_deg == 1) {
152 auto a = get_coeff(poly, 1);
153 auto b = get_coeff(poly, 0);
154
155 auto neg_b = SymbolicExpr::multiply(b, SymbolicExpr::number(-1));
156 auto a_inv = SymbolicExpr::power(a, SymbolicExpr::number(-1));
157
158 bool flip = false;
159
160 try {
161 auto a_simp = a->simplify();
162 if (auto num_a = std::dynamic_pointer_cast<NumberNode>(a_simp->root)) {
163 if (std::holds_alternative<double>(num_a->value)) {
164 if (std::get<double>(num_a->value) < 0) flip = true;
165 } else if (std::holds_alternative<BigInt>(num_a->value)) {
166 if (std::get<BigInt>(num_a->value).IsNegative()) flip = true;
167 } else if (std::holds_alternative<Rational>(num_a->value)) {
168 if (std::get<Rational>(num_a->value).get_numerator().IsNegative()) flip = true;
169 }
170 }
171 } catch(...) {}
172
173 RelationalNode::Op new_op = rel->op;
174 if (flip) {
175 switch(rel->op) {
176 case RelationalNode::Op::LT: new_op = RelationalNode::Op::GT; break;
177 case RelationalNode::Op::GT: new_op = RelationalNode::Op::LT; break;
178 case RelationalNode::Op::LEQ: new_op = RelationalNode::Op::GEQ; break;
179 case RelationalNode::Op::GEQ: new_op = RelationalNode::Op::LEQ; break;
180 default: break;
181 }
182 }
183
184 auto rhs = SymbolicExpr::multiply(neg_b, a_inv)->simplify();
185 auto var_node = std::make_shared<VariableNode>(var_name);
186 auto res_node = std::make_shared<RelationalNode>(var_node, rhs->root, new_op);
187
188 return { std::make_shared<SymbolicExpr>(res_node) };
189 }
190 }
191 return {};
192 }
193 }
194
195 return lamina::solve_dispatch(simplified_eq, var_name, lamina::SolveOptions{});
196}

198std::vector<std::map<std::string, std::shared_ptr<SymbolicExpr>>> SymbolicExpr::solve_system(const std::vector<std::shared_ptr<SymbolicExpr>>& equations, const std::vector<std::string>& vars) {
199 size_t n = vars.size();
200 size_t m = equations.size();
201 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> A(m, std::vector<std::shared_ptr<SymbolicExpr>>(n + 1));
202
203 for(size_t i=0; i<m; ++i) {
204 auto eq = equations[i]->expand();
205
206 auto C = eq;
207 for(const auto& v : vars) C = C->substitute(v, SymbolicExpr::number(0));
208 C = C->simplify();
209 A[i][n] = SymbolicExpr::multiply(C, SymbolicExpr::number(-1));
210
211 for(size_t j=0; j<n; ++j) {
212 auto p = symbolic_to_poly<SymbolicPolyCoeff>(eq, vars[j]);
213 if (p.degree() >= 1) {
214 A[i][j] = get_coeff(p, 1);
215 } else {
216 A[i][j] = SymbolicExpr::number(0);
217 }
218 }
219 }
220
221 std::vector<size_t> pivot_col_for_row;
222 int sign;
223 gaussian_eliminate(A, m, n, pivot_col_for_row, sign);
224
225 std::map<std::string, std::shared_ptr<SymbolicExpr>> solution;
226 for(const auto& v : vars) solution[v] = SymbolicExpr::number(0);
227
228 for(size_t r=0; r<m; ++r) {
229 size_t pivot_col = pivot_col_for_row[r];
230 if (pivot_col != (size_t)-1) {
231 solution[vars[pivot_col]] = A[r][n];
232 }
233 }
234 return { solution };
235}

244std::shared_ptr<SymbolicExpr> SymbolicExpr::determinant(const std::shared_ptr<SymbolicExpr>& mat) {
245 if (!mat || mat->get_type() != Type::Matrix) return SymbolicExpr::number(0);
246 auto mat_node = std::dynamic_pointer_cast<MatrixNode>(mat->root);
247 if (!mat_node || mat_node->rows != mat_node->cols) return SymbolicExpr::number(0);
248
249 size_t n = mat_node->rows;
250 if (n == 1) return std::make_shared<SymbolicExpr>(mat_node->get(0,0));
251 if (n == 2) {
252 auto a = std::make_shared<SymbolicExpr>(mat_node->get(0,0));
253 auto b = std::make_shared<SymbolicExpr>(mat_node->get(0,1));
254 auto c = std::make_shared<SymbolicExpr>(mat_node->get(1,0));
255 auto d = std::make_shared<SymbolicExpr>(mat_node->get(1,1));
256 return SymbolicExpr::add(SymbolicExpr::multiply(a,d), SymbolicExpr::multiply(SymbolicExpr::multiply(b,c), SymbolicExpr::number(-1)))->simplify();
257 }
258
259 if (n > 3) {
260 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> A(n, std::vector<std::shared_ptr<SymbolicExpr>>(n));
261 for(size_t i=0; i<n; ++i) {
262 for(size_t j=0; j<n; ++j) {
263 A[i][j] = std::make_shared<SymbolicExpr>(mat_node->get(i,j));
264 }
265 }
266
267 int sign = 1;
268 auto det = SymbolicExpr::number(1);
269
270 for (size_t col = 0; col < n; ++col) {
271 size_t pivot_row = col;
272 while (pivot_row < n && A[pivot_row][col]->is_zero()) {
273 pivot_row++;
274 }
275 if (pivot_row == n) return SymbolicExpr::number(0);
276
277 if (pivot_row != col) {
278 std::swap(A[col], A[pivot_row]);
279 sign = -sign;
280 }
281
282 auto pivot = A[col][col];
283 det = SymbolicExpr::multiply(det, pivot)->simplify();
284 auto pivot_inv = SymbolicExpr::power(pivot, SymbolicExpr::number(-1));
285
286 for (size_t r = col + 1; r < n; ++r) {
287 auto factor = A[r][col];
288 if (!factor->is_zero()) {
289 auto mult = SymbolicExpr::multiply(factor, pivot_inv)->simplify();
290 auto neg_mult = SymbolicExpr::multiply(mult, SymbolicExpr::number(-1));
291 for (size_t c = col + 1; c < n; ++c) {
292 auto term = SymbolicExpr::multiply(neg_mult, A[col][c]);
293 A[r][c] = SymbolicExpr::add(A[r][c], term)->simplify();
294 }
295 A[r][col] = SymbolicExpr::number(0);
296 }
297 }
298 }
299
300 if (sign == -1) {
301 det = SymbolicExpr::multiply(det, SymbolicExpr::number(-1))->simplify();
302 }
303 return det;
304 }
305
306 std::vector<std::shared_ptr<SymbolicExpr>> terms;
307 for(size_t c=0; c<n; ++c) {
308 auto elem = std::make_shared<SymbolicExpr>(mat_node->get(0,c));
309 if (elem->is_zero()) continue;
310
311 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> minor_data;
312 for(size_t r=1; r<n; ++r) {
313 std::vector<std::shared_ptr<SymbolicExpr>> row;
314 for(size_t k=0; k<n; ++k) {
315 if (k == c) continue;
316 row.push_back(std::make_shared<SymbolicExpr>(mat_node->get(r,k)));
317 }
318 minor_data.push_back(row);
319 }
320 auto minor_mat = SymbolicExpr::matrix(minor_data);
321 auto minor_det = SymbolicExpr::determinant(minor_mat);
322
323 auto term = SymbolicExpr::multiply(elem, minor_det);
324 if (c % 2 == 1) term = SymbolicExpr::multiply(term, SymbolicExpr::number(-1));
325 terms.push_back(term);
326 }
327
328 if (terms.empty()) return SymbolicExpr::number(0);
329 auto result = terms[0];
330 for(size_t k=1; k<terms.size(); ++k) result = SymbolicExpr::add(result, terms[k]);
331 return result->simplify();
332}

347 } else {

349 }

354 return SymbolicExpr::determinant(poly_mat);

380std::vector<std::pair<std::shared_ptr<SymbolicExpr>, std::vector<std::shared_ptr<SymbolicExpr>>>> SymbolicExpr::eigenvectors(const std::shared_ptr<SymbolicExpr>& mat) {
381 auto evals_expr = eigenvalues(mat);
382
383 std::vector<std::pair<std::shared_ptr<SymbolicExpr>, std::vector<std::shared_ptr<SymbolicExpr>>>> result;
384
385 if (!evals_expr || (evals_expr->get_type() != SymbolicExpr::Type::Matrix && evals_expr->get_type() != SymbolicExpr::Type::Vector)) {
386 return {};
387 }
388
389 auto mat_node = std::dynamic_pointer_cast<MatrixNode>(evals_expr->root);
390 size_t num_evals = mat_node->cols;
391
392 auto A_node = std::dynamic_pointer_cast<MatrixNode>(mat->root);
393 size_t n = A_node->rows;
394
395 for(size_t i=0; i<num_evals; ++i) {
396 auto lambda_node = mat_node->get(0, i);
397 auto lambda = std::make_shared<SymbolicExpr>(lambda_node);
398
399 std::vector<std::shared_ptr<SymbolicExpr>> equations;
400 std::vector<std::string> vars;
401 for(size_t k=0; k<n; ++k) vars.push_back("v" + std::to_string(k));
402
403 for(size_t i=0; i<n; ++i) {
404 std::vector<std::shared_ptr<SymbolicExpr>> terms;
405 for(size_t j=0; j<n; ++j) {
406 auto a_ij = std::make_shared<SymbolicExpr>(A_node->get(i,j));
407 std::shared_ptr<SymbolicExpr> coeff;
408 if (i == j) {
409 coeff = SymbolicExpr::add(a_ij, SymbolicExpr::multiply(lambda, SymbolicExpr::number(-1)));
410 } else {
411 coeff = a_ij;
412 }
413
414 auto var = SymbolicExpr::variable(vars[j]);
415 terms.push_back(SymbolicExpr::multiply(coeff, var));
416 }
417 auto row_eq = terms[0];
418 for(size_t k=1; k<terms.size(); ++k) row_eq = SymbolicExpr::add(row_eq, terms[k]);
419 equations.push_back(row_eq->simplify());
420 }
421
422 auto sols = solve_system(equations, vars);
423
424 std::vector<std::shared_ptr<SymbolicExpr>> eigenvec;
425 for(const auto& v : vars) eigenvec.push_back(sols[0].at(v));
426
427 bool is_non_zero = false;
428 for(auto& x : eigenvec) if(!x->is_zero()) is_non_zero = true;
429
430 if (is_non_zero) {
431 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> col_vec_data;
432 for(auto& val : eigenvec) col_vec_data.push_back({val});
433 result.push_back({lambda, {SymbolicExpr::matrix(col_vec_data)}});
434 } else {
435 if (n > 1) {
436 bool found_vec = false;
437 for(int free_var_idx = n-1; free_var_idx >= 0 && !found_vec; --free_var_idx) {
438 std::vector<std::shared_ptr<SymbolicExpr>> sub_eqs;
439 std::vector<std::string> sub_vars;
440
441 for(size_t k=0; k<n; ++k) {
442 auto eq_sub = equations[k]->substitute(vars[free_var_idx], SymbolicExpr::number(1))->simplify();
443 if (!eq_sub->is_zero()) {
444 sub_eqs.push_back(eq_sub);
445 }
446 }
447
448 for(size_t k=0; k<n; ++k) {
449 if (k != (size_t)free_var_idx) sub_vars.push_back(vars[k]);
450 }
451
452 auto sub_sols = solve_system(sub_eqs, sub_vars);
453 if (!sub_sols.empty()) {
454 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> col_vec_data(n);
455 for(size_t k=0; k<n; ++k) {
456 if (k == (size_t)free_var_idx) col_vec_data[k] = {SymbolicExpr::number(1)};
457 else col_vec_data[k] = {sub_sols[0].at(vars[k])};
458 }
459
460 auto vec_expr = SymbolicExpr::matrix(col_vec_data);
461 result.push_back({lambda, {vec_expr}});
462 found_vec = true;
463 }
464 }
465 }
466 }
467 }
468
469 return result;
470}

490std::shared_ptr<SymbolicExpr> SymbolicExpr::rref(const std::shared_ptr<SymbolicExpr>& mat) {
491 if (!mat || (mat->get_type() != Type::Matrix && mat->get_type() != Type::Vector)) return mat;
492
493 auto m_node = std::dynamic_pointer_cast<MatrixNode>(mat->root);
494 size_t rows = m_node->rows;
495 size_t cols = m_node->cols;
496
497 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> data(rows, std::vector<std::shared_ptr<SymbolicExpr>>(cols));
498 for(size_t i=0; i<rows; ++i) {
499 for(size_t j=0; j<cols; ++j) {
500 data[i][j] = std::make_shared<SymbolicExpr>(m_node->get(i,j));
501 }
502 }
503
504 size_t lead = 0;
505 for (size_t r = 0; r < rows; ++r) {
506 if (cols <= lead) break;
507
508 size_t i = r;
509 while (true) {
510 if (i >= rows) {
511 i = r;
512 lead++;
513 if (cols == lead) goto end_loops;
514 continue;
515 }
516 auto val = data[i][lead]->simplify();
517 if (!val->get_number_value_is_zero()) {
518 break;
519 }
520 i++;
521 }
522
523 std::swap(data[i], data[r]);
524
525 auto pivot = data[r][lead];
526 auto pivot_inv = SymbolicExpr::power(pivot, SymbolicExpr::number(-1));
527
528 for (size_t j = 0; j < cols; ++j) {
529 data[r][j] = SymbolicExpr::multiply(data[r][j], pivot_inv)->simplify();
530 }
531
532 for (size_t k = 0; k < rows; ++k) {
533 if (k != r) {
534 auto factor = data[k][lead];
535 if (factor->simplify()->get_number_value_is_zero()) continue;
536
537 auto neg_factor = SymbolicExpr::multiply(factor, SymbolicExpr::number(-1));
538 for (size_t j = 0; j < cols; ++j) {
539 auto term = SymbolicExpr::multiply(neg_factor, data[r][j]);
540 data[k][j] = SymbolicExpr::add(data[k][j], term)->simplify();
541 }
542 }
543 }
544 lead++;
545 }
546 end_loops:;
547
548 return SymbolicExpr::matrix(data);
549}

551std::shared_ptr<SymbolicExpr> SymbolicExpr::inverse(const std::shared_ptr<SymbolicExpr>& mat) {
552 if (!mat || mat->get_type() != Type::Matrix) return nullptr;
553
554 auto m_node = std::dynamic_pointer_cast<MatrixNode>(mat->root);
555 size_t n = m_node->rows;
556 if (n != m_node->cols) return nullptr;
557
558 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> aug_data(n, std::vector<std::shared_ptr<SymbolicExpr>>(2*n));
559
560 for(size_t i=0; i<n; ++i) {
561 for(size_t j=0; j<n; ++j) {
562 aug_data[i][j] = std::make_shared<SymbolicExpr>(m_node->get(i,j));
563 }
564 for(size_t j=0; j<n; ++j) {
565 aug_data[i][n+j] = (i==j ? SymbolicExpr::number(1) : SymbolicExpr::number(0));
