28bool QueryInterface::is_unhandled_type(const std::shared_ptr<SymbolicNode>& node) const {
29 if (!node) return true;
30 if (std::dynamic_pointer_cast<MatrixNode>(node)) return true;
31 if (std::dynamic_pointer_cast<RelationalNode>(node)) return true;
32 if (std::dynamic_pointer_cast<LogicalNode>(node)) return true;
33 return false;
34}

44bool QueryInterface::is_infinity_node(const std::shared_ptr<SymbolicNode>& node) const {
45 // Direct infinity: FunctionNode with FuncType::Infinity
46 if (auto func = std::dynamic_pointer_cast<FunctionNode>(node)) {
47 return func->type == FunctionNode::FuncType::Infinity;
48 }
49 // Negative infinity: MultiplyNode(-1, Infinity)
50 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
51 if (mul->operands.size() == 2) {
52 for (const auto& op : mul->operands) {
53 if (auto func = std::dynamic_pointer_cast<FunctionNode>(op)) {
54 if (func->type == FunctionNode::FuncType::Infinity) {
55 return true;
56 }
57 }
58 }
59 }
60 }
61 return false;
62}

64int QueryInterface::get_infinity_sign(const std::shared_ptr<SymbolicNode>& node) const {
65 // Direct infinity node → positive infinity
66 if (auto func = std::dynamic_pointer_cast<FunctionNode>(node)) {
67 if (func->type == FunctionNode::FuncType::Infinity) {
68 return +1;
69 }
70 }
71 // Negative infinity: MultiplyNode containing -1 and Infinity
72 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
73 bool has_infinity = false;
74 bool has_neg_one = false;
75 for (const auto& op : mul->operands) {
76 if (auto func = std::dynamic_pointer_cast<FunctionNode>(op)) {
77 if (func->type == FunctionNode::FuncType::Infinity) {
78 has_infinity = true;
79 }
80 }
81 if (auto num = std::dynamic_pointer_cast<NumberNode>(op)) {
82 // Check if it's -1
83 if (std::holds_alternative<BigInt>(num->value)) {
84 if (std::get<BigInt>(num->value) == BigInt(-1)) {
85 has_neg_one = true;
86 }
87 } else if (std::holds_alternative<Rational>(num->value)) {
88 if (std::get<Rational>(num->value) == Rational(-1)) {
89 has_neg_one = true;
90 }
91 } else if (std::holds_alternative<lmmc_real_t>(num->value)) {
92 lmmc_real_t v = std::get<lmmc_real_t>(num->value);
93 if (v == -1.0) {
94 has_neg_one = true;
95 }
96 }
97 }
98 }
99 if (has_infinity && has_neg_one) return -1;
100 if (has_infinity) return +1; // Infinity multiplied by something else positive
101 }
102 return 0; // indeterminate
103}

109Tribool QueryInterface::query_positive(const SymbolicExpr& expr) const {
110 // Null root or unhandled types → Unknown (Req 10.10)
111 if (is_unhandled_type(expr.root)) {
112 return Tribool::Unknown;
113 }
114
115 // Check for NaN (Req 10.11): Unknown for sign queries
116 if (auto num = std::dynamic_pointer_cast<NumberNode>(expr.root)) {
117 if (is_nan_number(*num)) {
118 return Tribool::Unknown;
119 }
120 }
121
122 // Check for Infinity (Req 10.11): derive sign from infinity's sign
123 if (is_infinity_node(expr.root)) {
124 int sign = get_infinity_sign(expr.root);
125 if (sign > 0) return Tribool::True;
126 if (sign < 0) return Tribool::False;
127 return Tribool::Unknown;
128 }
129
130 // Delegate to InferenceEngine for all other node types
131 // (NumberNode, VariableNode, AddNode, MultiplyNode, PowerNode, FunctionNode)
132 InferenceEngine engine(ctx_);
133 return engine.query_positive(expr);
134}

188Tribool QueryInterface::query_real(const SymbolicExpr& expr) const {
189 // Null root or unhandled types → Unknown (Req 10.10)
190 if (is_unhandled_type(expr.root)) {
191 return Tribool::Unknown;
192 }
193
194 // Check for NaN (Req 10.11): NaN is not a real number in the usual sense,
195 // but the requirement says Unknown for sign queries on NaN.
196 // For query_real on NaN, return Unknown (it's not a well-defined real value).
197 if (auto num = std::dynamic_pointer_cast<NumberNode>(expr.root)) {
198 if (is_nan_number(*num)) {
199 return Tribool::Unknown;
200 }
201 }
202
203 // Infinity is not a real number (it's extended real)
204 if (is_infinity_node(expr.root)) {
205 return Tribool::Unknown;
206 }
207
208 // Delegate to InferenceEngine
209 InferenceEngine engine(ctx_);
210 return engine.query_real(expr);
211}
