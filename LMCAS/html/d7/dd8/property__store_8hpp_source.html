28/**
29 * @brief Stores and validates mathematical properties for named symbols.
30 *
31 * Supports domain hierarchy implication, sign implication, contradiction
32 * detection, and cross-constraint checking between domains and signs.
33 */
34class LAMINA_API PropertyStore {
35public:
36 /// Declare domain for a symbol. Throws std::invalid_argument on contradiction.
37 void declare_domain(const std::string& symbol, Domain domain);
38
39 /// Declare sign for a symbol. Throws std::invalid_argument on contradiction.
40 void declare_sign(const std::string& symbol, Sign sign);
41
42 /// Declare parity for a symbol. Throws std::invalid_argument on contradiction.
43 void declare_parity(const std::string& symbol, Parity parity);
44
45 /// Declare boundedness for a symbol, optionally with interval bounds.
46 void declare_bounded(const std::string& symbol, Boundedness bounded,
47 std::optional<Interval> bounds = std::nullopt);
48
49 /// Query the most specific domain for a symbol (default: Complex).
50 Domain get_domain(const std::string& symbol) const;
51
52 /// Query all signs stored for a symbol.
53 std::unordered_set<Sign, SignHash> get_signs(const std::string& symbol) const;
54
55 /// Query parity for a symbol (default: Unknown).
56 Parity get_parity(const std::string& symbol) const;
57
58 /// Query boundedness for a symbol (default: Unknown).
59 Boundedness get_boundedness(const std::string& symbol) const;
60
61 /// Get stored interval bounds (if any).
62 std::optional<Interval> get_bounds(const std::string& symbol) const;
63
64 /// Check if symbol has a specific sign (explicit or implied).
65 bool has_sign(const std::string& symbol, Sign sign) const;
66
67 /// Check if symbol has at least the given domain specificity.
68 bool has_domain(const std::string& symbol, Domain domain) const;
69
70private:
71 struct SymbolProperties {
72 Domain most_specific_domain = Domain::Complex;
73 std::unordered_set<Sign, SignHash> signs;
74 Parity parity = Parity::Unknown;
75 Boundedness boundedness = Boundedness::Unknown;
76 std::optional<Interval> bounds;
77 };
78
79 std::unordered_map<std::string, SymbolProperties> properties_;
80
81 /// Check domain-sign cross-constraints. Throws on contradiction.
82 void check_domain_sign_consistency(const std::string& symbol,
83 const SymbolProperties& props,
84 Domain new_domain);
85
86 /// Check sign contradiction pairs. Throws on contradiction.
87 void check_sign_contradiction(const std::string& symbol,
88 const std::unordered_set<Sign, SignHash>& existing,
89 Sign new_sign);
90
91 /// Get all signs implied by a given sign.
92 std::unordered_set<Sign, SignHash> get_implied_signs(Sign sign) const;
93
94 /// Get ancestor domains (less specific) for a given domain.
95 std::vector<Domain> get_ancestor_domains(Domain domain) const;
96
97 /// Get the specificity level of a domain (higher = more specific).
98 int domain_specificity(Domain domain) const;
99};
