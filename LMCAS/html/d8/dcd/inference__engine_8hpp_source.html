26/**
27 * @brief Derives properties of composite expressions by recursively analyzing sub-expressions.
28 *
29 * The InferenceEngine implements arithmetic inference rules for addition, multiplication,
30 * power, and function expressions. It determines sign, domain, and boundedness properties
31 * based on the properties of operands/arguments.
32 *
33 * Sign inference rules:
34 * - Addition: uniform sign → same sign on sum; mixed/unknown → Unknown
35 * - Multiplication: zero → Zero; parity of negatives determines sign; unknown → Unknown
36 * - Power: positive base + real exponent → Positive; even exponent → NonNegative; etc.
37 * - Functions: exp → Positive; abs → NonNegative; sin/cos → Bounded[-1,1]; etc.
38 *
39 * Domain inference rules:
40 * - Addition: all Integer → Integer; all Real → Real
41 * - Multiplication: all Integer → Integer; all Real/Integer → Real
42 * - Power: Real base + integer exponent → Real
43 * - Functions: exp/sin/cos/abs/sqrt/ln/tan with Real argument → Real
44 */
45class LAMINA_API InferenceEngine {
46public:
47 /**
48 * @brief Construct an InferenceEngine bound to an AssumptionContext.
49 * @param ctx The AssumptionContext used to query sub-expression properties
50 */
51 explicit InferenceEngine(const AssumptionContext& ctx);
52
53 /// @name Public query methods
54 /// These query sign/domain properties for arbitrary expressions by dispatching
55 /// to the appropriate inference method based on the expression's root node type.
56 /// @{
57
58 Tribool query_positive(const SymbolicExpr& expr) const;
59 Tribool query_negative(const SymbolicExpr& expr) const;
60 Tribool query_nonnegative(const SymbolicExpr& expr) const;
61 Tribool query_nonpositive(const SymbolicExpr& expr) const;
62 Tribool query_real(const SymbolicExpr& expr) const;
63 Tribool query_integer(const SymbolicExpr& expr) const;
64 Tribool query_nonzero(const SymbolicExpr& expr) const;
65
66 /// @}
67
68 /**
69 * @brief Propagate interval bounds through an expression tree.

75 /**
76 * @brief Apply monotonicity deduction rules when a new relation is added.
77 *
78 * For relations like x > y where both operands have appropriate domain assumptions,
79 * deduces new relations (e.g., ln(x) > ln(y) when both are Positive).
80 *
81 * @param rel The newly added relation
82 * @param store The RelationStore to add deduced relations to
83 * @param prop_store The PropertyStore for checking domain assumptions
