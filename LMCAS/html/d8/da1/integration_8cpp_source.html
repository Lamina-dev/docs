42static bool has_integral_node_check(const std::shared_ptr<SymbolicNode>& node) {
43 if (!node) return false;
44 if (auto fn = std::dynamic_pointer_cast<FunctionNode>(node)) {
45 if (fn->type == FunctionNode::FuncType::Calculus_Integral) return true;
46 for (auto& arg : fn->arguments)
47 if (has_integral_node_check(arg)) return true;
48 } else if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
49 for (auto& op : add->operands) if (has_integral_node_check(op)) return true;
50 } else if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
51 for (auto& op : mul->operands) if (has_integral_node_check(op)) return true;
52 } else if (auto pow = std::dynamic_pointer_cast<PowerNode>(node)) {
53 if (has_integral_node_check(pow->base)) return true;
54 if (has_integral_node_check(pow->exponent)) return true;
55 }
56 return false;
57}

108 // Condition: wildcard `_u` is bound to the integration variable itself.
109 auto u_is_var = [](const std::string& wc) {
110 return [wc](const MatchMap& m, const std::string& var) -> bool {
111 auto it = m.find(wc);
112 if (it == m.end()) return false;
113 auto v = std::dynamic_pointer_cast<VariableNode>(it->second.root);
114 return v && v->name == var;
115 };
116 };
117
118 // Condition: `_u` is the integration variable AND `_a` does not depend on it.
119 auto u_is_var_a_indep = [](const std::string& u_wc, const std::string& a_wc) {
120 return [u_wc, a_wc](const MatchMap& m, const std::string& var) -> bool {
121 auto it_u = m.find(u_wc);
122 if (it_u == m.end()) return false;
123 auto v = std::dynamic_pointer_cast<VariableNode>(it_u->second.root);
124 if (!v || v->name != var) return false;
125 auto it_a = m.find(a_wc);
126 if (it_a == m.end()) return false;
127 return !depends_on_var(it_a->second.root, var);
128 };

667std::shared_ptr<SymbolicExpr> TableLookupStrategy::try_integrate(
668 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
669
670 auto all_entries = ctx.table().get_all_sorted();
671
672 for (const auto* entry : all_entries) {
673 MatchMap bindings;
674 if (Matcher::match(entry->pattern, expr, entry->wildcards, bindings)) {
675
676 // The matcher allows partial matches on commutative operations
677 // (Add/Multiply): unmatched operands are bound to __Add_REST__ /
678 // __Mul_REST__ for use by rewrite rules. For integration table
679 // lookup that behavior is wrong - a pattern like (1 + _u^2)^-1
680 // would otherwise match (1 + x^2 + x + x^3)^-1 with _u=x and the
681 // rest x + x^3 silently dropped, producing arctan(x) for an
682 // integrand that has nothing to do with arctan. Require an exact
683 // match (no leftover) so the result genuinely equals the
684 // integrand under this binding.
685 if (bindings.find("__Add_REST__") != bindings.end() ||
686 bindings.find("__Mul_REST__") != bindings.end()) {
687 continue;
688 }
689
690 if (entry->condition && !entry->condition(bindings, var)) {
691 continue;
692 }
693
694 SymbolicExpr result = Matcher::replace(entry->result, bindings, false);
695 auto simplified = result.simplify();
696 if (simplified && !has_integral_node_check(simplified->root)) {
697 return simplified;
698 }
699 }
700 }
701 return nullptr;
702}

705 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
706
707 if (auto v_node = std::dynamic_pointer_cast<VariableNode>(expr.root)) {
708 if (v_node->name == var) {
709 return SymbolicExpr::multiply(
710 SymbolicExpr::power(make_expr_ptr(expr), SymbolicExpr::number(2)),
711 sym_rational(1, 2));
712 }
713 }
714
715 if (auto p_node = std::dynamic_pointer_cast<PowerNode>(expr.root)) {
716 SymbolicExpr base(p_node->base);
717 SymbolicExpr exp_expr(p_node->exponent);
718
719 if (auto b_var = std::dynamic_pointer_cast<VariableNode>(base.root)) {
720 if (b_var->name == var && !valid_dependency(exp_expr, var)) {

722 if (n_plus_1->is_zero()) {
723
724 return SymbolicExpr::ln(make_expr_ptr(base));
725 }

727 SymbolicExpr::power(make_expr_ptr(base), n_plus_1),

729 }
730 }

735std::shared_ptr<SymbolicExpr> SubstitutionStrategy::try_integrate(
736 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
737
738 std::vector<std::shared_ptr<SymbolicNode>> ops;
739 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(expr.root)) {
740 ops = mul->operands;
741 } else {
742 ops.push_back(expr.root);
743 }
744
745 for (size_t i = 0; i < ops.size(); ++i) {
746 SymbolicExpr candidate_term(ops[i]);
747 SymbolicExpr u;
748 bool possible = false;
749
750 if (auto pow = std::dynamic_pointer_cast<PowerNode>(candidate_term.root)) {
751 u = SymbolicExpr(pow->base);
752 possible = true;
753 } else if (auto func = std::dynamic_pointer_cast<FunctionNode>(candidate_term.root)) {
754 if (!func->arguments.empty()) {
755 u = SymbolicExpr(func->arguments[0]);
756 possible = true;
757 }
758 }
759
760 if (possible && valid_dependency(u, var)) {
761 auto d_ptr = u.differentiate(var);
762 if (!d_ptr) continue;
763 auto du = d_ptr->simplify();
764 if (du->is_zero()) continue;
765
766 auto f_u = candidate_term;
767 auto term_times_du = SymbolicExpr::multiply(make_expr_ptr(f_u), make_expr_ptr(*du));
768 auto ratio = SymbolicExpr::divide(make_expr_ptr(expr), term_times_du)->simplify();
769
770 if (!valid_dependency(*ratio, var)) {
771 std::shared_ptr<SymbolicExpr> prim = nullptr;
772
773 if (auto pow = std::dynamic_pointer_cast<PowerNode>(candidate_term.root)) {
774 SymbolicExpr n(pow->exponent);
775 auto np1 = SymbolicExpr::add(make_expr_ptr(n), SymbolicExpr::number(1))->simplify();
776 if (np1->is_zero()) {
777 prim = SymbolicExpr::ln(make_expr_ptr(u));
778 } else {
779 prim = SymbolicExpr::divide(
780 SymbolicExpr::power(make_expr_ptr(u), np1), np1);
781 }
782 } else if (auto func = std::dynamic_pointer_cast<FunctionNode>(candidate_term.root)) {
783 if (func->type == FunctionNode::FuncType::Cos) {
784 prim = SymbolicExpr::sin(make_expr_ptr(u));
785 } else if (func->type == FunctionNode::FuncType::Sin) {
786 prim = SymbolicExpr::multiply(SymbolicExpr::number(-1), SymbolicExpr::cos(make_expr_ptr(u)));
787 } else if (func->type == FunctionNode::FuncType::Exp) {
788 prim = SymbolicExpr::exp(make_expr_ptr(u));
789 }
790 }
791
792 if (prim) {
793 return SymbolicExpr::multiply(ratio, prim);
794 }
795 }
796 }
797 }
798 return nullptr;
799}

801// ---------------------------------------------------------------
802// LinearSubstitutionStrategy
803// ---------------------------------------------------------------
804//
805// Recognises an integrand whose outermost wrapper is a single-argument
806// FunctionNode (e.g. sin/cos/exp/sec(...)) or a PowerNode (e.g. (2x+1)^5,
807// sec(2x+1)^2), and whose inner argument decomposes as a*var + b with a, b
808// constant w.r.t. var and a != 0. It then rewrites the integrand into the
809// "base form" (substituting the inner argument by a fresh dummy variable),
810// looks the base form up via the existing table, and reconstructs the answer
811// as (1/a) * F(a*var + b) where F is the antiderivative returned by the table.
812
813bool LinearSubstitutionStrategy::extract_linear_arg(
814 const SymbolicExpr& arg,
815 const std::string& var,
816 std::shared_ptr<SymbolicExpr>& a_out,
817 std::shared_ptr<SymbolicExpr>& b_out) {
818
819 // Argument must actually depend on the integration variable.
820 if (!depends_on_var(arg.root, var)) return false;
821
822 Polynomial<SymbolicPolyCoeff> poly;
823 try {
824 poly = symbolic_to_poly<SymbolicPolyCoeff>(make_expr_ptr(arg), var);
825 } catch (...) {
826 return false;
827 }
828
829 // Need to be exactly degree 1 in var: arg = a * var + b with a, b constant.
830 if (poly.degree() != 1) return false;
831 if (poly.coeffs.size() < 2) return false;
832
833 auto a_expr = poly.coeffs[1].val;
834 auto b_expr = poly.coeffs[0].val;
835 if (!a_expr || !b_expr) return false;
836
837 // Coefficients must not depend on the integration variable.
838 if (depends_on_var(a_expr->root, var)) return false;
839 if (depends_on_var(b_expr->root, var)) return false;
840
841 auto a_simp = a_expr->simplify();
842 auto b_simp = b_expr->simplify();
843 if (!a_simp) a_simp = a_expr;
844 if (!b_simp) b_simp = b_expr;
845
846 // a must be demonstrably non-zero (refuse if simplification yields zero).
847 if (a_simp->is_zero()) return false;
848
849 a_out = a_simp;
850 b_out = b_simp;
851 return true;
852}

854std::shared_ptr<SymbolicExpr> LinearSubstitutionStrategy::try_integrate(
855 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
856
857 using FT = FunctionNode::FuncType;
858
859 // ---------------------------------------------------------------
860 // Step 1: identify the outer wrapper and the inner argument we'll
861 // try to express as a*var + b.
862 // ---------------------------------------------------------------
863 enum class Wrapper { None, Function, PowerOfFunction, PowerOfLinear };
864 Wrapper kind = Wrapper::None;
865
866 std::shared_ptr<FunctionNode> fn; // outer FunctionNode (Function case)
867 std::shared_ptr<PowerNode> pn; // outer PowerNode (Power* cases)
868 std::shared_ptr<FunctionNode> base_fn; // inner FunctionNode of a PowerNode
869 SymbolicExpr arg_expr; // the linear-candidate sub-expression
870
871 if ((fn = std::dynamic_pointer_cast<FunctionNode>(expr.root))) {
872 // Skip wrappers that are not "real" single-argument functions.
873 switch (fn->type) {
874 case FT::Calculus_Integral:
875 case FT::Infinity:
876 case FT::Limit:
877 case FT::RootOf:
878 case FT::Atan2:
879 case FT::Log:
880 return nullptr;
881 default:
882 break;
883 }
884 if (fn->arguments.size() != 1) return nullptr;
885 arg_expr = SymbolicExpr(fn->arguments[0]);
886 kind = Wrapper::Function;
887 } else if ((pn = std::dynamic_pointer_cast<PowerNode>(expr.root))) {
888 // Exponent must be independent of the integration variable so that
889 // substituting the linear argument back is sound.
890 SymbolicExpr exp_expr(pn->exponent);
891 if (depends_on_var(exp_expr.root, var)) return nullptr;
892
893 if ((base_fn = std::dynamic_pointer_cast<FunctionNode>(pn->base))) {
894 switch (base_fn->type) {
895 case FT::Calculus_Integral:
896 case FT::Infinity:
897 case FT::Limit:
898 case FT::RootOf:
899 case FT::Atan2:
900 case FT::Log:
901 return nullptr;
902 default:
903 break;
904 }
905 if (base_fn->arguments.size() != 1) return nullptr;
906 arg_expr = SymbolicExpr(base_fn->arguments[0]);
907 kind = Wrapper::PowerOfFunction;
908 } else {
909 // Treat the whole base as the linear-candidate (e.g. (2x+1)^n).
910 arg_expr = SymbolicExpr(pn->base);
911 kind = Wrapper::PowerOfLinear;
912 }
913 } else {
914 return nullptr;
915 }
916
917 // ---------------------------------------------------------------
918 // Step 2: extract a and b from arg = a*var + b.
919 // ---------------------------------------------------------------
920 std::shared_ptr<SymbolicExpr> a_coeff, b_coeff;
921 if (!extract_linear_arg(arg_expr, var, a_coeff, b_coeff)) return nullptr;
922
923 // Degenerate case: arg == var (a = 1, b = 0). The plain TableLookup /
924 // PowerRule strategies would have handled this earlier in the chain;
925 // returning nullptr lets the next strategy try.
926 bool a_is_one = a_coeff && a_coeff->is_one();
927 bool b_is_zero = b_coeff && b_coeff->is_zero();
928 if (a_is_one && b_is_zero) return nullptr;
929
930 // ---------------------------------------------------------------
931 // Step 3: build the "base-form" test expression with a fresh dummy
932 // variable in place of the linear argument.
933 // ---------------------------------------------------------------
934 // Use a name unlikely to clash with anything in user input or table entries.
935 const std::string dummy_name = "__lin_sub_u__";
936 auto dummy_var = SymbolicExpr::variable(dummy_name);
937
938 std::shared_ptr<SymbolicExpr> test_expr;
939 if (kind == Wrapper::Function) {
940 std::vector<std::shared_ptr<SymbolicNode>> new_args{dummy_var->root};
941 test_expr = std::make_shared<SymbolicExpr>(
942 std::make_shared<FunctionNode>(fn->type, new_args));
943 } else if (kind == Wrapper::PowerOfFunction) {
944 std::vector<std::shared_ptr<SymbolicNode>> new_args{dummy_var->root};
945 auto new_fn = std::make_shared<FunctionNode>(base_fn->type, new_args);
946 test_expr = std::make_shared<SymbolicExpr>(
947 std::make_shared<PowerNode>(new_fn, pn->exponent));
948 } else { // Wrapper::PowerOfLinear
949 test_expr = std::make_shared<SymbolicExpr>(
950 std::make_shared<PowerNode>(dummy_var->root, pn->exponent));
951 }
952 if (!test_expr) return nullptr;
953
954 // ---------------------------------------------------------------
955 // Step 4: look up the base form in the table, treating dummy_name as
956 // the integration variable. We deliberately use only the
957 // table-lookup strategy here so that we don't recurse back
958 // through the full strategy chain.
959 // ---------------------------------------------------------------
960 TableLookupStrategy table_only;
961 auto F_dummy = table_only.try_integrate(*test_expr, dummy_name, ctx, depth);
962 if (!F_dummy) return nullptr;
963
964 // ---------------------------------------------------------------
965 // Step 5: substitute dummy_name -> (a*var + b) in the antiderivative
966 // and multiply by 1/a.
967 // ---------------------------------------------------------------
968 auto F_substituted = F_dummy->substitute(dummy_name, make_expr_ptr(arg_expr));
969 if (!F_substituted) return nullptr;
970
971 auto inv_a = SymbolicExpr::power(a_coeff, SymbolicExpr::number(-1));
972 auto result = SymbolicExpr::multiply(inv_a, F_substituted);
973
974 auto simplified = result->simplify();
975 if (simplified && !has_integral_node_check(simplified->root)) {
976 return simplified;
977 }
978 return result;
979}

981std::shared_ptr<SymbolicExpr> PartialFractionStrategy::try_integrate(
982 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
983
984 std::shared_ptr<SymbolicExpr> den = nullptr;
985
986 if (auto p = std::dynamic_pointer_cast<PowerNode>(expr.root)) {
987 double exp_val = 0;
988 bool is_inv = false;
989 if (auto num_node = std::dynamic_pointer_cast<NumberNode>(p->exponent)) {
990 if (std::holds_alternative<lmmc_real_t>(num_node->value))
991 exp_val = std::get<lmmc_real_t>(num_node->value);
992 else if (std::holds_alternative<Rational>(num_node->value))
993 exp_val = std::get<Rational>(num_node->value).to_double();
994 else if (std::holds_alternative<BigInt>(num_node->value))
995 exp_val = std::get<BigInt>(num_node->value).to_double();
996 int eq;
997 lmmc_double_nearly_equal_tol(exp_val, -1.0, 1e-9, 1e-9, &eq);
998 if (eq) is_inv = true;
999 }
1000 if (is_inv) {
1001 den = make_expr_ptr(SymbolicExpr(p->base));
1002 }
1003 }
1004
1005 if (!den) return nullptr;
1006
1007 try {
1008 Polynomial<SymbolicPolyCoeff> Q = symbolic_to_poly<SymbolicPolyCoeff>(den, var);
1009
1010 if (Q.degree() == 2) {
1011 SymbolicExpr c_expr = *(Q.coeffs[0].val);
1012 SymbolicExpr b_expr = *(Q.coeffs[1].val);
1013 SymbolicExpr a_expr = *(Q.coeffs[2].val);
1014
1015 if (!a_expr.is_number() || !b_expr.is_number() || !c_expr.is_number()) {
1016 return nullptr;
1017 }
1018
1019 double a = a_expr.to_numeric();
1020 double b = b_expr.to_numeric();
1021 double c = c_expr.to_numeric();
1022
1023 int eq_a;
1024 lmmc_double_nearly_equal_tol(a, 0.0, 1e-9, 1e-9, &eq_a);
1025 if (eq_a) return nullptr;
1026
1027 double delta = b * b - 4 * a * c;
1028 int eq_delta;
1029 lmmc_double_nearly_equal_tol(delta, 0.0, 1e-9, 1e-9, &eq_delta);
1030
1031 if (!eq_delta && delta > 0) {
1032 double sqrt_delta = std::sqrt(delta);
1033 auto scalar = SymbolicExpr::number(1.0 / sqrt_delta);
1034 auto two_a = SymbolicExpr::number(2.0 * a);
1035 auto b_num = SymbolicExpr::number(b);
1036 auto two_a_x = SymbolicExpr::multiply(make_expr_ptr(*two_a), SymbolicExpr::variable(var));
1037 auto two_a_x_plus_b = SymbolicExpr::add(make_expr_ptr(*two_a_x), make_expr_ptr(*b_num));
1038 auto term1_arg = sym_sub(*two_a_x_plus_b, *SymbolicExpr::number(sqrt_delta));
1039 auto term2_arg = SymbolicExpr::add(make_expr_ptr(*two_a_x_plus_b), SymbolicExpr::number(sqrt_delta));
1040 auto term1 = SymbolicExpr::ln(make_expr_ptr(*term1_arg));
1041 auto term2 = SymbolicExpr::ln(make_expr_ptr(*term2_arg));
1042 return SymbolicExpr::multiply(scalar, sym_sub(*term1, *term2));
1043 } else if (!eq_delta && delta < 0) {
1044 double sqrt_neg_delta = std::sqrt(-delta);
1045 auto scalar = SymbolicExpr::number(2.0 / sqrt_neg_delta);
1046 auto two_a = SymbolicExpr::number(2.0 * a);
1047 auto b_num = SymbolicExpr::number(b);
1048 auto num = SymbolicExpr::add(
1049 SymbolicExpr::multiply(make_expr_ptr(*two_a), SymbolicExpr::variable(var)),
1050 make_expr_ptr(*b_num));
1051 auto inner = SymbolicExpr::divide(make_expr_ptr(*num), SymbolicExpr::number(sqrt_neg_delta));
1052 return SymbolicExpr::multiply(scalar, make_arctan(inner));
1053 }
1054 }
1055 } catch (...) {}
1056
1057 return nullptr;
1058}

1060std::shared_ptr<SymbolicExpr> IBPStrategy::try_integrate(
1061 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
1062
1063 std::vector<std::shared_ptr<SymbolicNode>> ops;
1064 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(expr.root)) {
1065 ops = mul->operands;
1066 } else {
1067 ops.push_back(expr.root);
1068 }
1069
1070 int best_u_idx = -1;
1071 int best_score = 100;
1072
1073 auto get_score = [&](const std::shared_ptr<SymbolicNode>& node) -> int {
1074 SymbolicExpr e(node);
1075 if (auto fn = std::dynamic_pointer_cast<FunctionNode>(node)) {
1076 if (fn->type == FunctionNode::FuncType::Ln || fn->type == FunctionNode::FuncType::Log) return 1;
1077 if (fn->type == FunctionNode::FuncType::ArcSin || fn->type == FunctionNode::FuncType::ArcTan) return 2;
1078 if (fn->type == FunctionNode::FuncType::Sin || fn->type == FunctionNode::FuncType::Cos) return 4;
1079 if (fn->type == FunctionNode::FuncType::Exp) return 5;
1080 }
1081 if (!valid_dependency(e, var)) return 10;
1082 if (std::dynamic_pointer_cast<VariableNode>(node)) return 3;
1083 if (std::dynamic_pointer_cast<PowerNode>(node)) return 3;
1084 return 10;
1085 };
1086
1087 if (ops.size() == 1) {
1088 int s = get_score(ops[0]);
1089 if (s <= 2) best_u_idx = 0;
1090 } else {
1091 for (size_t i = 0; i < ops.size(); ++i) {
1092 if (!valid_dependency(SymbolicExpr(ops[i]), var)) continue;
1093 int s = get_score(ops[i]);
1094 if (s < best_score) {
1095 best_score = s;
1096 best_u_idx = (int)i;
1097 }
1098 }
1099 }
1100
1101 if (best_u_idx == -1) return nullptr;
1102
1103 SymbolicExpr u(ops[best_u_idx]);
1104
1105 std::vector<std::shared_ptr<SymbolicNode>> dv_ops;
1106 for (size_t i = 0; i < ops.size(); ++i) {
1107 if ((int)i != best_u_idx) dv_ops.push_back(ops[i]);
1108 }
1109
1110 std::shared_ptr<SymbolicExpr> dv;
1111 if (dv_ops.empty()) dv = SymbolicExpr::number(1);
1112 else if (dv_ops.size() == 1) dv = make_expr_ptr(SymbolicExpr(dv_ops[0]));
1113 else dv = std::make_shared<SymbolicExpr>(std::make_shared<MultiplyNode>(dv_ops));
1114
1115 auto v = ctx.integrate_recursive(*dv, var, depth + 1);
1116 if (has_integral_node_check(v->root)) return nullptr;
1117
1118 auto du_ptr = u.differentiate(var);
1119 if (!du_ptr) return nullptr;
1120 auto du = make_expr_ptr(*du_ptr->simplify());
1121
1122 auto uv = SymbolicExpr::multiply(make_expr_ptr(u), v);
1123 auto vdu = SymbolicExpr::multiply(v, du);
1124 auto int_vdu = ctx.integrate_recursive(*vdu, var, depth + 1);
1125
1126 return sym_sub(*uv, *int_vdu);
1127}

1129// ---------------------------------------------------------------
1130// TrigCombinationStrategy
1131// ---------------------------------------------------------------
1132//
1133// Handles three families of trigonometric integrands whose argument is
1134// exactly the integration variable (linear arguments are handled earlier
1135// by LinearSubstitutionStrategy):
1136//
1137// 1. sin^m(var) * cos^n(var) (m,n integers >= 0, m+n <= 8)
1138// 2. tan^n(var) (n integer in [2,8])
1139// 3. sec^n(var) (n even integer in [2,8])
1140//
1141// The sin/cos case dispatches on parity:
1142// - At least one odd power -> peel off one factor, apply 1 = sin^2 + cos^2
1143// to convert the remaining (even) part to a polynomial in u = cos/sin,
1144// and integrate term-by-term.
1145// - Both even -> apply the half-angle identities
1146// sin^2(c x) = (1 - cos(2c x))/2, cos^2(c x) = (1 + cos(2c x))/2,
1147// expand the resulting binomial product into a polynomial in cos(2c x),
1148// and recurse on each cos^k(2c x) term (the total degree halves each
1149// time, so recursion terminates quickly).
1150// tan^n uses tan^n = tan^(n-2)*(sec^2 - 1) recursively, with bases
1151// tan^0 = 1 -> x and tan^1 -> -ln(cos(x)).
1152// Even-power sec^n uses the reduction
1153// int sec^n = sec^(n-2)*tan/(n-1) + (n-2)/(n-1) * int sec^(n-2)
1154// with base sec^2 -> tan(x).
1155

1158// Build sin(c*var) / cos(c*var) where c is a non-zero integer scale.
1159std::shared_ptr<SymbolicExpr> make_sin_scaled(long long c, const std::string& var) {
1160 auto v = SymbolicExpr::variable(var);
1161 if (c == 1) return SymbolicExpr::sin(v);
1162 auto cx = SymbolicExpr::multiply(SymbolicExpr::number(static_cast<long long>(c)), v);
1163 return SymbolicExpr::sin(cx);
1164}

1178// Binomial coefficient C(n,k) for small n; safe for n up to ~62.
1179long long binomial_ll(int n, int k) {
1180 if (k < 0 || k > n) return 0;
1181 if (k == 0 || k == n) return 1;
1182 if (k > n - k) k = n - k;
1183 long long c = 1;
1184 for (int i = 0; i < k; ++i) {
1185 c = c * static_cast<long long>(n - i) / static_cast<long long>(i + 1);
1186 }
1187 return c;
1188}

1190// Match a single-argument FunctionNode whose argument is the integration
1191// variable itself. Returns 0 (sin), 1 (cos), 2 (tan), 3 (sec), or -1 on
1192// no match for this strategy.
1193int trig_match_of_var(const std::shared_ptr<SymbolicNode>& node, const std::string& var) {
1194 auto fn = std::dynamic_pointer_cast<FunctionNode>(node);
1195 if (!fn) return -1;
1196 if (fn->arguments.size() != 1) return -1;
1197 auto v = std::dynamic_pointer_cast<VariableNode>(fn->arguments[0]);
1198 if (!v || v->name != var) return -1;
1199 using FT = FunctionNode::FuncType;
1200 switch (fn->type) {
1201 case FT::Sin: return 0;
1202 case FT::Cos: return 1;
1203 case FT::Tan: return 2;
1204 case FT::Sec: return 3;
1205 default: return -1;
1206 }
1207}

1220// Match a single factor of the form trig(var) or trig(var)^k where trig is
1221// sin/cos/tan/sec and k is a non-negative integer. Returns true on success
1222// with the kind (0..3) and the integer power.
1223bool trig_extract_factor(const std::shared_ptr<SymbolicNode>& node,
1224 const std::string& var,
1225 int& kind_out, int& power_out) {

1227 if (kind >= 0) {

1233 if (!pn) return false;
1234 int base_kind = trig_match_of_var(pn->base, var);
1235 if (base_kind < 0) return false;
1236 int p = 0;
1237 if (!trig_extract_nonneg_int(pn->exponent, p)) return false;
1238 kind_out = base_kind;
1239 power_out = p;
1240 return true;
1241}

1318std::shared_ptr<SymbolicExpr> TrigCombinationStrategy::integrate_sin_m_cos_n(
1319 int m, int n, long long scale,
1320 const std::string& var, Integrator& ctx, int depth) {
1321
1322 if (depth > ctx.max_depth()) return nullptr;
1323 if (m < 0 || n < 0) return nullptr;
1324
1325 if (m == 0 && n == 0) {
1326 // Constant 1 with respect to var: integrand is essentially 1.
1327 return SymbolicExpr::variable(var);
1328 }
1329 if ((m % 2 == 1) || (n % 2 == 1)) {
1330 return integrate_odd_case(m, n, scale, var, ctx, depth);
1331 }
1332 return integrate_even_case(m, n, scale, var, ctx, depth);

1335std::shared_ptr<SymbolicExpr> TrigCombinationStrategy::integrate_odd_case(
1336 int m, int n, long long scale,
1337 const std::string& var, Integrator& ctx, int depth) {
1338
1339 if (depth > ctx.max_depth()) return nullptr;
1340 if (scale == 0) return nullptr;
1341
1342 // Pick which factor to peel:
1343 // m odd -> peel one sin, set u = cos(scale*x), du = -scale*sin(scale*x)dx
1344 // so sin(scale*x)dx = -du/scale.
1345 // remaining: sin^(m-1) cos^n = (1-u^2)^k * u^n, k=(m-1)/2.
1346 // n odd -> peel one cos, set u = sin(scale*x), du = +scale*cos(scale*x)dx
1347 // so cos(scale*x)dx = du/scale.
1348 // remaining: sin^m cos^(n-1) = u^m * (1-u^2)^k, k=(n-1)/2.
1349 bool peel_sin = (m % 2 == 1);
1350 int k = 0;
1351 int other_pow = 0;
1352 long long sign_factor = 1; // includes the sign from du sign.
1353 bool u_is_cos = false;
1354
1355 if (peel_sin) {
1356 k = (m - 1) / 2;
1357 other_pow = n;
1358 sign_factor = -1;
1359 u_is_cos = true;
1360 } else {
1361 k = (n - 1) / 2;
1362 other_pow = m;
1363 sign_factor = 1;
1364 u_is_cos = false;
1365 }
1366
1367 auto u_expr = u_is_cos ? make_cos_scaled(scale, var)
1368 : make_sin_scaled(scale, var);
1369
1370 // Result = sum_{i=0..k} C(k,i)*(-1)^i * u^(2i+other_pow+1) / (scale*(2i+other_pow+1)) * sign_factor
1371 std::vector<std::shared_ptr<SymbolicNode>> add_terms;
1372 for (int i = 0; i <= k; ++i) {
1373 long long bin = binomial_ll(k, i);
1374 long long alt_sign = ((i % 2) == 0) ? 1 : -1;
1375 long long num = sign_factor * alt_sign * bin;
1376 int u_pow = 2 * i + other_pow + 1; // always >= 1
1377 long long den = scale * static_cast<long long>(u_pow);
1378 if (den == 0) return nullptr;
1379
1380 std::shared_ptr<SymbolicExpr> u_to_pow;
1381 if (u_pow == 1) {
1382 u_to_pow = u_expr;
1383 } else {
1384 u_to_pow = SymbolicExpr::power(u_expr, SymbolicExpr::number(u_pow));
1385 }
1386 // Normalize sign so the rational denominator is positive.
1387 if (den < 0) { num = -num; den = -den; }
1388 auto coeff = SymbolicExpr::number(Rational(BigInt(num), BigInt(den)));
1389 auto term = SymbolicExpr::multiply(coeff, u_to_pow);
1390 add_terms.push_back(term->root);
1391 }
1392
1393 if (add_terms.empty()) return SymbolicExpr::number(0);
1394 if (add_terms.size() == 1) {
1395 return std::make_shared<SymbolicExpr>(add_terms[0]);
1396 }
1397 return std::make_shared<SymbolicExpr>(std::make_shared<AddNode>(add_terms));
1398}

1400std::shared_ptr<SymbolicExpr> TrigCombinationStrategy::integrate_even_case(
1401 int m, int n, long long scale,
1402 const std::string& var, Integrator& ctx, int depth) {
1403
1404 if (depth > ctx.max_depth()) return nullptr;
1405 if ((m % 2 != 0) || (n % 2 != 0)) return nullptr;
1406 int p = m / 2;
1407 int q = n / 2;
1408 int K = p + q;
1409
1410 // sin^(2p)(c*x)*cos^(2q)(c*x)
1411 // = (1/2^(p+q)) * sum_{i,j} C(p,i)(-1)^i C(q,j) cos^(i+j)(2c*x)
1412 // Compute coefficients a[k] = sum_{i+j=k} C(p,i)*C(q,j)*(-1)^i for k=0..K.
1413 std::vector<long long> a(K + 1, 0);
1414 for (int i = 0; i <= p; ++i) {
1415 long long bp = binomial_ll(p, i);
1416 long long si = ((i % 2) == 0) ? 1 : -1;
1417 for (int j = 0; j <= q; ++j) {
1418 long long bq = binomial_ll(q, j);
1419 a[i + j] += si * bp * bq;
1420 }
1421 }
1422
1423 long long pow2 = 1;
1424 for (int t = 0; t < K; ++t) pow2 *= 2;
1425 if (pow2 == 0) pow2 = 1;
1426
1427 std::vector<std::shared_ptr<SymbolicNode>> add_terms;
1428 for (int k = 0; k <= K; ++k) {
1429 if (a[k] == 0) continue;
1430 // Recurse: integrate cos^k(2c*x) at the new scale 2c.
1431 auto inner = integrate_sin_m_cos_n(0, k, scale * 2, var, ctx, depth + 1);
1432 if (!inner) return nullptr;
1433 long long num = a[k];
1434 long long den = pow2;
1435 if (den < 0) { num = -num; den = -den; }
1436 auto coeff = SymbolicExpr::number(Rational(BigInt(num), BigInt(den)));
1437 auto term = SymbolicExpr::multiply(coeff, inner);
1438 add_terms.push_back(term->root);
1439 }
1440
1441 if (add_terms.empty()) return SymbolicExpr::number(0);
1442 if (add_terms.size() == 1) {
1443 return std::make_shared<SymbolicExpr>(add_terms[0]);
1444 }
1445 return std::make_shared<SymbolicExpr>(std::make_shared<AddNode>(add_terms));
1446}

1448std::shared_ptr<SymbolicExpr> TrigCombinationStrategy::integrate_tan_power(
1449 int n, const std::string& var, Integrator& ctx, int depth) {
1450
1451 if (depth > ctx.max_depth()) return nullptr;
1452 if (n < 0) return nullptr;
1453 auto v = SymbolicExpr::variable(var);
1454
1455 if (n == 0) {
1456 // int 1 dx = x
1457 return v;
1458 }
1459 if (n == 1) {
1460 // int tan(x) dx = -ln(cos(x))
1461 auto cos_x = SymbolicExpr::cos(v);
1462 auto ln_cos = SymbolicExpr::ln(cos_x);
1463 return SymbolicExpr::multiply(SymbolicExpr::number(-1), ln_cos);
1464 }
1465
1466 // n >= 2: tan^n = tan^(n-2)*(sec^2 - 1)
1467 // int tan^n = tan^(n-1)/(n-1) - int tan^(n-2)
1468 auto tan_x = SymbolicExpr::tan(v);
1469 std::shared_ptr<SymbolicExpr> tan_pow_term;
1470 if (n - 1 == 1) {
1471 tan_pow_term = tan_x;
1472 } else {
1473 tan_pow_term = SymbolicExpr::power(tan_x, SymbolicExpr::number(n - 1));
1474 }
1475 auto first = SymbolicExpr::multiply(trig_rational(1, n - 1), tan_pow_term);
1476
1477 auto rest = integrate_tan_power(n - 2, var, ctx, depth + 1);
1478 if (!rest) return nullptr;
1479 auto neg_rest = SymbolicExpr::multiply(SymbolicExpr::number(-1), rest);
1480 return SymbolicExpr::add(first, neg_rest);
1481}
1482
1483std::shared_ptr<SymbolicExpr> TrigCombinationStrategy::integrate_sec_power(
1484 int n, const std::string& var, Integrator& ctx, int depth) {
1485
1486 if (depth > ctx.max_depth()) return nullptr;
1487 if (n < 2 || (n % 2) != 0) return nullptr;
1488
1489 using FT = FunctionNode::FuncType;
1490 auto v = SymbolicExpr::variable(var);
1491 auto tan_x = SymbolicExpr::tan(v);
1492 auto make_sec_x = [&]() -> std::shared_ptr<SymbolicExpr> {
1493 return std::make_shared<SymbolicExpr>(
1494 std::make_shared<FunctionNode>(FT::Sec,
1495 std::vector<std::shared_ptr<SymbolicNode>>{v->root}));
1496 };
1497
1498 if (n == 2) {
1499 return tan_x;
1500 }
1501
1502 auto sec_x = make_sec_x();
1503 std::shared_ptr<SymbolicExpr> sec_pow;
1504 if (n - 2 == 1) {
1505 sec_pow = sec_x;
1506 } else {
1507 sec_pow = SymbolicExpr::power(sec_x, SymbolicExpr::number(n - 2));
1508 }
1509
1510 // First term: sec^(n-2)(x) * tan(x) / (n-1)
1511 auto inner = SymbolicExpr::multiply(sec_pow, tan_x);
1512 auto first = SymbolicExpr::multiply(trig_rational(1, n - 1), inner);
1513
1514 // Second term: ((n-2)/(n-1)) * int sec^(n-2)
1515 auto rest = integrate_sec_power(n - 2, var, ctx, depth + 1);
1516 if (!rest) return nullptr;
1517 auto coeff = SymbolicExpr::number(Rational(BigInt(static_cast<long long>(n - 2)),
1518 BigInt(static_cast<long long>(n - 1))));
1519 auto second = SymbolicExpr::multiply(coeff, rest);
1520
1521 return SymbolicExpr::add(first, second);
1522}

1542 strategies_.push_back(std::move(strategy));
1543 } else {
1544 strategies_.insert(strategies_.begin() + position, std::move(strategy));
1545 }

1563 for (size_t i = 0; i < cycle_state_.history.size(); ++i) {

1565 if (!valid_dependency(*ratio, var)) {

1572void Integrator::resolve_cycle(std::shared_ptr<SymbolicExpr>& result, size_t cycle_idx) {
1573 std::string cycle_var = "INT_CYCLE_" + std::to_string(cycle_idx);
1574 if (valid_dependency(*result, cycle_var)) {
1575 auto B = result->differentiate(cycle_var)->simplify();
1576 auto A = result->substitute(cycle_var, SymbolicExpr::number(0))->simplify();
1577 auto one_minus_B = sym_sub(*SymbolicExpr::number(1), *B)->simplify();
1578 if (!one_minus_B->is_zero()) {
1579 result = SymbolicExpr::divide(A, one_minus_B);
1580 }
1581 }

1584std::shared_ptr<SymbolicExpr> Integrator::apply_linearity(
1585 const SymbolicExpr& expr, const std::string& var) {
1586
1587 auto simp_expr = expr.simplify();
1588
1589 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(simp_expr->root)) {
1590 std::vector<std::shared_ptr<SymbolicNode>> constants;
1591 std::vector<std::shared_ptr<SymbolicNode>> dependents;
1592
1593 for (auto& op : mul->operands) {
1594 SymbolicExpr term(op);
1595 if (!valid_dependency(term, var)) {
1596 constants.push_back(op);
1597 } else {
1598 dependents.push_back(op);
1599 }
1600 }
1601
1602 if (!constants.empty() && dependents.size() < mul->operands.size()) {
1603 SymbolicExpr const_part = (constants.size() == 1) ?
1604 SymbolicExpr(constants[0]) : SymbolicExpr(std::make_shared<MultiplyNode>(constants));
1605 SymbolicExpr dep_part = (dependents.empty()) ?
1606 *SymbolicExpr::number(1) :
1607 ((dependents.size() == 1) ? SymbolicExpr(dependents[0]) : SymbolicExpr(std::make_shared<MultiplyNode>(dependents)));
1608
1609 auto int_part = integrate(dep_part, var);
1610 return SymbolicExpr::multiply(std::make_shared<SymbolicExpr>(const_part), std::make_shared<SymbolicExpr>(int_part));
1611 }
1612 }
1613
1614 if (auto add = std::dynamic_pointer_cast<AddNode>(simp_expr->root)) {
1615 std::vector<std::shared_ptr<SymbolicNode>> results;
1616 for (auto& op : add->operands) {
1617 SymbolicExpr term(op);
1618 auto int_term = integrate(term, var);
1619 results.push_back(int_term.root);
1620 }
1621 return std::make_shared<SymbolicExpr>(std::make_shared<AddNode>(results));
1622 }
1623
1624 return nullptr;
1625}

1627std::shared_ptr<SymbolicExpr> Integrator::dispatch_strategies(
1628 const SymbolicExpr& expr, const std::string& var, int depth) {
1629
1630 for (auto& strategy : strategies_) {
1631 auto result = strategy->try_integrate(expr, var, *this, depth);
1632 if (result) {
1633 err_stream << "[Integration] Strategy '" << strategy->name() << "' succeeded\n";
1634 return result;
1635 }
1636 }
1637 return nullptr;
1638}
1639
1640std::shared_ptr<SymbolicExpr> Integrator::integrate_recursive(
1641 const SymbolicExpr& expr, const std::string& var, int depth) {
1642
1643 if (depth > max_depth_) {
1644 return make_integral_node(expr, var);
1645 }
1646
1647 auto cycle_result = check_cycle(expr, var);
1648 if (cycle_result) return cycle_result;
1649
1650 cycle_state_.history.push_back(expr);
1651 size_t my_idx = cycle_state_.history.size() - 1;
1652
1653 if (expr.is_number() || !valid_dependency(expr, var)) {
1654 cycle_state_.history.pop_back();
1655 return SymbolicExpr::multiply(make_expr_ptr(expr), SymbolicExpr::variable(var));
1656 }
1657
1658 auto result = dispatch_strategies(expr, var, depth);
1659
1660 cycle_state_.history.pop_back();
1661
1662 if (!result) {
1663 return make_integral_node(expr, var);
1664 }
1665
1666 resolve_cycle(result, my_idx);
1667
1668 return result;
1669}

1671SymbolicExpr Integrator::integrate(const SymbolicExpr& expr, const std::string& var_name) {
1672
1673 cycle_state_.history.clear();
1674
1675 auto linear_result = apply_linearity(expr, var_name);
1676 if (linear_result) return *linear_result;
1677
1678 return *integrate_recursive(expr, var_name, 0);
1679}

1681SymbolicExpr Integrator::integrate_def(const SymbolicExpr& expr, const std::string& var_name,
1682 const SymbolicExpr& lower, const SymbolicExpr& upper) {
1683
1684 SymbolicExpr simp_expr_val = *expr.simplify();
1685 bool is_inv_x = false;
1686
1687 if (auto pow = std::dynamic_pointer_cast<PowerNode>(simp_expr_val.root)) {
1688 if (auto v = std::dynamic_pointer_cast<VariableNode>(pow->base)) {
1689 if (v->name == var_name) {
1690 if (auto en = std::dynamic_pointer_cast<NumberNode>(pow->exponent)) {
1691 int eq_minus_one = 0;
1692 if (std::holds_alternative<lmmc_real_t>(en->value)) {
1693 lmmc_double_nearly_equal_tol(std::get<lmmc_real_t>(en->value), -1.0, 1e-9, 1e-9, &eq_minus_one);
1694 }
1695 if ((std::holds_alternative<lmmc_real_t>(en->value) && eq_minus_one != 0) ||
1696 (std::holds_alternative<BigInt>(en->value) && std::get<BigInt>(en->value).to_int() == -1) ||
1697 (std::holds_alternative<Rational>(en->value) && std::get<Rational>(en->value).to_double() == -1.0)) {
1698 is_inv_x = true;
1699 }
1700 }
1701 }
1702 }
1703 }
1704
1705 bool numeric_bounds = (lower.root && std::dynamic_pointer_cast<NumberNode>(lower.root)) &&
1706 (upper.root && std::dynamic_pointer_cast<NumberNode>(upper.root));
1707
1708 if (is_inv_x && numeric_bounds) {
1709 double l_val = lower.to_numeric();
1710 double u_val = upper.to_numeric();
1711 int eq_l, eq_u;
1712 lmmc_double_nearly_equal_tol(l_val, 0.0, 1e-9, 1e-9, &eq_l);
1713 lmmc_double_nearly_equal_tol(u_val, 0.0, 1e-9, 1e-9, &eq_u);
1714 if (!eq_l && l_val < 0 && !eq_u && u_val > 0) {
1715 auto t = std::make_shared<SymbolicExpr>(*SymbolicExpr::variable("t"));
1716 auto zero = std::make_shared<SymbolicExpr>(*SymbolicExpr::number(0));
1717 auto int_left = integrate_def(expr, var_name, lower, *t);
1718 auto lim_left = int_left.limit("t", zero, "-");
1719 auto int_right = integrate_def(expr, var_name, *t, upper);
1720 auto lim_right = int_right.limit("t", zero, "+");
1721 if (lim_left && lim_right) {
1722 return *SymbolicExpr::add(lim_left, lim_right);
1723 }
1724 }
1725 }
1726
1727 SymbolicExpr indefinite = integrate(expr, var_name);
1728
1729 if (auto func = std::dynamic_pointer_cast<FunctionNode>(indefinite.root)) {
1730 if (func->type == FunctionNode::FuncType::Calculus_Integral) {
1731 std::vector<std::shared_ptr<SymbolicNode>> args;
1732 args.push_back(expr.root);
1733 args.push_back(SymbolicExpr::variable(var_name)->root);
1734 args.push_back(lower.root);
1735 args.push_back(upper.root);
1736 return SymbolicExpr(std::make_shared<FunctionNode>(FunctionNode::FuncType::Calculus_Integral, args));
1737 }
1738 }
1739
1740 auto F_b = indefinite.substitute(var_name, make_expr_ptr(upper));
1741 auto F_a = indefinite.substitute(var_name, make_expr_ptr(lower));
1742 auto result = sym_sub(*F_b, *F_a);
1743 return *result->simplify();
1744}

1746// ---------------------------------------------------------------
1747// RationalDecompositionStrategy
1748// ---------------------------------------------------------------
1749//
1750// Handles rational integrands P(x)/Q(x) where deg(Q) >= 3 (lower-degree
1751// quadratic cases are caught earlier by PartialFractionStrategy). The
1752// algorithm follows the textbook recipe:
1753//
1754// 1. extract_rational : factor the integrand into (numerator polynomial,
1755// denominator polynomial) over Q. Returns false if the input is not a
1756// rational function (e.g. contains sin, exp, ln, irrational powers).
1757// 2. poly_divide : long-divide P by Q when deg(P) >= deg(Q). The
1758// polynomial quotient is integrated term-by-term via the power rule.
1759// 3. factor_denominator : square-free factor Q, peel off rational linear
1760// roots, leaving each leftover factor either linear or an irreducible
1761// quadratic. Refuses to handle higher-degree irreducible factors.
1762// 4. solve_coefficients : set up the partial-fraction ansatz with one
1763// unknown per linear power and two unknowns per irreducible quadratic
1764// power, expand into a linear system whose unknowns are the partial
1765// fraction coefficients, and solve via gaussian_eliminate.
1766// 5. integrate_term : integrate each component analytically.
1767//
1768// On any failure that is not "expression is not rational" we fall back to
1769// returning an unevaluated integral node so that the rest of the pipeline
1770// is preserved.
1771
1772namespace {
1773
1774// Canonical builder utilities used throughout the rational strategy.
1775inline std::shared_ptr<SymbolicExpr> rd_num_rat(const Rational& r) {
1776 return SymbolicExpr::number(r);
1777}
1778
1779inline std::shared_ptr<SymbolicExpr> rd_num_int(long long v) {
1780 return SymbolicExpr::number(BigInt(v));
1781}
1782
1783// Build (var - r) as a SymbolicExpr.
1784inline std::shared_ptr<SymbolicExpr> rd_var_minus(const std::string& var, const Rational& r) {
1785 auto v = SymbolicExpr::variable(var);
1786 if (r == Rational(0)) return v;
1787 auto neg_r = SymbolicExpr::number(Rational(0) - r);
1788 return SymbolicExpr::add(v, neg_r);
1789}

1791// Test whether a rational-coefficient poly is the zero polynomial.
1792inline bool rd_is_zero_poly(const Polynomial<Rational>& p) {
1793 if (p.coeffs.empty()) return true;
1794 for (const auto& c : p.coeffs) if (!(c == Rational(0))) return false;
1795 return true;
1796}

1798// Convert a Polynomial<Rational> into a SymbolicExpr in the given variable.
1799inline std::shared_ptr<SymbolicExpr> rd_poly_to_sym(
1800 const Polynomial<Rational>& p, const std::string& var) {
1801 if (rd_is_zero_poly(p)) return SymbolicExpr::number(0);
1802 std::vector<std::shared_ptr<SymbolicExpr>> terms;
1803 auto v = SymbolicExpr::variable(var);
1804 for (size_t i = 0; i < p.coeffs.size(); ++i) {
1805 if (p.coeffs[i] == Rational(0)) continue;
1806 std::shared_ptr<SymbolicExpr> t;
1807 if (i == 0) {
1808 t = rd_num_rat(p.coeffs[i]);
1809 } else if (i == 1) {
1810 if (p.coeffs[i] == Rational(1)) {
1811 t = v;
1812 } else {
1813 t = SymbolicExpr::multiply(rd_num_rat(p.coeffs[i]), v);
1814 }
1815 } else {
1816 auto pw = SymbolicExpr::power(v, rd_num_int(static_cast<long long>(i)));
1817 if (p.coeffs[i] == Rational(1)) {
1818 t = pw;
1819 } else {
1820 t = SymbolicExpr::multiply(rd_num_rat(p.coeffs[i]), pw);
1821 }
1822 }
1823 terms.push_back(t);
1824 }
1825 if (terms.empty()) return SymbolicExpr::number(0);
1826 if (terms.size() == 1) return terms[0];
1827 auto res = terms[0];
1828 for (size_t i = 1; i < terms.size(); ++i) res = SymbolicExpr::add(res, terms[i]);
1829 return res;
1830}

1832// Recursively decompose a SymbolicNode into (numerator-polys, denominator-polys)
1833// over the rationals. Each polynomial is assumed to be a polynomial in `var`
1834// with rational coefficients. Returns false on any non-rational sub-expression
1835// (functions, irrational/symbolic exponents, variables other than the integration
1836// variable). Multiplication aggregates by appending; division comes from
1837// PowerNode with negative integer exponent. Numbers and the integration
1838// variable are converted directly via symbolic_to_poly.
1839bool rd_collect_rational(const std::shared_ptr<SymbolicNode>& node,
1840 const std::string& var,
1841 std::vector<Polynomial<Rational>>& num,
1842 std::vector<Polynomial<Rational>>& den) {
1843 if (!node) return false;
1844
1845 // A NumberNode is a constant; convert directly.
1846 if (auto n = std::dynamic_pointer_cast<NumberNode>(node)) {
1847 Polynomial<Rational> p =
1848 symbolic_to_poly<Rational>(std::make_shared<SymbolicExpr>(n), var);
1849 num.push_back(p);
1850 return true;
1851 }
1852
1853 // The integration variable, or any other variable that's actually a constant.
1854 if (auto v = std::dynamic_pointer_cast<VariableNode>(node)) {
1855 if (v->name == var) {
1856 num.push_back(Polynomial<Rational>({Rational(0), Rational(1)}, var));
1857 return true;
1858 }
1859 // Unknown symbolic constant: cannot lift safely into rational poly.
1860 return false;
1861 }
1862
1863 if (auto add = std::dynamic_pointer_cast<AddNode>(node)) {
1864 // Addition: combine all summands over a common denominator.
1865 // We collect each operand as P_i/Q_i and assemble
1866 // sum P_i * (prod_{j != i} Q_j) / prod_j Q_j
1867 std::vector<std::pair<std::vector<Polynomial<Rational>>,
1868 std::vector<Polynomial<Rational>>>> parts;
1869 parts.reserve(add->operands.size());
1870 for (const auto& op : add->operands) {
1871 std::vector<Polynomial<Rational>> sub_num, sub_den;
1872 if (!rd_collect_rational(op, var, sub_num, sub_den)) return false;
1873 parts.push_back({std::move(sub_num), std::move(sub_den)});
1874 }
1875
1876 // Combine: numerator = sum_i (prod_k num_k_i) * prod_{j != i} (prod_k den_k_j)
1877 // denominator = prod_i (prod_k den_k_i)
1878 Polynomial<Rational> total_num(var); // start as 0
1879 Polynomial<Rational> total_den({Rational(1)}, var);
1880
1881 // Pre-compute Q_i (denominator product per term) and total denominator.
1882 std::vector<Polynomial<Rational>> Q_per_term;
1883 Q_per_term.reserve(parts.size());
1884 for (const auto& [pn, pd] : parts) {
1885 Polynomial<Rational> q({Rational(1)}, var);
1886 for (const auto& d : pd) q = q * d;
1887 Q_per_term.push_back(q);
1888 total_den = total_den * q;
1889 }
1890
1891 for (size_t i = 0; i < parts.size(); ++i) {
1892 Polynomial<Rational> p({Rational(1)}, var);
1893 for (const auto& nn : parts[i].first) p = p * nn;
1894 // multiply by prod_{j != i} Q_j
1895 for (size_t j = 0; j < parts.size(); ++j) {
1896 if (j == i) continue;
1897 p = p * Q_per_term[j];
1898 }
1899 total_num = total_num + p;
1900 }
1901
1902 num.push_back(total_num);
1903 den.push_back(total_den);
1904 return true;
1905 }
1906
1907 if (auto mul = std::dynamic_pointer_cast<MultiplyNode>(node)) {
1908 for (const auto& op : mul->operands) {
1909 if (!rd_collect_rational(op, var, num, den)) return false;
1910 }
1911 return true;
1912 }
1913
1914 if (auto pw = std::dynamic_pointer_cast<PowerNode>(node)) {
1915 // Only integer exponents (negative or non-negative) are allowed.
1916 auto en = std::dynamic_pointer_cast<NumberNode>(pw->exponent);
1917 if (!en) return false;
1918
1919 long long exp_v = 0;
1920 bool ok = false;
1921 if (std::holds_alternative<BigInt>(en->value)) {
1922 const auto& bi = std::get<BigInt>(en->value);
1923 // Bound to keep matrix sizes sane.
1924 int v = bi.to_int();
1925 if (v >= -64 && v <= 64) { ok = true; exp_v = v; }
1926 } else if (std::holds_alternative<Rational>(en->value)) {
1927 const auto& r = std::get<Rational>(en->value);
1928 if (r.is_integer()) {
1929 int v = r.to_BigInt().to_int();
1930 if (v >= -64 && v <= 64) { ok = true; exp_v = v; }
1931 }
1932 } else if (std::holds_alternative<lmmc_real_t>(en->value)) {
1933 lmmc_real_t d = std::get<lmmc_real_t>(en->value);
1934 if (std::isfinite(d) && d == std::floor(d) && d >= -64.0 && d <= 64.0) {
1935 ok = true; exp_v = static_cast<long long>(d);
1936 }
1937 }
1938 if (!ok) return false;
1939
1940 // Build the base polynomial.
1941 std::vector<Polynomial<Rational>> bn, bd;
1942 if (!rd_collect_rational(pw->base, var, bn, bd)) return false;
1943
1944 // base_num = product of bn ; base_den = product of bd
1945 Polynomial<Rational> base_num({Rational(1)}, var);
1946 for (const auto& p : bn) base_num = base_num * p;
1947 Polynomial<Rational> base_den({Rational(1)}, var);
1948 for (const auto& p : bd) base_den = base_den * p;
1949
1950 if (exp_v == 0) {
1951 num.push_back(Polynomial<Rational>({Rational(1)}, var));
1952 return true;
1953 }
1954 long long e_abs = std::llabs(exp_v);
1955 Polynomial<Rational> n_pow({Rational(1)}, var);
1956 Polynomial<Rational> d_pow({Rational(1)}, var);
1957 for (long long i = 0; i < e_abs; ++i) {
1958 n_pow = n_pow * base_num;
1959 d_pow = d_pow * base_den;
1960 }
1961 if (exp_v > 0) {
1962 num.push_back(n_pow);
1963 den.push_back(d_pow);
1964 } else {
1965 // (P/Q)^(-k) = (Q/P)^k
1966 num.push_back(d_pow);
1967 den.push_back(n_pow);
1968 }
1969 return true;
1970 }
1971
1972 // FunctionNode etc. -> not a rational function in `var`.
1973 return false;
1974}

1985bool RationalDecompositionStrategy::extract_rational(
1986 const SymbolicExpr& expr, const std::string& var,
1987 Polynomial<Rational>& P_out, Polynomial<Rational>& Q_out) {
1988
1989 std::vector<Polynomial<Rational>> nums, dens;
1990 if (!rd_collect_rational(expr.root, var, nums, dens)) return false;
1991
1992 Polynomial<Rational> P({Rational(1)}, var);
1993 for (const auto& p : nums) P = P * p;
1994 Polynomial<Rational> Q({Rational(1)}, var);
1995 for (const auto& p : dens) Q = Q * p;
1996
1997 if (rd_is_zero_poly(Q)) return false; // 0 in denominator, refuse
1998
1999 // Reduce by GCD to keep things small (and keep variable name consistent).
2000 if (!rd_is_zero_poly(P)) {
2001 Polynomial<Rational> g = Polynomial<Rational>::gcd(P, Q);
2002 if (!rd_is_zero_poly(g) && g.degree() >= 1) {
2003 auto pq = P.div_mod(g);
2004 auto qq = Q.div_mod(g);
2005 if (rd_is_zero_poly(pq.second) && rd_is_zero_poly(qq.second)) {
2006 P = pq.first;
2007 Q = qq.first;
2008 }
2009 }
2010 }
2011
2012 // Make Q monic (rescale numerator accordingly).
2013 if (Q.degree() >= 0) {
2014 Rational lc = Q.lead_coeff();
2015 if (!(lc == Rational(1)) && !(lc == Rational(0))) {
2016 Polynomial<Rational> P_scaled(var);
2017 P_scaled.coeffs.reserve(P.coeffs.size());
2018 for (const auto& c : P.coeffs) P_scaled.coeffs.push_back(c / lc);
2019 Polynomial<Rational> Q_scaled(var);
2020 Q_scaled.coeffs.reserve(Q.coeffs.size());
2021 for (const auto& c : Q.coeffs) Q_scaled.coeffs.push_back(c / lc);
2022 P_scaled.trim();
2023 Q_scaled.trim();
2024 P = P_scaled;
2025 Q = Q_scaled;
2026 }
2027 }
2028
2029 P_out = P;
2030 Q_out = Q;
2031 return true;
2032}

2042
2043bool RationalDecompositionStrategy::factor_denominator(
2044 const Polynomial<Rational>& Q,
2045 std::vector<std::pair<Polynomial<Rational>, int>>& factors_out) {
2046 factors_out.clear();
2047 if (Q.degree() <= 0) return false;
2048
2049 // Step 1: square-free factorization to peel off multiplicities.
2050 auto sqfree = square_free_factorization(Q);
2051 if (sqfree.empty()) return false;
2052
2053 // Step 2: for each square-free factor, peel rational roots, leaving an
2054 // irreducible-quadratic cofactor at most. Anything that cannot be
2055 // reduced to factors of degree <= 2 forces this step to fail.
2056 for (auto& [piece, mult] : sqfree) {
2057 Polynomial<Rational> current = piece.make_monic();
2058
2059 if (current.degree() == 0) {
2060 // pure constant factor; nothing to integrate, ignore
2061 continue;
2062 }
2063 if (current.degree() == 1 || current.degree() == 2) {
2064 // Irreducible quadratics can stay as-is; quadratic discriminant
2065 // determines splitting later in solve_coefficients/integrate_term.
2066 // For deg 2 we still try rational roots to split into two linears
2067 // if possible.
2068 if (current.degree() == 2) {
2069 auto roots = find_rational_roots(current);
2070 if (!roots.empty()) {
2071 for (const auto& r : roots) {
2072 Polynomial<Rational> linear({Rational(0) - r, Rational(1)}, current.variable_name);
2073 auto qr = current.div_mod(linear);
2074 if (!rd_is_zero_poly(qr.second)) {
2075 return false;
2076 }
2077 factors_out.push_back({linear, mult});
2078 current = qr.first;
2079 if (current.degree() == 0) break;
2080 }
2081 }
2082 if (current.degree() == 2) {
2083 // Confirmed irreducible quadratic over Q.
2084 factors_out.push_back({current.make_monic(), mult});
2085 current = Polynomial<Rational>({Rational(1)}, current.variable_name);
2086 } else if (current.degree() == 1) {
2087 factors_out.push_back({current.make_monic(), mult});
2088 current = Polynomial<Rational>({Rational(1)}, current.variable_name);
2089 }
2090 continue;
2091 }
2092 // degree 1
2093 factors_out.push_back({current, mult});
2094 continue;
2095 }
2096
2097 // degree >= 3: try to split off rational linear factors.
2098 auto roots = find_rational_roots(current);
2099 for (const auto& r : roots) {
2100 Polynomial<Rational> linear({Rational(0) - r, Rational(1)}, current.variable_name);
2101 auto qr = current.div_mod(linear);
2102 if (rd_is_zero_poly(qr.second)) {
2103 factors_out.push_back({linear, mult});
2104 current = qr.first;
2105 if (current.degree() <= 0) break;
2106 }
2107 }
2108
2109 if (current.degree() <= 0) continue;
2110 if (current.degree() == 1) {
2111 factors_out.push_back({current.make_monic(), mult});
2112 continue;
2113 }
2114 if (current.degree() == 2) {
2115 factors_out.push_back({current.make_monic(), mult});
2116 continue;
2117 }
2118 // Higher-degree irreducible factor over Q: cannot handle in this
2119 // strategy. Caller will return an unevaluated integral.
2120 return false;
2121 }
2122 return true;
2123}

2125bool RationalDecompositionStrategy::solve_coefficients(
2126 const Polynomial<Rational>& P,
2127 const Polynomial<Rational>& Q,
2128 const std::vector<std::pair<Polynomial<Rational>, int>>& factors,
2129 std::vector<Polynomial<Rational>>& numerators_out) {
2130
2131 numerators_out.clear();
2132
2133 // Map each (factor index, power) to a list of unknowns.
2134 // Linear factor (x - r)^k contributes k unknowns A_1..A_k (constant numerators).
2135 // Quadratic factor q(x)^k contributes 2k unknowns: (B_l, C_l) per power l.
2136 //
2137 // We also remember, for each (factor index, power l), the partial-fraction
2138 // term's *full* numerator polynomial (constant or linear) as a function of
2139 // the unknowns. Then we multiply each term by the missing portion of Q to
2140 // get a polynomial whose coefficients (in x) are linear combinations of the
2141 // unknowns; equating to P gives the linear system.
2142
2143 // Total number of unknowns.
2144 size_t num_unknowns = 0;
2145 std::vector<std::pair<int, int>> ifac_ipow; // (factor_index, power index l in [1..mult])
2146 std::vector<int> kind_per_unknown_block; // 1 -> linear, 2 -> quadratic
2147 std::vector<size_t> unknown_offset_per_term;
2148 unknown_offset_per_term.reserve(factors.size());
2149
2150 for (size_t i = 0; i < factors.size(); ++i) {
2151 unknown_offset_per_term.push_back(num_unknowns);
2152 const auto& [fpoly, mult] = factors[i];
2153 if (fpoly.degree() == 1) {
2154 num_unknowns += static_cast<size_t>(mult);
2155 } else if (fpoly.degree() == 2) {
2156 num_unknowns += static_cast<size_t>(2 * mult);
2157 } else {
2158 return false;
2159 }
2160 }
2161 if (num_unknowns == 0) return false;
2162
2163 const std::string& var = Q.variable_name;
2164 int N = Q.degree();
2165 if (N < 0) return false;
2166
2167 // Build the matrix: each column is the contribution (coefficient vector
2168 // of x^0..x^{N-1}) of one unknown to the LHS polynomial; each row is one
2169 // equation indexed by the power of x.
2170 // Number of equations = N (size of polynomial of degree < N for the RHS P
2171 // when properly partial-fractioned: deg(P) < N).
2172 size_t rows = static_cast<size_t>(N);
2173 size_t cols = num_unknowns + 1; // augmented column for P
2174
2175 // Initialize augmented matrix as Rational.
2176 std::vector<std::vector<Rational>> M(rows, std::vector<Rational>(cols, Rational(0)));
2177
2178 // RHS column: coefficients of P (degree < N expected; if deg(P) >= N
2179 // then we have a problem — caller must run poly_divide first).
2180 if (P.degree() >= N) return false;
2181 for (size_t k = 0; k < rows; ++k) {
2182 Rational c = (k < P.coeffs.size()) ? P.coeffs[k] : Rational(0);
2183 M[k][num_unknowns] = c;
2184 }
2185
2186 // For each factor, for each multiplicity power l, build the contribution
2187 // poly = numerator_term * (Q / fpoly^l). Each unknown in this term
2188 // contributes one column.
2189 for (size_t i = 0; i < factors.size(); ++i) {
2190 const auto& [fpoly, mult] = factors[i];
2191 // Q / fpoly^l for l = 1..mult; build incrementally: start with Q/fpoly^mult
2192 // by dividing Q by fpoly mult times.
2193 Polynomial<Rational> remaining = Q;
2194 for (int t = 0; t < mult; ++t) {
2195 auto qr = remaining.div_mod(fpoly);
2196 if (!rd_is_zero_poly(qr.second)) {
2197 // Q does not divide cleanly by this power of factor -> bug.
2198 return false;
2199 }
2200 remaining = qr.first;
2201 }
2202 // remaining now equals Q / fpoly^mult.
2203
2204 // Now build for each power l = 1..mult:
2205 // term coefficient pattern is multiplied by Q / fpoly^l
2206 // = remaining * fpoly^(mult - l)
2207 Polynomial<Rational> fp_pow({Rational(1)}, var);
2208 // fp_pow starts as fpoly^0 = 1; we will iterate l from mult down to 1
2209 // so we increment fp_pow by * fpoly each time after using it.
2210 // Pre-build the array of coefficient polynomials for each l in 1..mult.
2211 std::vector<Polynomial<Rational>> q_over_fpow_l(mult + 1, Polynomial<Rational>(var));
2212 // index by l from 1..mult: q_over_fpow_l[l] = remaining * fpoly^(mult - l)
2213 Polynomial<Rational> cur({Rational(1)}, var);
2214 for (int l = mult; l >= 1; --l) {
2215 // when l = mult: cur = fpoly^0 = 1, factor = remaining * 1 = remaining
2216 // when decreasing l: factor = remaining * fpoly^(mult - l)
2217 Polynomial<Rational> factor_poly = remaining * cur;
2218 q_over_fpow_l[l] = factor_poly;
2219 cur = cur * fpoly;
2220 }
2221
2222 size_t col_off = unknown_offset_per_term[i];
2223
2224 if (fpoly.degree() == 1) {
2225 // Linear factor. Each l contributes one unknown A_l (constant
2226 // numerator). Coefficient column = q_over_fpow_l[l].
2227 for (int l = 1; l <= mult; ++l) {
2228 const auto& qpl = q_over_fpow_l[l];
2229 size_t this_col = col_off + static_cast<size_t>(l - 1);
2230 for (size_t k = 0; k < rows; ++k) {
2231 Rational c = (k < qpl.coeffs.size()) ? qpl.coeffs[k] : Rational(0);
2232 M[k][this_col] = c;
2233 }
2234 }
2235 } else if (fpoly.degree() == 2) {
2236 // Quadratic factor. Each l contributes two unknowns (B_l, C_l).
2237 // Numerator at power l = B_l * x + C_l ; the coefficient column for
2238 // B_l is q_over_fpow_l[l] shifted by 1 (multiplied by x), for C_l
2239 // is q_over_fpow_l[l] (multiplied by 1).
2240 for (int l = 1; l <= mult; ++l) {
2241 const auto& qpl = q_over_fpow_l[l];
2242 size_t base_col = col_off + static_cast<size_t>(2 * (l - 1));
2243 size_t b_col = base_col; // for B_l
2244 size_t c_col = base_col + 1; // for C_l
2245 // C_l contributes qpl directly.
2246 for (size_t k = 0; k < rows; ++k) {
2247 Rational c = (k < qpl.coeffs.size()) ? qpl.coeffs[k] : Rational(0);
2248 M[k][c_col] = c;
2249 }
2250 // B_l contributes qpl * x = qpl with index shifted by 1.
2251 for (size_t k = 0; k < rows; ++k) {
2252 if (k == 0) {
2253 M[k][b_col] = Rational(0);
2254 } else {
2255 size_t src = k - 1;
2256 Rational c = (src < qpl.coeffs.size()) ? qpl.coeffs[src] : Rational(0);
2257 M[k][b_col] = c;
2258 }
2259 }
2260 }
2261 }
2262 }
2263
2264 // Solve via Gaussian elimination on Rational matrix.
2265 size_t m = rows;
2266 size_t n = cols;
2267 int sign = 1;
2268 (void)sign;
2269 size_t pivot_row = 0;
2270 std::vector<size_t> pivot_col_for_row(m, static_cast<size_t>(-1));
2271 for (size_t col = 0; col + 1 < n && pivot_row < m; ++col) {
2272 size_t max_row = pivot_row;
2273 bool found = false;
2274 while (max_row < m) {
2275 if (!(M[max_row][col] == Rational(0))) { found = true; break; }
2276 ++max_row;
2277 }
2278 if (!found) continue;
2279 if (max_row != pivot_row) std::swap(M[pivot_row], M[max_row]);
2280 pivot_col_for_row[pivot_row] = col;
2281 Rational pivot = M[pivot_row][col];
2282 // Normalize pivot row.
2283 for (size_t k = col; k < n; ++k) {
2284 M[pivot_row][k] = M[pivot_row][k] / pivot;
2285 }
2286 // Eliminate in other rows.
2287 for (size_t i = 0; i < m; ++i) {
2288 if (i == pivot_row) continue;
2289 Rational f = M[i][col];
2290 if (f == Rational(0)) continue;
2291 for (size_t k = col; k < n; ++k) {
2292 M[i][k] = M[i][k] - f * M[pivot_row][k];
2293 }
2294 }
2295 ++pivot_row;
2296 }
2297
2298 // Check for inconsistency: any row with all-zero LHS but nonzero RHS.
2299 for (size_t r = 0; r < m; ++r) {
2300 bool all_zero_lhs = true;
2301 for (size_t c = 0; c + 1 < n; ++c) {
2302 if (!(M[r][c] == Rational(0))) { all_zero_lhs = false; break; }
2303 }
2304 if (all_zero_lhs && !(M[r][n - 1] == Rational(0))) return false;
2305 }
2306
2307 // Extract solution: requires that each unknown has a pivot column.
2308 std::vector<Rational> sol(num_unknowns, Rational(0));
2309 std::vector<bool> pivoted(num_unknowns, false);
2310 for (size_t r = 0; r < m; ++r) {
2311 size_t pc = pivot_col_for_row[r];
2312 if (pc == static_cast<size_t>(-1)) continue;
2313 if (pc >= num_unknowns) continue;
2314 sol[pc] = M[r][n - 1];
2315 pivoted[pc] = true;
2316 }
2317 for (size_t i = 0; i < num_unknowns; ++i) {
2318 if (!pivoted[i]) return false; // underdetermined -> can't form unique partial fractions
2319 }
2320
2321 // Re-package solution into per-(factor, power) numerator polynomials.
2322 numerators_out.reserve(num_unknowns);
2323 for (size_t i = 0; i < factors.size(); ++i) {
2324 const auto& [fpoly, mult] = factors[i];
2325 size_t col_off = unknown_offset_per_term[i];
2326 if (fpoly.degree() == 1) {
2327 for (int l = 1; l <= mult; ++l) {
2328 Rational A = sol[col_off + static_cast<size_t>(l - 1)];
2329 numerators_out.push_back(Polynomial<Rational>({A}, Q.variable_name));
2330 }
2331 } else { // degree 2
2332 for (int l = 1; l <= mult; ++l) {
2333 size_t base_col = col_off + static_cast<size_t>(2 * (l - 1));
2334 Rational B = sol[base_col];
2335 Rational C = sol[base_col + 1];
2336 numerators_out.push_back(
2337 Polynomial<Rational>({C, B}, Q.variable_name));
2338 }
2339 }
2340 }
2341 return true;
2342}

2344std::shared_ptr<SymbolicExpr> RationalDecompositionStrategy::integrate_term(
2345 const Polynomial<Rational>& numerator,
2346 const Polynomial<Rational>& factor,
2347 int power, const std::string& var) {
2348
2349 if (rd_is_zero_poly(numerator)) return SymbolicExpr::number(0);
2350
2351 if (factor.degree() == 1) {
2352 // factor = x - r (monic linear). r = -coeffs[0].
2353 Rational r = Rational(0) - factor.coeffs[0];
2354 Rational A = (numerator.coeffs.size() > 0) ? numerator.coeffs[0] : Rational(0);
2355 if (A == Rational(0)) return SymbolicExpr::number(0);
2356
2357 if (power == 1) {
2358 // A * ln|x - r|. We use ln(x - r) here (matching existing strategies
2359 // which omit the absolute value).
2360 auto inner = rd_var_minus(var, r);
2361 return SymbolicExpr::multiply(rd_num_rat(A), SymbolicExpr::ln(inner));
2362 }
2363 // power >= 2: -A / ((power-1) * (x - r)^(power-1))
2364 Rational coeff = Rational(0) - A / Rational(BigInt(power - 1));
2365 auto base = rd_var_minus(var, r);
2366 auto exp = rd_num_int(power - 1);
2367 auto pw = SymbolicExpr::power(base, exp);
2368 auto inv_pw = SymbolicExpr::power(pw, rd_num_int(-1));
2369 return SymbolicExpr::multiply(rd_num_rat(coeff), inv_pw);
2370 }
2371
2372 if (factor.degree() == 2) {
2373 // factor = x^2 + p*x + q (monic quadratic). discriminant = p^2 - 4q.
2374 Rational q = (factor.coeffs.size() > 0) ? factor.coeffs[0] : Rational(0);
2375 Rational p = (factor.coeffs.size() > 1) ? factor.coeffs[1] : Rational(0);
2376
2377 // Verify it's irreducible: discriminant must be < 0. (If it's >= 0
2378 // we should have factored it; but be defensive.)
2379 Rational disc = p * p - Rational(4) * q;
2380
2381 // numerator = B*x + C
2382 Rational B = (numerator.coeffs.size() > 1) ? numerator.coeffs[1] : Rational(0);
2383 Rational C = (numerator.coeffs.size() > 0) ? numerator.coeffs[0] : Rational(0);
2384
2385 if (power >= 2) {
2386 // Higher powers of irreducible quadratics: leave as unevaluated
2387 // integral; this triggers the safe fallback in try_integrate.
2388 // Build the symbolic integrand for the unevaluated node.
2389 auto num_sym = rd_poly_to_sym(numerator, var);
2390 auto den_sym_base = rd_poly_to_sym(factor, var);
2391 auto den_pw = SymbolicExpr::power(den_sym_base, rd_num_int(power));
2392 auto inv_den = SymbolicExpr::power(den_pw, rd_num_int(-1));
2393 auto integrand = SymbolicExpr::multiply(num_sym, inv_den);
2394 std::vector<std::shared_ptr<SymbolicNode>> args;
2395 args.push_back(integrand->root);
2396 args.push_back(SymbolicExpr::variable(var)->root);
2397 return std::make_shared<SymbolicExpr>(
2398 std::make_shared<FunctionNode>(FunctionNode::FuncType::Calculus_Integral, args));
2399 }
2400
2401 // power == 1: ∫ (B x + C) / (x^2 + p x + q) dx
2402 // Split numerator: (B x + C) = (B/2) * (2x + p) + (C - B p / 2)
2403 // ∫ (B/2)(2x+p)/(x^2+px+q) dx = (B/2) * ln(x^2+px+q)
2404 // ∫ (C - B p / 2) / (x^2 + p x + q) dx
2405 // = (C - B p / 2) * (2 / sqrt(4q - p^2)) * arctan( (2x + p) / sqrt(4q - p^2) )
2406 std::shared_ptr<SymbolicExpr> result = SymbolicExpr::number(0);
2407
2408 Rational B_half = B / Rational(2);
2409 if (!(B_half == Rational(0))) {
2410 auto den_sym = rd_poly_to_sym(factor, var);
2411 auto ln_part = SymbolicExpr::ln(den_sym);
2412 auto term_log = SymbolicExpr::multiply(rd_num_rat(B_half), ln_part);
2413 result = SymbolicExpr::add(result, term_log);
2414 }
2415
2416 Rational atan_coeff = C - (B * p) / Rational(2);
2417 if (!(atan_coeff == Rational(0))) {
2418 // s2 = -disc = 4q - p^2 (positive for irreducible factor).
2419 Rational s2 = Rational(0) - disc;
2420 // Build sqrt(s2) symbolically (works even when s2 isn't a perfect square).
2421 auto s2_sym = rd_num_rat(s2);
2422 auto sqrt_s2 = SymbolicExpr::sqrt(s2_sym);
2423 // 2x + p
2424 auto two_x = SymbolicExpr::multiply(rd_num_int(2), SymbolicExpr::variable(var));
2425 std::shared_ptr<SymbolicExpr> two_x_plus_p;
2426 if (p == Rational(0)) {
2427 two_x_plus_p = two_x;
2428 } else {
2429 two_x_plus_p = SymbolicExpr::add(two_x, rd_num_rat(p));
2430 }
2431 auto inv_sqrt = SymbolicExpr::power(sqrt_s2, rd_num_int(-1));
2432 auto arctan_arg = SymbolicExpr::multiply(two_x_plus_p, inv_sqrt);
2433 auto atan_part = make_arctan(arctan_arg);
2434 auto two = rd_num_int(2);
2435 auto coeff_part = SymbolicExpr::multiply(
2436 SymbolicExpr::multiply(rd_num_rat(atan_coeff), two), inv_sqrt);
2437 auto term_atan = SymbolicExpr::multiply(coeff_part, atan_part);
2438 result = SymbolicExpr::add(result, term_atan);
2439 }
2440
2441 return result;
2442 }
2443
2444 // Should not reach here (factor degree > 2 is rejected by factor_denominator).
2445 return SymbolicExpr::number(0);
2446}

2448std::shared_ptr<SymbolicExpr> RationalDecompositionStrategy::try_integrate(
2449 const SymbolicExpr& expr, const std::string& var, Integrator& ctx, int depth) {
2450 (void)ctx;
2451 (void)depth;
2452
2453 Polynomial<Rational> P, Q;
2454 try {
2455 if (!extract_rational(expr, var, P, Q)) {
2456 return nullptr; // not rational -> let next strategy try
2457 }
2458 } catch (...) {
2459 return nullptr;
2460 }
2461
2462 // Defensive: if Q is zero or constant, this is not the right strategy.
2463 if (rd_is_zero_poly(Q) || Q.degree() < 1) return nullptr;
2464
2465 // Don't fight with the simpler PartialFractionStrategy on degree <= 2:
2466 // letting RationalDecomposition handle them too is safe, but yields
2467 // bulkier outputs for trivial cases. We accept degree >= 3 here; for
2468 // degrees 1 and 2 we let later strategies (PartialFraction / direct
2469 // table) take over by returning nullptr.
2470 if (Q.degree() < 3) return nullptr;
2471 try {
2472 // Long division if needed.
2473 Polynomial<Rational> quot, rem;
2474 if (P.degree() >= Q.degree()) {
2475 poly_divide(P, Q, quot, rem);
2476 } else {
2477 quot = Polynomial<Rational>(Q.variable_name);
2478 rem = P;
2479 }
2480
2481 // Factor denominator.
2482 std::vector<std::pair<Polynomial<Rational>, int>> factors;
2483 if (!factor_denominator(Q, factors)) {
2484 // Cannot factor over Q -> return unevaluated integral node.
2485 return Integrator::depends_on(expr, var)
2486 ? std::make_shared<SymbolicExpr>(
2487 std::make_shared<FunctionNode>(
2488 FunctionNode::FuncType::Calculus_Integral,
2489 std::vector<std::shared_ptr<SymbolicNode>>{
2490 expr.root,
2491 SymbolicExpr::variable(var)->root}))
2492 : nullptr;
2493 }
2494 if (factors.empty()) {
2495 // Means Q is constant after factoring, so really there's nothing left;
2496 // integrate quot only.
2497 if (rd_is_zero_poly(quot)) return SymbolicExpr::number(0);
2498 // ∫ quot(x) dx = poly_integral(quot)
2499 // fall through; handled below
2500 }
2501
2502 // Solve coefficients on the proper part rem / Q.
2503 std::vector<Polynomial<Rational>> numerators;
2504 if (!rd_is_zero_poly(rem)) {
2505 if (!solve_coefficients(rem, Q, factors, numerators)) {
2506 return std::make_shared<SymbolicExpr>(
2507 std::make_shared<FunctionNode>(
2508 FunctionNode::FuncType::Calculus_Integral,
2509 std::vector<std::shared_ptr<SymbolicNode>>{
2510 expr.root,
2511 SymbolicExpr::variable(var)->root}));
2512 }
2513 }
2514
2515 // Build the result piece by piece.
2516 std::shared_ptr<SymbolicExpr> result = SymbolicExpr::number(0);
2517
2518 // Polynomial part from long division.
2519 if (!rd_is_zero_poly(quot)) {
2520 // Antiderivative of x^k is x^(k+1) / (k+1).
2521 for (size_t k = 0; k < quot.coeffs.size(); ++k) {
2522 if (quot.coeffs[k] == Rational(0)) continue;
2523 Rational coeff = quot.coeffs[k] / Rational(BigInt(static_cast<long long>(k + 1)));
2524 std::shared_ptr<SymbolicExpr> term;
2525 auto v = SymbolicExpr::variable(var);
2526 auto pw = SymbolicExpr::power(v, rd_num_int(static_cast<long long>(k + 1)));
2527 if (coeff == Rational(1)) {
2528 term = pw;
2529 } else {
2530 term = SymbolicExpr::multiply(rd_num_rat(coeff), pw);
2531 }
2532 result = SymbolicExpr::add(result, term);
2533 }
2534 }
2535
2536 // Partial-fraction terms.
2537 if (!numerators.empty()) {
2538 size_t idx = 0;
2539 for (size_t i = 0; i < factors.size(); ++i) {
2540 const auto& [fpoly, mult] = factors[i];
2541 for (int l = 1; l <= mult; ++l, ++idx) {
2542 if (idx >= numerators.size()) break;
2543 auto term = integrate_term(numerators[idx], fpoly, l, var);
2544 if (!term) {
2545 // Fallback: unevaluated integral over the original.
2546 return std::make_shared<SymbolicExpr>(
2547 std::make_shared<FunctionNode>(
2548 FunctionNode::FuncType::Calculus_Integral,
2549 std::vector<std::shared_ptr<SymbolicNode>>{
2550 expr.root,
2551 SymbolicExpr::variable(var)->root}));
2552 }
2553 result = SymbolicExpr::add(result, term);
2554 }
2555 }
2556 }
2557
2558 auto simplified = result->simplify();
2559 if (simplified) {
2560 // If the simplified form contains an unevaluated integral marker
2561 // (e.g. high-power irreducible quadratic), prefer keeping the
2562 // un-simplified result so downstream consumers can still read the
2563 // partial-fraction form.
2564 return simplified;
2565 }
2566 return result;
2567 } catch (...) {
2568 // Any unexpected runtime failure -> return unevaluated integral.
2569 return std::make_shared<SymbolicExpr>(
2570 std::make_shared<FunctionNode>(
2571 FunctionNode::FuncType::Calculus_Integral,
2572 std::vector<std::shared_ptr<SymbolicNode>>{
2573 expr.root,
2574 SymbolicExpr::variable(var)->root}));
2575 }
2576}

2578// ---------------------------------------------------------------
2579// SpecialFunctionStrategy
2580// ---------------------------------------------------------------
2581//
2582// Recognises a small, fixed set of integrand shapes whose antiderivatives are
2583// not elementary and instead must be expressed via the special functions erf,
2584// Ei, Si, Ci, Li. The inner argument of each pattern is required to be the
2585// integration variable itself (or a quadratic c*var^2 with rational c, for the
2586// erf branch). More general arguments are reduced first by SubstitutionStrategy
2587// or LinearSubstitutionStrategy upstream and so are not handled here.
2588//
2589// Patterns recognised (var = x):
2590// exp(-x^2) -> (sqrt(pi) / 2) * erf(x)
2591// exp(-a*x^2) -> (sqrt(pi) / (2*sqrt(a))) * erf(sqrt(a)*x)
2592// where a is a constant w.r.t. var (assumed positive).
2593// exp(x) / x -> Ei(x)
2594// sin(x) / x -> Si(x)
2595// cos(x) / x -> Ci(x)
2596// 1 / ln(x) -> Li(x)
2597//
2598// Builders use FunctionNode with the dedicated FuncType enum values added in
2599// task 1.1, so that print_visitor / differentiation_visitor render and

2711 const std::string& var,

2806// MultipleIntegralEngine
2807// ---------------------------------------------------------------
2808//
2809// Evaluates iterated (multiple) integrals by sequentially applying
2810// `Integrator::integrate` (for indefinite steps) or
2811// `Integrator::integrate_def` (for definite steps), going from the innermost
2812// step (index 0) to the outermost step (last index).
2813//
2814// If the integrand does not depend on a particular variable that has
2815// definite bounds [a, b], we short-circuit the integration step to a
2816// straightforward multiplication by (b - a) so we do not turn a constant
2817// expression into a fresh formal integral.
2818//
2819// Whenever a single step yields an expression that still contains an
2820// unevaluated `Calculus_Integral` node, we stop further iteration and
2821// return the partially-integrated result.
2822
2823bool MultipleIntegralEngine::validate(const std::vector<IntegrationStep>& steps) const {
2824 if (steps.empty() || steps.size() > 3) return false;
2825
2826 for (size_t i = 0; i < steps.size(); ++i) {
2827 const auto& s = steps[i];
2828 if (s.variable.empty()) return false;
2829
2830 // Bounds must be either both null (indefinite) or both set (definite).
2831 const bool has_lower = static_cast<bool>(s.lower);
2832 const bool has_upper = static_cast<bool>(s.upper);
2833 if (has_lower != has_upper) return false;
2834
2835 // Reject duplicate variables.
2836 for (size_t j = 0; j < i; ++j) {
2837 if (steps[j].variable == s.variable) return false;
2838 }
2839 }
2840 return true;
2841}
2842
2843std::shared_ptr<SymbolicExpr> MultipleIntegralEngine::evaluate(
2844 const SymbolicExpr& integrand,
2845 const std::vector<IntegrationStep>& steps,
2846 Integrator& integrator) {
2847
2848 if (!validate(steps)) return nullptr;
2849
2850 auto current = std::make_shared<SymbolicExpr>(integrand);
2851
2852 for (const auto& step : steps) {
2853 if (!current) return nullptr;
2854
2855 // If a previous step already produced an unevaluated integral node,
2856 // stop and propagate that partial result.
2857 if (has_integral_node_check(current->root)) {
2858 return current;
2859 }
2860
2861 const bool definite = static_cast<bool>(step.lower) && static_cast<bool>(step.upper);
2862
2863 if (!definite) {
2864 // Indefinite single-variable integration.
2865 SymbolicExpr res = integrator.integrate(*current, step.variable);
2866 current = std::make_shared<SymbolicExpr>(res);
2867 } else {
2868 // Definite single-variable integration. If the integrand is
2869 // independent of the variable, short-circuit to current*(upper-lower).
2870 if (!Integrator::depends_on(*current, step.variable)) {
2871 auto diff = sym_sub(*step.upper, *step.lower);
2872 auto product = SymbolicExpr::multiply(current, diff);
2873 auto simp = product->simplify();
2874 current = simp ? simp : product;
2875 } else {
2876 SymbolicExpr res = integrator.integrate_def(
2877 *current, step.variable, *step.lower, *step.upper);
2878 auto res_ptr = std::make_shared<SymbolicExpr>(res);
2879 auto simp = res_ptr->simplify();
2880 current = simp ? simp : res_ptr;
2881 }
2882 }
2883 }
2884
2885 return current;
2886}
