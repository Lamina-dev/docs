27/**
28 * @brief Scoped assumption context with push/pop management.
29 *
30 * Maintains a stack of scopes, each containing a PropertyStore and RelationStore.
31 * Queries use read-through semantics: searching from the top scope down to root,
32 * returning the first scope that has a declaration for the queried symbol.
33 * Child scope declarations shadow parent declarations without modifying them.
34 * Popping a scope restores all query results to their pre-push values.
35 */
36class LAMINA_API AssumptionContext {
37public:
38 AssumptionContext();
39
40 // --- Scope management ---
41
42 /// Push a new scope onto the stack.
43 void push();
44
45 /// Pop the current scope. Throws std::runtime_error if at root scope.
46 void pop();
47
48 /// Return the current nesting depth (number of scopes on the stack).
49 int depth() const;
50
51 // --- Direct access to current (top) scope stores ---
52
53 /// Access the current scope's PropertyStore (mutable, for declarations).
54 PropertyStore& current_properties();
55
56 /// Access the current scope's PropertyStore (const).
57 const PropertyStore& current_properties() const;
58
59 /// Access the current scope's RelationStore (mutable).
60 RelationStore& current_relations();
61
62 /// Access the current scope's RelationStore (const).
63 const RelationStore& current_relations() const;
64
65 // --- Read-through query methods ---
66 // These search from the top scope down to root, returning the first
67 // scope's value that has a declaration for the symbol. This provides
68 // shadowing semantics: child declarations override parent declarations.
69
70 /// Check if a symbol has a specific sign (read-through all scopes).
71 bool has_sign(const std::string& symbol, Sign sign) const;
72
73 /// Check if a symbol has at least the given domain specificity (read-through).
74 bool has_domain(const std::string& symbol, Domain domain) const;
75
76 /// Get the most specific domain for a symbol (read-through).
77 Domain get_domain(const std::string& symbol) const;
78
79 /// Get all signs for a symbol (read-through).
80 std::unordered_set<Sign, SignHash> get_signs(const std::string& symbol) const;
81
82 /// Get parity for a symbol (read-through).
83 Parity get_parity(const std::string& symbol) const;
84
85 /// Get boundedness for a symbol (read-through).
86 Boundedness get_boundedness(const std::string& symbol) const;
87
88 /// Get interval bounds for a symbol (read-through).
89 std::optional<Interval> get_bounds(const std::string& symbol) const;
90
91 // --- Convenience declaration API ---
92
93 /// Declare domain for a variable in the current scope.
94 /// Throws std::invalid_argument if variable name is empty.
95 void assume_domain(const std::string& variable, Domain domain);
96
97 /// Declare sign for a variable in the current scope.
98 /// Throws std::invalid_argument if variable name is empty.
99 void assume_sign(const std::string& variable, Sign sign);
100
101 /// Store a relational constraint. The expression's root must be a RelationalNode.
102 /// Throws std::invalid_argument if expression is null/empty or root is not RelationalNode.
103 void assume(const SymbolicExpr& relation);
104
105 // --- Convenience query API (delegates to QueryInterface) ---
106
107 /// Query whether the expression is positive (> 0).
108 /// Throws std::invalid_argument if expression has null root.
109 Tribool is_positive(const SymbolicExpr& expr) const;
110
111 /// Query whether the expression is negative (< 0).
112 /// Throws std::invalid_argument if expression has null root.
113 Tribool is_negative(const SymbolicExpr& expr) const;
114
115 /// Query whether the expression is non-negative (>= 0).
116 /// Throws std::invalid_argument if expression has null root.
117 Tribool is_nonnegative(const SymbolicExpr& expr) const;
118
119 /// Query whether the expression is real.
120 /// Throws std::invalid_argument if expression has null root.
121 Tribool is_real(const SymbolicExpr& expr) const;
122
123 /// Query whether the expression is an integer.
124 /// Throws std::invalid_argument if expression has null root.
125 Tribool is_integer(const SymbolicExpr& expr) const;
126
127 /// Query whether the expression is non-zero (!= 0).
128 /// Throws std::invalid_argument if expression has null root.
129 Tribool is_nonzero(const SymbolicExpr& expr) const;
130
131private:
132 struct Scope {
133 PropertyStore properties;
134 RelationStore relations;
135 };
136
137 std::vector<Scope> scope_stack_;
138};
