63ParametricSolver::solve_linear_parametric(
64 const std::vector<std::shared_ptr<SymbolicExpr>>& equations,

67{
68 size_t m = equations.size();
69 size_t n = unknowns.size();
70
71 if (m == 0 || n == 0) {
72
73 std::map<std::string, std::shared_ptr<SymbolicExpr>> solution;
74 for (const auto& var : unknowns) {
75 solution[var] = SymbolicExpr::variable(var);
76 }

82

84 for (size_t j = 0; j < n; ++j) {
85 A[i][j] = extract_coefficient(equations[i], unknowns[j]);
86 }
87

93 std::vector<size_t> pivot_rows;
94 size_t current_row = 0;
95
96 for (size_t col = 0; col < n && current_row < m; ++col) {
97
98 size_t pivot_row = current_row;
99 bool found_pivot = false;
100 while (pivot_row < m) {
101 auto simplified = A[pivot_row][col]->simplify();
102 A[pivot_row][col] = simplified;
103 if (!simplified->is_zero()) {
104 found_pivot = true;
105 break;
106 }
107 pivot_row++;
108 }
109
110 if (!found_pivot) {
111
112 continue;
113 }
114
115 if (pivot_row != current_row) {
116 std::swap(A[pivot_row], A[current_row]);
117 std::swap(b[pivot_row], b[current_row]);
118 }
119
120 auto pivot = A[current_row][col];
121
122 for (size_t r = 0; r < m; ++r) {
123 if (r == current_row) continue;
124 auto entry = A[r][col]->simplify();
125 if (entry->is_zero()) continue;
126
127 auto factor = SymbolicExpr::divide(entry, pivot)->simplify();
128
129 for (size_t k = col; k < n; ++k) {
130 auto term = SymbolicExpr::multiply(factor, A[current_row][k]);
131 A[r][k] = SymbolicExpr::add(A[r][k], SymbolicExpr::multiply(term, SymbolicExpr::number(-1)))->simplify();
132 }
133
134 auto b_term = SymbolicExpr::multiply(factor, b[current_row]);
135 b[r] = SymbolicExpr::add(b[r], SymbolicExpr::multiply(b_term, SymbolicExpr::number(-1)))->simplify();
136 }
137
138 pivot_cols.push_back(col);
139 pivot_rows.push_back(current_row);
140 current_row++;
141 }
142
143 for (size_t r = current_row; r < m; ++r) {
144 auto b_simplified = b[r]->simplify();
145 if (!b_simplified->is_zero()) {
146
147 return {};
148 }
149 }
150
151 std::map<std::string, std::shared_ptr<SymbolicExpr>> solution;
152
153 std::set<size_t> pivot_col_set(pivot_cols.begin(), pivot_cols.end());
154 for (size_t col = 0; col < n; ++col) {
155 if (pivot_col_set.find(col) == pivot_col_set.end()) {
156
157 solution[unknowns[col]] = SymbolicExpr::variable(unknowns[col]);
158 }
159 }
160
161 for (size_t k = 0; k < pivot_cols.size(); ++k) {
162 size_t col = pivot_cols[k];
163 size_t row = pivot_rows[k];
164
165 auto value = b[row];
166
167 for (size_t j = col + 1; j < n; ++j) {
168 if (pivot_col_set.find(j) != pivot_col_set.end()) continue;
169 auto coeff = A[row][j]->simplify();
170 if (!coeff->is_zero()) {
171
172 auto term = SymbolicExpr::multiply(coeff, SymbolicExpr::variable(unknowns[j]));
173 value = SymbolicExpr::add(value, SymbolicExpr::multiply(term, SymbolicExpr::number(-1)));
174 }
175 }
176
177 auto pivot_val = A[row][col];
178 value = SymbolicExpr::divide(value, pivot_val)->simplify();
179
180 solution[unknowns[col]] = value;
181 }
182
183 return { solution };
184}

267 auto solve_rec = [&](auto&& self, int var_pos,
268 const std::map<std::string, std::shared_ptr<SymbolicExpr>>& partial)
269 -> std::vector<std::map<std::string, std::shared_ptr<SymbolicExpr>>> {
270
271 std::vector<std::shared_ptr<SymbolicExpr>> reduced;
272 reduced.reserve(basis.size());
273
274 for (const auto& p : basis) {
275 auto r = substitute_all(p, partial);
276 if (!r) continue;
277 if (r->is_zero()) continue;
278
279 bool depends = false;
280 for (int i = 0; i <= var_pos && i < (int)unknowns.size(); ++i) {
281 if (contains(*r, unknowns[i])) {
282 depends = true;
283 break;
284 }
285 }
286
287 if (!depends) {
288
289 if (r->is_number() && !r->is_zero()) {
290 return {};
291 }
292
293 continue;
294 }
295
296 reduced.push_back(r);
297 }
298
299 if (var_pos < 0) {
300 return { partial };
301 }
302
303 const auto& curr_var = unknowns[var_pos];
304 bool curr_var_appears = false;
305 std::shared_ptr<SymbolicExpr> target = nullptr;
306 int best_deg = std::numeric_limits<int>::max();
307
308 for (const auto& r : reduced) {
309 if (!contains(*r, curr_var)) continue;
310 curr_var_appears = true;
311
312 bool has_other = false;
313 for (int i = 0; i < var_pos; ++i) {
314 if (contains(*r, unknowns[i])) {
315 has_other = true;
316 break;
317 }
318 }
319 if (has_other) continue;
320
321 auto poly = symbolic_to_poly<SymbolicPolyCoeff>(r, curr_var);
322 int deg = poly.degree();
323 if (deg >= 1 && deg < best_deg) {
324 best_deg = deg;
325 target = r;
326 }
327 }
328
329 if (!curr_var_appears) {
330 auto next_partial = partial;
331 next_partial[curr_var] = SymbolicExpr::variable(curr_var);
332 return self(self, var_pos - 1, next_partial);
333 }
334
335 if (!target) {
336 auto next_partial = partial;
337 next_partial[curr_var] = SymbolicExpr::variable(curr_var);
338 return self(self, var_pos - 1, next_partial);
339 }
340
341 auto roots = SymbolicExpr::solve(target, curr_var);
342 if (roots.empty()) {
343
344 auto next_partial = partial;
345 next_partial[curr_var] = SymbolicExpr::variable(curr_var);
346 return self(self, var_pos - 1, next_partial);
347 }
348
349 std::vector<std::map<std::string, std::shared_ptr<SymbolicExpr>>> results;
350 for (const auto& r : roots) {
351 auto next_partial = partial;
352 next_partial[curr_var] = r;
353 auto sub_res = self(self, var_pos - 1, next_partial);
354 results.insert(results.end(), sub_res.begin(), sub_res.end());
355 }
356 return results;
357 };
358
359 std::map<std::string, std::shared_ptr<SymbolicExpr>> empty;
360 return solve_rec(solve_rec, static_cast<int>(unknowns.size()) - 1, empty);
361}

388static std::shared_ptr<SymbolicExpr> compute_determinant(
389 const std::vector<std::vector<std::shared_ptr<SymbolicExpr>>>& matrix,
390 size_t n)
391{
392 if (n == 0) return SymbolicExpr::number(0);
393 if (n == 1) return matrix[0][0];
394
395 if (n == 2) {
396 auto ad = SymbolicExpr::multiply(matrix[0][0], matrix[1][1]);
397 auto bc = SymbolicExpr::multiply(matrix[0][1], matrix[1][0]);
398
399 auto neg_bc = SymbolicExpr::multiply(bc, SymbolicExpr::number(-1));
400 return SymbolicExpr::add(ad, neg_bc)->simplify();
401 }
402
403 if (n == 3) {
404
405 auto a = matrix[0][0], b = matrix[0][1], c = matrix[0][2];
406 auto d = matrix[1][0], e = matrix[1][1], f = matrix[1][2];
407 auto g = matrix[2][0], h = matrix[2][1], i = matrix[2][2];
408
409 auto ei = SymbolicExpr::multiply(e, i);
410 auto fh = SymbolicExpr::multiply(f, h);
411 auto ei_fh = SymbolicExpr::add(ei, SymbolicExpr::multiply(fh, SymbolicExpr::number(-1)));
412
413 auto di = SymbolicExpr::multiply(d, i);
414 auto fg = SymbolicExpr::multiply(f, g);
415 auto di_fg = SymbolicExpr::add(di, SymbolicExpr::multiply(fg, SymbolicExpr::number(-1)));
416
417 auto dh = SymbolicExpr::multiply(d, h);
418 auto eg = SymbolicExpr::multiply(e, g);
419 auto dh_eg = SymbolicExpr::add(dh, SymbolicExpr::multiply(eg, SymbolicExpr::number(-1)));
420
421 auto term1 = SymbolicExpr::multiply(a, ei_fh);
422 auto term2 = SymbolicExpr::multiply(b, di_fg);
423 auto term3 = SymbolicExpr::multiply(c, dh_eg);
424
425 auto neg_term2 = SymbolicExpr::multiply(term2, SymbolicExpr::number(-1));
426 auto sum12 = SymbolicExpr::add(term1, neg_term2);
427 return SymbolicExpr::add(sum12, term3)->simplify();
428 }
429
430 auto det = SymbolicExpr::number(0);
431 for (size_t j = 0; j < n; ++j) {
432
433 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> minor(n - 1, std::vector<std::shared_ptr<SymbolicExpr>>(n - 1));
434 for (size_t row = 1; row < n; ++row) {
435 size_t minor_col = 0;
436 for (size_t col = 0; col < n; ++col) {
437 if (col == j) continue;
438 minor[row - 1][minor_col] = matrix[row][col];
439 minor_col++;
440 }
441 }
442
443 auto cofactor = compute_determinant(minor, n - 1);
444 auto term = SymbolicExpr::multiply(matrix[0][j], cofactor);
445
446 if (j % 2 == 0) {
447 det = SymbolicExpr::add(det, term);
448 } else {
449 det = SymbolicExpr::add(det, SymbolicExpr::multiply(term, SymbolicExpr::number(-1)));
450 }
451 }
452 return det->simplify();
453}

468PiecewiseSolution ParametricSolver::solve_system_piecewise(
469 const std::vector<std::shared_ptr<SymbolicExpr>>& equations,
470 const std::vector<std::string>& unknowns,
471 const std::vector<std::string>& parameters)
472{
473 PiecewiseSolution result;
474 size_t m = equations.size();
475 size_t n = unknowns.size();
476
477 if (m == 0 || n == 0) {
478 return result;
479 }
480
481 std::vector<std::string> effective_params;
482 std::set<std::string> unknown_set(unknowns.begin(), unknowns.end());
483 for (const auto& p : parameters) {
484 if (unknown_set.find(p) == unknown_set.end()) {
485 effective_params.push_back(p);
486 }
487 }
488
489 if (m != n) {
490 auto solutions = solve_system(equations, unknowns, parameters);
491 if (!solutions.empty()) {
492 PiecewiseSolution::Case generic_case;
493 generic_case.condition = SymbolicExpr::number(1);
494 generic_case.solutions = solutions;
495 result.cases.push_back(generic_case);
496 }
497 return result;
498 }
499
500 if (!is_linear_in_unknowns(equations, unknowns)) {
501 auto solutions = solve_system(equations, unknowns, parameters);
502 if (!solutions.empty()) {
503 PiecewiseSolution::Case generic_case;
504 generic_case.condition = SymbolicExpr::number(1);
505 generic_case.solutions = solutions;
506 result.cases.push_back(generic_case);
507 }
508 return result;
509 }
510
511 std::vector<std::vector<std::shared_ptr<SymbolicExpr>>> A(n, std::vector<std::shared_ptr<SymbolicExpr>>(n));
512 for (size_t i = 0; i < n; ++i) {
513 for (size_t j = 0; j < n; ++j) {
514 A[i][j] = extract_coefficient(equations[i], unknowns[j]);
515 }
516 }
517
518 auto det = compute_determinant(A, n);
519
520 if (det->is_zero()) {
521
522 auto solutions = solve_system(equations, unknowns, parameters);
523 PiecewiseSolution::Case degenerate_case;
524
525 degenerate_case.condition = SymbolicExpr::number(0);
526 degenerate_case.solutions = solutions;
527 result.cases.push_back(degenerate_case);
528 } else if (!depends_on_parameters(det, effective_params)) {
529
530 auto solutions = solve_system(equations, unknowns, parameters);
531 PiecewiseSolution::Case generic_case;
532
533 generic_case.condition = det;
534 generic_case.solutions = solutions;
535 result.cases.push_back(generic_case);
536 } else {
537
538 auto generic_solutions = solve_system(equations, unknowns, parameters);
539 PiecewiseSolution::Case generic_case;
540 generic_case.condition = det;
541 generic_case.solutions = generic_solutions;
542 result.cases.push_back(generic_case);
543
544 PiecewiseSolution::Case degenerate_case;
545 degenerate_case.condition = SymbolicExpr::number(0);
546
547 degenerate_case.solutions = generic_solutions;
548 result.cases.push_back(degenerate_case);
549 }
550
551 return result;
552}
