3 * @brief Implementation of the AssumptionContext class.

5 * Provides scoped push/pop management with read-through query semantics.
6 * Each scope has its own PropertyStore and RelationStore. Queries search
7 * from the top scope down to root, with child declarations shadowing parent.

70bool AssumptionContext::has_sign(const std::string& symbol, Sign sign) const {
71 // Search from top scope down to root.
72 // Return the result from the first scope that has sign info for this symbol.
73 for (auto it = scope_stack_.rbegin(); it != scope_stack_.rend(); ++it) {
74 if (!it->properties.get_signs(symbol).empty()) {
75 return it->properties.has_sign(symbol, sign);
76 }
77 }
78 // No scope has sign info for this symbol
79 return false;
80}

82bool AssumptionContext::has_domain(const std::string& symbol, Domain domain) const {
83 // Search from top scope down to root.
84 // Return the result from the first scope that has domain info for this symbol.
85 for (auto it = scope_stack_.rbegin(); it != scope_stack_.rend(); ++it) {
86 if (it->properties.get_domain(symbol) != Domain::Complex) {
87 return it->properties.has_domain(symbol, domain);
88 }
89 }
90 // No scope has domain info — default is Complex.
91 // has_domain checks if the symbol has at least the given specificity.
92 return domain == Domain::Complex;
93}

163void AssumptionContext::assume(const SymbolicExpr& relation) {
164 if (!relation.root) {
165 throw std::invalid_argument("assume: expression must not be null/empty");
166 }
167 auto rel_node = std::dynamic_pointer_cast<RelationalNode>(relation.root);
168 if (!rel_node) {
169 throw std::invalid_argument("assume: expression root must be a RelationalNode");
170 }
171 // Extract lhs, rhs, and op from the RelationalNode and store in RelationStore
172 SymbolicExpr lhs(rel_node->left);
173 SymbolicExpr rhs(rel_node->right);
174 scope_stack_.back().relations.add_relation(lhs, rhs, rel_node->op,
175 scope_stack_.back().properties);
176}
