27// True iff `ep` is a finite-valued endpoint whose value is a concrete number
28// node (BigInt / Rational / lmmc_real_t). Endpoints with a symbolic value
29// (e.g. depending on parameters) cannot be compared numerically, so the
30// normalization / merging logic must not collapse them based on `to_numeric()`,
31// which silently returns 0.0 for unevaluable expressions.
32static bool endpoint_is_numeric(const Endpoint& ep) {
33 if (ep.is_neg_infinity || ep.is_pos_infinity) return true;
34 if (!ep.value || !ep.value->root) return false;
35 return ep.value->is_number();
36}

40 if (ep.is_pos_infinity) return std::numeric_limits<double>::infinity();
41 if (ep.value) return ep.value->to_numeric();
42 return 0.0;

75bool Interval::is_empty() const {
76

78 if (lower.is_pos_infinity) return true;

151static int compare_endpoints_lower(const Endpoint& a, const Endpoint& b) {
152
153 if (a.is_neg_infinity && b.is_neg_infinity) return 0;
154 if (a.is_neg_infinity) return -1;
155 if (b.is_neg_infinity) return 1;
156
157 if (a.is_pos_infinity && b.is_pos_infinity) return 0;
158 if (a.is_pos_infinity) return 1;
159 if (b.is_pos_infinity) return -1;
160
161 bool a_num = endpoint_is_numeric(a);
162 bool b_num = endpoint_is_numeric(b);
163 if (!a_num || !b_num) {
164 // 不要用 to_string() 给符号端点造伪数值序——后续 intersect()/normalize()
165 // 会把这里的返回值当成真实大小关系，进而把不可比的端点错误合并。
166 // 这里只在两侧"结构相等"时返回 0；否则用 SymbolicNode::compare 给出
167 // 一个确定但纯结构性的 tie-breaker，仅用于 std::sort 的严格弱序需求。
168 if (a_num) return -1;
169 if (b_num) return 1;
170 if (a.value && b.value && a.value->root && b.value->root) {
171 int c = a.value->root->compare(*b.value->root);
172 if (c != 0) return c;
173 } else if (a.value != b.value) {
174 return (a.value < b.value) ? -1 : 1;
175 }
176 if (a.is_open && !b.is_open) return 1;
177 if (!a.is_open && b.is_open) return -1;
178 return 0;
179 }
180
181 double va = endpoint_numeric_value(a);
182 double vb = endpoint_numeric_value(b);
183
184 if (va < vb) return -1;
185 if (va > vb) return 1;
186
187 if (a.is_open && !b.is_open) return 1;
188 if (!a.is_open && b.is_open) return -1;
189 return 0;
190}

229static bool can_merge(const Interval& a, const Interval& b) {
230
231 if (a.upper.is_pos_infinity) return true;
232 if (b.lower.is_neg_infinity) return true;
233
234 if (a.upper.is_neg_infinity || b.lower.is_pos_infinity) return false;
235
236 // If either side is symbolic we cannot prove the intervals are adjacent
237 // or overlapping numerically; refuse to merge.
238 if (!endpoint_is_numeric(a.upper) || !endpoint_is_numeric(b.lower)) {
239 return false;
240 }
241
242 double au = endpoint_numeric_value(a.upper);
243 double bl = endpoint_numeric_value(b.lower);
244
245 if (au > bl) return true;
246 if (au == bl) {
247
248 return !a.upper.is_open || !b.lower.is_open;
249 }
250 return false;
251}

258void IntervalUnion::normalize() {
259
260 intervals_.erase(
261 std::remove_if(intervals_.begin(), intervals_.end(),
262 [](const Interval& iv) { return iv.is_empty(); }),
263 intervals_.end());
264
265 if (intervals_.empty()) return;
266
267 std::sort(intervals_.begin(), intervals_.end(),
268 [](const Interval& a, const Interval& b) {
269 return compare_endpoints_lower(a.lower, b.lower) < 0;
270 });
271
272 std::vector<Interval> merged;
273 merged.push_back(intervals_[0]);
274
275 for (size_t i = 1; i < intervals_.size(); ++i) {
276 Interval& current = merged.back();
277 const Interval& next = intervals_[i];
278
279 if (can_merge(current, next)) {
280
281 current.upper = max_upper(current.upper, next.upper);
282 } else {
283 merged.push_back(next);
284 }
285 }
286
287 intervals_ = std::move(merged);
288}

290IntervalUnion IntervalUnion::intersect(const IntervalUnion& other) const {
291 if (intervals_.empty() || other.intervals_.empty()) {
292 return IntervalUnion::empty();
293 }
294
295 std::vector<Interval> result;
296 size_t i = 0, j = 0;
297
298 while (i < intervals_.size() && j < other.intervals_.size()) {
299 const Interval& a = intervals_[i];
300 const Interval& b = other.intervals_[j];
301
302 Endpoint lo = (compare_endpoints_lower(a.lower, b.lower) >= 0) ? a.lower : b.lower;
303
304 Endpoint hi = (compare_endpoints_upper(a.upper, b.upper) <= 0) ? a.upper : b.upper;
305
306 Interval candidate{lo, hi};
307 if (!candidate.is_empty()) {
308 result.push_back(candidate);
309 }
310
311 if (compare_endpoints_upper(a.upper, b.upper) < 0) {
312 ++i;
313 } else {
314 ++j;
315 }
316 }
317
318 IntervalUnion res;
319 res.intervals_ = std::move(result);
320 return res;
321}

370 gap_upper.is_neg_infinity = false;
371 gap_upper.is_pos_infinity = false;

374 if (!gap.is_empty()) {

462static std::shared_ptr<SymbolicExpr> parse_numeric_value(const std::string& str, size_t& pos) {
463 size_t start = pos;
464 bool has_sign = false;
465 bool has_digit = false;
466 bool has_dot = false;
467
468 if (pos < str.size() && (str[pos] == '-' || str[pos] == '+')) {
469 has_sign = true;
470 ++pos;
471 }
472
473 while (pos < str.size() && std::isdigit(static_cast<unsigned char>(str[pos]))) {
474 has_digit = true;
475 ++pos;
476 }
477
478 if (pos < str.size() && str[pos] == '.') {
479 has_dot = true;
480 ++pos;
481 while (pos < str.size() && std::isdigit(static_cast<unsigned char>(str[pos]))) {
482 has_digit = true;
483 ++pos;
484 }
485 }
486
487 if (!has_digit) {
488 pos = start;
489 return nullptr;
490 }
491
492 std::string num_str = str.substr(start, pos - start);
493 if (has_dot) {
494 double val = std::stod(num_str);
495 return SymbolicExpr::number(val);
496 } else {
497
498 long long val = std::stoll(num_str);
499 return SymbolicExpr::number(val);
500 }
501}

503static bool parse_endpoint_value(const std::string& str, size_t& pos, Endpoint& ep) {
504 pos = skip_ws(str, pos);
505
506 if (starts_with_at(str, pos, "-\xe2\x88\x9e")) {
507 ep.is_neg_infinity = true;
508 ep.is_pos_infinity = false;
509 ep.is_open = true;
510 ep.value = nullptr;
511 pos += 4;
512 return true;
513 }
514
515 if (starts_with_at(str, pos, "+\xe2\x88\x9e")) {
516 ep.is_pos_infinity = true;
517 ep.is_neg_infinity = false;
518 ep.is_open = true;
519 ep.value = nullptr;
520 pos += 4;
521 return true;
522 }
523
524 auto val = parse_numeric_value(str, pos);
525 if (!val) return false;
526
527 ep.value = val;
528 ep.is_neg_infinity = false;
529 ep.is_pos_infinity = false;
530
531 return true;
532}

534static bool parse_single_interval(const std::string& str, size_t& pos, Interval& iv) {
535 pos = skip_ws(str, pos);
536 if (pos >= str.size()) return false;
537
538 char lower_bracket = str[pos];
539 if (lower_bracket != '(' && lower_bracket != '[') return false;
540 bool lower_open = (lower_bracket == '(');
541 ++pos;
542
543 Endpoint lower;
544 if (!parse_endpoint_value(str, pos, lower)) return false;
545
546 if (!lower.is_neg_infinity && !lower.is_pos_infinity) {
547 lower.is_open = lower_open;
548 }
549
550 pos = skip_ws(str, pos);
551 if (pos >= str.size() || str[pos] != ',') return false;
552 ++pos;
553
554 Endpoint upper;
555 if (!parse_endpoint_value(str, pos, upper)) return false;
556
557 pos = skip_ws(str, pos);
558 if (pos >= str.size()) return false;
559 char upper_bracket = str[pos];
560 if (upper_bracket != ')' && upper_bracket != ']') return false;
561 bool upper_open = (upper_bracket == ')');
562 ++pos;
563
564 if (!upper.is_neg_infinity && !upper.is_pos_infinity) {
565 upper.is_open = upper_open;
566 }
567
568 iv.lower = lower;
569 iv.upper = upper;
570 return true;
571}

573std::optional<IntervalUnion> IntervalUnion::parse(const std::string& str) {
574 if (str.empty()) return std::nullopt;
575
576 if (str == "\xe2\x88\x85") {
577 return IntervalUnion::empty();
578 }
579
580 std::vector<Interval> intervals;
581 size_t pos = 0;
582
583 Interval iv;
584 if (!parse_single_interval(str, pos, iv)) return std::nullopt;
585 intervals.push_back(iv);
586
587 while (pos < str.size()) {
588 pos = skip_ws(str, pos);
589 if (pos >= str.size()) break;
590
591 if (!starts_with_at(str, pos, "\xe2\x88\xaa")) {
592 return std::nullopt;
593 }
594 pos += 3;
595
596 pos = skip_ws(str, pos);
597
598 Interval next_iv;
599 if (!parse_single_interval(str, pos, next_iv)) return std::nullopt;
600 intervals.push_back(next_iv);
601 }
602
603 return IntervalUnion(std::move(intervals));

606std::shared_ptr<SymbolicExpr> IntervalUnion::to_expr(const std::string& var) const {
607
608 if (intervals_.empty()) {
609 return nullptr;
610 }
611
612 auto var_node = SymbolicExpr::variable(var)->root;
613
614 auto interval_to_expr = [&](const Interval& iv) -> std::shared_ptr<SymbolicNode> {
615 std::shared_ptr<SymbolicNode> lower_cond = nullptr;
616 std::shared_ptr<SymbolicNode> upper_cond = nullptr;
617
618 if (!iv.lower.is_neg_infinity) {
619 auto bound = iv.lower.value ? iv.lower.value->root : std::make_shared<NumberNode>(0.0);
620 RelationalNode::Op op = iv.lower.is_open ? RelationalNode::Op::GT : RelationalNode::Op::GEQ;
621 lower_cond = std::make_shared<RelationalNode>(var_node, bound, op);
622 }
623
624 if (!iv.upper.is_pos_infinity) {
625 auto bound = iv.upper.value ? iv.upper.value->root : std::make_shared<NumberNode>(0.0);
626 RelationalNode::Op op = iv.upper.is_open ? RelationalNode::Op::LT : RelationalNode::Op::LEQ;
627 upper_cond = std::make_shared<RelationalNode>(var_node, bound, op);
628 }
629
630 if (lower_cond && upper_cond) {
631 return std::make_shared<LogicalNode>(lower_cond, upper_cond, LogicalNode::Op::And);
632 } else if (lower_cond) {
633 return lower_cond;
634 } else if (upper_cond) {
635 return upper_cond;
636 } else {
637
638 return nullptr;
639 }
640 };
641
642 if (intervals_.size() == 1 && intervals_[0].is_entire_line()) {
643
644 return nullptr;
645 }
646
647 if (intervals_.size() == 1) {
648 auto node = interval_to_expr(intervals_[0]);
649 if (!node) return nullptr;
650 return std::make_shared<SymbolicExpr>(node);
651 }
652
653 auto result = interval_to_expr(intervals_[0]);
654 for (size_t i = 1; i < intervals_.size(); ++i) {
655 auto next = interval_to_expr(intervals_[i]);
656 if (result && next) {
657 result = std::make_shared<LogicalNode>(result, next, LogicalNode::Op::Or);
658 } else if (next) {
659 result = next;
660 }
661
662 }
663
664 if (!result) return nullptr;
665 return std::make_shared<SymbolicExpr>(result);
666}
