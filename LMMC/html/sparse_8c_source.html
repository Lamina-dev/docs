11static int lmmc_mul_overflow_size(size_t a, size_t b, size_t* out) {
12 if (a == 0 || b == 0) {
13 *out = 0;
14 return 0;
15 }
16 if (a > ((size_t)-1) / b) {
17 return 1;
18 }
19 *out = a * b;
20 return 0;
21}

23static lmmc_status_t lmmc_sparse_validate(const lmmc_sparse_mat_t* sparse) {
24 size_t i = 0;
25 size_t outer_size = 0;
26 size_t inner_limit = 0;
27
28 if (sparse == NULL || sparse->row_ptr == NULL) {
29 return LMMC_STATUS_INVALID_ARGUMENT;
30 }
31 if (sparse->rows == 0 || sparse->cols == 0) {
32 return LMMC_STATUS_INVALID_ARGUMENT;
33 }
34 if (sparse->nnz > 0 && (sparse->col_idx == NULL || sparse->values == NULL)) {
35 return LMMC_STATUS_INVALID_ARGUMENT;
36 }
37
38 if (sparse->format == LMMC_SPARSE_CSR) {
39 outer_size = sparse->rows;
40 inner_limit = sparse->cols;
41 } else {
42 outer_size = sparse->cols;
43 inner_limit = sparse->rows;
44 }
45
46 if (sparse->row_ptr[0] != 0 || sparse->row_ptr[outer_size] != sparse->nnz) {
47 return LMMC_STATUS_INVALID_ARGUMENT;
48 }
49
50 for (i = 0; i < outer_size; ++i) {
51 if (sparse->row_ptr[i] > sparse->row_ptr[i + 1]) {
52 return LMMC_STATUS_INVALID_ARGUMENT;
53 }
54 }
55
56 for (i = 0; i < sparse->nnz; ++i) {
57 if (sparse->col_idx[i] >= inner_limit) {
58 return LMMC_STATUS_INVALID_ARGUMENT;
59 }
60 }
61 return LMMC_STATUS_OK;
62}

64static lmmc_status_t lmmc_sparse_create(size_t rows, size_t cols, size_t nnz, lmmc_sparse_format_t format, lmmc_sparse_mat_t* out_sparse) {
65 size_t outer_size = (format == LMMC_SPARSE_CSR) ? rows : cols;
66 size_t outer_ptr_bytes = 0;
67 size_t idx_bytes = 0;
68 size_t val_bytes = 0;
69 size_t* outer_ptr = NULL;
70 size_t* inner_idx = NULL;
71 lmmc_real_t* values = NULL;
72
73 if (out_sparse == NULL || rows == 0 || cols == 0) {
74 return LMMC_STATUS_INVALID_ARGUMENT;
75 }
76
77 if (lmmc_mul_overflow_size(outer_size + 1, sizeof(size_t), &outer_ptr_bytes) ||
78 lmmc_mul_overflow_size(nnz, sizeof(size_t), &idx_bytes) ||
79 lmmc_mul_overflow_size(nnz, sizeof(lmmc_real_t), &val_bytes)) {
80 return LMMC_STATUS_INVALID_ARGUMENT;
81 }
82
83 outer_ptr = (size_t*)lmmc_alloc(outer_ptr_bytes);
84 if (outer_ptr == NULL) {
85 return LMMC_STATUS_ALLOCATION_FAILED;
86 }
87 memset(outer_ptr, 0, outer_ptr_bytes);
88
89 if (nnz > 0) {
90 inner_idx = (size_t*)lmmc_alloc(idx_bytes);
91 values = (lmmc_real_t*)lmmc_alloc(val_bytes);
92 if (inner_idx == NULL || values == NULL) {
93 if (inner_idx != NULL) lmmc_free(inner_idx);
94 if (values != NULL) lmmc_free(values);
95 lmmc_free(outer_ptr);
96 return LMMC_STATUS_ALLOCATION_FAILED;
97 }
98 memset(inner_idx, 0, idx_bytes);
99 memset(values, 0, val_bytes);
100 for (size_t i = 0; i < nnz; ++i) {
101 LMMC_REAL_INIT(&values[i]);
102 }
103 }
104
105 out_sparse->rows = rows;
106 out_sparse->cols = cols;
107 out_sparse->nnz = nnz;
108 out_sparse->row_ptr = outer_ptr;
109 out_sparse->col_idx = inner_idx;
110 out_sparse->values = values;
111 out_sparse->format = format;
112 out_sparse->owns_data = 1;
113 return LMMC_STATUS_OK;
114}

215lmmc_status_t lmmc_sparse_from_dense(const lmmc_mat_t* dense, lmmc_real_t eps, lmmc_sparse_mat_t* out_sparse) {
216 size_t i = 0, j = 0, nz = 0;
217 lmmc_status_t st = LMMC_STATUS_OK;
218 if (dense == NULL || out_sparse == NULL || dense->data == NULL) {
219 return LMMC_STATUS_INVALID_ARGUMENT;
220 }
221
222 lmmc_real_t zero; LMMC_REAL_INIT(&zero);
223 lmmc_real_t abs_v; LMMC_REAL_INIT(&abs_v);
224 LMMC_REAL_SET_D(&zero, 0.0);
225
226 if (LMMC_REAL_CMP(&eps, &zero) < 0) {
227 LMMC_REAL_CLEAR(&zero);
228 LMMC_REAL_CLEAR(&abs_v);
229 return LMMC_STATUS_INVALID_ARGUMENT;
230 }
231
232 for (i = 0; i < dense->rows; ++i) {
233 for (j = 0; j < dense->cols; ++j) {
234 LMMC_REAL_ABS(&abs_v, &dense->data[i * dense->stride + j]);
235 if (LMMC_REAL_CMP(&abs_v, &eps) > 0) ++nz;
236 }
237 }
238
239 st = lmmc_sparse_create_csr(dense->rows, dense->cols, nz, out_sparse);
240 if (st != LMMC_STATUS_OK) {
241 LMMC_REAL_CLEAR(&zero);
242 LMMC_REAL_CLEAR(&abs_v);
243 return st;
244 }
245
246 nz = 0;
247 out_sparse->row_ptr[0] = 0;
248 for (i = 0; i < dense->rows; ++i) {
249 for (j = 0; j < dense->cols; ++j) {
250 lmmc_real_t* v_ptr = &dense->data[i * dense->stride + j];
251 LMMC_REAL_ABS(&abs_v, v_ptr);
252 if (LMMC_REAL_CMP(&abs_v, &eps) > 0) {
253 out_sparse->col_idx[nz] = j;
254 LMMC_REAL_SET(&out_sparse->values[nz], v_ptr);
255 ++nz;
256 }
257 }
258 out_sparse->row_ptr[i + 1] = nz;
259 }
260
261 LMMC_REAL_CLEAR(&zero);
262 LMMC_REAL_CLEAR(&abs_v);
263 return LMMC_STATUS_OK;
264}

303lmmc_status_t lmmc_sparse_mat_vec_mul(const lmmc_sparse_mat_t* sparse, const lmmc_vec_t* x, lmmc_vec_t* y) {

305 lmmc_status_t st = lmmc_sparse_validate(sparse);

307 return LMMC_STATUS_INVALID_ARGUMENT;

319 lmmc_real_t sum; LMMC_REAL_INIT(&sum);

329 size_t end = row_ptr[i + 1];

333 LMMC_REAL_MUL(&tmp_mul, &vals[p], &x_data[col_idx[p]]);

343 LMMC_REAL_SET(&sum, &tmp_sum);

352 LMMC_REAL_SET(&sum, &tmp_sum);
353 }
354 LMMC_REAL_SET(&y_data[i], &sum);
355 }
356 } else {
357 st = lmmc_vec_fill(y, zero);

359 LMMC_REAL_CLEAR(&sum);

361 LMMC_REAL_CLEAR(&tmp_sum);
362 LMMC_REAL_CLEAR(&zero);

364 }
365 for (i = 0; i < sparse->cols; ++i) {
366 size_t start = row_ptr[i];
367 size_t end = row_ptr[i + 1];
368 for (p = start; p < end; ++p) {
369 LMMC_REAL_MUL(&tmp_mul, &vals[p], &x_data[i]);
370 LMMC_REAL_ADD(&tmp_sum, &y_data[col_idx[p]], &tmp_mul);
371 LMMC_REAL_SET(&y_data[col_idx[p]], &tmp_sum);
372 }
373 }
374 }
375
376 LMMC_REAL_CLEAR(&sum);
377 LMMC_REAL_CLEAR(&tmp_mul);
378 LMMC_REAL_CLEAR(&tmp_sum);
379 LMMC_REAL_CLEAR(&zero);

384 size_t i = 0, p = 0;
385 size_t* next = NULL;
386 lmmc_status_t st = lmmc_sparse_validate(sparse);
387
388 if (st != LMMC_STATUS_OK || out_transposed == NULL) return LMMC_STATUS_INVALID_ARGUMENT;
389
390 st = lmmc_sparse_create(sparse->cols, sparse->rows, sparse->nnz, sparse->format, out_transposed);
391 if (st != LMMC_STATUS_OK) return st;
392
393 size_t outer_size_src = (sparse->format == LMMC_SPARSE_CSR) ? sparse->rows : sparse->cols;
394 size_t inner_size_src = (sparse->format == LMMC_SPARSE_CSR) ? sparse->cols : sparse->rows;
395
396 for (p = 0; p < sparse->nnz; ++p) {
397 ++out_transposed->row_ptr[sparse->col_idx[p] + 1];
398 }
399
400 for (i = 0; i < inner_size_src; ++i) {
401 out_transposed->row_ptr[i + 1] += out_transposed->row_ptr[i];
402 }
403
404 if (out_transposed->nnz == 0) return LMMC_STATUS_OK;
405
406 next = (size_t*)lmmc_alloc(inner_size_src * sizeof(size_t));
407 if (next == NULL) {
408 lmmc_sparse_destroy(out_transposed);
409 return LMMC_STATUS_ALLOCATION_FAILED;
410 }
411 memcpy(next, out_transposed->row_ptr, inner_size_src * sizeof(size_t));
412
413 for (i = 0; i < outer_size_src; ++i) {
414 for (p = sparse->row_ptr[i]; p < sparse->row_ptr[i + 1]; ++p) {
415 size_t inner = sparse->col_idx[p];
416 size_t dst = next[inner]++;
417 out_transposed->col_idx[dst] = i;
418 LMMC_REAL_SET(&out_transposed->values[dst], &sparse->values[p]);
419 }
420 }
421
422 lmmc_free(next);
423 return LMMC_STATUS_OK;
424}

545 size_t *c_row_ptr = NULL;
546 size_t *c_col_idx = NULL;
547 lmmc_real_t *c_values = NULL;
548 size_t nnz_est = 0;
549 size_t i, j, k, p1, p2;
550
551 if (a == NULL || b == NULL || c == NULL) return LMMC_STATUS_INVALID_ARGUMENT;
552 if (a->cols != b->rows) return LMMC_STATUS_DIMENSION_MISMATCH;
553
554 if (a->format == LMMC_SPARSE_CSC) {
555 st = lmmc_sparse_to_csr(a, &a_csr);
556 if (st != LMMC_STATUS_OK) return st;
557 pa = &a_csr;
558 }
559 if (b->format == LMMC_SPARSE_CSC) {
560 st = lmmc_sparse_to_csr(b, &b_csr);
561 if (st != LMMC_STATUS_OK) {
562 lmmc_sparse_destroy(&a_csr);
563 return st;
564 }
565 pb = &b_csr;
566 }
567
568 marker = (size_t*)lmmc_alloc(pb->cols * sizeof(size_t));
569 c_row_ptr = (size_t*)lmmc_alloc((pa->rows + 1) * sizeof(size_t));
570 if (marker == NULL || c_row_ptr == NULL) {
571 st = LMMC_STATUS_ALLOCATION_FAILED;
572 goto cleanup;
573 }
574 memset(marker, 0xFF, pb->cols * sizeof(size_t));
575 memset(c_row_ptr, 0, (pa->rows + 1) * sizeof(size_t));
576
577 nnz_est = 0;
578 for (i = 0; i < pa->rows; ++i) {
579 for (p1 = pa->row_ptr[i]; p1 < pa->row_ptr[i + 1]; ++p1) {
580 k = pa->col_idx[p1];
581 for (p2 = pb->row_ptr[k]; p2 < pb->row_ptr[k + 1]; ++p2) {
582 j = pb->col_idx[p2];
583 if (marker[j] != i) {
584 marker[j] = i;
585 nnz_est++;
586 }
587 }
588 }
589 c_row_ptr[i + 1] = nnz_est;
590 }
591
592 c_col_idx = (size_t*)lmmc_alloc(nnz_est * sizeof(size_t));
593 c_values = (lmmc_real_t*)lmmc_alloc(nnz_est * sizeof(lmmc_real_t));
594 accumulator = (lmmc_real_t*)lmmc_alloc(pb->cols * sizeof(lmmc_real_t));
595 if ((nnz_est > 0 && (c_col_idx == NULL || c_values == NULL)) || accumulator == NULL) {
596 st = LMMC_STATUS_ALLOCATION_FAILED;
597 goto cleanup;
598 }
599 memset(marker, 0xFF, pb->cols * sizeof(size_t));
600 for (size_t act_i = 0; act_i < pb->cols; act_i++) {
601 LMMC_REAL_INIT(&accumulator[act_i]);
602 LMMC_REAL_SET_D(&accumulator[act_i], 0.0);
603 }
604 for (size_t act_i = 0; act_i < nnz_est; act_i++) {
605 LMMC_REAL_INIT(&c_values[act_i]);
606 }
607
608 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
609 lmmc_real_t tmp_sum; LMMC_REAL_INIT(&tmp_sum);
610
611 size_t current_nnz = 0;
612 for (i = 0; i < pa->rows; ++i) {
613 size_t row_start = current_nnz;
614 for (p1 = pa->row_ptr[i]; p1 < pa->row_ptr[i + 1]; ++p1) {
615 k = pa->col_idx[p1];
616 for (p2 = pb->row_ptr[k]; p2 < pb->row_ptr[k + 1]; ++p2) {
617 j = pb->col_idx[p2];
618 if (marker[j] != i) {
619 marker[j] = i;
620 c_col_idx[current_nnz++] = j;
621 }
622 LMMC_REAL_MUL(&tmp_mul, &pa->values[p1], &pb->values[p2]);
623 LMMC_REAL_ADD(&tmp_sum, &accumulator[j], &tmp_mul);
624 LMMC_REAL_SET(&accumulator[j], &tmp_sum);
625 }
626 }
627 for (p1 = row_start; p1 < current_nnz; ++p1) {
628 j = c_col_idx[p1];
629 LMMC_REAL_SET(&c_values[p1], &accumulator[j]);
630 LMMC_REAL_SET_D(&accumulator[j], 0.0);
631 }
632 }
633
634 LMMC_REAL_CLEAR(&tmp_mul);
635 LMMC_REAL_CLEAR(&tmp_sum);
636
637 c->rows = pa->rows;
638 c->cols = pb->cols;
639 c->nnz = nnz_est;
640 c->row_ptr = c_row_ptr;
641 c->col_idx = c_col_idx;
642 c->values = c_values;
643 c->format = LMMC_SPARSE_CSR;
644 c->owns_data = 1;
645
646 c_row_ptr = NULL;
647 c_col_idx = NULL;
648 c_values = NULL;
649
650cleanup:
651 if (marker) lmmc_free(marker);
652 if (accumulator) {
653 for (size_t act_i = 0; act_i < pb->cols; act_i++) {
654 LMMC_REAL_CLEAR(&accumulator[act_i]);
655 }
656 lmmc_free(accumulator);
657 }
658 if (c_row_ptr) lmmc_free(c_row_ptr);
659 if (c_col_idx) lmmc_free(c_col_idx);
660 if (c_values) {
661 for (size_t act_i = 0; act_i < nnz_est; act_i++) {
662 LMMC_REAL_CLEAR(&c_values[act_i]);
663 }
664 lmmc_free(c_values);
665 }
666 lmmc_sparse_destroy(&a_csr);
667 lmmc_sparse_destroy(&b_csr);
668 return st;
669}

1065lmmc_status_t lmmc_sparse_coo_to_csr(
1066 const lmmc_sparse_coo_t* coo,
1067 lmmc_sparse_mat_t* out_csr
1068) {
1069 lmmc_coo_sort_entry_t* entries = NULL;
1070 size_t unique_nnz = 0;
1071 size_t i;
1072 lmmc_status_t st;
1073
1074 if (coo == NULL || out_csr == NULL) {
1075 return LMMC_STATUS_INVALID_ARGUMENT;
1076 }
1077
1078
1079 if (coo->nnz == 0) {
1080 return lmmc_sparse_create_csr(coo->rows, coo->cols, 0, out_csr);
1081 }
1082
1083
1084 entries = (lmmc_coo_sort_entry_t*)lmmc_alloc(coo->nnz * sizeof(lmmc_coo_sort_entry_t));
1085 if (entries == NULL) {
1086 return LMMC_STATUS_ALLOCATION_FAILED;
1087 }
1088
1089 for (i = 0; i < coo->nnz; ++i) {
1090 entries[i].row = coo->row_idx[i];
1091 entries[i].col = coo->col_idx[i];
1092 entries[i].orig_idx = i;
1093 }
1094
1095
1096 qsort(entries, coo->nnz, sizeof(lmmc_coo_sort_entry_t), lmmc_coo_cmp_row_col);
1097
1098
1099 unique_nnz = 1;
1100 for (i = 1; i < coo->nnz; ++i) {
1101 if (entries[i].row != entries[i - 1].row || entries[i].col != entries[i - 1].col) {
1102 unique_nnz++;
1103 }
1104 }

1309lmmc_status_t lmmc_sparse_add(
1310 lmmc_real_t alpha, const lmmc_sparse_mat_t* a,
1311 lmmc_real_t beta, const lmmc_sparse_mat_t* b,
1312 lmmc_sparse_mat_t* out_c
1313) {
1314 lmmc_sparse_mat_t a_csr = {0};
1315 lmmc_sparse_mat_t b_csr = {0};
1316 const lmmc_sparse_mat_t* pa = a;
1317 const lmmc_sparse_mat_t* pb = b;
1318 lmmc_status_t st = LMMC_STATUS_OK;
1319 size_t i;
1320 size_t nnz_c = 0;
1321 size_t* c_row_ptr = NULL;
1322 size_t* c_col_idx = NULL;
1323 lmmc_real_t* c_values = NULL;
1324
1325 if (a == NULL || b == NULL || out_c == NULL) {
1326 return LMMC_STATUS_INVALID_ARGUMENT;
1327 }
1328 if (a->rows != b->rows || a->cols != b->cols) {
1329 return LMMC_STATUS_DIMENSION_MISMATCH;
1330 }
1331
1332 st = lmmc_sparse_validate(a);
1333 if (st != LMMC_STATUS_OK) return LMMC_STATUS_INVALID_ARGUMENT;
1334 st = lmmc_sparse_validate(b);
1335 if (st != LMMC_STATUS_OK) return LMMC_STATUS_INVALID_ARGUMENT;
1336
1337
1338 if (a->format == LMMC_SPARSE_CSC) {
1339 st = lmmc_sparse_to_csr(a, &a_csr);
1340 if (st != LMMC_STATUS_OK) return st;
1341 pa = &a_csr;
1342 }
1343 if (b->format == LMMC_SPARSE_CSC) {
1344 st = lmmc_sparse_to_csr(b, &b_csr);
1345 if (st != LMMC_STATUS_OK) {
1346 lmmc_sparse_destroy(&a_csr);
1347 return st;
1348 }
1349 pb = &b_csr;
1350 }
1351
1352
1353 c_row_ptr = (size_t*)lmmc_alloc((pa->rows + 1) * sizeof(size_t));
1354 if (c_row_ptr == NULL) {
1355 st = LMMC_STATUS_ALLOCATION_FAILED;
1356 goto cleanup;
1357 }
1358 c_row_ptr[0] = 0;
1359
1360 for (i = 0; i < pa->rows; ++i) {
1361 size_t a_start = pa->row_ptr[i];
1362 size_t a_end = pa->row_ptr[i + 1];
1363 size_t b_start = pb->row_ptr[i];

1366 size_t bi = b_start;
1367 size_t row_nnz = 0;
1368
1369 while (ai < a_end && bi < b_end) {
1370 if (pa->col_idx[ai] < pb->col_idx[bi]) {
1371 ai++;
1372 row_nnz++;
1373 } else if (pa->col_idx[ai] > pb->col_idx[bi]) {
1374 bi++;
1375 row_nnz++;
1376 } else {
1377
1378 ai++;
1379 bi++;
1380 row_nnz++;
1381 }
1382 }
1383 row_nnz += (a_end - ai) + (b_end - bi);
1384 nnz_c += row_nnz;
1385 c_row_ptr[i + 1] = nnz_c;
1386 }
1387
1388
1389 if (nnz_c > 0) {
1390 c_col_idx = (size_t*)lmmc_alloc(nnz_c * sizeof(size_t));
1391 c_values = (lmmc_real_t*)lmmc_alloc(nnz_c * sizeof(lmmc_real_t));
1392 if (c_col_idx == NULL || c_values == NULL) {
1393 st = LMMC_STATUS_ALLOCATION_FAILED;
1394 goto cleanup;
1395 }
1396 for (size_t k = 0; k < nnz_c; ++k) {
1397 LMMC_REAL_INIT(&c_values[k]);
1398 }
1399 }
1400
1401
1402 {
1403 lmmc_real_t tmp_a; LMMC_REAL_INIT(&tmp_a);
1404 lmmc_real_t tmp_b; LMMC_REAL_INIT(&tmp_b);
1405 lmmc_real_t tmp_sum; LMMC_REAL_INIT(&tmp_sum);
1406 size_t c_idx = 0;
1407
1408 for (i = 0; i < pa->rows; ++i) {
1409 size_t a_start = pa->row_ptr[i];
1410 size_t a_end = pa->row_ptr[i + 1];
1411 size_t b_start = pb->row_ptr[i];
1412 size_t b_end = pb->row_ptr[i + 1];
1413 size_t ai = a_start;
1414 size_t bi = b_start;
1415
1416 while (ai < a_end && bi < b_end) {
1417 if (pa->col_idx[ai] < pb->col_idx[bi]) {
1418
1419 c_col_idx[c_idx] = pa->col_idx[ai];
1420 LMMC_REAL_MUL(&c_values[c_idx], &alpha, &pa->values[ai]);
1421 c_idx++;
1422 ai++;
1423 } else if (pa->col_idx[ai] > pb->col_idx[bi]) {
1424
1425 c_col_idx[c_idx] = pb->col_idx[bi];
1426 LMMC_REAL_MUL(&c_values[c_idx], &beta, &pb->values[bi]);
1427 c_idx++;
1428 bi++;
1429 } else {
1430
1431 c_col_idx[c_idx] = pa->col_idx[ai];
1432 LMMC_REAL_MUL(&tmp_a, &alpha, &pa->values[ai]);
1433 LMMC_REAL_MUL(&tmp_b, &beta, &pb->values[bi]);
1434 LMMC_REAL_ADD(&c_values[c_idx], &tmp_a, &tmp_b);
1435 c_idx++;
1436 ai++;
1437 bi++;
1438 }
1439 }
1440
1441 while (ai < a_end) {
1442 c_col_idx[c_idx] = pa->col_idx[ai];
1443 LMMC_REAL_MUL(&c_values[c_idx], &alpha, &pa->values[ai]);
1444 c_idx++;
1445 ai++;
1446 }
1447
1448 while (bi < b_end) {
1449 c_col_idx[c_idx] = pb->col_idx[bi];
1450 LMMC_REAL_MUL(&c_values[c_idx], &beta, &pb->values[bi]);
1451 c_idx++;
1452 bi++;
1453 }
1454 }
1455
1456 LMMC_REAL_CLEAR(&tmp_a);
1457 LMMC_REAL_CLEAR(&tmp_b);
1458 LMMC_REAL_CLEAR(&tmp_sum);
1459 }
1460
1461
1462 out_c->rows = pa->rows;
1463 out_c->cols = pa->cols;
1464 out_c->nnz = nnz_c;
1465 out_c->row_ptr = c_row_ptr;
1466 out_c->col_idx = c_col_idx;
1467 out_c->values = c_values;
1468 out_c->format = LMMC_SPARSE_CSR;
1469 out_c->owns_data = 1;
1470
1471
1472 c_row_ptr = NULL;
1473 c_col_idx = NULL;
1474 c_values = NULL;
1475 st = LMMC_STATUS_OK;
1476
1477cleanup:
1478 if (c_row_ptr) lmmc_free(c_row_ptr);
1479 if (c_col_idx) lmmc_free(c_col_idx);
1480 if (c_values) {
1481 for (size_t k = 0; k < nnz_c; ++k) {
1482 LMMC_REAL_CLEAR(&c_values[k]);
1483 }
1484 lmmc_free(c_values);
1485 }
1486 lmmc_sparse_destroy(&a_csr);
1487 lmmc_sparse_destroy(&b_csr);
1488 return st;
1489}

1699 return LMMC_STATUS_INVALID_ARGUMENT;
1700 }

1702 return LMMC_STATUS_INVALID_ARGUMENT;

1746lmmc_status_t lmmc_sparse_dense_to_bsr(const lmmc_mat_t* dense, size_t block_size,
1747 lmmc_real_t eps, lmmc_sparse_bsr_t* out) {
1748 size_t block_rows, block_cols;
1749 size_t bi, bj, li, lj;
1750 size_t nnz_blocks = 0;
1751 size_t nz_idx = 0;
1752 lmmc_status_t st;
1753 lmmc_real_t abs_v;
1754 lmmc_real_t zero;
1755
1756 if (dense == NULL || out == NULL || dense->data == NULL) {
1757 return LMMC_STATUS_INVALID_ARGUMENT;
1758 }
1759 if (block_size == 0) {
1760 return LMMC_STATUS_INVALID_ARGUMENT;
1761 }
1762 if (dense->rows % block_size != 0 || dense->cols % block_size != 0) {
1763 return LMMC_STATUS_INVALID_ARGUMENT;
1764 }
1765
1766 LMMC_REAL_INIT(&abs_v);
1767 LMMC_REAL_INIT(&zero);
1768 LMMC_REAL_SET_D(&zero, 0.0);
1769
1770 if (LMMC_REAL_CMP(&eps, &zero) < 0) {
1771 LMMC_REAL_CLEAR(&abs_v);
1772 LMMC_REAL_CLEAR(&zero);
1773 return LMMC_STATUS_INVALID_ARGUMENT;
1774 }
1775
1776 block_rows = dense->rows / block_size;
1777 block_cols = dense->cols / block_size;
1778
1779 /* 第一遍：计算非零块数量 */
1780 for (bi = 0; bi < block_rows; ++bi) {
1781 for (bj = 0; bj < block_cols; ++bj) {
1782 int block_nonzero = 0;
1783 for (li = 0; li < block_size && !block_nonzero; ++li) {
1784 for (lj = 0; lj < block_size && !block_nonzero; ++lj) {
1785 size_t row = bi * block_size + li;
1786 size_t col = bj * block_size + lj;
1787 LMMC_REAL_ABS(&abs_v, &dense->data[row * dense->stride + col]);
1788 if (LMMC_REAL_CMP(&abs_v, &eps) > 0) {
1789 block_nonzero = 1;
1790 }
1791 }
1792 }
1793 if (block_nonzero) {
1794 ++nnz_blocks;
1795 }
1796 }
1797 }
1798
1799 /* 分配 BSR 结构 */
1800 st = lmmc_sparse_bsr_create(block_rows, block_cols, block_size, nnz_blocks, out);
1801 if (st != LMMC_STATUS_OK) {
1802 LMMC_REAL_CLEAR(&abs_v);
1803 LMMC_REAL_CLEAR(&zero);
1804 return st;
1805 }
1806
1807 /* 第二遍：填充数据 */
1808 nz_idx = 0;
1809 out->row_ptr[0] = 0;
1810 for (bi = 0; bi < block_rows; ++bi) {
1811 for (bj = 0; bj < block_cols; ++bj) {
1812 int block_nonzero = 0;
1813 for (li = 0; li < block_size && !block_nonzero; ++li) {
1814 for (lj = 0; lj < block_size && !block_nonzero; ++lj) {
1815 size_t row = bi * block_size + li;
1816 size_t col = bj * block_size + lj;
1817 LMMC_REAL_ABS(&abs_v, &dense->data[row * dense->stride + col]);
1818 if (LMMC_REAL_CMP(&abs_v, &eps) > 0) {
1819 block_nonzero = 1;
1820 }
1821 }
1822 }
1823 if (block_nonzero) {
1824 size_t block_offset = nz_idx * block_size * block_size;
1825 out->col_idx[nz_idx] = bj;
1826
1827 /* 复制块内所有元素 */
1828 for (li = 0; li < block_size; ++li) {
1829 for (lj = 0; lj < block_size; ++lj) {
1830 size_t row = bi * block_size + li;
1831 size_t col = bj * block_size + lj;
1832 size_t blk_idx = block_offset + li * block_size + lj;
1833 LMMC_REAL_SET(&out->values[blk_idx],
1834 &dense->data[row * dense->stride + col]);
1835 }
1836 }
1837 ++nz_idx;
1838 }
1839 }
1840 out->row_ptr[bi + 1] = nz_idx;
1841 }
1842
1843 LMMC_REAL_CLEAR(&abs_v);
1844 LMMC_REAL_CLEAR(&zero);
1845 return LMMC_STATUS_OK;
1846}

1867lmmc_status_t lmmc_sparse_sym_csr_from_csr(const lmmc_sparse_mat_t* full,
1868 lmmc_sparse_sym_half_t half, lmmc_sparse_sym_csr_t* out) {
1869 size_t i, p;
1870 size_t n;
1871 size_t nnz_half = 0;
1872 size_t nz_idx = 0;
1873 size_t row_ptr_bytes, col_idx_bytes, val_bytes;
1874 lmmc_status_t st;
1875
1876 if (full == NULL || out == NULL) {
1877 return LMMC_STATUS_INVALID_ARGUMENT;
1878 }
1879
1880 st = lmmc_sparse_validate(full);
1881 if (st != LMMC_STATUS_OK) {
1882 return LMMC_STATUS_INVALID_ARGUMENT;
1883 }
1884
1885 if (full->rows != full->cols) {
1886 return LMMC_STATUS_INVALID_ARGUMENT;
1887 }
1888 if (full->format != LMMC_SPARSE_CSR) {
1889 return LMMC_STATUS_INVALID_ARGUMENT;
1890 }
1891
1892 n = full->rows;
1893
1894 /* 第一遍：计算半三角中的非零元数量 */
1895 for (i = 0; i < n; ++i) {
1896 for (p = full->row_ptr[i]; p < full->row_ptr[i + 1]; ++p) {
1897 size_t j = full->col_idx[p];
1898 if (half == LMMC_SPARSE_SYM_UPPER) {
1899 if (j >= i) ++nnz_half;
1900 } else {
1901 if (j <= i) ++nnz_half;
1902 }
1903 }
1904 }
1905
1906 /* 分配内存 */
1907 if (lmmc_mul_overflow_size(n + 1, sizeof(size_t), &row_ptr_bytes) ||
1908 lmmc_mul_overflow_size(nnz_half, sizeof(size_t), &col_idx_bytes) ||
1909 lmmc_mul_overflow_size(nnz_half, sizeof(lmmc_real_t), &val_bytes)) {
1910 return LMMC_STATUS_INVALID_ARGUMENT;
1911 }
1912
1913 out->row_ptr = (size_t*)lmmc_alloc(row_ptr_bytes);
1914 if (out->row_ptr == NULL) {
1915 return LMMC_STATUS_ALLOCATION_FAILED;
1916 }
1917
1918 out->col_idx = NULL;
1919 out->values = NULL;
1920
1921 if (nnz_half > 0) {
1922 out->col_idx = (size_t*)lmmc_alloc(col_idx_bytes);
1923 if (out->col_idx == NULL) {
1924 lmmc_free(out->row_ptr);
1925 out->row_ptr = NULL;
1926 return LMMC_STATUS_ALLOCATION_FAILED;
1927 }
1928
1929 out->values = (lmmc_real_t*)lmmc_alloc(val_bytes);
1930 if (out->values == NULL) {
1931 lmmc_free(out->col_idx);
1932 lmmc_free(out->row_ptr);
1933 out->col_idx = NULL;
1934 out->row_ptr = NULL;
1935 return LMMC_STATUS_ALLOCATION_FAILED;
1936 }
1937 }
1938
1939 /* 第二遍：填充数据 */
1940 nz_idx = 0;
1941 out->row_ptr[0] = 0;
1942 for (i = 0; i < n; ++i) {
1943 for (p = full->row_ptr[i]; p < full->row_ptr[i + 1]; ++p) {
1944 size_t j = full->col_idx[p];
1945 int keep = 0;
1946 if (half == LMMC_SPARSE_SYM_UPPER) {
1947 keep = (j >= i);
1948 } else {
1949 keep = (j <= i);
1950 }
1951 if (keep) {
1952 out->col_idx[nz_idx] = j;
1953 LMMC_REAL_SET(&out->values[nz_idx], &full->values[p]);
1954 ++nz_idx;
1955 }
1956 }
1957 out->row_ptr[i + 1] = nz_idx;
1958 }
1959
1960 out->n = n;
1961 out->nnz = nnz_half;
1962 out->half = half;
1963 out->owns_data = 1;
1964 return LMMC_STATUS_OK;
1965}

1967lmmc_status_t lmmc_sparse_sym_spmv(const lmmc_sparse_sym_csr_t* A,
1968 const lmmc_vec_t* x, lmmc_vec_t* y) {
1969 size_t i, p;
1970 lmmc_real_t zero;
1971 lmmc_real_t prod;
1972
1973 if (A == NULL || x == NULL || y == NULL) {
1974 return LMMC_STATUS_INVALID_ARGUMENT;
1975 }
1976 if (x->data == NULL || y->data == NULL) {
1977 return LMMC_STATUS_INVALID_ARGUMENT;
1978 }
1979 if (A->row_ptr == NULL) {
1980 return LMMC_STATUS_INVALID_ARGUMENT;
1981 }
1982 if (A->nnz > 0 && (A->col_idx == NULL || A->values == NULL)) {
1983 return LMMC_STATUS_INVALID_ARGUMENT;
1984 }
1985 if (x->size != A->n || y->size != A->n) {
1986 return LMMC_STATUS_DIMENSION_MISMATCH;
1987 }
1988
1989 /* 清零输出向量 */
1990 LMMC_REAL_INIT(&zero);
1991 LMMC_REAL_INIT(&prod);
1992 LMMC_REAL_SET_D(&zero, 0.0);
1993
1994 for (i = 0; i < A->n; ++i) {
1995 LMMC_REAL_SET_D(&y->data[i], 0.0);
1996 }
1997
1998 /* 对称 SpMV：利用对称性，每个非对角元贡献两次 */
1999 for (i = 0; i < A->n; ++i) {
2000 for (p = A->row_ptr[i]; p < A->row_ptr[i + 1]; ++p) {
2001 size_t j = A->col_idx[p];
2002 LMMC_REAL_MUL(&prod, &A->values[p], &x->data[j]);
2003
2004 /* y[i] += A[i,j] * x[j] */
2005 {
2006 lmmc_real_t tmp;
2007 LMMC_REAL_INIT(&tmp);
2008 LMMC_REAL_ADD(&tmp, &y->data[i], &prod);
2009 LMMC_REAL_SET(&y->data[i], &tmp);
2010 LMMC_REAL_CLEAR(&tmp);
2011 }
2012
2013 /* 对称贡献：y[j] += A[i,j] * x[i]（仅非对角元） */
2014 if (i != j) {
2015 lmmc_real_t prod2;
2016 lmmc_real_t tmp2;
2017 LMMC_REAL_INIT(&prod2);
2018 LMMC_REAL_INIT(&tmp2);
2019 LMMC_REAL_MUL(&prod2, &A->values[p], &x->data[i]);
2020 LMMC_REAL_ADD(&tmp2, &y->data[j], &prod2);
2021 LMMC_REAL_SET(&y->data[j], &tmp2);
2022 LMMC_REAL_CLEAR(&prod2);
2023 LMMC_REAL_CLEAR(&tmp2);
2024 }
2025 }
2026 }
2027
2028 LMMC_REAL_CLEAR(&zero);
2029 LMMC_REAL_CLEAR(&prod);
2030 return LMMC_STATUS_OK;
2031}
