14static int lmmc_mul_overflow_size(size_t a, size_t b, size_t* out) {
15 if (a == 0 || b == 0) {
16 *out = 0;
17 return 0;
18 }
19 if (a > ((size_t)-1) / b) {
20 return 1;
21 }
22 *out = a * b;
23 return 0;
24}

27 size_t n_elem = 0;

49 out_mat->cols = cols;

51 out_mat->data = data;
52 out_mat->owns_data = 1;
53 return LMMC_STATUS_OK;

133lmmc_status_t lmmc_mat_gemm(lmmc_real_t alpha, const lmmc_mat_t* A, int transA,
134 const lmmc_mat_t* B, int transB, lmmc_real_t beta, lmmc_mat_t* C) {
135 if (A == NULL || B == NULL || C == NULL ||
136 A->data == NULL || B->data == NULL || C->data == NULL) {
137 return LMMC_STATUS_INVALID_ARGUMENT;
138 }
139
140 /* Determine effective dimensions of op(A) and op(B) */
141 size_t M = transA ? A->cols : A->rows;
142 size_t K_A = transA ? A->rows : A->cols;
143 size_t K_B = transB ? B->cols : B->rows;
144 size_t N = transB ? B->rows : B->cols;
145
146 /* Dimension validation */
147 if (K_A != K_B) {
148 return LMMC_STATUS_DIMENSION_MISMATCH;
149 }
150 if (C->rows != M || C->cols != N) {
151 return LMMC_STATUS_DIMENSION_MISMATCH;
152 }
153
154 size_t K = K_A;
155
156 const lmmc_real_t* restrict a_data = A->data;
157 const lmmc_real_t* restrict b_data = B->data;
158 lmmc_real_t* restrict c_data = C->data;
159
160 size_t a_stride = A->stride;
161 size_t b_stride = B->stride;
162 size_t c_stride = C->stride;
163
164#ifdef LMMC_USE_BLAS
165 {
166 /* Route through BLAS dgemm with transpose support */
167 lmmc_blas_dgemm_ex(transA, transB,
168 M, N, K,
169 alpha,
170 a_data, a_stride,
171 b_data, b_stride,
172 beta,
173 c_data, c_stride);
174 return LMMC_STATUS_OK;
175 }
176#endif
177
178 /* Scale C by beta */
179 if (beta == 0.0) {
180 for (size_t i = 0; i < M; ++i) {
181 for (size_t j = 0; j < N; ++j) {
182 LMMC_REAL_SET_D(&c_data[i * c_stride + j], 0.0);
183 }
184 }
185 } else if (beta != 1.0) {
186 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
187 for (size_t i = 0; i < M; ++i) {
188 for (size_t j = 0; j < N; ++j) {
189 LMMC_REAL_MUL(&tmp_mul, &beta, &c_data[i * c_stride + j]);
190 LMMC_REAL_SET(&c_data[i * c_stride + j], &tmp_mul);
191 }
192 }
193 LMMC_REAL_CLEAR(&tmp_mul);
194 }
195
196 /* If alpha is zero, we're done */
197 if (alpha == 0.0) {
198 return LMMC_STATUS_OK;
199 }
200
201 /* Native triple-loop: C += alpha * op(A) * op(B) */
202 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
203 lmmc_real_t tmp_sum; LMMC_REAL_INIT(&tmp_sum);
204 lmmc_real_t alpha_a_ik; LMMC_REAL_INIT(&alpha_a_ik);
205
206 size_t bs = 64;
207 for (size_t ii = 0; ii < M; ii += bs) {
208 size_t i_end = (ii + bs > M) ? M : ii + bs;
209 for (size_t kk = 0; kk < K; kk += bs) {
210 size_t k_end = (kk + bs > K) ? K : kk + bs;
211 for (size_t jj = 0; jj < N; jj += bs) {
212 size_t j_end = (jj + bs > N) ? N : jj + bs;
213 for (size_t i = ii; i < i_end; ++i) {
214 for (size_t k = kk; k < k_end; ++k) {
215 /* Get op(A)[i,k] */
216 lmmc_real_t a_ik;
217 if (transA) {
218 a_ik = a_data[k * a_stride + i];
219 } else {
220 a_ik = a_data[i * a_stride + k];
221 }
222 /* alpha * op(A)[i,k] */
223 LMMC_REAL_MUL(&alpha_a_ik, &alpha, &a_ik);
224
225 for (size_t j = jj; j < j_end; ++j) {
226 /* Get op(B)[k,j] */
227 lmmc_real_t b_kj;
228 if (transB) {
229 b_kj = b_data[j * b_stride + k];
230 } else {
231 b_kj = b_data[k * b_stride + j];
232 }
233 LMMC_REAL_MUL(&tmp_mul, &alpha_a_ik, &b_kj);
234 LMMC_REAL_ADD(&tmp_sum, &c_data[i * c_stride + j], &tmp_mul);
235 LMMC_REAL_SET(&c_data[i * c_stride + j], &tmp_sum);
236 }
237 }
238 }
239 }
240 }
241 }
242
243 LMMC_REAL_CLEAR(&tmp_mul);
244 LMMC_REAL_CLEAR(&tmp_sum);
245 LMMC_REAL_CLEAR(&alpha_a_ik);
246 return LMMC_STATUS_OK;
247}

372lmmc_status_t lmmc_mat_gemv(lmmc_real_t alpha, const lmmc_mat_t* A, int transA,
373 const lmmc_vec_t* x, lmmc_real_t beta, lmmc_vec_t* y) {
374 if (A == NULL || x == NULL || y == NULL ||
375 A->data == NULL || x->data == NULL || y->data == NULL) {
376 return LMMC_STATUS_INVALID_ARGUMENT;
377 }
378
379 /* Determine effective dimensions of op(A) */
380 size_t M = transA ? A->cols : A->rows;
381 size_t N = transA ? A->rows : A->cols;
382
383 /* Dimension validation */
384 if (N != x->size || M != y->size) {
385 return LMMC_STATUS_DIMENSION_MISMATCH;
386 }
387
388 const lmmc_real_t* restrict a_data = A->data;
389 size_t a_stride = A->stride;
390
391#ifdef LMMC_USE_BLAS
392 {
393 char trans_char = transA ? 'T' : 'N';
394 lmmc_blas_dgemv(trans_char, A->rows, A->cols,
395 alpha,
396 a_data, a_stride,
397 x->data, 1,
398 beta,
399 y->data, 1);
400 return LMMC_STATUS_OK;
401 }
402#endif
403
404 /* Scale y by beta */
405 if (beta == 0.0) {
406 for (size_t i = 0; i < M; ++i) {
407 LMMC_REAL_SET_D(&y->data[i], 0.0);
408 }
409 } else if (beta != 1.0) {
410 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
411 for (size_t i = 0; i < M; ++i) {
412 LMMC_REAL_MUL(&tmp_mul, &beta, &y->data[i]);
413 LMMC_REAL_SET(&y->data[i], &tmp_mul);
414 }
415 LMMC_REAL_CLEAR(&tmp_mul);
416 }
417
418 /* If alpha is zero, we're done */
419 if (alpha == 0.0) {
420 return LMMC_STATUS_OK;
421 }
422
423 /* Native loop: y += alpha * op(A) * x */
424 lmmc_real_t sum; LMMC_REAL_INIT(&sum);
425 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
426 lmmc_real_t tmp_sum; LMMC_REAL_INIT(&tmp_sum);
427 lmmc_real_t alpha_sum; LMMC_REAL_INIT(&alpha_sum);
428
429 for (size_t i = 0; i < M; ++i) {
430 LMMC_REAL_SET_D(&sum, 0.0);
431 for (size_t j = 0; j < N; ++j) {
432 lmmc_real_t a_ij;
433 if (transA) {
434 a_ij = a_data[j * a_stride + i];
435 } else {
436 a_ij = a_data[i * a_stride + j];
437 }
438 LMMC_REAL_MUL(&tmp_mul, &a_ij, &x->data[j]);
439 LMMC_REAL_ADD(&tmp_sum, &sum, &tmp_mul);
440 LMMC_REAL_SET(&sum, &tmp_sum);
441 }
442 /* y[i] += alpha * sum */
443 LMMC_REAL_MUL(&alpha_sum, &alpha, &sum);
444 LMMC_REAL_ADD(&tmp_sum, &y->data[i], &alpha_sum);
445 LMMC_REAL_SET(&y->data[i], &tmp_sum);
446 }
447
448 LMMC_REAL_CLEAR(&sum);
449 LMMC_REAL_CLEAR(&tmp_mul);
450 LMMC_REAL_CLEAR(&tmp_sum);
451 LMMC_REAL_CLEAR(&alpha_sum);
452 return LMMC_STATUS_OK;
453}

766lmmc_status_t lmmc_mat_det(const lmmc_mat_t* a, lmmc_real_t* out_det) {
767 if (a == NULL || out_det == NULL || a->data == NULL) {
768 return LMMC_STATUS_INVALID_ARGUMENT;
769 }
770 if (a->rows != a->cols) {
771 return LMMC_STATUS_INVALID_ARGUMENT;
772 }
773
774 size_t n = a->rows;
775
776
777 if (n == 1) {
778 LMMC_REAL_SET(out_det, &a->data[0]);
779 return LMMC_STATUS_OK;
780 }
781
782
783 if (n == 2) {
784 lmmc_real_t ad; LMMC_REAL_INIT(&ad);
785 lmmc_real_t bc; LMMC_REAL_INIT(&bc);
786 lmmc_real_t result; LMMC_REAL_INIT(&result);
787
788 LMMC_REAL_MUL(&ad, &a->data[0 * a->stride + 0], &a->data[1 * a->stride + 1]);
789 LMMC_REAL_MUL(&bc, &a->data[0 * a->stride + 1], &a->data[1 * a->stride + 0]);
790 LMMC_REAL_SUB(&result, &ad, &bc);
791 LMMC_REAL_SET(out_det, &result);
792
793 LMMC_REAL_CLEAR(&ad);
794 LMMC_REAL_CLEAR(&bc);
795 LMMC_REAL_CLEAR(&result);
796 return LMMC_STATUS_OK;
797 }
798
799
800 size_t n_elem = n * n;
801 size_t n_bytes = n_elem * sizeof(lmmc_real_t);
802 lmmc_real_t* lu = (lmmc_real_t*)lmmc_alloc(n_bytes);
803 if (lu == NULL) {
804 return LMMC_STATUS_ALLOCATION_FAILED;
805 }
806
807
808 for (size_t i = 0; i < n; ++i) {
809 for (size_t j = 0; j < n; ++j) {
810 lu[i * n + j] = a->data[i * a->stride + j];
811 }
812 }
813
814 int sign = 1;
815
816 lmmc_real_t abs_val; LMMC_REAL_INIT(&abs_val);
817 lmmc_real_t max_val; LMMC_REAL_INIT(&max_val);
818 lmmc_real_t factor; LMMC_REAL_INIT(&factor);
819 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul);
820 lmmc_real_t tmp_sub; LMMC_REAL_INIT(&tmp_sub);
821
822 for (size_t k = 0; k < n; ++k) {
823
824 size_t pivot_row = k;
825 LMMC_REAL_ABS(&max_val, &lu[k * n + k]);
826
827 for (size_t i = k + 1; i < n; ++i) {
828 LMMC_REAL_ABS(&abs_val, &lu[i * n + k]);
829 if (LMMC_REAL_CMP(&abs_val, &max_val) > 0) {
830 LMMC_REAL_SET(&max_val, &abs_val);
831 pivot_row = i;
832 }
833 }
834
835
836 lmmc_real_t zero; LMMC_REAL_INIT(&zero);
837 LMMC_REAL_SET_D(&zero, 0.0);
838 if (LMMC_REAL_CMP(&max_val, &zero) == 0) {
839 LMMC_REAL_SET_D(out_det, 0.0);
840 LMMC_REAL_CLEAR(&zero);
841 LMMC_REAL_CLEAR(&abs_val);
842 LMMC_REAL_CLEAR(&max_val);
843 LMMC_REAL_CLEAR(&factor);
844 LMMC_REAL_CLEAR(&tmp_mul);
845 LMMC_REAL_CLEAR(&tmp_sub);
846 lmmc_free(lu);
847 return LMMC_STATUS_OK;
848 }
849 LMMC_REAL_CLEAR(&zero);
850
851
852 if (pivot_row != k) {
853 for (size_t j = 0; j < n; ++j) {
854 lmmc_real_t tmp = lu[k * n + j];
855 lu[k * n + j] = lu[pivot_row * n + j];
856 lu[pivot_row * n + j] = tmp;
857 }
858 sign = -sign;
859 }
860
861
862 for (size_t i = k + 1; i < n; ++i) {
863 LMMC_REAL_DIV(&factor, &lu[i * n + k], &lu[k * n + k]);
864 LMMC_REAL_SET(&lu[i * n + k], &factor);
865
866 for (size_t j = k + 1; j < n; ++j) {
867 LMMC_REAL_MUL(&tmp_mul, &factor, &lu[k * n + j]);
868 LMMC_REAL_SUB(&tmp_sub, &lu[i * n + j], &tmp_mul);
869 LMMC_REAL_SET(&lu[i * n + j], &tmp_sub);
870 }
871 }
872 }
873
874
875 lmmc_real_t det; LMMC_REAL_INIT(&det);
876 LMMC_REAL_SET_D(&det, (double)sign);
877
878 lmmc_real_t tmp_prod; LMMC_REAL_INIT(&tmp_prod);
879 for (size_t i = 0; i < n; ++i) {
880 LMMC_REAL_MUL(&tmp_prod, &det, &lu[i * n + i]);
881 LMMC_REAL_SET(&det, &tmp_prod);
882 }
883 LMMC_REAL_SET(out_det, &det);
884
885 LMMC_REAL_CLEAR(&det);
886 LMMC_REAL_CLEAR(&tmp_prod);
887 LMMC_REAL_CLEAR(&abs_val);
888 LMMC_REAL_CLEAR(&max_val);
889 LMMC_REAL_CLEAR(&factor);
890 LMMC_REAL_CLEAR(&tmp_mul);
891 LMMC_REAL_CLEAR(&tmp_sub);
892 lmmc_free(lu);
893
894 return LMMC_STATUS_OK;
895}

897lmmc_status_t lmmc_mat_inv(const lmmc_mat_t* A, lmmc_mat_t* A_inv) {
898 lmmc_mat_t lu_mat;
899 size_t* pivots = NULL;
900 lmmc_vec_t ei;
901 lmmc_vec_t col;
902 size_t n, i, j;
903 lmmc_status_t status;
904
905 if (A == NULL || A_inv == NULL || A->data == NULL || A_inv->data == NULL) {
906 return LMMC_STATUS_INVALID_ARGUMENT;
907 }
908 if (A->rows != A->cols) {
909 return LMMC_STATUS_INVALID_ARGUMENT;
910 }
911 if (A_inv->rows != A->rows || A_inv->cols != A->cols) {
912 return LMMC_STATUS_DIMENSION_MISMATCH;
913 }
914
915 n = A->rows;
916
917 /* Copy A into workspace for LU decomposition */
918 status = lmmc_mat_create(n, n, &lu_mat);
919 if (status != LMMC_STATUS_OK) {
920 return status;
921 }
922 status = lmmc_mat_copy(A, &lu_mat);
923 if (status != LMMC_STATUS_OK) {
924 lmmc_mat_destroy(&lu_mat);
925 return status;
926 }
927
928 /* Allocate pivot array */
929 pivots = (size_t*)lmmc_alloc(n * sizeof(size_t));
930 if (pivots == NULL) {
931 lmmc_mat_destroy(&lu_mat);
932 return LMMC_STATUS_ALLOCATION_FAILED;
933 }
934
935 /* LU decompose with partial pivoting */
936 status = lmmc_lu_decompose_inplace(&lu_mat, pivots, NULL);
937 if (status != LMMC_STATUS_OK) {
938 /* Singular or other failure — leave A_inv unmodified */
939 lmmc_free(pivots);
940 lmmc_mat_destroy(&lu_mat);
941 return status;
942 }
943
944 /* Create temporary vectors for solving */
945 status = lmmc_vec_create(n, &ei);
946 if (status != LMMC_STATUS_OK) {
947 lmmc_free(pivots);
948 lmmc_mat_destroy(&lu_mat);
949 return status;
950 }
951 status = lmmc_vec_create(n, &col);
952 if (status != LMMC_STATUS_OK) {
953 lmmc_vec_destroy(&ei);
954 lmmc_free(pivots);
955 lmmc_mat_destroy(&lu_mat);
956 return status;
957 }
958
959 /* Solve A * col_i = e_i for each column of the identity */
960 for (i = 0; i < n; ++i) {
961 /* Set up e_i (standard basis vector) */
962 for (j = 0; j < n; ++j) {
963 LMMC_REAL_SET_D(&ei.data[j], 0.0);
964 }
965 LMMC_REAL_SET_D(&ei.data[i], 1.0);
966
967 /* Solve using LU factors */
968 status = lmmc_lu_solve(&lu_mat, pivots, &ei, &col);
969 if (status != LMMC_STATUS_OK) {
970 lmmc_vec_destroy(&col);
971 lmmc_vec_destroy(&ei);
972 lmmc_free(pivots);
973 lmmc_mat_destroy(&lu_mat);
974 return status;
975 }
976
977 /* Copy solution into column i of A_inv */
978 for (j = 0; j < n; ++j) {
979 LMMC_REAL_SET(&A_inv->data[j * A_inv->stride + i], &col.data[j]);
980 }
981 }
982
983 lmmc_vec_destroy(&col);
984 lmmc_vec_destroy(&ei);
985 lmmc_free(pivots);
986 lmmc_mat_destroy(&lu_mat);
987
988 return LMMC_STATUS_OK;
989}

991lmmc_status_t lmmc_solve_triangular(const lmmc_mat_t* T, int upper, int diag_unit,
992 const lmmc_vec_t* b, lmmc_vec_t* x) {
993 size_t n, i, j;
994 lmmc_real_t sum, tmp, diag_val, abs_diag, eps;
995
996 if (T == NULL || b == NULL || x == NULL || T->data == NULL || b->data == NULL || x->data == NULL) {
997 return LMMC_STATUS_INVALID_ARGUMENT;
998 }
999 if (T->rows != T->cols) {
1000 return LMMC_STATUS_INVALID_ARGUMENT;
1001 }
1002 if (b->size != T->rows || x->size != T->rows) {
1003 return LMMC_STATUS_DIMENSION_MISMATCH;
1004 }
1005
1006 n = T->rows;
1007 LMMC_REAL_INIT(&sum);
1008 LMMC_REAL_INIT(&tmp);
1009 LMMC_REAL_INIT(&diag_val);
1010 LMMC_REAL_INIT(&abs_diag);
1011 LMMC_REAL_INIT(&eps);
1012 LMMC_REAL_SET_D(&eps, 1e-15);
1013
1014 if (upper) {
1015 /* Back substitution for upper triangular */
1016 for (i = n; i-- > 0;) {
1017 LMMC_REAL_SET(&sum, &b->data[i]);
1018 for (j = i + 1; j < n; ++j) {
1019 LMMC_REAL_MUL(&tmp, &T->data[i * T->stride + j], &x->data[j]);
1020 LMMC_REAL_SUB(&sum, &sum, &tmp);
1021 }
1022 if (diag_unit) {
1023 LMMC_REAL_SET(&x->data[i], &sum);
1024 } else {
1025 LMMC_REAL_SET(&diag_val, &T->data[i * T->stride + i]);
1026 LMMC_REAL_ABS(&abs_diag, &diag_val);
1027 if (LMMC_REAL_CMP(&abs_diag, &eps) <= 0) {
1028 LMMC_REAL_CLEAR(&sum);
1029 LMMC_REAL_CLEAR(&tmp);
1030 LMMC_REAL_CLEAR(&diag_val);
1031 LMMC_REAL_CLEAR(&abs_diag);
1032 LMMC_REAL_CLEAR(&eps);
1033 return LMMC_STATUS_SINGULAR_MATRIX;
1034 }
1035 LMMC_REAL_DIV(&x->data[i], &sum, &diag_val);
1036 }
1037 }
1038 } else {
1039 /* Forward substitution for lower triangular */
1040 for (i = 0; i < n; ++i) {
1041 LMMC_REAL_SET(&sum, &b->data[i]);
1042 for (j = 0; j < i; ++j) {
1043 LMMC_REAL_MUL(&tmp, &T->data[i * T->stride + j], &x->data[j]);
1044 LMMC_REAL_SUB(&sum, &sum, &tmp);
1045 }
1046 if (diag_unit) {
1047 LMMC_REAL_SET(&x->data[i], &sum);
1048 } else {
1049 LMMC_REAL_SET(&diag_val, &T->data[i * T->stride + i]);
1050 LMMC_REAL_ABS(&abs_diag, &diag_val);
1051 if (LMMC_REAL_CMP(&abs_diag, &eps) <= 0) {
1052 LMMC_REAL_CLEAR(&sum);
1053 LMMC_REAL_CLEAR(&tmp);
1054 LMMC_REAL_CLEAR(&diag_val);
1055 LMMC_REAL_CLEAR(&abs_diag);
1056 LMMC_REAL_CLEAR(&eps);
1057 return LMMC_STATUS_SINGULAR_MATRIX;
1058 }
1059 LMMC_REAL_DIV(&x->data[i], &sum, &diag_val);
1060 }
1061 }
1062 }
1063
1064 LMMC_REAL_CLEAR(&sum);
1065 LMMC_REAL_CLEAR(&tmp);
1066 LMMC_REAL_CLEAR(&diag_val);
1067 LMMC_REAL_CLEAR(&abs_diag);
1068 LMMC_REAL_CLEAR(&eps);
1069
1070 return LMMC_STATUS_OK;
1071}
