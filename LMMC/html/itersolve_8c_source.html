63static lmmc_status_t lmmc_gmres_back_substitute(
64 const lmmc_real_t* h,
65 size_t ld,
66 const lmmc_real_t* rhs,
67 size_t dim,
68 lmmc_real_t* out_y
69) {
70 size_t ii = 0;
71 lmmc_real_t sum;
72 lmmc_real_t diag;
73 lmmc_real_t tmp_mul;
74 lmmc_real_t tmp_sub;
75 lmmc_real_t abs_diag;
76 lmmc_real_t eps_30;
77
78 LMMC_REAL_INIT(&sum);
79 LMMC_REAL_INIT(&diag);
80 LMMC_REAL_INIT(&tmp_mul);
81 LMMC_REAL_INIT(&tmp_sub);
82 LMMC_REAL_INIT(&abs_diag);
83 LMMC_REAL_INIT(&eps_30);
84
85 LMMC_REAL_SET_D(&eps_30, 1e-30);
86
87 if (h == NULL || rhs == NULL || out_y == NULL || ld == 0 || dim == 0) {
88 LMMC_REAL_CLEAR(&eps_30);
89 LMMC_REAL_CLEAR(&abs_diag);
90 LMMC_REAL_CLEAR(&tmp_sub);
91 LMMC_REAL_CLEAR(&tmp_mul);
92 LMMC_REAL_CLEAR(&diag);
93 LMMC_REAL_CLEAR(&sum);
94 return LMMC_STATUS_INVALID_ARGUMENT;
95 }
96
97 for (ii = dim; ii > 0; --ii) {
98 size_t i = ii - 1;
99 size_t j = 0;
100
101 LMMC_REAL_SET(&sum, &rhs[i]);
102 LMMC_REAL_SET_D(&diag, 0.0);
103
104 for (j = i + 1; j < dim; ++j) {
105 LMMC_REAL_MUL(&tmp_mul, &h[i * ld + j], &out_y[j]);
106 LMMC_REAL_SUB(&tmp_sub, &sum, &tmp_mul);
107 LMMC_REAL_SET(&sum, &tmp_sub);
108 }
109
110 LMMC_REAL_SET(&diag, &h[i * ld + i]);
111 LMMC_REAL_ABS(&abs_diag, &diag);
112
113 if (!lmmc_is_finite_number(sum) || !lmmc_is_finite_number(diag) || LMMC_REAL_CMP(&abs_diag, &eps_30) <= 0) {
114 LMMC_REAL_CLEAR(&eps_30);
115 LMMC_REAL_CLEAR(&abs_diag);
116 LMMC_REAL_CLEAR(&tmp_sub);
117 LMMC_REAL_CLEAR(&tmp_mul);
118 LMMC_REAL_CLEAR(&diag);
119 LMMC_REAL_CLEAR(&sum);
120 return LMMC_STATUS_NUMERICAL_FAILURE;
121 }
122
123 LMMC_REAL_DIV(&out_y[i], &sum, &diag);
124 if (!lmmc_is_finite_number(out_y[i])) {
125 LMMC_REAL_CLEAR(&eps_30);
126 LMMC_REAL_CLEAR(&abs_diag);
127 LMMC_REAL_CLEAR(&tmp_sub);
128 LMMC_REAL_CLEAR(&tmp_mul);
129 LMMC_REAL_CLEAR(&diag);
130 LMMC_REAL_CLEAR(&sum);
131 return LMMC_STATUS_NUMERICAL_FAILURE;
132 }
133 }
134
135 LMMC_REAL_CLEAR(&eps_30);
136 LMMC_REAL_CLEAR(&abs_diag);
137 LMMC_REAL_CLEAR(&tmp_sub);
138 LMMC_REAL_CLEAR(&tmp_mul);
139 LMMC_REAL_CLEAR(&diag);
140 LMMC_REAL_CLEAR(&sum);
141 return LMMC_STATUS_OK;
142}

164 max_iter = problem_size * 20;
165 if (max_iter < 100) {

179 out_cfg->log_cb = NULL;
180 out_cfg->log_user_data = NULL;
181 out_cfg->apply_op = NULL;
182 out_cfg->op_user_data = NULL;

186static lmmc_status_t validate_and_init_itersolve_config(
187 const lmmc_sparse_mat_t* a,
188 const lmmc_vec_t* b,
189 const lmmc_precond_t* precond,
190 const lmmc_itersolve_config_t* cfg,
191 lmmc_vec_t* x,
192 lmmc_itersolve_config_t* out_cfg,
193 lmmc_itersolve_result_t* out_result
194) {
195 lmmc_real_t zero; LMMC_REAL_INIT(&zero); LMMC_REAL_SET_D(&zero, 0.0);
196 lmmc_status_t st = LMMC_STATUS_OK;
197
198 if (a == NULL || b == NULL || x == NULL || a->row_ptr == NULL || b->data == NULL || x->data == NULL) {
199 st = LMMC_STATUS_INVALID_ARGUMENT;
200 goto cleanup;
201 }
202 if (a->rows == 0 || a->cols == 0 || b->size == 0 || x->size == 0) {
203 st = LMMC_STATUS_INVALID_ARGUMENT;
204 goto cleanup;
205 }
206 if (a->rows != a->cols || b->size != a->rows || x->size != a->cols) {
207 st = LMMC_STATUS_DIMENSION_MISMATCH;
208 goto cleanup;
209 }
210
211 if (cfg != NULL) {
212 *out_cfg = *cfg;
213 } else {
214 st = lmmc_itersolve_default_config(b->size, out_cfg);
215 if (st != LMMC_STATUS_OK) goto cleanup;
216 }
217
218 if (!lmmc_is_finite_number(out_cfg->abs_tol) || !lmmc_is_finite_number(out_cfg->rel_tol) ||
219 LMMC_REAL_CMP(&out_cfg->abs_tol, &zero) < 0 || LMMC_REAL_CMP(&out_cfg->rel_tol, &zero) < 0 || out_cfg->max_iter == 0) {
220 st = LMMC_STATUS_INVALID_ARGUMENT;
221 goto cleanup;
222 }
223
224 if (precond != NULL && precond->size != b->size) {
225 st = LMMC_STATUS_DIMENSION_MISMATCH;
226 goto cleanup;
227 }
228
229 if (out_result != NULL) {
230 out_result->converged = 0;
231 out_result->num_iter = 0;
232 LMMC_REAL_INIT(&out_result->initial_residual_norm);
233 LMMC_REAL_INIT(&out_result->final_residual_norm);
234 LMMC_REAL_SET_D(&out_result->initial_residual_norm, 0.0);
235 LMMC_REAL_SET_D(&out_result->final_residual_norm, 0.0);
236 }
237
238cleanup:
239 LMMC_REAL_CLEAR(&zero);
240 return st;
241}

243static lmmc_status_t lmmc_gmres_arnoldi_step(
244 const lmmc_sparse_mat_t* a, const lmmc_precond_t* precond,
245 lmmc_vec_t* basis, lmmc_real_t* h, size_t j, size_t restart,
246 lmmc_vec_t* ax, lmmc_vec_t* w, lmmc_real_t* out_h_next
247) {
248 size_t ii = 0;
249 lmmc_status_t st = lmmc_sparse_mat_vec_mul(a, &basis[j], ax);
250 if (st != LMMC_STATUS_OK) goto end;
251
252 st = lmmc_apply_precond_or_identity(precond, ax, w);
253 if (st != LMMC_STATUS_OK) goto end;
254
255 for (ii = 0; ii <= j; ++ii) {
256 lmmc_real_t hij; LMMC_REAL_INIT(&hij); LMMC_REAL_SET_D(&hij, 0.0);
257 st = lmmc_vec_dot_checked(&basis[ii], w, &hij);
258 if (st != LMMC_STATUS_OK) {
259 LMMC_REAL_CLEAR(&hij); goto end;
260 }
261 LMMC_REAL_SET(&h[ii * restart + j], &hij);
262
263
264 lmmc_real_t neg_hij; LMMC_REAL_INIT(&neg_hij);
265 LMMC_REAL_NEG(&neg_hij, &hij);
266 st = lmmc_vec_axpy(neg_hij, &basis[ii], w);
267 LMMC_REAL_CLEAR(&neg_hij);
268 if (st != LMMC_STATUS_OK) {
269 LMMC_REAL_CLEAR(&hij); goto end;
270 }
271 LMMC_REAL_CLEAR(&hij);
272 }
273
274 st = lmmc_vec_norm2_checked(w, out_h_next);
275 if (st == LMMC_STATUS_OK) {
276 LMMC_REAL_SET(&h[(j + 1) * restart + j], out_h_next);
277 }
278
279end:
280 return st;
281}

283static lmmc_status_t lmmc_gmres_apply_givens(
284 lmmc_real_t* h, size_t j, size_t restart,
285 lmmc_real_t* cs, lmmc_real_t* sn, lmmc_real_t* g,
286 lmmc_real_t eps_30, int* out_breakdown
287) {
288 size_t ii = 0;
289 *out_breakdown = 0;
290 for (ii = 0; ii < j; ++ii) {
291 lmmc_real_t h0; LMMC_REAL_INIT(&h0); LMMC_REAL_SET(&h0, &h[ii * restart + j]);
292 lmmc_real_t h1; LMMC_REAL_INIT(&h1); LMMC_REAL_SET(&h1, &h[(ii + 1) * restart + j]);
293
294 lmmc_real_t m1, m2; LMMC_REAL_INIT(&m1); LMMC_REAL_INIT(&m2);
295 LMMC_REAL_MUL(&m1, &cs[ii], &h0);
296 LMMC_REAL_MUL(&m2, &sn[ii], &h1);
297 LMMC_REAL_ADD(&h[ii * restart + j], &m1, &m2);
298
299 lmmc_real_t neg_sn; LMMC_REAL_INIT(&neg_sn);
300 LMMC_REAL_NEG(&neg_sn, &sn[ii]);
301 LMMC_REAL_MUL(&m1, &neg_sn, &h0);
302 LMMC_REAL_MUL(&m2, &cs[ii], &h1);
303 LMMC_REAL_ADD(&h[(ii + 1) * restart + j], &m1, &m2);
304
305 LMMC_REAL_CLEAR(&neg_sn);
306 LMMC_REAL_CLEAR(&m2); LMMC_REAL_CLEAR(&m1);
307 LMMC_REAL_CLEAR(&h1); LMMC_REAL_CLEAR(&h0);
308 }
309
310 {
311 lmmc_real_t hj; LMMC_REAL_INIT(&hj); LMMC_REAL_SET(&hj, &h[j * restart + j]);
312 lmmc_real_t hsub; LMMC_REAL_INIT(&hsub); LMMC_REAL_SET(&hsub, &h[(j + 1) * restart + j]);
313
314 lmmc_real_t denom_loc; LMMC_REAL_INIT(&denom_loc);
315 lmmc_real_t m1, m2, a_sum;
316 LMMC_REAL_INIT(&m1); LMMC_REAL_INIT(&m2); LMMC_REAL_INIT(&a_sum);
317 LMMC_REAL_MUL(&m1, &hj, &hj);
318 LMMC_REAL_MUL(&m2, &hsub, &hsub);
319 LMMC_REAL_ADD(&a_sum, &m1, &m2);
320 LMMC_REAL_SQRT(&denom_loc, &a_sum);
321
322 lmmc_real_t gtmp; LMMC_REAL_INIT(&gtmp); LMMC_REAL_SET_D(&gtmp, 0.0);
323
324 if (!lmmc_is_finite_number(denom_loc)) {
325 LMMC_REAL_CLEAR(&a_sum); LMMC_REAL_CLEAR(&m2); LMMC_REAL_CLEAR(&m1);
326 LMMC_REAL_CLEAR(&gtmp); LMMC_REAL_CLEAR(&denom_loc);
327 LMMC_REAL_CLEAR(&hsub); LMMC_REAL_CLEAR(&hj);
328 return LMMC_STATUS_NUMERICAL_FAILURE;
329 }
330 if (LMMC_REAL_CMP(&denom_loc, &eps_30) <= 0) {
331 LMMC_REAL_CLEAR(&a_sum); LMMC_REAL_CLEAR(&m2); LMMC_REAL_CLEAR(&m1);
332 LMMC_REAL_CLEAR(&gtmp); LMMC_REAL_CLEAR(&denom_loc);
333 LMMC_REAL_CLEAR(&hsub); LMMC_REAL_CLEAR(&hj);
334 *out_breakdown = 1;
335 return LMMC_STATUS_OK;
336 }
337
338 LMMC_REAL_DIV(&cs[j], &hj, &denom_loc);
339 LMMC_REAL_DIV(&sn[j], &hsub, &denom_loc);
340 if (!lmmc_is_finite_number(cs[j]) || !lmmc_is_finite_number(sn[j])) {
341 LMMC_REAL_CLEAR(&a_sum); LMMC_REAL_CLEAR(&m2); LMMC_REAL_CLEAR(&m1);
342 LMMC_REAL_CLEAR(&gtmp); LMMC_REAL_CLEAR(&denom_loc);
343 LMMC_REAL_CLEAR(&hsub); LMMC_REAL_CLEAR(&hj);
344 return LMMC_STATUS_NUMERICAL_FAILURE;
345 }
346
347 LMMC_REAL_MUL(&m1, &cs[j], &hj);
348 LMMC_REAL_MUL(&m2, &sn[j], &hsub);
349 LMMC_REAL_ADD(&h[j * restart + j], &m1, &m2);
350 LMMC_REAL_SET_D(&h[(j + 1) * restart + j], 0.0);
351
352 LMMC_REAL_MUL(&gtmp, &cs[j], &g[j]);
353
354 lmmc_real_t neg_sn; LMMC_REAL_INIT(&neg_sn);
355 LMMC_REAL_NEG(&neg_sn, &sn[j]);
356 LMMC_REAL_MUL(&g[j + 1], &neg_sn, &g[j]);
357 LMMC_REAL_CLEAR(&neg_sn);
358
359 LMMC_REAL_SET(&g[j], &gtmp);
360
361 LMMC_REAL_CLEAR(&a_sum); LMMC_REAL_CLEAR(&m2); LMMC_REAL_CLEAR(&m1);
362 LMMC_REAL_CLEAR(&gtmp); LMMC_REAL_CLEAR(&denom_loc);
363 LMMC_REAL_CLEAR(&hsub); LMMC_REAL_CLEAR(&hj);
364 }
365 return LMMC_STATUS_OK;
366}

368lmmc_status_t lmmc_cg_solve(
369 const lmmc_sparse_mat_t* a,
370 const lmmc_vec_t* b,
371 const lmmc_precond_t* precond,
372 const lmmc_itersolve_config_t* cfg,
373 lmmc_vec_t* x,
374 lmmc_itersolve_result_t* out_result
375) {
376 lmmc_itersolve_config_t local_cfg = {0};
377 lmmc_vec_t r = {0};
378 lmmc_vec_t z = {0};
379 lmmc_vec_t p = {0};
380 lmmc_vec_t ap = {0};
381 lmmc_status_t st = LMMC_STATUS_OK;
382
383 lmmc_real_t norm_b; LMMC_REAL_INIT(&norm_b); LMMC_REAL_SET_D(&norm_b, 0.0);
384 lmmc_real_t norm_r; LMMC_REAL_INIT(&norm_r); LMMC_REAL_SET_D(&norm_r, 0.0);
385 lmmc_real_t threshold; LMMC_REAL_INIT(&threshold); LMMC_REAL_SET_D(&threshold, 0.0);
386 lmmc_real_t rho; LMMC_REAL_INIT(&rho); LMMC_REAL_SET_D(&rho, 0.0);
387 lmmc_real_t denom; LMMC_REAL_INIT(&denom); LMMC_REAL_SET_D(&denom, 0.0);
388 lmmc_real_t alpha; LMMC_REAL_INIT(&alpha); LMMC_REAL_SET_D(&alpha, 0.0);
389 lmmc_real_t rho_new; LMMC_REAL_INIT(&rho_new); LMMC_REAL_SET_D(&rho_new, 0.0);
390 lmmc_real_t beta; LMMC_REAL_INIT(&beta); LMMC_REAL_SET_D(&beta, 0.0);
391 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul); LMMC_REAL_SET_D(&tmp_mul, 0.0);
392 lmmc_real_t tmp_add; LMMC_REAL_INIT(&tmp_add); LMMC_REAL_SET_D(&tmp_add, 0.0);
393 lmmc_real_t zero; LMMC_REAL_INIT(&zero); LMMC_REAL_SET_D(&zero, 0.0);
394 lmmc_real_t eps_30; LMMC_REAL_INIT(&eps_30); LMMC_REAL_SET_D(&eps_30, 1e-30);
395 lmmc_real_t abs_val; LMMC_REAL_INIT(&abs_val); LMMC_REAL_SET_D(&abs_val, 0.0);
396
397 size_t iter_count = 0;
398 int converged = 0;
399
400 st = validate_and_init_itersolve_config(a, b, precond, cfg, x, &local_cfg, out_result);
401 if (st != LMMC_STATUS_OK) goto cleanup;
402
403 st = lmmc_vec_create(b->size, &r);
404 if (st != LMMC_STATUS_OK) goto cleanup;
405 st = lmmc_vec_create(b->size, &z);
406 if (st != LMMC_STATUS_OK) goto cleanup;
407 st = lmmc_vec_create(b->size, &p);
408 if (st != LMMC_STATUS_OK) goto cleanup;
409 st = lmmc_vec_create(b->size, &ap);
410 if (st != LMMC_STATUS_OK) goto cleanup;
411
412 st = lmmc_sparse_mat_vec_mul(a, x, &ap);
413 if (st != LMMC_STATUS_OK) goto cleanup;
414
415 {
416 size_t i = 0;
417 for (i = 0; i < b->size; ++i) {
418 LMMC_REAL_SUB(&r.data[i], &b->data[i], &ap.data[i]);
419 }
420 }
421
422 st = lmmc_vec_norm2_checked(b, &norm_b);
423 if (st != LMMC_STATUS_OK) goto cleanup;
424
425 st = lmmc_vec_norm2_checked(&r, &norm_r);
426 if (st != LMMC_STATUS_OK) goto cleanup;
427
428 LMMC_REAL_MUL(&tmp_mul, &local_cfg.rel_tol, &norm_b);
429 LMMC_REAL_ADD(&threshold, &local_cfg.abs_tol, &tmp_mul);
430
431 if (!lmmc_is_finite_number(threshold)) {
432 st = LMMC_STATUS_NUMERICAL_FAILURE;
433 goto cleanup;
434 }
435
436 if (out_result != NULL) {
437 LMMC_REAL_SET(&out_result->initial_residual_norm, &norm_r);
438 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
439 }
440
441 lmmc_itersolve_do_log(&local_cfg, 0, norm_r);
442
443 if (LMMC_REAL_CMP(&norm_r, &threshold) <= 0) {
444 converged = 1;
445 iter_count = 0;
446 goto cleanup;
447 }
448
449 st = lmmc_apply_precond_or_identity(precond, &r, &z);
450 if (st != LMMC_STATUS_OK) goto cleanup;
451
452 st = lmmc_vec_dot_checked(&r, &z, &rho);
453 if (st != LMMC_STATUS_OK) goto cleanup;
454
455 LMMC_REAL_ABS(&abs_val, &rho);
456 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
457 st = LMMC_STATUS_NUMERICAL_FAILURE;
458 goto cleanup;
459 }
460
461 st = lmmc_vec_copy(&z, &p);
462 if (st != LMMC_STATUS_OK) goto cleanup;
463
464 {
465 size_t iter = 0;
466 for (iter = 0; iter < local_cfg.max_iter; ++iter) {
467 size_t i = 0;
468
469 st = lmmc_sparse_mat_vec_mul(a, &p, &ap);
470 if (st != LMMC_STATUS_OK) goto cleanup;
471
472 st = lmmc_vec_dot_checked(&p, &ap, &denom);
473 if (st != LMMC_STATUS_OK) goto cleanup;
474
475 LMMC_REAL_ABS(&abs_val, &denom);
476 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
477 st = LMMC_STATUS_NUMERICAL_FAILURE;
478 goto cleanup;
479 }
480
481 LMMC_REAL_DIV(&alpha, &rho, &denom);
482 if (!lmmc_is_finite_number(alpha)) {
483 st = LMMC_STATUS_NUMERICAL_FAILURE;
484 goto cleanup;
485 }
486
487 for (i = 0; i < x->size; ++i) {
488 LMMC_REAL_MUL(&tmp_mul, &alpha, &p.data[i]);
489 LMMC_REAL_ADD(&tmp_add, &x->data[i], &tmp_mul);
490 LMMC_REAL_SET(&x->data[i], &tmp_add);
491 if (!lmmc_is_finite_number(x->data[i])) {
492 st = LMMC_STATUS_NUMERICAL_FAILURE;
493 goto cleanup;
494 }
495 }
496
497
498 {
499 lmmc_real_t neg_alpha; LMMC_REAL_INIT(&neg_alpha);
500 LMMC_REAL_NEG(&neg_alpha, &alpha);
501 st = lmmc_vec_axpy(neg_alpha, &ap, &r);
502 LMMC_REAL_CLEAR(&neg_alpha);
503 if (st != LMMC_STATUS_OK) goto cleanup;
504 }
505
506 st = lmmc_vec_norm2_checked(&r, &norm_r);
507 if (st != LMMC_STATUS_OK) goto cleanup;
508
509 iter_count = iter + 1;
510 if (out_result != NULL) {
511 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
512 }
513
514 lmmc_itersolve_do_log(&local_cfg, iter_count, norm_r);
515
516 if (LMMC_REAL_CMP(&norm_r, &threshold) <= 0) {
517 converged = 1;
518 break;
519 }
520
521 st = lmmc_apply_precond_or_identity(precond, &r, &z);
522 if (st != LMMC_STATUS_OK) goto cleanup;
523
524 st = lmmc_vec_dot_checked(&r, &z, &rho_new);
525 if (st != LMMC_STATUS_OK) goto cleanup;
526
527 LMMC_REAL_ABS(&abs_val, &rho_new);
528 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
529 st = LMMC_STATUS_NUMERICAL_FAILURE;
530 goto cleanup;
531 }
532
533 LMMC_REAL_DIV(&beta, &rho_new, &rho);
534 if (!lmmc_is_finite_number(beta)) {
535 st = LMMC_STATUS_NUMERICAL_FAILURE;
536 goto cleanup;
537 }
538
539 for (i = 0; i < p.size; ++i) {
540 LMMC_REAL_MUL(&tmp_mul, &beta, &p.data[i]);
541 LMMC_REAL_ADD(&tmp_add, &z.data[i], &tmp_mul);
542 LMMC_REAL_SET(&p.data[i], &tmp_add);
543 if (!lmmc_is_finite_number(p.data[i])) {
544 st = LMMC_STATUS_NUMERICAL_FAILURE;
545 goto cleanup;
546 }
547 }
548
549 LMMC_REAL_SET(&rho, &rho_new);
550 LMMC_REAL_ABS(&abs_val, &rho);
551 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
552 st = LMMC_STATUS_NUMERICAL_FAILURE;
553 goto cleanup;
554 }
555 }
556 }
557
558cleanup:
559 if (out_result != NULL) {
560 out_result->converged = converged;
561 out_result->num_iter = iter_count;
562 if (lmmc_is_finite_number(norm_r)) {
563 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
564 }
565 }
566
567 LMMC_REAL_CLEAR(&abs_val);
568 LMMC_REAL_CLEAR(&eps_30);
569 LMMC_REAL_CLEAR(&zero);
570 LMMC_REAL_CLEAR(&tmp_add);
571 LMMC_REAL_CLEAR(&tmp_mul);
572 LMMC_REAL_CLEAR(&beta);
573 LMMC_REAL_CLEAR(&rho_new);
574 LMMC_REAL_CLEAR(&alpha);
575 LMMC_REAL_CLEAR(&denom);
576 LMMC_REAL_CLEAR(&rho);
577 LMMC_REAL_CLEAR(&threshold);
578 LMMC_REAL_CLEAR(&norm_r);
579 LMMC_REAL_CLEAR(&norm_b);
580
581 lmmc_vec_destroy(&ap);
582 lmmc_vec_destroy(&p);
583 lmmc_vec_destroy(&z);
584 lmmc_vec_destroy(&r);
585
586 return st;
587}

589static lmmc_status_t lmmc_bicgstab_compute_alpha(
590 const lmmc_sparse_mat_t* a, const lmmc_precond_t* precond,
591 const lmmc_vec_t* p, lmmc_vec_t* y, lmmc_vec_t* v,
592 const lmmc_vec_t* r_hat, lmmc_real_t rho_hat, lmmc_real_t eps_30,
593 lmmc_real_t* out_alpha
594) {
595 lmmc_status_t st = LMMC_STATUS_OK;
596 lmmc_real_t denom; LMMC_REAL_INIT(&denom);
597 lmmc_real_t abs_val; LMMC_REAL_INIT(&abs_val);
598
599 st = lmmc_apply_precond_or_identity(precond, p, y);
600 if (st != LMMC_STATUS_OK) goto end;
601
602 st = lmmc_sparse_mat_vec_mul(a, y, v);
603 if (st != LMMC_STATUS_OK) goto end;
604
605 st = lmmc_vec_dot_checked(r_hat, v, &denom);
606 if (st != LMMC_STATUS_OK) goto end;
607
608 LMMC_REAL_ABS(&abs_val, &denom);
609 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
610 st = LMMC_STATUS_NUMERICAL_FAILURE; goto end;
611 }
612
613 LMMC_REAL_DIV(out_alpha, &rho_hat, &denom);
614 if (!lmmc_is_finite_number(*out_alpha)) {
615 st = LMMC_STATUS_NUMERICAL_FAILURE; goto end;
616 }
617
618end:
619 LMMC_REAL_CLEAR(&abs_val);
620 LMMC_REAL_CLEAR(&denom);
621 return st;
622}

624static lmmc_status_t lmmc_bicgstab_compute_omega(
625 const lmmc_sparse_mat_t* a, const lmmc_precond_t* precond,
626 const lmmc_vec_t* s, lmmc_vec_t* z, lmmc_vec_t* t,
627 lmmc_real_t eps_30, lmmc_real_t* out_omega

631 lmmc_real_t abs_val; LMMC_REAL_INIT(&abs_val);
632

635
636 st = lmmc_sparse_mat_vec_mul(a, z, t);
637 if (st != LMMC_STATUS_OK) goto end;
638
639 st = lmmc_vec_dot_checked(t, s, &ts);
640 if (st != LMMC_STATUS_OK) goto end;
641
642 st = lmmc_vec_dot_checked(t, t, &tt);
643 if (st != LMMC_STATUS_OK) goto end;
644
645 LMMC_REAL_ABS(&abs_val, &tt);
646 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
647 st = LMMC_STATUS_NUMERICAL_FAILURE; goto end;
648 }
649
650 LMMC_REAL_DIV(out_omega, &ts, &tt);
651 if (!lmmc_is_finite_number(*out_omega)) {
652 st = LMMC_STATUS_NUMERICAL_FAILURE; goto end;
653 }
654
655end:
656 LMMC_REAL_CLEAR(&abs_val);
657 LMMC_REAL_CLEAR(&tt);
658 LMMC_REAL_CLEAR(&ts);
659 return st;

662lmmc_status_t lmmc_bicgstab_solve(
663 const lmmc_sparse_mat_t* a,
664 const lmmc_vec_t* b,
665 const lmmc_precond_t* precond,
666 const lmmc_itersolve_config_t* cfg,
667 lmmc_vec_t* x,
668 lmmc_itersolve_result_t* out_result
669) {
670 lmmc_itersolve_config_t local_cfg = {0};
671 lmmc_vec_t r = {0}, r_hat = {0}, p = {0}, v = {0}, s = {0}, t = {0}, y = {0}, z = {0}, ax = {0};
672 lmmc_status_t st = LMMC_STATUS_OK;
673
674 lmmc_real_t norm_b; LMMC_REAL_INIT(&norm_b); LMMC_REAL_SET_D(&norm_b, 0.0);
675 lmmc_real_t norm_r; LMMC_REAL_INIT(&norm_r); LMMC_REAL_SET_D(&norm_r, 0.0);
676 lmmc_real_t threshold; LMMC_REAL_INIT(&threshold); LMMC_REAL_SET_D(&threshold, 0.0);
677 lmmc_real_t rho; LMMC_REAL_INIT(&rho); LMMC_REAL_SET_D(&rho, 1.0);
678 lmmc_real_t alpha; LMMC_REAL_INIT(&alpha); LMMC_REAL_SET_D(&alpha, 1.0);
679 lmmc_real_t omega; LMMC_REAL_INIT(&omega); LMMC_REAL_SET_D(&omega, 1.0);
680 lmmc_real_t rho_hat; LMMC_REAL_INIT(&rho_hat); LMMC_REAL_SET_D(&rho_hat, 0.0);
681 lmmc_real_t beta; LMMC_REAL_INIT(&beta); LMMC_REAL_SET_D(&beta, 0.0);
682 lmmc_real_t denom; LMMC_REAL_INIT(&denom); LMMC_REAL_SET_D(&denom, 0.0);
683 lmmc_real_t ts; LMMC_REAL_INIT(&ts); LMMC_REAL_SET_D(&ts, 0.0);
684 lmmc_real_t tt; LMMC_REAL_INIT(&tt); LMMC_REAL_SET_D(&tt, 0.0);
685 lmmc_real_t norm_s; LMMC_REAL_INIT(&norm_s); LMMC_REAL_SET_D(&norm_s, 0.0);
686
687 lmmc_real_t tmp_mul; LMMC_REAL_INIT(&tmp_mul); LMMC_REAL_SET_D(&tmp_mul, 0.0);
688 lmmc_real_t tmp_add; LMMC_REAL_INIT(&tmp_add); LMMC_REAL_SET_D(&tmp_add, 0.0);
689 lmmc_real_t tmp_sub; LMMC_REAL_INIT(&tmp_sub); LMMC_REAL_SET_D(&tmp_sub, 0.0);
690 lmmc_real_t zero; LMMC_REAL_INIT(&zero); LMMC_REAL_SET_D(&zero, 0.0);
691 lmmc_real_t eps_30; LMMC_REAL_INIT(&eps_30); LMMC_REAL_SET_D(&eps_30, 1e-30);
692 lmmc_real_t abs_val; LMMC_REAL_INIT(&abs_val); LMMC_REAL_SET_D(&abs_val, 0.0);
693
694 size_t iter_count = 0;
695 int converged = 0;
696
697 st = validate_and_init_itersolve_config(a, b, precond, cfg, x, &local_cfg, out_result);
698 if (st != LMMC_STATUS_OK) goto cleanup;
699
700 st = lmmc_vec_create(b->size, &r); if (st != LMMC_STATUS_OK) goto cleanup;
701 st = lmmc_vec_create(b->size, &r_hat); if (st != LMMC_STATUS_OK) goto cleanup;
702 st = lmmc_vec_create(b->size, &p); if (st != LMMC_STATUS_OK) goto cleanup;
703 st = lmmc_vec_create(b->size, &v); if (st != LMMC_STATUS_OK) goto cleanup;
704 st = lmmc_vec_create(b->size, &s); if (st != LMMC_STATUS_OK) goto cleanup;
705 st = lmmc_vec_create(b->size, &t); if (st != LMMC_STATUS_OK) goto cleanup;
706 st = lmmc_vec_create(b->size, &y); if (st != LMMC_STATUS_OK) goto cleanup;
707 st = lmmc_vec_create(b->size, &z); if (st != LMMC_STATUS_OK) goto cleanup;
708 st = lmmc_vec_create(b->size, &ax); if (st != LMMC_STATUS_OK) goto cleanup;
709
710 st = lmmc_sparse_mat_vec_mul(a, x, &ax);
711 if (st != LMMC_STATUS_OK) goto cleanup;
712
713 {
714 size_t i = 0;
715 for (i = 0; i < b->size; ++i) {
716 LMMC_REAL_SUB(&r.data[i], &b->data[i], &ax.data[i]);
717 LMMC_REAL_SET_D(&p.data[i], 0.0);
718 LMMC_REAL_SET_D(&v.data[i], 0.0);
719 }
720 }
721
722 st = lmmc_vec_copy(&r, &r_hat);
723 if (st != LMMC_STATUS_OK) goto cleanup;
724
725 st = lmmc_vec_norm2_checked(b, &norm_b);
726 if (st != LMMC_STATUS_OK) goto cleanup;
727 st = lmmc_vec_norm2_checked(&r, &norm_r);
728 if (st != LMMC_STATUS_OK) goto cleanup;
729
730 LMMC_REAL_MUL(&tmp_mul, &local_cfg.rel_tol, &norm_b);
731 LMMC_REAL_ADD(&threshold, &local_cfg.abs_tol, &tmp_mul);
732
733 if (!lmmc_is_finite_number(threshold)) {
734 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
735 }
736
737 if (out_result != NULL) {
738 LMMC_REAL_SET(&out_result->initial_residual_norm, &norm_r);
739 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
740 }
741
742 lmmc_itersolve_do_log(&local_cfg, 0, norm_r);
743
744 if (LMMC_REAL_CMP(&norm_r, &threshold) <= 0) {
745 converged = 1;
746 iter_count = 0;
747 goto cleanup;
748 }
749
750 {
751 size_t iter = 0;
752 for (iter = 0; iter < local_cfg.max_iter; ++iter) {
753 size_t i = 0;
754
755 st = lmmc_vec_dot_checked(&r_hat, &r, &rho_hat);
756 if (st != LMMC_STATUS_OK) goto cleanup;
757
758 LMMC_REAL_ABS(&abs_val, &rho_hat);
759 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
760 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
761 }
762
763 if (iter == 0) {
764 st = lmmc_vec_copy(&r, &p);
765 if (st != LMMC_STATUS_OK) goto cleanup;
766 } else {
767 LMMC_REAL_ABS(&abs_val, &omega);
768 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
769 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
770 }
771
772 lmmc_real_t tmp_div1; LMMC_REAL_INIT(&tmp_div1);
773 lmmc_real_t tmp_div2; LMMC_REAL_INIT(&tmp_div2);
774 LMMC_REAL_DIV(&tmp_div1, &rho_hat, &rho);
775 LMMC_REAL_DIV(&tmp_div2, &alpha, &omega);
776 LMMC_REAL_MUL(&beta, &tmp_div1, &tmp_div2);
777 LMMC_REAL_CLEAR(&tmp_div2);
778 LMMC_REAL_CLEAR(&tmp_div1);
779
780 if (!lmmc_is_finite_number(beta)) {
781 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
782 }
783
784 for (i = 0; i < p.size; ++i) {
785 LMMC_REAL_MUL(&tmp_mul, &omega, &v.data[i]);
786 LMMC_REAL_SUB(&tmp_sub, &p.data[i], &tmp_mul);
787 LMMC_REAL_MUL(&tmp_mul, &beta, &tmp_sub);
788 LMMC_REAL_ADD(&tmp_add, &r.data[i], &tmp_mul);
789 LMMC_REAL_SET(&p.data[i], &tmp_add);
790
791 if (!lmmc_is_finite_number(p.data[i])) {
792 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
793 }
794 }
795 }
796
797 st = lmmc_bicgstab_compute_alpha(a, precond, &p, &y, &v, &r_hat, rho_hat, eps_30, &alpha);
798 if (st != LMMC_STATUS_OK) goto cleanup;
799
800
801 st = lmmc_vec_copy(&r, &s);
802 if (st != LMMC_STATUS_OK) goto cleanup;
803 {
804 lmmc_real_t neg_alpha; LMMC_REAL_INIT(&neg_alpha);
805 LMMC_REAL_NEG(&neg_alpha, &alpha);
806 st = lmmc_vec_axpy(neg_alpha, &v, &s);
807 LMMC_REAL_CLEAR(&neg_alpha);
808 if (st != LMMC_STATUS_OK) goto cleanup;
809 }
810
811 st = lmmc_vec_norm2_checked(&s, &norm_s);
812 if (st != LMMC_STATUS_OK) goto cleanup;
813
814 if (LMMC_REAL_CMP(&norm_s, &threshold) <= 0) {
815 for (i = 0; i < x->size; ++i) {
816 LMMC_REAL_MUL(&tmp_mul, &alpha, &y.data[i]);
817 LMMC_REAL_ADD(&tmp_add, &x->data[i], &tmp_mul);
818 LMMC_REAL_SET(&x->data[i], &tmp_add);
819 if (!lmmc_is_finite_number(x->data[i])) {
820 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
821 }
822 }
823 LMMC_REAL_SET(&norm_r, &norm_s);
824 converged = 1;
825 iter_count = iter + 1;
826 if (out_result != NULL) {
827 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
828 }
829 lmmc_itersolve_do_log(&local_cfg, iter_count, norm_r);
830 break;
831 }
832
833 st = lmmc_bicgstab_compute_omega(a, precond, &s, &z, &t, eps_30, &omega);
834 if (st != LMMC_STATUS_OK) goto cleanup;
835
836 for (i = 0; i < x->size; ++i) {
837 LMMC_REAL_MUL(&tmp_mul, &alpha, &y.data[i]);
838 lmmc_real_t tmp_mul2; LMMC_REAL_INIT(&tmp_mul2);
839 LMMC_REAL_MUL(&tmp_mul2, &omega, &z.data[i]);
840
841 LMMC_REAL_ADD(&tmp_add, &tmp_mul, &tmp_mul2);
842 lmmc_real_t tmp_add2; LMMC_REAL_INIT(&tmp_add2);
843 LMMC_REAL_ADD(&tmp_add2, &x->data[i], &tmp_add);
844 LMMC_REAL_SET(&x->data[i], &tmp_add2);
845 LMMC_REAL_CLEAR(&tmp_add2);
846 LMMC_REAL_CLEAR(&tmp_mul2);
847
848 if (!lmmc_is_finite_number(x->data[i])) {
849 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
850 }
851 }
852
853
854 st = lmmc_vec_copy(&s, &r);
855 if (st != LMMC_STATUS_OK) goto cleanup;
856 {
857 lmmc_real_t neg_omega; LMMC_REAL_INIT(&neg_omega);
858 LMMC_REAL_NEG(&neg_omega, &omega);
859 st = lmmc_vec_axpy(neg_omega, &t, &r);
860 LMMC_REAL_CLEAR(&neg_omega);
861 if (st != LMMC_STATUS_OK) goto cleanup;
862 }
863
864 st = lmmc_vec_norm2_checked(&r, &norm_r);
865 if (st != LMMC_STATUS_OK) goto cleanup;
866
867 iter_count = iter + 1;
868 if (out_result != NULL) {
869 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
870 }
871
872 lmmc_itersolve_do_log(&local_cfg, iter_count, norm_r);
873
874 if (LMMC_REAL_CMP(&norm_r, &threshold) <= 0) {
875 converged = 1;
876 break;
877 }
878
879 LMMC_REAL_ABS(&abs_val, &omega);
880 if (LMMC_REAL_CMP(&abs_val, &eps_30) <= 0) {
881 st = LMMC_STATUS_NUMERICAL_FAILURE; goto cleanup;
882 }
883
884 LMMC_REAL_SET(&rho, &rho_hat);
885 }
886 }
887
888cleanup:
889 if (out_result != NULL) {
890 out_result->converged = converged;
891 out_result->num_iter = iter_count;
892 if (lmmc_is_finite_number(norm_r)) {
893 LMMC_REAL_SET(&out_result->final_residual_norm, &norm_r);
894 }
895 }
896
897 LMMC_REAL_CLEAR(&abs_val);
898 LMMC_REAL_CLEAR(&eps_30);
899 LMMC_REAL_CLEAR(&zero);
900 LMMC_REAL_CLEAR(&tmp_sub);
901 LMMC_REAL_CLEAR(&tmp_add);
902 LMMC_REAL_CLEAR(&tmp_mul);
903 LMMC_REAL_CLEAR(&norm_s);
904 LMMC_REAL_CLEAR(&tt);
905 LMMC_REAL_CLEAR(&ts);
906 LMMC_REAL_CLEAR(&denom);
907 LMMC_REAL_CLEAR(&beta);
908 LMMC_REAL_CLEAR(&rho_hat);
909 LMMC_REAL_CLEAR(&omega);
910 LMMC_REAL_CLEAR(&alpha);
911 LMMC_REAL_CLEAR(&rho);
912 LMMC_REAL_CLEAR(&threshold);
913 LMMC_REAL_CLEAR(&norm_r);
914 LMMC_REAL_CLEAR(&norm_b);
915
916 lmmc_vec_destroy(&ax);
917 lmmc_vec_destroy(&z);
918 lmmc_vec_destroy(&y);
919 lmmc_vec_destroy(&t);
920 lmmc_vec_destroy(&s);
921 lmmc_vec_destroy(&v);
922 lmmc_vec_destroy(&p);
923 lmmc_vec_destroy(&r_hat);
924 lmmc_vec_destroy(&r);
925
926 return st;
927}

1010
1011 st = lmmc_vec_norm2_checked(b, &norm_b);

1014 st = lmmc_sparse_mat_vec_mul(a, x, &ax);
1015 if (st != LMMC_STATUS_OK) goto cleanup;
1016 for (i = 0; i < n; ++i) {
1017 LMMC_REAL_SUB(&r.data[i], &b->data[i], &ax.data[i]);
1018 }

1020 st = lmmc_vec_norm2_checked(&r, &norm_r);
1021 if (st != LMMC_STATUS_OK) goto cleanup;
