11static int lmmc_mul_overflow_size(size_t a, size_t b, size_t* out) {
12 if (a == 0 || b == 0) {
13 *out = 0;
14 return 0;
15 }
16 if (a > ((size_t)-1) / b) {
17 return 1;
18 }
19 *out = a * b;
20 return 0;
21}

571 return LMMC_STATUS_DIMENSION_MISMATCH;
572 }
573
574 if (axis == 0) {
575 for (j = 0; j < tensor->dim1; ++j) {
576 for (k = 0; k < tensor->dim2; ++k) {
577 LMMC_REAL_SET_D(&sum, 0.0);
578 for (i = 0; i < tensor->dim0; ++i) {
579 lmmc_real_t* v = &tensor->data[i * tensor->stride0 + j * tensor->stride1 + k * tensor->stride2];
580 if (!lmmc_is_finite_number(v)) {
581 LMMC_REAL_CLEAR(&sum);
582 LMMC_REAL_CLEAR(&tmp_sum);
583 return LMMC_STATUS_NUMERICAL_FAILURE;
584 }
585 LMMC_REAL_ADD(&tmp_sum, &sum, v);
586 LMMC_REAL_SET(&sum, &tmp_sum);
587 if (!lmmc_is_finite_number(&sum)) {
588 LMMC_REAL_CLEAR(&sum);
589 LMMC_REAL_CLEAR(&tmp_sum);
590 return LMMC_STATUS_NUMERICAL_FAILURE;
591 }
592 }
593 LMMC_REAL_SET(&out_matrix->data[j * out_matrix->stride + k], &sum);
594 }
595 }
596 } else if (axis == 1) {
597 for (i = 0; i < tensor->dim0; ++i) {
598 for (k = 0; k < tensor->dim2; ++k) {
599 LMMC_REAL_SET_D(&sum, 0.0);
600 for (j = 0; j < tensor->dim1; ++j) {
601 lmmc_real_t* v = &tensor->data[i * tensor->stride0 + j * tensor->stride1 + k * tensor->stride2];
602 if (!lmmc_is_finite_number(v)) {
603 LMMC_REAL_CLEAR(&sum);
604 LMMC_REAL_CLEAR(&tmp_sum);
605 return LMMC_STATUS_NUMERICAL_FAILURE;
606 }
607 LMMC_REAL_ADD(&tmp_sum, &sum, v);
608 LMMC_REAL_SET(&sum, &tmp_sum);
609 if (!lmmc_is_finite_number(&sum)) {
610 LMMC_REAL_CLEAR(&sum);
611 LMMC_REAL_CLEAR(&tmp_sum);
612 return LMMC_STATUS_NUMERICAL_FAILURE;
613 }
614 }
615 LMMC_REAL_SET(&out_matrix->data[i * out_matrix->stride + k], &sum);
616 }
617 }
618 } else {
619 for (i = 0; i < tensor->dim0; ++i) {
620 for (j = 0; j < tensor->dim1; ++j) {
621 LMMC_REAL_SET_D(&sum, 0.0);
622 for (k = 0; k < tensor->dim2; ++k) {
623 lmmc_real_t* v = &tensor->data[i * tensor->stride0 + j * tensor->stride1 + k * tensor->stride2];
624 if (!lmmc_is_finite_number(v)) {
625 LMMC_REAL_CLEAR(&sum);
626 LMMC_REAL_CLEAR(&tmp_sum);
627 return LMMC_STATUS_NUMERICAL_FAILURE;
628 }
629 LMMC_REAL_ADD(&tmp_sum, &sum, v);
630 LMMC_REAL_SET(&sum, &tmp_sum);
631 if (!lmmc_is_finite_number(&sum)) {
632 LMMC_REAL_CLEAR(&sum);
633 LMMC_REAL_CLEAR(&tmp_sum);
634 return LMMC_STATUS_NUMERICAL_FAILURE;
635 }
636 }
637 LMMC_REAL_SET(&out_matrix->data[i * out_matrix->stride + j], &sum);
638 }
639 }
640 }
641
642 LMMC_REAL_CLEAR(&sum);
643 LMMC_REAL_CLEAR(&tmp_sum);
644 return LMMC_STATUS_OK;
645}

759 size_t i;
760 size_t expected_stride;
761 if (t == NULL || t->ndim == 0) return 0;
762
763 expected_stride = 1;
764 for (i = t->ndim; i > 0; --i) {
765 if (t->strides[i - 1] != expected_stride) return 0;
766 expected_stride *= t->dims[i - 1];
767 }

771lmmc_status_t lmmc_tensor_create(size_t ndim, const size_t* dims, lmmc_tensor_nd_t* out) {
772 size_t total = 1;
773 size_t bytes = 0;
774 size_t i;
775 lmmc_real_t* data = NULL;
776
777 if (out == NULL || dims == NULL || ndim == 0 || ndim > LMMC_TENSOR_MAX_NDIM) {
778 return LMMC_STATUS_INVALID_ARGUMENT;
779 }
780
781 for (i = 0; i < ndim; ++i) {
782 if (dims[i] == 0) {
783 return LMMC_STATUS_INVALID_ARGUMENT;
784 }
785 /* Overflow check */
786 if (total > ((size_t)-1) / dims[i]) {
787 return LMMC_STATUS_INVALID_ARGUMENT;
788 }
789 total *= dims[i];
790 }
791
792 /* Overflow check for bytes */
793 if (total > ((size_t)-1) / sizeof(lmmc_real_t)) {
794 return LMMC_STATUS_INVALID_ARGUMENT;
795 }
796 bytes = total * sizeof(lmmc_real_t);
797
798 data = (lmmc_real_t*)lmmc_alloc(bytes);
799 if (data == NULL) {
800 return LMMC_STATUS_ALLOCATION_FAILED;
801 }
802 memset(data, 0, bytes);
803
804 for (i = 0; i < total; ++i) {
805 LMMC_REAL_INIT(&data[i]);
806 }
807
808 out->ndim = ndim;
809 memset(out->dims, 0, sizeof(out->dims));
810 memset(out->strides, 0, sizeof(out->strides));
811
812 for (i = 0; i < ndim; ++i) {
813 out->dims[i] = dims[i];
814 }
815
816 /* Row-major strides */
817 out->strides[ndim - 1] = 1;
818 for (i = ndim - 1; i > 0; --i) {
819 out->strides[i - 1] = out->strides[i] * out->dims[i];
820 }
821
822 out->data = data;
823 out->owns_data = 1;
824 return LMMC_STATUS_OK;
825}

871lmmc_status_t lmmc_tensor_permute(const lmmc_tensor_nd_t* in, const size_t* perm, lmmc_tensor_nd_t* out) {
872 size_t i, ndim;
873 size_t new_dims[LMMC_TENSOR_MAX_NDIM];
874 int seen[LMMC_TENSOR_MAX_NDIM] = {0};
875 size_t total = 1;
876 size_t bytes;
877 lmmc_real_t* data = NULL;
878 size_t idx[LMMC_TENSOR_MAX_NDIM] = {0};
879
880 if (in == NULL || in->data == NULL || perm == NULL || out == NULL) {
881 if (out != NULL) { out->data = NULL; out->owns_data = 0; }
882 return LMMC_STATUS_INVALID_ARGUMENT;
883 }
884 ndim = in->ndim;
885 if (ndim == 0 || ndim > LMMC_TENSOR_MAX_NDIM) {
886 out->data = NULL; out->owns_data = 0;
887 return LMMC_STATUS_INVALID_ARGUMENT;
888 }
889
890 /* Validate permutation */
891 for (i = 0; i < ndim; ++i) {
892 if (perm[i] >= ndim || seen[perm[i]]) {
893 out->data = NULL; out->owns_data = 0;
894 return LMMC_STATUS_INVALID_ARGUMENT;
895 }
896 seen[perm[i]] = 1;
897 new_dims[i] = in->dims[perm[i]];
898 }
899
900 /* Compute total elements */
901 for (i = 0; i < ndim; ++i) {
902 total *= new_dims[i];
903 }
904 bytes = total * sizeof(lmmc_real_t);
905
906 data = (lmmc_real_t*)lmmc_alloc(bytes);
907 if (data == NULL) {
908 out->data = NULL; out->owns_data = 0;
909 return LMMC_STATUS_ALLOCATION_FAILED;
910 }
911 memset(data, 0, bytes);
912 for (i = 0; i < total; ++i) {
913 LMMC_REAL_INIT(&data[i]);
914 }
915
916 /* Set up output tensor metadata */
917 out->ndim = ndim;
918 memset(out->dims, 0, sizeof(out->dims));
919 memset(out->strides, 0, sizeof(out->strides));
920 for (i = 0; i < ndim; ++i) {
921 out->dims[i] = new_dims[i];
922 }
923 out->strides[ndim - 1] = 1;
924 for (i = ndim - 1; i > 0; --i) {
925 out->strides[i - 1] = out->strides[i] * out->dims[i];
926 }
927 out->data = data;
928 out->owns_data = 1;
929
930 /* Copy data with permuted indexing */
931 /* Iterate over all elements of the input tensor */
932 memset(idx, 0, sizeof(idx));
933 for (;;) {
934 /* Compute source offset */
935 size_t src_offset = 0;
936 size_t dst_offset = 0;
937 size_t perm_idx[LMMC_TENSOR_MAX_NDIM];
938
939 for (i = 0; i < ndim; ++i) {
940 src_offset += idx[i] * in->strides[i];
941 }
942 /* Compute permuted index: out[perm_idx] where perm_idx[j] = idx[perm[j]] */
943 for (i = 0; i < ndim; ++i) {
944 perm_idx[i] = idx[perm[i]];
945 }
946 for (i = 0; i < ndim; ++i) {
947 dst_offset += perm_idx[i] * out->strides[i];
948 }
949
950 LMMC_REAL_SET(&out->data[dst_offset], &in->data[src_offset]);
951
952 /* Increment multi-index (input order) */
953 i = ndim;
954 while (i > 0) {
955 --i;
956 idx[i]++;
957 if (idx[i] < in->dims[i]) break;
958 idx[i] = 0;
959 if (i == 0) goto permute_done;
960 }
961 }
962permute_done:
963 return LMMC_STATUS_OK;
964}

966lmmc_status_t lmmc_tensor_contract(const lmmc_tensor_nd_t* a, const lmmc_tensor_nd_t* b,
967 const size_t* axes_a, const size_t* axes_b, size_t naxes, lmmc_tensor_nd_t* out) {
968
969 size_t i, j, k;
970 size_t out_ndim;
971 size_t out_dims[LMMC_TENSOR_MAX_NDIM];
972 /* Track which axes are contracted */
973 int a_contracted[LMMC_TENSOR_MAX_NDIM] = {0};
974 int b_contracted[LMMC_TENSOR_MAX_NDIM] = {0};
975 size_t a_free_axes[LMMC_TENSOR_MAX_NDIM];
976 size_t b_free_axes[LMMC_TENSOR_MAX_NDIM];
977 size_t n_a_free = 0, n_b_free = 0;
978 size_t total_out = 1;
979 size_t bytes;
980 lmmc_real_t* data = NULL;
981
982 if (a == NULL || a->data == NULL || b == NULL || b->data == NULL || out == NULL) {
983 if (out != NULL) { out->data = NULL; out->owns_data = 0; }
984 return LMMC_STATUS_INVALID_ARGUMENT;
985 }
986 if (naxes > a->ndim || naxes > b->ndim) {
987 out->data = NULL; out->owns_data = 0;
988 return LMMC_STATUS_INVALID_ARGUMENT;
989 }
990 if (naxes > 0 && (axes_a == NULL || axes_b == NULL)) {
991 out->data = NULL; out->owns_data = 0;
992 return LMMC_STATUS_INVALID_ARGUMENT;
993 }
994
995 /* Validate axes */
996 for (i = 0; i < naxes; ++i) {
997 if (axes_a[i] >= a->ndim || axes_b[i] >= b->ndim) {
998 out->data = NULL; out->owns_data = 0;
999 return LMMC_STATUS_INVALID_ARGUMENT;
1000 }
1001 if (a_contracted[axes_a[i]] || b_contracted[axes_b[i]]) {
1002 out->data = NULL; out->owns_data = 0;
1003 return LMMC_STATUS_INVALID_ARGUMENT;
1004 }
1005 if (a->dims[axes_a[i]] != b->dims[axes_b[i]]) {
1006 out->data = NULL; out->owns_data = 0;
1007 return LMMC_STATUS_INVALID_ARGUMENT;
1008 }
1009 a_contracted[axes_a[i]] = 1;
1010 b_contracted[axes_b[i]] = 1;
1011 }
1012
1013 /* Determine free axes */
1014 for (i = 0; i < a->ndim; ++i) {
1015 if (!a_contracted[i]) {
1016 a_free_axes[n_a_free++] = i;
1017 }
1018 }
1019 for (i = 0; i < b->ndim; ++i) {
1020 if (!b_contracted[i]) {
1021 b_free_axes[n_b_free++] = i;
1022 }
1023 }
1024
1025 out_ndim = n_a_free + n_b_free;
1026 if (out_ndim == 0) out_ndim = 1; /* scalar result stored as 1-D tensor of size 1 */
1027 if (out_ndim > LMMC_TENSOR_MAX_NDIM) {
1028 out->data = NULL; out->owns_data = 0;
1029 return LMMC_STATUS_INVALID_ARGUMENT;
1030 }
1031
1032 /* Build output dims */
1033 j = 0;
1034 for (i = 0; i < n_a_free; ++i) {
1035 out_dims[j++] = a->dims[a_free_axes[i]];
1036 }
1037 for (i = 0; i < n_b_free; ++i) {
1038 out_dims[j++] = b->dims[b_free_axes[i]];
1039 }
1040 if (n_a_free + n_b_free == 0) {
1041 out_dims[0] = 1;
1042 }
1043
1044 /* Allocate output */
1045 total_out = 1;
1046 for (i = 0; i < out_ndim; ++i) {
1047 total_out *= out_dims[i];
1048 }
1049 bytes = total_out * sizeof(lmmc_real_t);
1050 data = (lmmc_real_t*)lmmc_alloc(bytes);
1051 if (data == NULL) {
1052 out->data = NULL; out->owns_data = 0;
1053 return LMMC_STATUS_ALLOCATION_FAILED;
1054 }
1055 memset(data, 0, bytes);
1056 for (i = 0; i < total_out; ++i) {
1057 LMMC_REAL_INIT(&data[i]);
1058 }
1059
1060 out->ndim = out_ndim;
1061 memset(out->dims, 0, sizeof(out->dims));
1062 memset(out->strides, 0, sizeof(out->strides));
1063 for (i = 0; i < out_ndim; ++i) {
1064 out->dims[i] = out_dims[i];
1065 }
1066 out->strides[out_ndim - 1] = 1;
1067 for (i = out_ndim - 1; i > 0; --i) {
1068 out->strides[i - 1] = out->strides[i] * out->dims[i];
1069 }
1070 out->data = data;
1071 out->owns_data = 1;
1072
1073 /* Perform contraction via nested loops */
1074 {
1075 /* Compute contraction dimension product */
1076 size_t contract_size = 1;
1077 for (i = 0; i < naxes; ++i) {
1078 contract_size *= a->dims[axes_a[i]];
1079 }
1080
1081 /* Iterate over all output elements */
1082 size_t out_idx[LMMC_TENSOR_MAX_NDIM] = {0};
1083 for (;;) {
1084 /* For this output element, sum over contracted indices */
1085 size_t contract_idx[LMMC_TENSOR_MAX_NDIM] = {0}; /* indices into contracted dims */
1086 lmmc_real_t sum;
1087 LMMC_REAL_INIT(&sum);
1088 LMMC_REAL_SET_D(&sum, 0.0);
1089
1090 for (k = 0; k < contract_size; ++k) {
1091 /* Compute a's full index */
1092 size_t a_full_idx[LMMC_TENSOR_MAX_NDIM] = {0};
1093 size_t b_full_idx[LMMC_TENSOR_MAX_NDIM] = {0};
1094 size_t a_offset = 0, b_offset = 0;
1095 lmmc_real_t va, vb, prod, new_sum;
1096 LMMC_REAL_INIT(&va);
1097 LMMC_REAL_INIT(&vb);
1098 LMMC_REAL_INIT(&prod);
1099 LMMC_REAL_INIT(&new_sum);
1100
1101 /* Fill a's free indices from output */
1102 for (i = 0; i < n_a_free; ++i) {
1103 a_full_idx[a_free_axes[i]] = out_idx[i];
1104 }
1105 /* Fill a's contracted indices */
1106 for (i = 0; i < naxes; ++i) {
1107 a_full_idx[axes_a[i]] = contract_idx[i];
1108 }
1109 /* Fill b's free indices from output */
1110 for (i = 0; i < n_b_free; ++i) {
1111 b_full_idx[b_free_axes[i]] = out_idx[n_a_free + i];
1112 }
1113 /* Fill b's contracted indices */
1114 for (i = 0; i < naxes; ++i) {
1115 b_full_idx[axes_b[i]] = contract_idx[i];
1116 }
1117
1118 /* Compute offsets */
1119 for (i = 0; i < a->ndim; ++i) {
1120 a_offset += a_full_idx[i] * a->strides[i];
1121 }
1122 for (i = 0; i < b->ndim; ++i) {
1123 b_offset += b_full_idx[i] * b->strides[i];
1124 }
1125
1126 LMMC_REAL_SET(&va, &a->data[a_offset]);
1127 LMMC_REAL_SET(&vb, &b->data[b_offset]);
1128 LMMC_REAL_MUL(&prod, &va, &vb);
1129 LMMC_REAL_ADD(&new_sum, &sum, &prod);
1130 LMMC_REAL_SET(&sum, &new_sum);
1131
1132 LMMC_REAL_CLEAR(&va);
1133 LMMC_REAL_CLEAR(&vb);
1134 LMMC_REAL_CLEAR(&prod);
1135 LMMC_REAL_CLEAR(&new_sum);
1136
1137 /* Increment contract_idx */
1138 if (naxes > 0) {
1139 size_t ci = naxes;
1140 while (ci > 0) {
1141 --ci;
1142 contract_idx[ci]++;
1143 if (contract_idx[ci] < a->dims[axes_a[ci]]) break;
1144 contract_idx[ci] = 0;
1145 if (ci == 0) break;
1146 }
1147 }
1148 }
1149
1150 /* Write sum to output */
1151 {
1152 size_t out_offset = 0;
1153 for (i = 0; i < out_ndim; ++i) {
1154 out_offset += out_idx[i] * out->strides[i];
1155 }
1156 LMMC_REAL_SET(&out->data[out_offset], &sum);
1157 }
1158 LMMC_REAL_CLEAR(&sum);
1159
1160 /* Increment out_idx */
1161 i = out_ndim;
1162 while (i > 0) {
1163 --i;
1164 out_idx[i]++;
1165 if (out_idx[i] < out->dims[i]) break;
1166 out_idx[i] = 0;
1167 if (i == 0) goto contract_done;
1168 }
1169 }
1170 }
1171contract_done:
1172 return LMMC_STATUS_OK;
1173}
