19static void lmmc_abs_inplace(lmmc_real_t* res, const lmmc_real_t* x) {
20 lmmc_real_t zero;

22 LMMC_REAL_SET_D(&zero, 0.0);

88 return LMMC_STATUS_NUMERICAL_FAILURE;

144lmmc_status_t lmmc_bisection_solve(
145 lmmc_scalar_func_t func,
146 void* user_data,
147 lmmc_real_t left,
148 lmmc_real_t right,
149 const lmmc_nonlinear_config_t* cfg,
150 lmmc_nonlinear_result_t* out_result
151) {
152 lmmc_nonlinear_config_t local_cfg = {0};
153 lmmc_real_t f_left = 0.0;
154 lmmc_real_t f_right = 0.0;
155 size_t iter = 0;
156
157 if (func == NULL || out_result == NULL) {
158 return LMMC_STATUS_INVALID_ARGUMENT;
159 }
160
161 lmmc_nonlinear_reset_result(out_result);
162
163 if (!lmmc_is_finite(&left) || !lmmc_is_finite(&right) || !(left < right)) {
164 return LMMC_STATUS_INVALID_ARGUMENT;
165 }
166
167 if (cfg != NULL) {
168 local_cfg = *cfg;
169 } else {
170 lmmc_status_t st_cfg = lmmc_nonlinear_default_config(&local_cfg);
171 if (st_cfg != LMMC_STATUS_OK) {
172 return st_cfg;
173 }
174 }
175
176 if (!lmmc_is_finite(&local_cfg.abs_tol) || !lmmc_is_finite(&local_cfg.rel_tol) ||
177 local_cfg.abs_tol < 0.0 || local_cfg.rel_tol < 0.0 || local_cfg.max_iter == 0) {
178 return LMMC_STATUS_INVALID_ARGUMENT;
179 }
180
181 f_left = func(left, user_data);
182 f_right = func(right, user_data);
183
184 if (!lmmc_is_finite(&f_left) || !lmmc_is_finite(&f_right)) {
185 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
186 return LMMC_STATUS_NUMERICAL_FAILURE;
187 }
188
189 out_result->root = left;
190 out_result->function_value = f_left;
191 out_result->residual_norm = lmmc_abs(f_left);
192
193 lmmc_nonlinear_do_log(&local_cfg, 0, out_result->root, out_result->function_value);
194
195 if (lmmc_abs(f_left) <= local_cfg.abs_tol) {
196 out_result->converged = 1;
197 return LMMC_STATUS_OK;
198 }
199
200 if (lmmc_abs(f_right) <= local_cfg.abs_tol) {
201 out_result->converged = 1;
202 out_result->root = right;
203 out_result->function_value = f_right;
204 out_result->residual_norm = lmmc_abs(f_right);
205 lmmc_nonlinear_do_log(&local_cfg, 0, out_result->root, out_result->function_value);
206 return LMMC_STATUS_OK;
207 }
208
209 if ((f_left > 0.0 && f_right > 0.0) || (f_left < 0.0 && f_right < 0.0)) {
210 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_INVALID_BRACKET;
211 return LMMC_STATUS_INVALID_ARGUMENT;
212 }
213
214 for (iter = 1; iter <= local_cfg.max_iter; ++iter) {
215 lmmc_real_t mid = 0.0;
216 lmmc_real_t f_mid = 0.0;
217 lmmc_real_t interval_width = 0.0;
218 lmmc_real_t x_scale = 0.0;
219 lmmc_real_t x_tol = 0.0;
220 lmmc_real_t sum_lr = 0.0;
221 lmmc_real_t half = 0.5;
222 lmmc_real_t abs_mid = 0.0;
223 lmmc_real_t one = 1.0;
224 lmmc_real_t half_width = 0.0;
225 lmmc_real_t tmp = 0.0;
226
227 LMMC_REAL_ADD(&sum_lr, &left, &right);
228 LMMC_REAL_MUL(&mid, &half, &sum_lr);
229 f_mid = func(mid, user_data);
230 LMMC_REAL_SUB(&interval_width, &right, &left);
231 abs_mid = lmmc_abs(mid);
232 x_scale = lmmc_max(abs_mid, one);
233 LMMC_REAL_MUL(&tmp, &local_cfg.rel_tol, &x_scale);
234 LMMC_REAL_ADD(&x_tol, &local_cfg.abs_tol, &tmp);
235
236 if (!lmmc_is_finite(&mid) || !lmmc_is_finite(&f_mid) ||
237 !lmmc_is_finite(&interval_width) || !lmmc_is_finite(&x_tol)) {
238 out_result->num_iter = iter;
239 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
240 return LMMC_STATUS_NUMERICAL_FAILURE;
241 }
242
243 out_result->num_iter = iter;
244 out_result->root = mid;
245 out_result->function_value = f_mid;
246 out_result->residual_norm = lmmc_abs(f_mid);
247
248 lmmc_nonlinear_do_log(&local_cfg, iter, out_result->root, out_result->function_value);
249
250 LMMC_REAL_MUL(&half_width, &half, &interval_width);
251 if (lmmc_abs(f_mid) <= local_cfg.abs_tol || half_width <= x_tol) {
252 out_result->converged = 1;
253 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NONE;
254 return LMMC_STATUS_OK;
255 }
256
257 if ((f_left > 0.0 && f_mid < 0.0) || (f_left < 0.0 && f_mid > 0.0)) {
258 right = mid;
259 f_right = f_mid;
260 } else {
261 left = mid;
262 f_left = f_mid;
263 }
264 }
265
266 out_result->converged = 0;
267 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_MAX_ITER;
268 return LMMC_STATUS_NUMERICAL_FAILURE;
269}

271lmmc_status_t lmmc_newton_solve(
272 lmmc_scalar_func_t func,
273 lmmc_scalar_dfunc_t dfunc,
274 void* user_data,
275 lmmc_real_t x0,
276 const lmmc_nonlinear_config_t* cfg,
277 lmmc_nonlinear_result_t* out_result
278) {
279 lmmc_nonlinear_config_t local_cfg = {0};
280 lmmc_real_t x = x0;
281 lmmc_real_t fx = 0.0;
282 size_t iter = 0;
283
284 if (func == NULL || out_result == NULL || !lmmc_is_finite(&x0)) {
285 return LMMC_STATUS_INVALID_ARGUMENT;
286 }
287
288 lmmc_nonlinear_reset_result(out_result);
289
290 {
291 lmmc_status_t st_cfg = lmmc_nonlinear_load_and_validate_config(cfg, &local_cfg);
292 if (st_cfg != LMMC_STATUS_OK) {
293 return st_cfg;
294 }
295 }
296
297 fx = func(x, user_data);
298 if (!lmmc_is_finite(&fx)) {
299 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
300 return LMMC_STATUS_NUMERICAL_FAILURE;
301 }
302
303 out_result->root = x;
304 out_result->function_value = fx;
305 out_result->residual_norm = lmmc_abs(fx);
306
307 lmmc_nonlinear_do_log(&local_cfg, 0, out_result->root, out_result->function_value);
308
309 if (out_result->residual_norm <= local_cfg.abs_tol) {
310 out_result->converged = 1;
311 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NONE;
312 return LMMC_STATUS_OK;
313 }
314
315 for (iter = 1; iter <= local_cfg.max_iter; ++iter) {
316 lmmc_real_t derivative = 0.0;
317 lmmc_real_t step = 0.0;
318 lmmc_real_t x_next = 0.0;
319 lmmc_real_t f_next = 0.0;
320 lmmc_real_t x_tol = 0.0;
321
322 if (dfunc != NULL) {
323 derivative = dfunc(x, user_data);
324 } else {
325 lmmc_real_t abs_x = lmmc_abs(x);
326 lmmc_real_t one = 1.0;
327 lmmc_real_t max_val = lmmc_max(abs_x, one);
328 lmmc_real_t h = 0.0;
329 lmmc_real_t f_plus = 0.0;
330 lmmc_real_t f_minus = 0.0;
331 lmmc_real_t x_plus_h = 0.0;
332 lmmc_real_t x_minus_h = 0.0;
333 lmmc_real_t f_diff = 0.0;
334 lmmc_real_t two_h = 0.0;
335 lmmc_real_t two = 2.0;
336
337 LMMC_REAL_MUL(&h, &local_cfg.derivative_step, &max_val);
338
339 if (!lmmc_is_finite(&h) || h <= 0.0) {
340 out_result->num_iter = iter;
341 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
342 return LMMC_STATUS_NUMERICAL_FAILURE;
343 }
344
345 LMMC_REAL_ADD(&x_plus_h, &x, &h);
346 LMMC_REAL_SUB(&x_minus_h, &x, &h);
347 f_plus = func(x_plus_h, user_data);
348 f_minus = func(x_minus_h, user_data);
349
350 if (!lmmc_is_finite(&f_plus) || !lmmc_is_finite(&f_minus)) {
351 out_result->num_iter = iter;
352 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
353 return LMMC_STATUS_NUMERICAL_FAILURE;
354 }
355
356 LMMC_REAL_SUB(&f_diff, &f_plus, &f_minus);
357 LMMC_REAL_MUL(&two_h, &two, &h);
358 LMMC_REAL_DIV(&derivative, &f_diff, &two_h);
359 }
360
361 if (!lmmc_is_finite(&derivative)) {
362 out_result->num_iter = iter;
363 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
364 return LMMC_STATUS_NUMERICAL_FAILURE;
365 }
366
367 {
368 /* Relative derivative threshold: max(|x| * DBL_EPSILON * 64, DBL_EPSILON * 64) */
369 lmmc_real_t abs_x_val = lmmc_abs(x);
370 lmmc_real_t deriv_floor = DBL_EPSILON * 64.0;
371 lmmc_real_t rel_threshold = abs_x_val * DBL_EPSILON * 64.0;
372 lmmc_real_t threshold = lmmc_max(rel_threshold, deriv_floor);
373 if (lmmc_abs(derivative) < threshold) {
374 out_result->num_iter = iter;
375 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_ZERO_DERIVATIVE;
376 return LMMC_STATUS_NUMERICAL_FAILURE;
377 }
378 }
379
380 LMMC_REAL_DIV(&step, &fx, &derivative);
381 LMMC_REAL_SUB(&x_next, &x, &step);
382
383 if (!lmmc_is_finite(&step) || !lmmc_is_finite(&x_next)) {
384 out_result->num_iter = iter;
385 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_SINGULAR_STEP;
386 return LMMC_STATUS_NUMERICAL_FAILURE;
387 }
388
389 f_next = func(x_next, user_data);
390 if (!lmmc_is_finite(&f_next)) {
391 out_result->num_iter = iter;
392 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
393 return LMMC_STATUS_NUMERICAL_FAILURE;
394 }
395
396 if (lmmc_nonlinear_x_tolerance(x_next, &local_cfg, &x_tol) != LMMC_STATUS_OK) {
397 out_result->num_iter = iter;
398 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NUMERICAL_ISSUE;
399 return LMMC_STATUS_NUMERICAL_FAILURE;
400 }
401
402 out_result->num_iter = iter;
403 out_result->root = x_next;
404 out_result->function_value = f_next;
405 out_result->residual_norm = lmmc_abs(f_next);
406
407 lmmc_nonlinear_do_log(&local_cfg, iter, out_result->root, out_result->function_value);
408
409 {
410 lmmc_real_t x_diff = 0.0;
411 LMMC_REAL_SUB(&x_diff, &x_next, &x);
412 if (out_result->residual_norm <= local_cfg.abs_tol || lmmc_abs(x_diff) <= x_tol) {
413 out_result->converged = 1;
414 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_NONE;
415 return LMMC_STATUS_OK;
416 }
417 }
418
419 if (lmmc_abs(step) < local_cfg.min_step) {
420 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_SINGULAR_STEP;
421 return LMMC_STATUS_NUMERICAL_FAILURE;
422 }
423
424 x = x_next;
425 fx = f_next;
426 }
427
428 out_result->converged = 0;
429 out_result->failure_reason = LMMC_NONLINEAR_FAILURE_MAX_ITER;
430 return LMMC_STATUS_NUMERICAL_FAILURE;
431}
