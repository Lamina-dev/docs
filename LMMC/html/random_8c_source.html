25static inline uint64_t splitmix64_next(uint64_t* state) {
26 uint64_t z = (*state += UINT64_C(0x9e3779b97f4a7c15));
27 z = (z ^ (z >> 30)) * UINT64_C(0xbf58476d1ce4e5b9);
28 z = (z ^ (z >> 27)) * UINT64_C(0x94d049bb133111eb);
29 return z ^ (z >> 31);
30}

37static uint64_t generate_default_seed(void) {
38 static _Atomic uint64_t seed_counter = 0;
39
40 uint64_t t = (uint64_t)time(NULL);
41 /* Use address of a local variable for ASLR entropy */
42 volatile int stack_var = 0;
43 uint64_t addr = (uint64_t)(uintptr_t)&stack_var;
44 uint64_t counter = atomic_fetch_add(&seed_counter, 1);
45
46 /* Mix all sources via SplitMix64 finalizer */
47 uint64_t mixed = t ^ addr ^ counter;
48 return splitmix64_next(&mixed);
49}

52static inline uint64_t xoshiro256ss_next(uint64_t* s) {
53 const uint64_t result = rotl(s[1] * 5, 7) * 9;
54 const uint64_t t = s[1] << 17;
55
56 s[2] ^= s[0];
57 s[3] ^= s[1];
58 s[1] ^= s[2];
59 s[0] ^= s[3];
60
61 s[2] ^= t;
62 s[3] = rotl(s[3], 45);
63
64 return result;
65}

159static void rng_apply_jump_poly(uint64_t* s, const uint64_t* poly) {
160 uint64_t s0 = 0, s1 = 0, s2 = 0, s3 = 0;
161 int i, b;
162
163 for (i = 0; i < 4; i++) {
164 for (b = 0; b < 64; b++) {
165 if (poly[i] & (UINT64_C(1) << b)) {
166 s0 ^= s[0];
167 s1 ^= s[1];
168 s2 ^= s[2];
169 s3 ^= s[3];
170 }
171 xoshiro256ss_next(s);
172 }
173 }
174
175 s[0] = s0;
176 s[1] = s1;
177 s[2] = s2;
178 s[3] = s3;
179}

235/* ============================================================
236 * Ziggurat normal sampler (256 rectangles)
237 * ============================================================
238 * Reference: Marsaglia & Tsang, "The Ziggurat Method for
239 * Generating Random Variables", JSS 2000.
240 *
241 * Tables are built for the right half of the standard normal
242 * (x >= 0). The sampler generates |x| and applies a random sign.
243 *
244 * Table layout:
245 * zig_xtab[0] = v/f(r) (width of base strip including tail)
246 * zig_xtab[1] = r (tail cutoff)
247 * zig_xtab[i] for i=2..255: decreasing x-coordinates
248 * zig_xtab[256] = 0 (peak of distribution)
249 *
250 * For 256 rectangles:
251 * r = 3.6541528853610088
252 * v = 0.00492867323399 (area of each rectangle in the half-normal)
253 */
254
255#define ZIG_N 256
256#define ZIG_R 3.6541528853610088
257#define ZIG_V 0.00492867323399
258
259static double zig_xtab[ZIG_N + 1]; /* x-coordinates of rectangle edges */
260static int zig_ready = 0;
261
262static inline double zig_pdf(double x) {
263 return exp(-0.5 * x * x);
264}
265
266static void zig_setup(void) {
267 int i;
268
269 if (zig_ready) return;
270
271 /* Build the table:
272 * xtab[1] = r (the tail start)
273 * xtab[i] for i=2..255: computed from equal-area property
274 * xtab[256] = 0 (the peak of the distribution)
275 * xtab[0] = v / f(r) (width of the base strip that includes the tail)
276 */
277 zig_xtab[256] = 0.0;
278 zig_xtab[1] = ZIG_R;
279
280 for (i = 2; i <= 255; i++) {
281 zig_xtab[i] = sqrt(-2.0 * log(ZIG_V / zig_xtab[i - 1] + zig_pdf(zig_xtab[i - 1])));
282 }
283 zig_xtab[0] = ZIG_V / zig_pdf(ZIG_R);
284
285 zig_ready = 1;
286}
287
292static double zig_sample_tail(uint64_t* state) {
293 double x, y, u1, u2;
294 for (;;) {
295 do {
296 u1 = u64_to_double01(xoshiro256ss_next(state));
297 } while (u1 == 0.0);
298 do {
299 u2 = u64_to_double01(xoshiro256ss_next(state));
300 } while (u2 == 0.0);
301
302 x = -log(u1) / ZIG_R;
303 y = -log(u2);
304
305 if (2.0 * y >= x * x) {
306 return x + ZIG_R;
307 }
308 }
309}

317static double ziggurat_rnor(uint64_t* state) {
318 uint64_t u, u2;
319 int i, sign;
320 double x;
321
322 if (!zig_ready) {
323 zig_setup();
324 }
325
326 for (;;) {
327 u = xoshiro256ss_next(state);
328 i = (int)(u & 0xFF); /* layer index: 0..255 */
329 sign = (u & 0x100) ? -1 : 1; /* bit 8 for sign */
330
331 /* Generate uniform x in [0, xtab[i]) using a fresh random number */
332 u2 = xoshiro256ss_next(state);
333 x = u64_to_double01(u2) * zig_xtab[i];
334
335 /* Fast accept: x < xtab[i+1] */
336 if (x < zig_xtab[i + 1]) {
337 return sign * x;
338 }
339
340 /* Layer 0 is special: it includes the tail */
341 if (i == 0) {
342 /* x is in [xtab[1], xtab[0]). Check if in rectangular part or tail */
343 if (x < zig_xtab[1]) {
344 return sign * x;
345 }
346 /* Need tail sample */
347 double tail = zig_sample_tail(state);
348 return sign * tail;
349 }
350
351 /* Wedge test: accept with probability (f(x) - f(xtab[i])) / (f(xtab[i+1]) - f(xtab[i])) */
352 {
353 double f_x = zig_pdf(x);
354 double f_outer = zig_pdf(zig_xtab[i]); /* f at outer edge (smaller f value) */
355 double f_inner = zig_pdf(zig_xtab[i + 1]); /* f at inner edge (larger f value) */
356 double u_wedge = u64_to_double01(xoshiro256ss_next(state));
357
358 if (u_wedge * (f_inner - f_outer) < (f_x - f_outer)) {
359 return sign * x;
360 }
361 }
362 }

430lmmc_status_t lmmc_rng_shuffle(
431 lmmc_rng_t* rng,
432 void* array,
433 size_t count,
434 size_t elem_size)
435{
436 size_t i, j;
437 unsigned char* arr;
438 unsigned char* tmp;
439
440 if (rng == NULL || array == NULL) {
441 return LMMC_STATUS_INVALID_ARGUMENT;
442 }
443 if (elem_size == 0) {
444 return LMMC_STATUS_INVALID_ARGUMENT;
445 }
446
447 if (count <= 1) {
448 return LMMC_STATUS_OK;
449 }
450
451 arr = (unsigned char*)array;
452 tmp = (unsigned char*)lmmc_alloc(elem_size);
453 if (tmp == NULL) {

455 }
456
457 for (i = count - 1; i > 0; i--) {
458 uint64_t r = xoshiro256ss_next(rng->state);
459 j = (size_t)(r % (i + 1));
460
461 if (i != j) {
462 memcpy(tmp, arr + i * elem_size, elem_size);
463 memcpy(arr + i * elem_size, arr + j * elem_size, elem_size);
464 memcpy(arr + j * elem_size, tmp, elem_size);
465 }
466 }
467
468 lmmc_free(tmp);
469 return LMMC_STATUS_OK;
470}

484lmmc_status_t lmmc_rng_gamma(
485 lmmc_rng_t* rng,
486 lmmc_real_t shape,
487 lmmc_real_t scale,
488 lmmc_real_t* out)
489{
490 double d, c, x, v, u;
491
492 if (rng == NULL || out == NULL) {
493 return LMMC_STATUS_INVALID_ARGUMENT;
494 }
495 if (shape <= 0.0 || scale <= 0.0) {
496 return LMMC_STATUS_INVALID_ARGUMENT;
497 }
498
499 /* For shape < 1, use: Gamma(shape) = Gamma(shape+1) * U^(1/shape) */
500 if (shape < 1.0) {
501 lmmc_real_t g;
502 lmmc_status_t st = lmmc_rng_gamma(rng, shape + 1.0, 1.0, &g);
503 if (st != LMMC_STATUS_OK) return st;
504
505 u = u64_to_double01(xoshiro256ss_next(rng->state));
506 while (u == 0.0) {
507 u = u64_to_double01(xoshiro256ss_next(rng->state));
508 }
509 *out = scale * g * pow(u, 1.0 / shape);
510 return LMMC_STATUS_OK;
511 }
512
513 /* Marsaglia-Tsang method for shape >= 1 */
514 d = shape - 1.0 / 3.0;
515 c = 1.0 / sqrt(9.0 * d);
516
517 for (;;) {
518 do {
519 x = rng_std_normal(rng->state);
520 v = 1.0 + c * x;
521 } while (v <= 0.0);
522
523 v = v * v * v;
524 u = u64_to_double01(xoshiro256ss_next(rng->state));
525
526 /* Squeeze test */
527 if (u < 1.0 - 0.0331 * (x * x) * (x * x)) {
528 *out = scale * d * v;
529 return LMMC_STATUS_OK;
530 }
531
532 /* Full acceptance check */
533 if (log(u) < 0.5 * x * x + d * (1.0 - v + log(v))) {
534 *out = scale * d * v;
535 return LMMC_STATUS_OK;
536 }
537 }
538}

652/* ============================================================
653 * Poisson distribution
654 * - Inversion for lambda < 10
655 * - PTRD (Hörmann) for lambda >= 10
656 * ============================================================ */
657
661static size_t poisson_inversion(uint64_t* state, double lambda) {
662 double L = exp(-lambda);
663 double p = 1.0;
664 size_t k = 0;
665
666 do {
667 k++;
668 p *= u64_to_double01(xoshiro256ss_next(state));
669 } while (p > L);
670
671 return k - 1;
672}
673
679static size_t poisson_ptrd(uint64_t* state, double lambda) {
680 double smu = sqrt(lambda);
681 double b = 0.931 + 2.53 * smu;
682 double a = -0.059 + 0.02483 * b;
683 double inv_alpha = 1.1239 + 1.1328 / (b - 3.4);
684 double vr = 0.9277 - 3.6224 / (b - 2.0);
685 double us, v, u, k_real;
686 int64_t k;
687
688 for (;;) {
689 v = u64_to_double01(xoshiro256ss_next(state));
690 if (v <= 0.86 * vr) {
691 u = v / vr - 0.43;
692 k_real = floor((2.0 * a / (0.5 - fabs(u)) + b) * u + lambda + 0.445);
693 if (k_real >= 0.0) return (size_t)k_real;
694 }
695
696 if (v >= vr) {
697 u = u64_to_double01(xoshiro256ss_next(state)) - 0.5;
698 } else {
699 u = v / vr - 0.93;
700 u = ((u >= 0.0) ? 0.5 : -0.5) - u;
701 v = u64_to_double01(xoshiro256ss_next(state)) * vr;
702 }
703
704 us = 0.5 - fabs(u);
705 if (us < 0.013 && v > us) continue;
706
707 k_real = floor((2.0 * a / us + b) * u + lambda + 0.445);
708 k = (int64_t)k_real;
709 if (k < 0) continue;
710
711 /* Acceptance: log(v * inv_alpha / (a/(us*us) + b)) <=
712 * -lambda + k*log(lambda) - lgamma(k+1) */
713 v = v * inv_alpha / (a / (us * us) + b);
714 {
715 /* Acceptance: log(v) <= k*log(lambda) - lambda - lgamma(k+1) */
716 double log_accept = (double)k * log(lambda) - lambda - lgamma((double)k + 1.0);
717 if (log(v) <= log_accept) return (size_t)k;
718 }
719 }
720}

722lmmc_status_t lmmc_rng_poisson(
723 lmmc_rng_t* rng,
724 lmmc_real_t lambda,
725 size_t* out)
726{
727 if (rng == NULL || out == NULL) {
728 return LMMC_STATUS_INVALID_ARGUMENT;
729 }
730 if (lambda <= 0.0) {
731 return LMMC_STATUS_INVALID_ARGUMENT;
732 }
733
734 if (lambda < 10.0) {
735 *out = poisson_inversion(rng->state, lambda);

737 *out = poisson_ptrd(rng->state, lambda);
738 }
739
740 return LMMC_STATUS_OK;

744/* ============================================================
745 * Binomial distribution
746 * For small n*min(p,1-p), use direct Bernoulli trials.
747 * For larger values, use the BTRD algorithm (Hörmann).
748 * ============================================================ */
749
750lmmc_status_t lmmc_rng_binomial(
751 lmmc_rng_t* rng,
752 size_t n,
753 lmmc_real_t p,
754 size_t* out)
755{
756 size_t i, count;
757 double pp, u;
758 int flip;
759
760 if (rng == NULL || out == NULL) {
761 return LMMC_STATUS_INVALID_ARGUMENT;
762 }
763 if (p < 0.0 || p > 1.0) {
764 return LMMC_STATUS_INVALID_ARGUMENT;
765 }
766
767 if (n == 0 || p == 0.0) {
768 *out = 0;
769 return LMMC_STATUS_OK;
770 }
771 if (p == 1.0) {
772 *out = n;
773 return LMMC_STATUS_OK;
774 }
775
776 /* Use symmetry: work with min(p, 1-p) */
777 flip = 0;
778 pp = p;
779 if (p > 0.5) {
780 pp = 1.0 - p;
781 flip = 1;
782 }
783
784 /* For small n*p, use direct Bernoulli trials */
785 if ((double)n * pp < 30.0) {
786 count = 0;
787 for (i = 0; i < n; i++) {
788 u = u64_to_double01(xoshiro256ss_next(rng->state));
789 if (u < pp) {
790 count++;
791 }
792 }
793 *out = flip ? (n - count) : count;
794 return LMMC_STATUS_OK;
795 }
796
797 /* For larger n*p, use normal approximation with continuity correction
798 * and rejection from Poisson. This is a simplified approach using
799 * the waiting-time / geometric method. */
800 {
801 /* Use the inverse-transform geometric method (Devroye) */
802 double log_q = log(1.0 - pp);
803 double sum = 0.0;
804 count = 0;
805
806 for (;;) {
807 u = u64_to_double01(xoshiro256ss_next(rng->state));
808 while (u == 0.0) {
809 u = u64_to_double01(xoshiro256ss_next(rng->state));
810 }
811 sum += floor(log(u) / log_q) + 1.0;
812 if (sum > (double)n) break;
813 count++;
814 }
815
816 *out = flip ? (n - count) : count;
817 return LMMC_STATUS_OK;
818 }
819}

822/* ============================================================
823 * Uniform integer distribution on [lo, hi]
824 * ============================================================ */
825
826lmmc_status_t lmmc_rng_int_uniform(
827 lmmc_rng_t* rng,
828 int64_t lo,
829 int64_t hi,
830 int64_t* out)
831{
832 uint64_t range, r, limit;
833
834 if (rng == NULL || out == NULL) {
835 return LMMC_STATUS_INVALID_ARGUMENT;
836 }
837 if (lo > hi) {
838 return LMMC_STATUS_INVALID_ARGUMENT;
839 }
840
841 if (lo == hi) {
842 *out = lo;
843 return LMMC_STATUS_OK;
844 }
845
846 /* Unbiased rejection sampling */
847 range = (uint64_t)(hi - lo) + 1;
848
849 if (range == 0) {
850 /* Full 64-bit range (overflow case: hi - lo + 1 == 2^64) */
851 *out = (int64_t)xoshiro256ss_next(rng->state);
852 return LMMC_STATUS_OK;
853 }
854
855 /* Reject values >= limit to avoid modulo bias */
856 limit = UINT64_MAX - (UINT64_MAX % range);
857
858 do {
859 r = xoshiro256ss_next(rng->state);
860 } while (r >= limit);
861
862 *out = lo + (int64_t)(r % range);
863 return LMMC_STATUS_OK;
864}
