11lmmc_status_t lmmc_quad_trapezoid(
12 lmmc_quad_func_t func,
13 void* user_data,
14 lmmc_real_t a,
15 lmmc_real_t b,
16 size_t n,
17 lmmc_real_t* out_result)
18{
19 if (func == NULL || out_result == NULL) {
20 return LMMC_STATUS_INVALID_ARGUMENT;
21 }
22 if (n == 0) {
23 return LMMC_STATUS_INVALID_ARGUMENT;
24 }
25 if (a >= b) {
26 return LMMC_STATUS_INVALID_ARGUMENT;
27 }
28
29 lmmc_real_t h = (b - a) / (lmmc_real_t)n;
30 lmmc_real_t sum = func(a, user_data) + func(b, user_data);
31
32 for (size_t i = 1; i < n; i++) {
33 lmmc_real_t x_i = a + (lmmc_real_t)i * h;
34 sum += 2.0 * func(x_i, user_data);
35 }
36
37 *out_result = (h / 2.0) * sum;
38 return LMMC_STATUS_OK;
39}

42lmmc_status_t lmmc_quad_simpson(
43 lmmc_quad_func_t func,
44 void* user_data,
45 lmmc_real_t a,
46 lmmc_real_t b,
47 size_t n,
48 lmmc_real_t* out_result)
49{
50 if (func == NULL || out_result == NULL) {
51 return LMMC_STATUS_INVALID_ARGUMENT;
52 }
53 if (n == 0 || n % 2 != 0) {
54 return LMMC_STATUS_INVALID_ARGUMENT;
55 }
56 if (a >= b) {
57 return LMMC_STATUS_INVALID_ARGUMENT;
58 }
59
60 lmmc_real_t h = (b - a) / (lmmc_real_t)n;
61 lmmc_real_t sum = func(a, user_data) + func(b, user_data);
62
63 for (size_t i = 1; i < n; i++) {
64 lmmc_real_t x_i = a + (lmmc_real_t)i * h;
65 if (i % 2 == 1) {
66 sum += 4.0 * func(x_i, user_data);
67 } else {
68 sum += 2.0 * func(x_i, user_data);
69 }
70 }
71
72 *out_result = (h / 3.0) * sum;
73 return LMMC_STATUS_OK;
74}

349lmmc_status_t lmmc_quad_gauss_legendre(
350 lmmc_quad_func_t func,
351 void* user_data,
352 lmmc_real_t a,
353 lmmc_real_t b,
354 size_t order,
355 lmmc_real_t* out_result)
356{
357 if (func == NULL || out_result == NULL) {
358 return LMMC_STATUS_INVALID_ARGUMENT;
359 }
360 if (order < LMMC_GL_MIN_ORDER || order > LMMC_GL_MAX_ORDER) {
361 return LMMC_STATUS_INVALID_ARGUMENT;
362 }
363 if (a >= b) {
364 return LMMC_STATUS_INVALID_ARGUMENT;
365 }
366
367 const lmmc_real_t* nodes = gl_nodes_table[order - LMMC_GL_MIN_ORDER];
368 const lmmc_real_t* weights = gl_weights_table[order - LMMC_GL_MIN_ORDER];
369
370 lmmc_real_t half_len = (b - a) / 2.0;
371 lmmc_real_t midpoint = (a + b) / 2.0;
372
373 lmmc_real_t sum = 0.0;
374 for (size_t i = 0; i < order; i++) {
375 lmmc_real_t x_i = half_len * nodes[i] + midpoint;
376 sum += weights[i] * func(x_i, user_data);
377 }
378
379 *out_result = half_len * sum;
380 return LMMC_STATUS_OK;
381}

412static lmmc_status_t gk15_adaptive_recursive(
413 lmmc_quad_func_t func, void* user_data,
414 lmmc_real_t a, lmmc_real_t b,
415 lmmc_real_t abs_tol, lmmc_real_t rel_tol,
416 size_t depth, size_t max_depth,
417 lmmc_real_t* out_value, lmmc_real_t* out_error, size_t* out_evals)
418{
419 lmmc_real_t half_len = (b - a) / 2.0;
420 lmmc_real_t midpoint = (a + b) / 2.0;
421
422 lmmc_real_t fvals[15];
423 for (int i = 0; i < 15; i++) {
424 lmmc_real_t x_i = half_len * gk15_nodes[i] + midpoint;
425 fvals[i] = func(x_i, user_data);
426 }
427 *out_evals = 15;
428
429 lmmc_real_t k15_sum = 0.0;
430 for (int i = 0; i < 15; i++) {
431 k15_sum += gk15_weights[i] * fvals[i];
432 }
433 lmmc_real_t k15_result = half_len * k15_sum;
434
435 lmmc_real_t g7_sum = 0.0;
436 for (int i = 0; i < 7; i++) {
437 g7_sum += g7_weights[i] * fvals[g7_kronrod_idx[i]];
438 }
439 lmmc_real_t g7_result = half_len * g7_sum;
440
441 lmmc_real_t error = lmmc_abs(k15_result - g7_result);
442 lmmc_real_t tolerance = lmmc_max(abs_tol, rel_tol * lmmc_abs(k15_result));
443
444 if (error <= tolerance) {
445 *out_value = k15_result;
446 *out_error = error;
447 return LMMC_STATUS_OK;
448 }
449
450 if (depth >= max_depth) {
451 *out_value = k15_result;
452 *out_error = error;
453 return LMMC_STATUS_WARNING_MAX_DEPTH;
454 }
455
456 lmmc_real_t mid = (a + b) / 2.0;
457 lmmc_real_t left_value, left_error, right_value, right_error;
458 size_t left_evals, right_evals;
459
460 /* Adaptive tolerance distribution: split tolerance proportionally */
461 lmmc_real_t sub_abs_tol = abs_tol / 2.0;
462
463 lmmc_status_t left_status = gk15_adaptive_recursive(
464 func, user_data, a, mid,
465 sub_abs_tol, rel_tol, depth + 1, max_depth,
466 &left_value, &left_error, &left_evals);
467
468 lmmc_status_t right_status = gk15_adaptive_recursive(
469 func, user_data, mid, b,
470 sub_abs_tol, rel_tol, depth + 1, max_depth,
471 &right_value, &right_error, &right_evals);
472
473 *out_value = left_value + right_value;
474 *out_error = left_error + right_error;
475 *out_evals += left_evals + right_evals;
476
477 if (left_status == LMMC_STATUS_WARNING_MAX_DEPTH ||
478 right_status == LMMC_STATUS_WARNING_MAX_DEPTH) {
479 return LMMC_STATUS_WARNING_MAX_DEPTH;
480 }
481 return LMMC_STATUS_OK;
482}

519/* ========================================================================
520 * Romberg 积分 (Richardson extrapolation on trapezoidal rule)
521 * ======================================================================== */
522
523lmmc_status_t lmmc_quad_romberg(
524 lmmc_quad_func_t f, void* ud,
525 lmmc_real_t a, lmmc_real_t b,
526 lmmc_real_t abs_tol, size_t max_iter,
527 lmmc_quad_result_t* out)
528{
529 if (f == NULL || out == NULL) {
530 return LMMC_STATUS_INVALID_ARGUMENT;
531 }
532 if (a >= b) {
533 return LMMC_STATUS_INVALID_ARGUMENT;
534 }
535 if (abs_tol < 1e-15 || abs_tol > 1e-1) {
536 return LMMC_STATUS_INVALID_ARGUMENT;
537 }
538 if (max_iter < 1 || max_iter > 1000000) {
539 return LMMC_STATUS_INVALID_ARGUMENT;
540 }
541
542 /* Cap Romberg table rows to avoid huge stack allocation */
543 size_t max_rows = max_iter;
544 if (max_rows > 30) max_rows = 30;
545
546 /* Two-row Romberg table */
547 lmmc_real_t prev[30];
548 lmmc_real_t curr[30];
549 size_t num_evals = 0;
550
551 /* R[0][0] = trapezoidal rule with 1 interval */
552 lmmc_real_t h = b - a;
553 prev[0] = h * (f(a, ud) + f(b, ud)) / 2.0;
554 num_evals = 2;
555
556 out->value = prev[0];
557 out->error = lmmc_abs(prev[0]);
558 out->num_evals = num_evals;
559
560 if (max_rows == 1) {
561 return LMMC_STATUS_OK;
562 }
563
564 for (size_t i = 1; i < max_rows; i++) {
565 /* Trapezoidal rule with 2^i intervals */
566 size_t n = (size_t)1 << i;
567 h = (b - a) / (lmmc_real_t)n;
568
569 /* Add new midpoints (only the odd-indexed points are new) */
570 lmmc_real_t sum_new = 0.0;
571 for (size_t k = 1; k <= n / 2; k++) {
572 lmmc_real_t x = a + (2.0 * (lmmc_real_t)k - 1.0) * h;
573 sum_new += f(x, ud);
574 num_evals++;
575 }
576 curr[0] = prev[0] / 2.0 + h * sum_new;
577
578 /* Richardson extrapolation */
579 lmmc_real_t pow4 = 1.0;
580 for (size_t j = 1; j <= i; j++) {
581 pow4 *= 4.0;
582 curr[j] = (pow4 * curr[j - 1] - prev[j - 1]) / (pow4 - 1.0);
583 }
584
585 /* Error estimate */
586 lmmc_real_t err = lmmc_abs(curr[i] - prev[i - 1]);
587 out->value = curr[i];
588 out->error = err;
589 out->num_evals = num_evals;
590
591 if (err <= abs_tol) {
592 return LMMC_STATUS_OK;
593 }
594
595 /* Copy curr to prev */
596 for (size_t j = 0; j <= i; j++) {
597 prev[j] = curr[j];
598 }
599 }
600
601 /* Reached max iterations - return best estimate */
602 return LMMC_STATUS_CONVERGENCE_FAILED;
603}

606/* ========================================================================
607 * Tanh-Sinh (Double Exponential) 积分
608 *
609 * The double-exponential transformation maps [a,b] to (-inf, inf):
610 * x(t) = mid + half_len * tanh(pi/2 * sinh(t))
611 * dx/dt = half_len * (pi/2) * cosh(t) / cosh^2(pi/2 * sinh(t))
612 *
613 * We compute the integral as h * sum_j w_j * f(x_j) where
614 * w_j = (pi/2) * cosh(j*h) / cosh^2(pi/2 * sinh(j*h))
615 *
616 * Convergence is checked by comparing successive levels (halving h).
617 * ======================================================================== */
618
619lmmc_status_t lmmc_quad_tanh_sinh(
620 lmmc_quad_func_t f, void* ud,
621 lmmc_real_t a, lmmc_real_t b,
622 lmmc_real_t abs_tol, size_t max_nodes,
623 lmmc_quad_result_t* out)
624{
625 if (f == NULL || out == NULL) {
626 return LMMC_STATUS_INVALID_ARGUMENT;
627 }
628 if (a >= b) {
629 return LMMC_STATUS_INVALID_ARGUMENT;
630 }
631 if (abs_tol < 1e-15 || abs_tol > 1e-1) {
632 return LMMC_STATUS_INVALID_ARGUMENT;
633 }
634 if (max_nodes < 1 || max_nodes > 1000000) {
635 return LMMC_STATUS_INVALID_ARGUMENT;
636 }
637
638 lmmc_real_t mid = (a + b) / 2.0;
639 lmmc_real_t half_len = (b - a) / 2.0;
640 lmmc_real_t pi_half = LMMC_CONST_PI / 2.0;
641
642 size_t num_evals = 0;
643 int converged = 0;
644
645 /*
646 * Tanh-Sinh quadrature: compute the integral using progressively
647 * smaller step sizes. At each level, compute the full trapezoidal
648 * sum from scratch (simple but correct approach).
649 */
650 lmmc_real_t prev_integral = 0.0;
651 lmmc_real_t integral = 0.0;
652 size_t max_levels = 10;
653
654 /* Step sizes: start small for good coverage near singularities */
655 lmmc_real_t h = 0.03125; /* 1/32 */
656
657 for (size_t level = 0; level < max_levels && num_evals < max_nodes; level++) {
658 lmmc_real_t sum = 0.0;
659
660 /* t = 0: midpoint contribution */
661 {
662 lmmc_real_t fmid = f(mid, ud);
663 num_evals++;
664 if (!isfinite(fmid)) fmid = 0.0;
665 sum += fmid * pi_half;
666 }
667
668 /* Symmetric pairs t = +/- j*h for j = 1, 2, 3, ... */
669 for (size_t j = 1; num_evals < max_nodes; j++) {
670 lmmc_real_t t = (lmmc_real_t)j * h;
671 lmmc_real_t sinh_t = sinh(t);
672 lmmc_real_t cosh_t = cosh(t);
673 lmmc_real_t u = pi_half * sinh_t;
674
675 if (u > 20.0) break;
676
677 /* For large u, cosh(u) ≈ exp(u)/2, so
678 * w = pi/2 * cosh_t / cosh^2(u) ≈ pi/2 * cosh_t * 4 * exp(-2u)
679 * Use this to avoid overflow in cosh(u) for large u.
680 */
681 lmmc_real_t w;
682 if (u > 6.0) {
683 /* Use exponential form to avoid overflow */
684 lmmc_real_t exp_neg_2u = exp(-2.0 * u);
685 w = pi_half * cosh_t * 4.0 * exp_neg_2u;
686 } else {
687 lmmc_real_t cosh_u = cosh(u);
688 w = pi_half * cosh_t / (cosh_u * cosh_u);
689 }
690
691 if (w < 1e-50) break;
692
693 lmmc_real_t tanh_u;
694 if (u > 6.0) {
695 tanh_u = 1.0 - 2.0 * exp(-2.0 * u);
696 } else {
697 tanh_u = tanh(u);
698 }
699 lmmc_real_t xp = mid + half_len * tanh_u;
700 lmmc_real_t xn = mid - half_len * tanh_u;
701
702 lmmc_real_t fp = f(xp, ud);
703 lmmc_real_t fn_val = f(xn, ud);
704 num_evals += 2;
705
706 if (!isfinite(fp)) fp = 0.0;
707 if (!isfinite(fn_val)) fn_val = 0.0;
708
709 sum += (fp + fn_val) * w;
710 }
711
712 integral = half_len * h * sum;
713
714 /* Check convergence (need at least 2 levels) */
715 if (level >= 1) {
716 lmmc_real_t error = lmmc_abs(integral - prev_integral);
717 out->value = integral;
718 out->error = error;
719 out->num_evals = num_evals;
720 if (error <= abs_tol) {
721 converged = 1;
722 break;
723 }
724 }
725
726 prev_integral = integral;
727 h /= 2.0;
728 }
729
730 if (!converged) {
731 out->value = integral;
732 out->error = lmmc_abs(integral - prev_integral);
733 out->num_evals = num_evals;
734 return LMMC_STATUS_CONVERGENCE_FAILED;
735 }
736 return LMMC_STATUS_OK;
737}

740/* ========================================================================
741 * Gauss-Hermite 求积 (weight: exp(-x^2), domain: (-inf, +inf))
742 *
743 * Uses Golub-Welsch algorithm: the nodes are eigenvalues of the symmetric
744 * tridiagonal Jacobi matrix for Hermite polynomials, and weights are
745 * derived from the first component of each eigenvector.
746 *
747 * For physicist's Hermite: H_n(x), weight exp(-x^2)
748 * Recurrence: x H_n = H_{n+1}/2 + n H_{n-1}
749 * Jacobi matrix: a_i = 0, b_i = sqrt(i/2) for i=1..n-1
750 * ======================================================================== */
751
752#define LMMC_GH_MAX_ORDER 20
753
765static void tridiag_eigvals_weights(
766 lmmc_real_t* diag, lmmc_real_t* subdiag,
767 lmmc_real_t* weights, size_t n, lmmc_real_t mu0)
768{
769 /* Store first row of eigenvector matrix (initially identity) */
770 lmmc_real_t z[LMMC_GH_MAX_ORDER];
771 for (size_t i = 0; i < n; i++) z[i] = (i == 0) ? 1.0 : 0.0;
772
773 /* Implicit QL algorithm with shifts */
774 for (size_t l = 0; l < n; l++) {
775 size_t iter = 0;
776 size_t m;
777 while (1) {
778 /* Find small subdiagonal element */
779 for (m = l; m < n - 1; m++) {
780 lmmc_real_t dd = lmmc_abs(diag[m]) + lmmc_abs(diag[m + 1]);
781 if (lmmc_abs(subdiag[m]) + dd == dd) break;
782 }
783 if (m == l) break;
784
785 if (iter++ >= 100) break; /* safety */
786
787 /* Wilkinson shift */
788 lmmc_real_t g = (diag[l + 1] - diag[l]) / (2.0 * subdiag[l]);
789 lmmc_real_t r = sqrt(g * g + 1.0);
790 lmmc_real_t sign_g = (g >= 0.0) ? 1.0 : -1.0;
791 g = diag[m] - diag[l] + subdiag[l] / (g + sign_g * r);
792
793 lmmc_real_t s = 1.0, c = 1.0, p = 0.0;
794 size_t i;
795 for (i = m; i > l; i--) {
796 lmmc_real_t fi = s * subdiag[i - 1];
797 lmmc_real_t bi = c * subdiag[i - 1];
798 if (lmmc_abs(fi) >= lmmc_abs(g)) {
799 c = g / fi;
800 r = sqrt(c * c + 1.0);
801 subdiag[i] = fi * r;
802 s = 1.0 / r;
803 c *= s;
804 } else {
805 s = fi / g;
806 r = sqrt(s * s + 1.0);
807 subdiag[i] = g * r;
808 c = 1.0 / r;
809 s *= c;
810 }
811
812 g = diag[i] - p;
813 r = (diag[i - 1] - g) * s + 2.0 * c * bi;
814 p = s * r;
815 diag[i] = g + p;
816 g = c * r - bi;
817
818 /* Track first row of eigenvector matrix */
819 lmmc_real_t zi = z[i];
820 z[i] = s * z[i - 1] + c * zi;
821 z[i - 1] = c * z[i - 1] - s * zi;
822 }
823 diag[l] -= p;
824 subdiag[l] = g;
825 if (m < n - 1) subdiag[m] = 0.0;
826 }
827 }
828
829 /* Weights = mu0 * z[i]^2 */
830 for (size_t i = 0; i < n; i++) {
831 weights[i] = mu0 * z[i] * z[i];
832 }
833}

835lmmc_status_t lmmc_quad_gauss_hermite(
836 lmmc_quad_func_t f, void* ud,
837 size_t order, lmmc_real_t* out)
838{
839 if (f == NULL || out == NULL) {
840 return LMMC_STATUS_INVALID_ARGUMENT;
841 }
842 if (order < 1 || order > LMMC_GH_MAX_ORDER) {
843 return LMMC_STATUS_INVALID_ARGUMENT;
844 }
845
846 /* Build Jacobi matrix for physicist's Hermite polynomials.
847 * Three-term recurrence: x H_n(x) = H_{n+1}(x)/2 + n*H_{n-1}(x)
848 * Normalized: p_n = H_n / ||H_n||, ||H_n||^2 = sqrt(pi) * 2^n * n!
849 * Jacobi matrix: alpha_i = 0, beta_i = sqrt(i/2) for i = 1..n-1
850 */
851 lmmc_real_t diag[LMMC_GH_MAX_ORDER];
852 lmmc_real_t subdiag[LMMC_GH_MAX_ORDER];
853 lmmc_real_t weights[LMMC_GH_MAX_ORDER];
854
855 for (size_t i = 0; i < order; i++) {
856 diag[i] = 0.0; /* alpha_i = 0 for Hermite */
857 }
858 for (size_t i = 0; i < order - 1; i++) {
859 subdiag[i] = sqrt((lmmc_real_t)(i + 1) / 2.0);
860 }
861
862 /* mu_0 = integral of exp(-x^2) over (-inf, inf) = sqrt(pi) */
863 lmmc_real_t mu0 = sqrt(LMMC_CONST_PI);
864
865 tridiag_eigvals_weights(diag, subdiag, weights, order, mu0);
866
867 /* Compute the quadrature sum: sum_i w_i * f(x_i) */
868 lmmc_real_t sum = 0.0;
869 for (size_t i = 0; i < order; i++) {
870 sum += weights[i] * f(diag[i], ud);
871 }
872
873 *out = sum;
874 return LMMC_STATUS_OK;
875}

878/* ========================================================================
879 * Gauss-Laguerre 求积 (weight: exp(-x), domain: [0, +inf))
880 *
881 * Uses Golub-Welsch algorithm with Laguerre polynomial recurrence.
882 * Three-term recurrence for monic Laguerre:
883 * x L_n(x) = L_{n+1}(x) + (2n+1) L_n(x) - n^2 L_{n-1}(x)
884 * Jacobi matrix: alpha_i = 2*i + 1, beta_i = i (for i = 1..n-1)
885 * mu_0 = integral of exp(-x) over [0, inf) = 1
886 * ======================================================================== */
887
888#define LMMC_GL_LAG_MAX_ORDER 20
889
890lmmc_status_t lmmc_quad_gauss_laguerre(
891 lmmc_quad_func_t f, void* ud,
892 size_t order, lmmc_real_t* out)
893{
894 if (f == NULL || out == NULL) {
895 return LMMC_STATUS_INVALID_ARGUMENT;
896 }
897 if (order < 1 || order > LMMC_GL_LAG_MAX_ORDER) {
898 return LMMC_STATUS_INVALID_ARGUMENT;
899 }
900
901 /* Build Jacobi matrix for Laguerre polynomials.
902 * alpha_i = 2*i + 1 (i = 0..n-1)
903 * beta_i = i (i = 1..n-1), so sqrt(beta_i) = sqrt(i)
904 */
905 lmmc_real_t diag[LMMC_GL_LAG_MAX_ORDER];
906 lmmc_real_t subdiag[LMMC_GL_LAG_MAX_ORDER];
907 lmmc_real_t weights[LMMC_GL_LAG_MAX_ORDER];
908
909 for (size_t i = 0; i < order; i++) {
910 diag[i] = 2.0 * (lmmc_real_t)i + 1.0;
911 }
912 for (size_t i = 0; i < order - 1; i++) {
913 subdiag[i] = (lmmc_real_t)(i + 1); /* sqrt(beta_{i+1}) = sqrt((i+1)^2) = i+1 */
914 }
915
916 /* mu_0 = integral of exp(-x) over [0, inf) = 1 */
917 lmmc_real_t mu0 = 1.0;
918
919 tridiag_eigvals_weights(diag, subdiag, weights, order, mu0);
920
921 /* Compute the quadrature sum: sum_i w_i * f(x_i) */
922 lmmc_real_t sum = 0.0;
923 for (size_t i = 0; i < order; i++) {
924 sum += weights[i] * f(diag[i], ud);
925 }
926
927 *out = sum;
928 return LMMC_STATUS_OK;
