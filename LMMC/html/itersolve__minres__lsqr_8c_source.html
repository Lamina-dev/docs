65/* Compute y = A^T * x for CSR sparse matrix */
66static lmmc_status_t matvec_transpose_csr(
67 const lmmc_sparse_mat_t* a,
68 const lmmc_vec_t* x_in,
69 lmmc_vec_t* y_out
70) {
71 size_t i, j;
72 for (i = 0; i < y_out->size; ++i)
73 y_out->data[i] = 0.0;
74 for (i = 0; i < a->rows; ++i) {
75 size_t start = a->row_ptr[i];
76 size_t end = a->row_ptr[i + 1];
77 for (j = start; j < end; ++j) {
78 size_t col = a->col_idx[j];
79 y_out->data[col] += a->values[j] * x_in->data[i];
80 }
81 }
82 return LMMC_STATUS_OK;
83}

85/* Compute y = A^T * x for CSC sparse matrix */
86static lmmc_status_t matvec_transpose_csc(
87 const lmmc_sparse_mat_t* a,
88 const lmmc_vec_t* x_in,
89 lmmc_vec_t* y_out
90) {
91 size_t i, j;
92 for (j = 0; j < y_out->size; ++j)
93 y_out->data[j] = 0.0;
94 /* CSC: row_ptr is col_ptr, col_idx is row_idx */
95 for (j = 0; j < a->cols; ++j) {
96 size_t start = a->row_ptr[j];
97 size_t end = a->row_ptr[j + 1];
98 double acc = 0.0;
99 for (i = start; i < end; ++i) {
100 size_t row = a->col_idx[i];
101 acc += a->values[i] * x_in->data[row];
102 }
103 y_out->data[j] = acc;
104 }
105 return LMMC_STATUS_OK;
106}

122lmmc_status_t lmmc_minres_solve(
123 const lmmc_sparse_mat_t* a,
124 const lmmc_vec_t* b,
125 const lmmc_precond_t* precond,
126 const lmmc_itersolve_config_t* cfg,
127 lmmc_vec_t* x,
128 lmmc_itersolve_result_t* out_result
129) {
130 lmmc_itersolve_config_t local_cfg;
131 lmmc_status_t st = LMMC_STATUS_OK;
132 size_t n, iter_count = 0;
133 int converged = 0;
134 double norm_r = 0.0, norm_b_val = 0.0, threshold = 0.0;
135
136 /* Lanczos vectors and work vectors */
137 lmmc_vec_t v_prev = {0}, v_curr = {0}, v_next = {0};
138 lmmc_vec_t w_prev = {0}, w_curr = {0};
139 lmmc_vec_t av = {0}, z_vec = {0};
140
141 /* MINRES scalars */
142 double alpha_k, beta_k, beta_kp1;
143 double cs_prev, sn_prev, cs_curr, sn_curr;
144 double phi_bar, epsilon_curr, delta_bar, gamma_val, phi_val;
145
146 /* Validate inputs */
147 if (b == NULL || x == NULL || b->data == NULL || x->data == NULL ||
148 b->size == 0 || x->size == 0) {
149 return LMMC_STATUS_INVALID_ARGUMENT;
150 }
151
152 if (cfg != NULL)
153 local_cfg = *cfg;
154 else {
155 st = lmmc_itersolve_default_config(b->size, &local_cfg);
156 if (st != LMMC_STATUS_OK) return st;
157 }
158
159 /* Dispatch validation */
160 if (local_cfg.apply_op != NULL && a != NULL)
161 return LMMC_STATUS_INVALID_ARGUMENT;
162 if (local_cfg.apply_op == NULL && a == NULL)
163 return LMMC_STATUS_INVALID_ARGUMENT;
164
165 /* Dimension checks */
166 if (a != NULL) {
167 if (a->rows != a->cols || b->size != a->rows || x->size != a->cols)
168 return LMMC_STATUS_DIMENSION_MISMATCH;
169 } else {
170 if (b->size != x->size)
171 return LMMC_STATUS_DIMENSION_MISMATCH;
172 }
173 if (precond != NULL && precond->size != b->size)
174 return LMMC_STATUS_DIMENSION_MISMATCH;
175 if (local_cfg.max_iter == 0)
176 return LMMC_STATUS_INVALID_ARGUMENT;
177
178 n = b->size;
179
180 if (out_result != NULL) {
181 out_result->converged = 0;
182 out_result->num_iter = 0;
183 out_result->initial_residual_norm = 0.0;
184 out_result->final_residual_norm = 0.0;
185 }
186
187 /* Allocate work vectors */
188 st = lmmc_vec_create(n, &v_prev); if (st != LMMC_STATUS_OK) goto minres_end;
189 st = lmmc_vec_create(n, &v_curr); if (st != LMMC_STATUS_OK) goto minres_end;
190 st = lmmc_vec_create(n, &v_next); if (st != LMMC_STATUS_OK) goto minres_end;
191 st = lmmc_vec_create(n, &w_prev); if (st != LMMC_STATUS_OK) goto minres_end;
192 st = lmmc_vec_create(n, &w_curr); if (st != LMMC_STATUS_OK) goto minres_end;
193 st = lmmc_vec_create(n, &av); if (st != LMMC_STATUS_OK) goto minres_end;
194 st = lmmc_vec_create(n, &z_vec); if (st != LMMC_STATUS_OK) goto minres_end;
195
196 /* Compute initial residual: r = b - A*x */
197 st = matvec_dispatch(a, &local_cfg, x, &av);
198 if (st != LMMC_STATUS_OK) goto minres_end;
199 {
200 size_t i;
201 for (i = 0; i < n; ++i)
202 v_curr.data[i] = b->data[i] - av.data[i];
203 }
204
205 /* Apply preconditioner */
206 st = apply_precond(precond, &v_curr, &z_vec);
207 if (st != LMMC_STATUS_OK) goto minres_end;
208
209 /* beta_1 = sqrt(r^T * z) */
210 {
211 double dot_rz;
212 st = vec_dot_checked(&v_curr, &z_vec, &dot_rz);
213 if (st != LMMC_STATUS_OK) goto minres_end;
214 if (dot_rz < 0.0) dot_rz = -dot_rz;
215 beta_k = sqrt(dot_rz);
216 }
217
218 st = vec_norm2_checked(b, &norm_b_val);
219 if (st != LMMC_STATUS_OK) goto minres_end;
220
221 norm_r = beta_k;
222 phi_bar = beta_k;
223 threshold = local_cfg.abs_tol + local_cfg.rel_tol * norm_b_val;
224
225 if (out_result != NULL) {
226 out_result->initial_residual_norm = norm_r;
227 out_result->final_residual_norm = norm_r;
228 }
229 do_log(&local_cfg, 0, norm_r);
230
231 if (norm_r <= threshold) {
232 converged = 1;
233 goto minres_end;
234 }
235 if (beta_k <= 1e-30) {
236 converged = 1;
237 goto minres_end;
238 }
239
240 /* Normalize v_curr = z / beta_k, zero out others */
241 {
242 size_t i;
243 for (i = 0; i < n; ++i) {
244 v_curr.data[i] = z_vec.data[i] / beta_k;
245 v_prev.data[i] = 0.0;
246 w_prev.data[i] = 0.0;
247 w_curr.data[i] = 0.0;
248 }
249 }
250
251 /* Initialize Givens rotation state */
252 cs_prev = 1.0; sn_prev = 0.0;
253 cs_curr = 1.0; sn_curr = 0.0;
254 epsilon_curr = 0.0;
255
256 /* Main MINRES iteration */
257 {
258 size_t iter;
259 for (iter = 0; iter < local_cfg.max_iter; ++iter) {
260 size_t i;
261 double cs_new, sn_new;
262
263 /* Lanczos step: av = A * v_curr */
264 st = matvec_dispatch(a, &local_cfg, &v_curr, &av);
265 if (st != LMMC_STATUS_OK) goto minres_end;
266
267 /* Apply preconditioner */
268 st = apply_precond(precond, &av, &z_vec);
269 if (st != LMMC_STATUS_OK) goto minres_end;
270
271 /* alpha_k = v_curr^T * z_vec */
272 st = vec_dot_checked(&v_curr, &z_vec, &alpha_k);
273 if (st != LMMC_STATUS_OK) goto minres_end;
274
275 /* v_next = z_vec - alpha_k * v_curr - beta_k * v_prev */
276 for (i = 0; i < n; ++i)
277 v_next.data[i] = z_vec.data[i] - alpha_k * v_curr.data[i] - beta_k * v_prev.data[i];
278
279 /* beta_{k+1} = sqrt(v_next^T * v_next) */
280 {
281 double dot_vv;
282 st = vec_dot_checked(&v_next, &v_next, &dot_vv);
283 if (st != LMMC_STATUS_OK) goto minres_end;
284 if (dot_vv < 0.0) dot_vv = -dot_vv;
285 beta_kp1 = sqrt(dot_vv);
286 }
287
288 /* Apply previous Givens rotation to get delta_bar */
289 epsilon_curr = sn_prev * beta_k;
290 delta_bar = cs_curr * alpha_k - sn_curr * epsilon_curr;
291
292 /* Compute new Givens rotation to eliminate beta_{k+1} */
293 gamma_val = sqrt(delta_bar * delta_bar + beta_kp1 * beta_kp1);
294
295 if (gamma_val <= 1e-30) {
296 st = LMMC_STATUS_NUMERICAL_FAILURE;
297 goto minres_end;
298 }
299
300 cs_new = delta_bar / gamma_val;
301 sn_new = beta_kp1 / gamma_val;
302
303 /* phi = cs_new * phi_bar */
304 phi_val = cs_new * phi_bar;
305 /* phi_bar = -sn_new * phi_bar */
306 phi_bar = -sn_new * phi_bar;
307
308 /* Update w: w_new = (v_curr - epsilon_curr*w_prev - delta_bar*w_curr) / gamma */
309 for (i = 0; i < n; ++i)
310 z_vec.data[i] = (v_curr.data[i] - epsilon_curr * w_prev.data[i] - delta_bar * w_curr.data[i]) / gamma_val;
311
312 /* Update solution: x = x + phi * w_new */
313 for (i = 0; i < n; ++i) {
314 x->data[i] += phi_val * z_vec.data[i];
315 if (!is_finite_val(x->data[i])) {
316 st = LMMC_STATUS_NUMERICAL_FAILURE;
317 goto minres_end;
318 }
319 }
320
321 /* Shift w vectors */
322 for (i = 0; i < n; ++i) {
323 w_prev.data[i] = w_curr.data[i];
324 w_curr.data[i] = z_vec.data[i];
325 }
326
327 /* Update rotations */
328 sn_prev = sn_curr;
329 cs_prev = cs_curr;
330 sn_curr = sn_new;
331 cs_curr = cs_new;
332
333 /* Residual norm estimate = |phi_bar| */
334 norm_r = fabs(phi_bar);
335 iter_count = iter + 1;
336 if (out_result != NULL)
337 out_result->final_residual_norm = norm_r;
338 do_log(&local_cfg, iter_count, norm_r);
339
340 if (norm_r <= threshold) { converged = 1; break; }
341
342 /* Normalize v_next and shift Lanczos vectors */
343 if (beta_kp1 <= 1e-30) break; /* Lucky breakdown */
344
345 for (i = 0; i < n; ++i) {
346 v_prev.data[i] = v_curr.data[i];
347 v_curr.data[i] = v_next.data[i] / beta_kp1;
348 }
349 beta_k = beta_kp1;
350 }
351 }
352
353minres_end:
354 if (out_result != NULL) {
355 out_result->converged = converged;
356 out_result->num_iter = iter_count;
357 out_result->final_residual_norm = norm_r;
358 }
359 lmmc_vec_destroy(&z_vec);
360 lmmc_vec_destroy(&av);
361 lmmc_vec_destroy(&w_curr);
362 lmmc_vec_destroy(&w_prev);
363 lmmc_vec_destroy(&v_next);
364 lmmc_vec_destroy(&v_curr);
365 lmmc_vec_destroy(&v_prev);
366 return st;
367}

371lmmc_status_t lmmc_lsqr_solve(
372 const lmmc_sparse_mat_t* a,
373 const lmmc_vec_t* b,
374 const lmmc_itersolve_config_t* cfg,
375 lmmc_vec_t* x,
376 lmmc_itersolve_result_t* out_result
377) {
378 lmmc_itersolve_config_t local_cfg;
379 lmmc_status_t st = LMMC_STATUS_OK;
380 size_t m, n_cols, iter_count = 0;
381 int converged = 0;
382 double norm_r = 0.0, norm_b_val = 0.0, threshold = 0.0;
383 double alpha_l, beta_l, rho_bar, phi_bar_l;
384 double rho_val, cs_l, sn_l, theta_l, phi_l;
385
386 lmmc_vec_t u = {0}, v = {0}, w = {0}, tmp_vec = {0}, av_tmp = {0};
387
388 /* Validate inputs */
389 if (b == NULL || x == NULL || b->data == NULL || x->data == NULL ||
390 b->size == 0 || x->size == 0)
391 return LMMC_STATUS_INVALID_ARGUMENT;
392
393 if (cfg != NULL)
394 local_cfg = *cfg;
395 else {
396 st = lmmc_itersolve_default_config(b->size, &local_cfg);
397 if (st != LMMC_STATUS_OK) return st;
398 }
399
400 /* Dispatch validation */
401 if (local_cfg.apply_op != NULL && a != NULL)
402 return LMMC_STATUS_INVALID_ARGUMENT;
403 if (local_cfg.apply_op == NULL && a == NULL)
404 return LMMC_STATUS_INVALID_ARGUMENT;
405 if (local_cfg.max_iter == 0)
406 return LMMC_STATUS_INVALID_ARGUMENT;
407
408 /* Determine dimensions */
409 if (a != NULL) {
410 m = a->rows;
411 n_cols = a->cols;
412 if (b->size != m || x->size != n_cols)
413 return LMMC_STATUS_DIMENSION_MISMATCH;
414 } else {
415 m = b->size;
416 n_cols = x->size;
417 }
418
419 if (out_result != NULL) {
420 out_result->converged = 0;
421 out_result->num_iter = 0;
422 out_result->initial_residual_norm = 0.0;
423 out_result->final_residual_norm = 0.0;
424 }
425
426 /* Allocate work vectors */
427 st = lmmc_vec_create(m, &u); if (st != LMMC_STATUS_OK) goto lsqr_end;
428 st = lmmc_vec_create(n_cols, &v); if (st != LMMC_STATUS_OK) goto lsqr_end;
429 st = lmmc_vec_create(n_cols, &w); if (st != LMMC_STATUS_OK) goto lsqr_end;
430 st = lmmc_vec_create(n_cols, &tmp_vec); if (st != LMMC_STATUS_OK) goto lsqr_end;
431 st = lmmc_vec_create(m, &av_tmp); if (st != LMMC_STATUS_OK) goto lsqr_end;
432
433 /* Initialize: u = b - A*x */
434 st = matvec_dispatch(a, &local_cfg, x, &av_tmp);
435 if (st != LMMC_STATUS_OK) goto lsqr_end;
436 {
437 size_t i;
438 for (i = 0; i < m; ++i)
439 u.data[i] = b->data[i] - av_tmp.data[i];
440 }
441
442 /* beta = ||u|| */
443 st = vec_norm2_checked(&u, &beta_l);
444 if (st != LMMC_STATUS_OK) goto lsqr_end;
445
446 st = vec_norm2_checked(b, &norm_b_val);
447 if (st != LMMC_STATUS_OK) goto lsqr_end;
448
449 norm_r = beta_l;
450 threshold = local_cfg.abs_tol + local_cfg.rel_tol * norm_b_val;
451
452 if (out_result != NULL) {
453 out_result->initial_residual_norm = norm_r;
454 out_result->final_residual_norm = norm_r;
455 }
456 do_log(&local_cfg, 0, norm_r);
457
458 if (norm_r <= threshold) { converged = 1; goto lsqr_end; }
459 if (beta_l <= 1e-30) { converged = 1; goto lsqr_end; }
460
461 /* Normalize u */
462 {
463 size_t i;
464 for (i = 0; i < m; ++i)
465 u.data[i] /= beta_l;
466 }
467
468 /* v = A^T * u */
469 st = matvec_transpose_dispatch(a, &u, &v);
470 if (st != LMMC_STATUS_OK) goto lsqr_end;
471
472 /* alpha = ||v|| */
473 st = vec_norm2_checked(&v, &alpha_l);
474 if (st != LMMC_STATUS_OK) goto lsqr_end;
475
476 if (alpha_l <= 1e-30) { converged = 1; goto lsqr_end; }
477
478 /* Normalize v */
479 {
480 size_t i;
481 for (i = 0; i < n_cols; ++i)
482 v.data[i] /= alpha_l;
483 }
484
485 /* w = v */
486 st = lmmc_vec_copy(&v, &w);
487 if (st != LMMC_STATUS_OK) goto lsqr_end;
488
489 /* Initialize */
490 rho_bar = alpha_l;
491 phi_bar_l = beta_l;
492
493 /* Main LSQR iteration */
494 {
495 size_t iter;
496 for (iter = 0; iter < local_cfg.max_iter; ++iter) {
497 size_t i;
498 double scale;
499
500 /* Bidiagonalization: u = A*v - alpha*u */
501 st = matvec_dispatch(a, &local_cfg, &v, &av_tmp);
502 if (st != LMMC_STATUS_OK) goto lsqr_end;
503 for (i = 0; i < m; ++i)
504 u.data[i] = av_tmp.data[i] - alpha_l * u.data[i];
505
506 /* beta = ||u|| */
507 st = vec_norm2_checked(&u, &beta_l);
508 if (st != LMMC_STATUS_OK) goto lsqr_end;
509
510 if (beta_l <= 1e-30) {
511 converged = 1;
512 iter_count = iter + 1;
513 break;
514 }
515
516 /* Normalize u */
517 for (i = 0; i < m; ++i)
518 u.data[i] /= beta_l;
519
520 /* v = A^T*u - beta*v */
521 st = matvec_transpose_dispatch(a, &u, &tmp_vec);
522 if (st != LMMC_STATUS_OK) goto lsqr_end;
523 for (i = 0; i < n_cols; ++i)
524 v.data[i] = tmp_vec.data[i] - beta_l * v.data[i];
525
526 /* alpha = ||v|| */
527 st = vec_norm2_checked(&v, &alpha_l);
528 if (st != LMMC_STATUS_OK) goto lsqr_end;
529
530 if (alpha_l <= 1e-30) {
531 converged = 1;
532 iter_count = iter + 1;
533 break;
534 }
535
536 /* Normalize v */
537 for (i = 0; i < n_cols; ++i)
538 v.data[i] /= alpha_l;
539
540 /* Givens rotation */
541 rho_val = sqrt(rho_bar * rho_bar + beta_l * beta_l);
542 if (rho_val <= 1e-30) {
543 st = LMMC_STATUS_NUMERICAL_FAILURE;
544 goto lsqr_end;
545 }
546
547 cs_l = rho_bar / rho_val;
548 sn_l = beta_l / rho_val;
549 theta_l = sn_l * alpha_l;
550 rho_bar = -cs_l * alpha_l;
551 phi_l = cs_l * phi_bar_l;
552 phi_bar_l = sn_l * phi_bar_l;
553
554 /* Update x: x += (phi/rho) * w */
555 scale = phi_l / rho_val;
556 for (i = 0; i < n_cols; ++i) {
557 x->data[i] += scale * w.data[i];
558 if (!is_finite_val(x->data[i])) {
559 st = LMMC_STATUS_NUMERICAL_FAILURE;
560 goto lsqr_end;
561 }
562 }
563
564 /* Update w: w = v - (theta/rho) * w */
565 scale = theta_l / rho_val;
566 for (i = 0; i < n_cols; ++i)
567 w.data[i] = v.data[i] - scale * w.data[i];
568
569 /* Residual norm estimate */
570 norm_r = fabs(phi_bar_l);
571 iter_count = iter + 1;
572 if (out_result != NULL)
573 out_result->final_residual_norm = norm_r;
574 do_log(&local_cfg, iter_count, norm_r);
575
576 if (norm_r <= threshold) { converged = 1; break; }
577 }
578 }
579
580lsqr_end:
581 if (out_result != NULL) {
582 out_result->converged = converged;
583 out_result->num_iter = iter_count;
584 out_result->final_residual_norm = norm_r;
585 }
586 lmmc_vec_destroy(&av_tmp);
587 lmmc_vec_destroy(&tmp_vec);
588 lmmc_vec_destroy(&w);
589 lmmc_vec_destroy(&v);
590 lmmc_vec_destroy(&u);
591 return st;
592}
