31static int fft_is_power_of_four(size_t n) {
32 if (n == 0) return 0;
33 if ((n & (n - 1)) != 0) return 0;
34 /* A power of 2 is a power of 4 iff the single set bit is at an even position */
35 return (n & 0x5555555555555555ULL) != 0;
36}

41static size_t fft_next_power_of_two(size_t n) {
42 size_t p = 1;
43 if (n == 0) return 1;
44 while (p < n) {
45 p <<= 1;
46 }
47 return p;
48}

55static void fft_bit_reverse(lmmc_real_t* real, lmmc_real_t* imag, size_t n) {
56 size_t i, j, bit;
57 lmmc_real_t tmp;
58
59 j = 0;
60 for (i = 1; i < n; ++i) {
61 bit = n >> 1;
62 while (j & bit) {
63 j ^= bit;
64 bit >>= 1;
65 }
66 j ^= bit;
67
68 if (i < j) {
69 tmp = real[i]; real[i] = real[j]; real[j] = tmp;
70 tmp = imag[i]; imag[i] = imag[j]; imag[j] = tmp;
71 }
72 }
73}

79static lmmc_status_t fft_radix2(lmmc_real_t* real, lmmc_real_t* imag, size_t n, int inverse) {
80 size_t len, i, j;
81 double angle_sign;
82
83 if (n <= 1) return LMMC_STATUS_OK;
84
85 fft_bit_reverse(real, imag, n);
86
87 angle_sign = inverse ? 1.0 : -1.0;
88
89 for (len = 2; len <= n; len <<= 1) {
90 double angle = angle_sign * 2.0 * LMMC_PI / (double)len;
91 double wlen_r = cos(angle);
92 double wlen_i = sin(angle);
93
94 for (i = 0; i < n; i += len) {
95 double w_r = 1.0, w_i = 0.0;
96 size_t half = len >> 1;
97
98 for (j = 0; j < half; ++j) {
99 size_t u_idx = i + j;
100 size_t v_idx = i + j + half;
101
102 /* twiddle * x[v] */
103 double t_r = w_r * real[v_idx] - w_i * imag[v_idx];
104 double t_i = w_r * imag[v_idx] + w_i * real[v_idx];
105
106 /* butterfly */
107 real[v_idx] = real[u_idx] - t_r;
108 imag[v_idx] = imag[u_idx] - t_i;
109 real[u_idx] = real[u_idx] + t_r;
110 imag[u_idx] = imag[u_idx] + t_i;
111
112 /* advance twiddle */
113 {
114 double new_w_r = w_r * wlen_r - w_i * wlen_i;
115 double new_w_i = w_r * wlen_i + w_i * wlen_r;
116 w_r = new_w_r;
117 w_i = new_w_i;
118 }
119 }
120 }
121 }
122
123 if (inverse) {
124 double scale = 1.0 / (double)n;
125 for (i = 0; i < n; ++i) {
126 real[i] *= scale;
127 imag[i] *= scale;
128 }
129 }
130
131 return LMMC_STATUS_OK;
132}

146static lmmc_status_t fft_bluestein(lmmc_real_t* real, lmmc_real_t* imag, size_t n, int inverse) {
147 size_t m; /* padded power-of-2 length */
148 size_t k;
149 double sign;
150 lmmc_real_t *a_r = NULL, *a_i = NULL;
151 lmmc_real_t *b_r = NULL, *b_i = NULL;
152 lmmc_status_t st = LMMC_STATUS_OK;
153
154 if (n <= 1) {
155 return LMMC_STATUS_OK;
156 }
157
158 /* Padded length: next power of 2 >= 2N - 1 */
159 m = fft_next_power_of_two(2 * n - 1);
160
161 /* Allocate working arrays */
162 a_r = (lmmc_real_t*)lmmc_alloc(m * sizeof(lmmc_real_t));
163 a_i = (lmmc_real_t*)lmmc_alloc(m * sizeof(lmmc_real_t));
164 b_r = (lmmc_real_t*)lmmc_alloc(m * sizeof(lmmc_real_t));
165 b_i = (lmmc_real_t*)lmmc_alloc(m * sizeof(lmmc_real_t));
166
167 if (a_r == NULL || a_i == NULL || b_r == NULL || b_i == NULL) {
168 st = LMMC_STATUS_ALLOCATION_FAILED;
169 goto cleanup;
170 }
171
172 /* Zero-fill */
173 memset(a_r, 0, m * sizeof(lmmc_real_t));
174 memset(a_i, 0, m * sizeof(lmmc_real_t));
175 memset(b_r, 0, m * sizeof(lmmc_real_t));
176 memset(b_i, 0, m * sizeof(lmmc_real_t));
177
178 /* Sign for forward/inverse: forward DFT uses exp(-2*pi*i*k*n/N),
179 * so chirp phase sign is -1 for forward, +1 for inverse. */
180 sign = inverse ? 1.0 : -1.0;
181
182 /* Build chirp-modulated input a[k] = x[k] * exp(sign * i * pi * k^2 / N)
183 * (Bluestein identity: modulate by exp(-i*pi*k^2/N) for forward) */
184 for (k = 0; k < n; ++k) {
185 double phase = sign * LMMC_PI * (double)(k * k) / (double)n;
186 double c = cos(phase);
187 double s = sin(phase);
188 /* a[k] = x[k] * exp(sign*i*pi*k^2/N) = x[k] * (c + i*s) */
189 a_r[k] = real[k] * c - imag[k] * s;
190 a_i[k] = imag[k] * c + real[k] * s;
191 }
192
193 /* Build convolution kernel b[k] = exp(-sign * i * pi * k^2 / N)
194 * with wrap-around for negative indices */
195 for (k = 0; k < n; ++k) {
196 double phase = -sign * LMMC_PI * (double)(k * k) / (double)n;
197 double c = cos(phase);
198 double s = sin(phase);
199 b_r[k] = c;
200 b_i[k] = s;
201 }
202 /* Wrap-around: b[m-k] = b[k] for k = 1..n-1 */
203 for (k = 1; k < n; ++k) {
204 b_r[m - k] = b_r[k];
205 b_i[m - k] = b_i[k];
206 }
207
208 /* FFT both a and b (forward, length m which is power of 2) */
209 st = fft_radix2(a_r, a_i, m, 0);
210 if (st != LMMC_STATUS_OK) goto cleanup;
211
212 st = fft_radix2(b_r, b_i, m, 0);
213 if (st != LMMC_STATUS_OK) goto cleanup;
214
215 /* Pointwise multiply: a = a * b */
216 for (k = 0; k < m; ++k) {
217 double tr = a_r[k] * b_r[k] - a_i[k] * b_i[k];
218 double ti = a_r[k] * b_i[k] + a_i[k] * b_r[k];
219 a_r[k] = tr;
220 a_i[k] = ti;
221 }
222
223 /* Inverse FFT of the product */
224 st = fft_radix2(a_r, a_i, m, 1);
225 if (st != LMMC_STATUS_OK) goto cleanup;
226
227 /* Extract result: X[k] = a[k] * exp(sign * i * pi * k^2 / N) */
228 for (k = 0; k < n; ++k) {
229 double phase = sign * LMMC_PI * (double)(k * k) / (double)n;
230 double c = cos(phase);
231 double s = sin(phase);
232 /* result[k] = a[k] * exp(sign*i*pi*k^2/N) = a[k] * (c + i*s) */
233 real[k] = a_r[k] * c - a_i[k] * s;
234 imag[k] = a_i[k] * c + a_r[k] * s;
235 }
236
237 /* For inverse FFT, normalize by 1/N */
238 if (inverse) {
239 double scale = 1.0 / (double)n;
240 for (k = 0; k < n; ++k) {
241 real[k] *= scale;
242 imag[k] *= scale;
243 }
244 }
245
246cleanup:
247 if (a_r) lmmc_free(a_r);
248 if (a_i) lmmc_free(a_i);
249 if (b_r) lmmc_free(b_r);
250 if (b_i) lmmc_free(b_i);
251 return st;
252}

256lmmc_status_t lmmc_fft(lmmc_real_t* real, lmmc_real_t* imag, size_t n, int inverse) {
257 if (real == NULL || imag == NULL || n == 0) {
258 return LMMC_STATUS_INVALID_ARGUMENT;
259 }
260 if (n == 1) {
261 /* DFT of length 1 is identity */
262 return LMMC_STATUS_OK;
263 }
264
265 /* Dispatch: power-of-4 -> radix-4, power-of-2 -> radix-2, else -> Bluestein */
266 if (fft_is_power_of_four(n)) {
267 return lmmc_fft_radix4(real, imag, n, inverse ? 1 : 0);
268 } else if (fft_is_power_of_two(n)) {
269 return fft_radix2(real, imag, n, inverse ? 1 : 0);
270 } else {
271 return fft_bluestein(real, imag, n, inverse ? 1 : 0);
272 }
273}

295 st = lmmc_fft_radix4_next_size(n, &nfft);
296 if (st != LMMC_STATUS_OK) {
297 return st;
298 }
299
300 /* Copy input data */
301 memcpy(real_out, real_in, n * sizeof(lmmc_real_t));
302 memcpy(imag_out, imag_in, n * sizeof(lmmc_real_t));
303
304 /* Zero-pad the remainder */
305 if (nfft > n) {
306 memset(real_out + n, 0, (nfft - n) * sizeof(lmmc_real_t));
307 memset(imag_out + n, 0, (nfft - n) * sizeof(lmmc_real_t));
308 }
309
310 *out_nfft = nfft;
311 return LMMC_STATUS_OK;
