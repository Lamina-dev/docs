22/*
23Evaluate in: -1, 0, +1, +2, +inf
24
25 <-s-><--n--><--n--><--n-->
26 |a3-|---a2-|---a1-|---a0-|
27 |-b1-|---b0-|
28 <-t--><--n-->
29
30v0 = a0 * b0 # A(0)*B(0)
31v1 = (a0+ a1+ a2+ a3)*(b0+ b1) # A(1)*B(1) ah <= 3 bh <= 1
32vm1 = (a0- a1+ a2- a3)*(b0- b1) # A(-1)*B(-1) |ah| <= 1 bh = 0
33v2 = (a0+2a1+4a2+8a3)*(b0+2b1) # A(2)*B(2) ah <= 14 bh <= 2
34vinf= a3 * b1 # A(inf)*B(inf)
35*/

37void lmmp_mul_toom42_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na, mp_srcptr restrict numb, mp_size_t nb) {
38 lmmp_param_assert(nb >= 20);
39 lmmp_param_assert(na <= 3 * nb);
40 lmmp_param_assert(5 * na >= 9 * nb);
41 TEMP_S_DECL;
42 mp_size_t n = na >= 2 * nb ? (na + 3) >> 2 : (nb + 1) >> 1, s = na - 3 * n, t = nb - n;
43 int vm1_neg;
44 mp_limb_t cy, vinf0, am1h;
45 mp_limb_t* restrict tp = SALLOC_TYPE(4 * n + 4, mp_limb_t);
46
47#define a0 numa
48#define a1 (numa + n)
49#define a2 (numa + 2 * n)
50#define a3 (numa + 3 * n)
51#define b0 numb
52#define b1 (numb + n)
53
54#define v0 dst //[dst,2*n]
55#define v1 (dst + 2 * n) //[dst+2*n,2*n+1]
56#define vinf (dst + 4 * n) //[dst+4*n,s+t]
57#define vm1 tp //[tp,2*n+1]
58#define v2 (tp + 2 * n + 2) //[tp+2*n+2,2*n+1]
59
60#define bm1 dst //[dst,n]
61#define am1 (dst + n) //[dst+n,n+1]
62#define ap1 tp //[tp,n+1]
63#define bp1 (tp + n + 1) //[tp+n+1,n+1]
64#define ap2 ap1 // same space
65#define bp2 bp1 // same space
66#define a13 bp1 // temporary use
67
68 // ap1,am1
69 ap1[n] = lmmp_add_n_(ap1, a0, a2, n);
70 a13[n] = lmmp_add_(a13, a1, n, a3, s);
71 vm1_neg = lmmp_cmp_(ap1, a13, n + 1) < 0;
72 if (vm1_neg)
73 lmmp_add_n_sub_n_(ap1, am1, a13, ap1, n + 1);
74 else
75 lmmp_add_n_sub_n_(ap1, am1, ap1, a13, n + 1);
76 am1h = am1[n]; // overlap with v1
77
78 // bp1,bm1
79 if (t == n) {
80 if (lmmp_cmp_(b0, b1, n) < 0) {
81 bp1[n] = lmmp_add_n_sub_n_(bp1, bm1, b1, b0, n) >> 1;
82 vm1_neg ^= 1;
83 } else {
84 bp1[n] = lmmp_add_n_sub_n_(bp1, bm1, b0, b1, n) >> 1;
85 }
86 } else {
87 if (lmmp_zero_q_(b0 + t, n - t) && lmmp_cmp_(b0, b1, t) < 0) {
88 cy = lmmp_add_n_sub_n_(bp1, bm1, b1, b0, t);
89 lmmp_zero(bm1 + t, n - t);
90 vm1_neg ^= 1;
91 } else {
92 cy = lmmp_add_n_sub_n_(bp1, bm1, b0, b1, t);
93 lmmp_sub_1_(bm1 + t, b0 + t, n - t, cy & 1);
94 }
95 bp1[n] = lmmp_add_1_(bp1 + t, b0 + t, n - t, cy >> 1);
96 }
97
98 // vinf=a3*b1
99 if (s > t)
100 lmmp_mul_(vinf, a3, s, b1, t);
101 else
102 lmmp_mul_(vinf, b1, t, a3, s);
103 vinf0 = vinf[0]; // overlap with v1
104 cy = vinf[1]; // overlap with v1
105
106 // v1=ap1*bp1
107 lmmp_mul_n_(v1, ap1, bp1, n + 1);
108 vinf[1] = cy; // restore, since v1[2*n+1]==0.
109
110 // ap2
111 cy = lmmp_addshl1_n_(ap2, a2, a3, s);
112 if (s != n)
113 cy = lmmp_add_1_(ap2 + s, a2 + s, n - s, cy);
114 cy = 2 * cy + lmmp_addshl1_n_(ap2, a1, ap2, n);
115 cy = 2 * cy + lmmp_addshl1_n_(ap2, a0, ap2, n);
116 ap2[n] = cy;
117
118 // bp2=bp1+b1
119 lmmp_add_(bp2, bp1, n + 1, b1, t);
120
121 // v2=ap2*bp2
122 lmmp_mul_n_(v2, ap2, bp2, n + 1);
123
124 // vm1=am1*bm1
125 lmmp_mul_n_(vm1, am1, bm1, n);
126 if (am1h)
127 vm1[2 * n] = lmmp_add_n_(vm1 + n, vm1 + n, bm1, n);
128 else
129 vm1[2 * n] = 0;
130
131 // v0=a0*b0
132 lmmp_mul_n_(v0, a0, b0, n);
133
134 lmmp_toom_interp5_(dst, v2, vm1, n, s + t, vm1_neg, vinf0);
135 TEMP_S_FREE;
136#undef a0
137#undef a1
138#undef a2
139#undef a3
140#undef b0
141#undef b1
142
143#undef v0
144#undef v1
145#undef vinf
146#undef vm1
147#undef v2
148
149#undef bm1
150#undef am1
151#undef ap1
152#undef bp1
153#undef ap2
154#undef bp2
155#undef a13
156}

402void lmmp_mul_toom42_unbalance_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na, mp_srcptr restrict numb, mp_size_t nb) {
403 lmmp_param_assert(na >= 3 * nb);
404 lmmp_param_assert(nb > 20);
405 TEMP_S_DECL;
406 mp_limb_t* restrict ws = SALLOC_TYPE(nb, mp_limb_t);
407
408 toom42_cache_t cache;
409 cache.numb = numb;
410 cache.n = (2 * nb + 3) >> 2;
411 cache.s = 2 * nb - 3 * cache.n;
412 cache.t = nb - cache.n;
413 cache.tp = SALLOC_TYPE(4 * cache.n + 4, mp_limb_t);
414 cache._bp1 = SALLOC_TYPE(2 * cache.n + 1, mp_limb_t);
415 cache._bm1 = cache._bp1 + cache.n + 1;
416
417 int flag = lmmp_mul_toom42_cache_init_(dst, numa, &cache);
418 dst += 2 * nb;
419 numa += 2 * nb;
420 na -= 2 * nb;
421 lmmp_copy(ws, dst, nb);
422 while (2 * na >= 5 * nb) {
423 lmmp_mul_toom42_cache_(dst, numa, &cache, flag);
424 if (lmmp_add_n_(dst, dst, ws, nb))
425 lmmp_inc(dst + nb);
426 dst += 2 * nb;
427 numa += 2 * nb;
428 na -= 2 * nb;
429 lmmp_copy(ws, dst, nb);
430 }
431 // 0.5 nb <= na < 2.5 nb
432 if (na >= nb)
433 lmmp_mul_(dst, numa, na, numb, nb);
434 else
435 lmmp_mul_(dst, numb, nb, numa, na);
436 if (lmmp_add_n_(dst, dst, ws, nb))
437 lmmp_inc(dst + nb);
438 TEMP_S_FREE;
439}
