14mp_size_t lmmp_pow_(mp_ptr restrict dst, mp_size_t rn, mp_srcptr restrict base, mp_size_t n, ulong exp) {
15 lmmp_param_assert(n > 0);
16 lmmp_param_assert(exp > 0);
17 lmmp_param_assert(base[n - 1] != 0);
18 if (exp == 1) {
19 lmmp_copy(dst, base, n);
20 return n;
21 } else if (exp == 2) {
22 lmmp_sqr_(dst, base, n);
23 rn = n << 1;
24 rn -= (dst[rn - 1] == 0);
25 return rn;
26 } else {
27 mp_size_t base_tz = 0;
28 while (*base == 0) {
29 ++base_tz;
30 ++base;
31 --n;
32 }
33 base_tz *= exp;
34 lmmp_zero(dst, base_tz);
35 dst += base_tz;
36 if (n == 1) {
37 if (exp <= POW_1_EXP_THRESHOLD) {
38 dst[0] = base[0];
39 rn = 1;
40 for (mp_size_t i = 1; i < exp; ++i) {
41 dst[rn] = lmmp_mul_1_(dst, dst, rn, base[0]);
42 ++rn;
43 rn -= (dst[rn - 1] == 0);
44 }
45 return rn + base_tz;
46 } else {
47 return lmmp_pow_1_(dst, rn, base[0], exp) + base_tz;
48 }
49 } else { /* n > 2 */
50 if (exp > POW_WIN2_EXP_THRESHOLD && n > POW_WIN2_N_THRESHOLD) {
51 if ((exp % 4 == 3) || (2 * lmmp_limb_popcnt_(exp) >= (lmmp_limb_bits_(exp)))) {
52 return lmmp_pow_win2_(dst, rn, base, n, exp) + base_tz;
53 }
54 }
55 if (exp & 1) {
56 return lmmp_pow_basecase_(dst, rn, base, n, exp) + base_tz;
57 }
58
59 int tz = lmmp_tailing_zeros_(exp);
60 TEMP_DECL;
61 mp_ptr restrict sq = TALLOC_TYPE((rn + 2) >> 1, mp_limb_t);
62 exp >>= tz;
63
64 if (tz & 1) {
65 if (exp == 1) {
66 lmmp_copy(sq, base, n);
67 rn = n;
68 } else {
69 mp_size_t rn1 = lmmp_pow_size_(base, n, exp);
70 rn = lmmp_pow_basecase_(sq, rn1, base, n, exp);
71 }
72 int i = 2;
73 for (; i <= tz; i += 2) {
74 lmmp_sqr_(dst, sq, rn);
75 rn <<= 1;
76 rn -= (dst[rn - 1] == 0);
77 lmmp_sqr_(sq, dst, rn);
78 rn <<= 1;
79 rn -= (sq[rn - 1] == 0);
80 }
81 lmmp_sqr_(dst, sq, rn);
82 rn <<= 1;
83 rn -= (dst[rn - 1] == 0);
84 } else {
85 if (exp == 1) {
86 lmmp_copy(dst, base, n);
87 rn = n;
88 } else {
89 mp_size_t rn1 = lmmp_pow_size_(base, n, exp);
90 rn = lmmp_pow_basecase_(dst, rn1, base, n, exp);
91 }
92 int i = 2;
93 for (; i <= tz; i += 2) {
94 lmmp_sqr_(sq, dst, rn);
95 rn <<= 1;
96 rn -= (sq[rn - 1] == 0);
97 lmmp_sqr_(dst, sq, rn);
98 rn <<= 1;
99 rn -= (dst[rn - 1] == 0);
100 }
101 }
102 TEMP_FREE;
103 return rn + base_tz;
104 }
105 }
