26mp_size_t lmmp_nPr_size_(ulong n, ulong r, mp_bitcnt_t* restrict bits) {
27 mp_size_t shl = n - lmmp_limb_popcnt_(n);
28 shl -= (n - r) - lmmp_limb_popcnt_(n - r);
29 *bits = shl;
30 if (n < DBL_2POW_MANT_DIG_) {
31 double ln_perm = lgamma(n + 1.0) - lgamma(n - r + 1.0);
32 double log2_perm = ln_perm / LOG2_;
33 mp_size_t rn = ceil(log2_perm / LIMB_BITS) + 2; /* more two limbs */
34 return rn;
35 } else {
36 // nPr = n! / (n-r)! < n^r
37 return lmmp_pow_1_size_(n, r);
38 }
39}

41static inline uint count_factors(fac_ptr fac, uint nfactors, uint n, uint r, uint p) {
42 uint pn = n;
43 uint e = 0;
44 ulong inv = MP_ULONG_MAX / p + 1;
45 while (pn > 0) {
46 _udiv32by32_q_preinv(pn, pn, inv);
47 e += pn;
48 }
49 pn = r;
50 while (pn > 0) {
51 _udiv32by32_q_preinv(pn, pn, inv);
52 e -= pn;
53 }
54 if (e > 0) {
55 fac[nfactors].f = p;
56 fac[nfactors++].j = e;
57 }
58 return nfactors;
59}

64static mp_size_t lmmp_odd_nPr_product_(mp_ptr restrict dst, mp_size_t rn, uint n, uint r) {
65 TEMP_DECL;
66 ulongp restrict limbs = TALLOC_TYPE(r / 2 + 1, ulong);
67 mp_size_t limbn = 0;
68 ulong t = 1, v;
69 mp_bitcnt_t cnt = 0;
70 for (ulong i = n - r + 1; i <= n; ++i) {
71 ctz_shl(v, i, cnt);
72 t *= v;
73 if (t > MP_UINT_MAX) {
74 limbs[limbn++] = t;
75 t = 1;
76 }
77 }
78 if (t != 1)
79 limbs[limbn++] = t;
80
81 mp_ptr restrict tp = TALLOC_TYPE(limbn * 2, mp_limb_t);
82 // 这里不能直接乘入dst，因为dst的大小可能小于limbn，导致溢出
83 rn = lmmp_elem_mul_ulong_(tp, limbs, limbn, tp + limbn);
84 lmmp_copy(dst, tp, rn);
85 TEMP_FREE;
86 return rn;
87}

89mp_size_t lmmp_odd_nPr_ushort_(mp_ptr restrict dst, mp_size_t rn, ulong n, ulong r) {
90 lmmp_param_assert(n >= r);
91 lmmp_param_assert(n <= NPR_SHORT_LIMIT);
92 if (n < ODD_FACTORIAL_SIZE) {
93 if (n == 0) {
94 dst[0] = 1;
95 } else if (n == r) {
96 dst[0] = odd_factorial[n - 1];
97 } else {
98 dst[0] = odd_factorial[n - 1] / odd_factorial[n - r - 1];
99 }
100 return 1;
101 } else if (r <= 10) {
102 dst[0] = 1;
103 rn = 1;
104 ulong t = 1, v;
105 ulong i = n - r + 1;
106 mp_bitcnt_t cnt = 0;
107 lmmp_debug_assert(n >= 3);
108 for (; i <= (ulong)n - 3; i += 3) {
109 t = i * (i + 1) * (i + 2);
110 ctz_shl(v, t, cnt);
111 mul_1(dst, rn, v);
112 }
113 t = 1;
114 for (; i <= n; ++i) {
115 t *= i;
116 }
117 ctz_shl(v, t, cnt);
118 if (v != 1) {
119 mul_1(dst, rn, v);
120 }
121 return rn;
122 } else if (n <= MP_UCHAR_MAX) {
123 lmmp_debug_assert(n >= 7);
124 lmmp_debug_assert(r >= 2);
125 dst[0] = 1;
126 rn = 1;
127 ulong t = 0, v;
128 ulong i = n - r + 1;
129 mp_bitcnt_t cnt;
130 for (; i <= (ulong)n - 7; i += 7) {
131 t = i * (i + 1) * (i + 2) * (i + 3) * (i + 4) * (i + 5) * (i + 6);
132 ctz_shl(v, t, cnt);
133 mul_1(dst, rn, v);
134 }
135 t = 1;
136 for (; i <= n; ++i) {
137 t *= i;
138 }
139 ctz_shl(v, t, cnt);
140 if (v != 1) {
141 mul_1(dst, rn, v);
142 }
143 return rn;
144 } else if (n <= 0xfff) {
145 TEMP_S_DECL;
146 ulongp restrict limbs = SALLOC_TYPE(r / 5 + 1, ulong);
147 mp_size_t limbn = 0;
148 ulong t, v;
149 ulong i = n - r + 1;
150 mp_bitcnt_t cnt;
151 lmmp_debug_assert(n >= 5);
152 for (; i <= (ulong)n - 5; i += 5) {
153 t = i * (i + 1) * (i + 2) * (i + 3) * (i + 4);
154 ctz_shl(v, t, cnt);
155 limbs[limbn++] = v;
156 }
157 t = 1;
158 for (; i <= n; ++i) {
159 t *= i;
160 }
161 ctz_shl(v, t, cnt);
162 if (v != 1)
163 limbs[limbn++] = v;
164 mp_ptr restrict tp = SALLOC_TYPE(limbn * 2, mp_limb_t);
165 // 这里不能直接乘入dst，因为dst的大小可能小于limbn，导致溢出
166 rn = lmmp_elem_mul_ulong_(tp, limbs, limbn, tp + limbn);
167 lmmp_copy(dst, tp, rn);
168 TEMP_S_FREE;
169 return rn;
170 } else if (rn < PERMUTATION_USHORT_MUL_THRESHOLD || n >= (PERMUTATION_USHORT_TIMES_THRESHOLD * r)) {
171 return lmmp_odd_nPr_product_(dst, rn, n, r);
172 } else {
173 TEMP_DECL;
174 uint primen = lmmp_prime_cnt16_(n);
175 uint nfactors = primen;
176 fac_ptr restrict fac = TALLOC_TYPE(nfactors, fac_t);
177 r = n - r;
178 nfactors = 0;
179 for (uint i = 1; i < primen; ++i) {
180 uint p = prime_short_table[i];
181 nfactors = count_factors(fac, nfactors, n, r, p);
182 }
183
184 rn = lmmp_factors_mul_(dst, rn, fac, nfactors);
185
186 TEMP_FREE;
187 return rn;
188 }
189}

191mp_size_t lmmp_odd_nPr_uint_(mp_ptr restrict dst, mp_size_t rn, ulong n, ulong r) {
192 lmmp_param_assert(n >= r);
193 lmmp_param_assert(n <= NPR_INT_LIMIT);
194 if (r <= 10) {
195 dst[0] = 1;
196 rn = 1;
197 ulong v;
198 mp_bitcnt_t cnt;
199 for (ulong i = n - r + 1; i <= n; ++i) {
200 ctz_shl(v, i, cnt);
201 mul_1(dst, rn, v);
202 }
203 return rn;
204 } else if (rn < PERMUTATION_UINT_MUL_THRESHOLD || n >= (PERMUTATION_UINT_TIMES_THRESHOLD * r)) {
205 return lmmp_odd_nPr_product_(dst, rn, n, r);
206 } else{
207 TEMP_B_DECL;
208
209 lmmp_prime_int_table_init_(n);
210 uint nfactors = lmmp_prime_size_(n);
211 fac_ptr restrict fac = BALLOC_TYPE(nfactors, fac_t);
212 r = n - r;
213 nfactors = 0;
214
215 prime_cache_t cache;
216 lmmp_prime_cache_init_(&cache, n);
217 while(cache.is_end == 0) {
218 lmmp_prime_cache_next_(&cache);
219 for (uint i = 0; i < cache.size; ++i) {
220 nfactors = count_factors(fac, nfactors, n, r, cache.pp[i]);
221 }
222 }
223 lmmp_prime_cache_free_(&cache);
224
225 rn = lmmp_factors_mul_(dst, rn, fac, nfactors);
226
227 TEMP_B_FREE;
228 return rn;
229 }

266mp_size_t lmmp_nPr_(mp_ptr restrict dst, mp_bitcnt_t bits, mp_size_t rn, ulong n, ulong r) {
267 lmmp_debug_assert(n >= r);
268 mp_size_t shw = bits / LIMB_BITS;
269 bits %= LIMB_BITS;
270 lmmp_zero(dst, shw);
271 if (n <= NPR_SHORT_LIMIT)
272 rn = lmmp_odd_nPr_ushort_(dst + shw, rn - shw, n, r);
273 else if (n <= NPR_INT_LIMIT)
274 rn = lmmp_odd_nPr_uint_(dst + shw, rn - shw, n, r);
275 else
276 rn = lmmp_odd_nPr_ulong_(dst + shw, rn - shw, n, r);
277
278 if (bits > 0) {
279 dst[shw + rn] = lmmp_shl_(dst + shw, dst + shw, rn, bits);
280 rn += shw + 1;
281 rn -= dst[rn - 1] == 0;
282 } else {
283 rn += shw;
284 }
285 return rn;
