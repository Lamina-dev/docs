17// 向上调整最小堆（插入元素后维护堆性质：len小的在上）
18static inline void heapifyUp(num_heap* restrict pq, size_t index) {
19 size_t parent = (index - 1) / 2;
20
21 // 最小堆：当前节点len < 父节点len 时交换
22 while (index > 0 && pq->head[index].n < pq->head[parent].n) {
23 swap_num_node(&pq->head[index], &pq->head[parent]);
24 index = parent;
25 parent = (index - 1) / 2;
26 }
27}

29// 向下调整最小堆
30static inline void heapifyDown(num_heap* restrict pq, size_t index) {
31 if (pq->size == 0) {
32 return;
33 }
34 lmmp_debug_assert(index < pq->size);
35
36 while (1) {
37 size_t smallest = index;
38 size_t left = 2 * index + 1;
39 size_t right = 2 * index + 2;
40
41 if (left < pq->size && pq->head[left].n < pq->head[smallest].n) {
42 smallest = left;
43 }
44 if (right < pq->size && pq->head[right].n < pq->head[smallest].n) {
45 smallest = right;
46 }
47
48 if (smallest == index) {
49 break;
50 }
51
52 swap_num_node(&pq->head[index], &pq->head[smallest]);
53 index = smallest;
54 }
55}

57void lmmp_num_heap_push_(num_heap* restrict pq, mp_ptr elem, mp_size_t n) {
58 lmmp_param_assert(pq->size < pq->cap);
59 pq->head[pq->size].num = elem;
60 pq->head[pq->size].n = n;
61 pq->size++;
62 heapifyUp(pq, pq->size - 1);

65bool lmmp_num_heap_pop_(num_heap* restrict pq, num_node_ptr restrict outElem) {
66 if (pq->size == 0) {
67 outElem->num = NULL;
68 outElem->n = 0;
69 return false;
70 }
71 *outElem = pq->head[0];
72 pq->head[0] = pq->head[pq->size - 1];
73 --(pq->size);
74 heapifyDown(pq, 0);
75 return true;
76}

78mp_ptr lmmp_num_heap_mul_(num_heap* restrict pq, mp_size_t* restrict rn) {
79 num_node numa, numb;
80 numa.num = NULL;
81 numb.num = NULL;
82 numa.n = 0;
83 numb.n = 0;
84
85 while (lmmp_num_heap_pop_(pq, &numa)) {
86 if (!lmmp_num_heap_pop_(pq, &numb)) {
87 break;
88 }
89 mp_ptr restrict prod = ALLOC_TYPE(numa.n + numb.n, mp_limb_t);
90 lmmp_mul_(prod, numb.num, numb.n, numa.num, numa.n);
91 lmmp_free(numa.num);
92 lmmp_free(numb.num);
93 numa.num = prod;
94 numa.n += numb.n;
95 numa.n -= numa.num[numa.n - 1] == 0 ? 1 : 0;
96 lmmp_num_heap_push_(pq, numa.num, numa.n);
97 }
98 *rn = numa.n;
99 return numa.num;
100}

102mp_size_t lmmp_elem_mul_ulong_(mp_ptr restrict dst, const ulongp restrict limbs, mp_size_t n, mp_ptr restrict tp) {
103 if (n < ELEM_MUL_BASECASE_THRESHOLD) {
104 lmmp_debug_assert(n > 0);
105 dst[0] = limbs[0];
106 mp_size_t rn = 1;
107 for (mp_size_t i = 1; i < n; i++) {
108 dst[rn] = lmmp_mul_1_(dst, dst, rn, limbs[i]);
109 rn++;
110 rn -= dst[rn - 1] == 0 ? 1 : 0;
111 }
112 return rn;
113 }
114 mp_size_t halfn = n / 2;
115 mp_size_t n1 = lmmp_elem_mul_ulong_(tp, limbs, halfn, dst);
116 mp_size_t n2 = lmmp_elem_mul_ulong_(tp + halfn, limbs + halfn, n - halfn, dst + halfn);
117 if (n1 > n2)
118 lmmp_mul_(dst, tp, n1, tp + halfn, n2);
119 else
120 lmmp_mul_(dst, tp + halfn, n2, tp, n1);
121 n = n1 + n2;
122 n -= dst[n - 1] == 0 ? 1 : 0;
123 return n;
124}
