15static inline uint64_t rotl64(uint64_t x, int b) {
16 b &= 63;

21 uint64_t seed = key == NULL ? 0 : key[0];
22 if (in == NULL || inlen == 0) {
23 seed += PRIME64_5;
24 seed += 0; // len = 0
25 seed ^= seed >> 33;
26 seed *= PRIME64_2;
27 seed ^= seed >> 29;
28 seed *= PRIME64_3;
29 seed ^= seed >> 32;
30 return seed;
31 }
