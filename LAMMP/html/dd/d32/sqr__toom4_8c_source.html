23/*
24Evaluate in: -1, -1/2, 0, +1/2, +1, +2, +inf
25
26 <-s--><--n--><--n--><--n-->
27 |-a3-|--a2--|--a1--|--a0--|
28
29 v0 = a0 ^2 # A(0)^2
30 v1 = ( a0+ a1+ a2+ a3)^2 # A(1)^2 ah <= 3
31 vm1 = ( a0- a1+ a2- a3)^2 # A(-1)^2 |ah| <= 1
32 v2 = ( a0+2a1+4a2+8a3)^2 # A(2)^2 ah <= 14
33 vh = (8a0+4a1+2a2+ a3)^2 # A(1/2)^2 ah <= 14
34 vmh = (8a0-4a1+2a2- a3)^2 # A(-1/2)^2 -4<=ah<=9
35 vinf= a3 ^2 # A(inf)^2
36*/

38void lmmp_sqr_toom4_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na) {
39 lmmp_param_assert(na > 0);
40 lmmp_param_assert(dst != NULL);
41 lmmp_param_assert(numa != NULL);
42 mp_size_t n, s;
43 mp_limb_t cy;
44
45#define a0 numa
46#define a1 (numa + n)
47#define a2 (numa + 2 * n)
48#define a3 (numa + 3 * n)
49
50 n = (na + 3) >> 2;
51 TEMP_S_DECL;
52 mp_ptr restrict scratch = SALLOC_TYPE(8 * n + 8, mp_limb_t);
53
54 s = na - 3 * n;
55
56 lmmp_debug_assert(0 < s && s <= n);
57
58 /* NOTE: The multiplications to v2, vm2, vh and vm1 overwrites the
59 * following limb, so these must be computed in order, and we need a
60 * one limb gap to tp. */
61#define v0 dst /* 2n */
62#define v1 (dst + 2 * n) /* 2n+1 */
63#define vinf (dst + 6 * n) /* s+t */
64#define v2 scratch /* 2n+1 */
65#define vm2 (scratch + 2 * n + 1) /* 2n+1 */
66#define vh (scratch + 4 * n + 2) /* 2n+1 */
67#define vm1 (scratch + 6 * n + 3) /* 2n+1 */
68#define tp (scratch + 8 * n + 5)
69
70 /* No overlap with v1 */
71#define apx dst /* n+1 */
72#define amx (dst + 4 * n + 2) /* n+1 */
73
74 /* Compute apx = a0 + 2 a1 + 4 a2 + 8 a3 and amx = a0 - 2 a1 + 4 a2 - 8 a3. */
75 lmmp_toom_eval_dgr3_pm2_(apx, amx, numa, n, s, tp);
76
77 lmmp_sqr_(v2, apx, n + 1); /* v2, 2n+1 limbs */
78 lmmp_sqr_(vm2, amx, n + 1); /* vm2, 2n+1 limbs */
79
80 /* Compute apx = 8 a0 + 4 a1 + 2 a2 + a3 = (((2*a0 + a1) * 2 + a2) * 2 + a3 */
81 cy = lmmp_addshl1_n_(apx, a1, a0, n);
82 cy = 2 * cy + lmmp_addshl1_n_(apx, a2, apx, n);
83 if (s < n) {
84 mp_limb_t cy2;
85 cy2 = lmmp_addshl1_n_(apx, a3, apx, s);
86 apx[n] = 2 * cy + lmmp_shl_(apx + s, apx + s, n - s, 1);
87 lmmp_inc_1(apx + s, cy2);
88 } else
89 apx[n] = 2 * cy + lmmp_addshl1_n_(apx, a3, apx, n);
90
91 lmmp_debug_assert(apx[n] < 15);
92
93 lmmp_sqr_(vh, apx, n + 1); /* vh, 2n+1 limbs */
94
95 /* Compute apx = a0 + a1 + a2 + a3 and amx = a0 - a1 + a2 - a3. */
96 lmmp_toom_eval_dgr3_pm1_(apx, amx, numa, n, s, tp);
97
98 lmmp_sqr_(v1, apx, n + 1); /* v1, 2n+1 limbs */
99 lmmp_sqr_(vm1, amx, n + 1); /* vm1, 2n+1 limbs */
100
101 lmmp_sqr_(v0, a0, n);
102 lmmp_sqr_(vinf, a3, s); /* vinf, 2s limbs */
103
104 lmmp_toom_interp7_(dst, n, (enum toom7_flags)0, vm2, vm1, v2, vh, 2 * s, tp);
105 TEMP_S_FREE;
106}
