23/*
24Evaluate in: -2, -1, 0, +1, +2, +inf
25
26 <-s-><--n--><--n--><--n-->
27 |a3-|---a2-|--a1--|--a0--|
28 |-b2-|--b1--|--b0--|
29 <-t--><--n--><--n-->
30
31 v0 = a0 * b0 # A(0)*B(0)
32 v1 = (a0+ a1+ a2+ a3)*(b0+ b1+ b2) # A(1)*B(1) ah <= 3 bh <= 2
33 vm1 = (a0- a1+ a2- a3)*(b0- b1+ b2) # A(-1)*B(-1) |ah| <= 1 |bh|<= 1
34 v2 = (a0+2a1+4a2+8a3)*(b0+2b1+4b2) # A(2)*B(2) ah <= 14 bh <= 6
35 vm2 = (a0-2a1+4a2-8a3)*(b0-2b1+4b2) # A(-2)*B(-2) |ah| <= 9 |bh|<= 4
36 vinf= a3 * b2 # A(inf)*B(inf)
37*/
38
39void lmmp_mul_toom43_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na, mp_srcptr restrict numb, mp_size_t nb) {
40 lmmp_param_assert(4 * na >= 5 * nb);
41 lmmp_param_assert(3 * na <= 5 * nb);
42 mp_size_t n, s, t;
43 enum toom6_flags flags;
44 mp_limb_t cy;
45
46#define a0 numa
47#define a1 (numa + n)
48#define a2 (numa + 2 * n)
49#define a3 (numa + 3 * n)
50#define b0 numb
51#define b1 (numb + n)
52#define b2 (numb + 2 * n)
53
54 n = 1 + (3 * na >= 4 * nb ? (na - 1) >> 2 : (nb - 1) / (mp_size_t)3);
55 TEMP_S_DECL;
56 mp_limb_t* restrict scratch = SALLOC_TYPE(6 * n + 6, mp_limb_t);
57
58 s = na - 3 * n;
59 t = nb - 2 * n;
60
61 lmmp_debug_assert(0 < s && s <= n);
62 lmmp_debug_assert(0 < t && t <= n);
63
64 /* This is true whenever na >= 25 or nb >= 19, I think. It
65 guarantees that we can fit 5 values of size n+1 in the product
66 area. */
67 lmmp_debug_assert(s + t >= 5);
68
69#define v0 dst /* 2n */
70#define vm1 (scratch) /* 2n+1 */
71#define v1 (dst + 2 * n) /* 2n+1 */
72#define vm2 (scratch + 2 * n + 1) /* 2n+1 */
73#define v2 (scratch + 4 * n + 2) /* 2n+1 */
74#define vinf (dst + 5 * n) /* s+t */
75#define bs1 dst /* n+1 */
76#define bsm1 (scratch + 2 * n + 2) /* n+1 */
77#define asm1 (scratch + 3 * n + 3) /* n+1 */
78#define asm2 (scratch + 4 * n + 4) /* n+1 */
79#define bsm2 (dst + n + 1) /* n+1 */
80#define bs2 (dst + 2 * n + 2) /* n+1 */
81#define as2 (dst + 3 * n + 3) /* n+1 */
82#define as1 (dst + 4 * n + 4) /* n+1 */
83
84 /* Total sccratch need is 6 * n + 3 + 1; we allocate one extra
85 limb, because products will overwrite 2n+2 limbs. */
86
87#define a0a2 scratch
88#define b0b2 scratch
89#define a1a3 asm1
90#define b1d bsm1
91
92 /* Compute as2 and asm2. */
93 flags = (enum toom6_flags)(toom6_vm2_neg & lmmp_toom_eval_dgr3_pm2_(as2, asm2, numa, n, s, a1a3));
94
95 /* Compute bs2 and bsm2. */
96 b1d[n] = lmmp_shl_(b1d, b1, n, 1); /* 2b1 */
97 cy = lmmp_shl_(b0b2, b2, t, 2); /* 4b2 */
98 cy += lmmp_add_n_(b0b2, b0b2, b0, t); /* 4b2 + b0 */
99 if (t != n)
100 cy = lmmp_add_1_(b0b2 + t, b0 + t, n - t, cy);
101 b0b2[n] = cy;
102
103 if (lmmp_cmp_(b0b2, b1d, n + 1) < 0) {
104 lmmp_add_n_sub_n_(bs2, bsm2, b1d, b0b2, n + 1);
105 flags = (enum toom6_flags)(flags ^ toom6_vm2_neg);
106 } else {
107 lmmp_add_n_sub_n_(bs2, bsm2, b0b2, b1d, n + 1);
108 }
109
110
111 /* Compute as1 and asm1. */
112 flags = (enum toom6_flags)(flags ^ (toom6_vm1_neg & lmmp_toom_eval_dgr3_pm1_(as1, asm1, numa, n, s, a0a2)));
113
114 /* Compute bs1 and bsm1. */
115 bsm1[n] = lmmp_add_(bsm1, b0, n, b2, t);
116 if (bsm1[n] == 0 && lmmp_cmp_(bsm1, b1, n) < 0) {
117 cy = lmmp_add_n_sub_n_(bs1, bsm1, b1, bsm1, n);
118 bs1[n] = cy >> 1;
119 flags = (enum toom6_flags)(flags ^ toom6_vm1_neg);
120 } else {
121 cy = lmmp_add_n_sub_n_(bs1, bsm1, bsm1, b1, n);
122 bs1[n] = bsm1[n] + (cy >> 1);
123 bsm1[n] -= cy & 1;
124 }
125
126 lmmp_debug_assert(as1[n] <= 3);
127 lmmp_debug_assert(bs1[n] <= 2);
128 lmmp_debug_assert(asm1[n] <= 1);
129 lmmp_debug_assert(bsm1[n] <= 1);
130 lmmp_debug_assert(as2[n] <= 14);
131 lmmp_debug_assert(bs2[n] <= 6);
132 lmmp_debug_assert(asm2[n] <= 9);
133 lmmp_debug_assert(bsm2[n] <= 4);
134
135 /* vm1, 2n+1 limbs */
136 lmmp_mul_n_(vm1, asm1, bsm1, n + 1); /* W4 */
137
138 /* vm2, 2n+1 limbs */
139 lmmp_mul_n_(vm2, asm2, bsm2, n + 1); /* W2 */
140
141 /* v2, 2n+1 limbs */
142 lmmp_mul_n_(v2, as2, bs2, n + 1); /* W1 */
143
144 /* v1, 2n+1 limbs */
145 lmmp_mul_n_(v1, as1, bs1, n + 1); /* W3 */
146
147 /* vinf, s+t limbs */ /* W0 */
148 if (s > t)
149 lmmp_mul_(vinf, a3, s, b2, t);
150 else
151 lmmp_mul_(vinf, b2, t, a3, s);
152
153 /* v0, 2n limbs */
154 lmmp_mul_n_(v0, numa, numb, n); /* W5 */
155
156 lmmp_toom_interp6_(dst, n, flags, vm1, vm2, v2, t + s);
157
158 TEMP_S_FREE;
159#undef v0
160#undef vm1
161#undef v1
162#undef vm2
163#undef v2

166#undef bs2

176#undef b1d
177#undef a0
178#undef a1
179#undef a2
180#undef a3
181#undef b0
182#undef b1
183#undef b2
184}
