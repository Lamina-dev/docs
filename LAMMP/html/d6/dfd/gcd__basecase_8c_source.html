11mp_size_t lmmp_gcd_basecase_(mp_ptr dst, mp_srcptr up, mp_size_t un, mp_srcptr vp, mp_size_t vn) {
12 lmmp_param_assert(un > 0 && vn > 0);
13 if (un < vn) {
14 LMMP_SWAP(up, vp, mp_srcptr);
15 LMMP_SWAP(un, vn, mp_size_t);
16 } else if (un == vn) {
17 int cmp = lmmp_cmp_(up, vp, un);
18 if (cmp < 0) {
19 LMMP_SWAP(up, vp, mp_srcptr);
20 } else if (cmp == 0) {
21 lmmp_copy(dst, up, un);
22 return un;
23 }
24 }
25 TEMP_DECL;
26#define an un
27#define bn vn
28 mp_ptr a = TALLOC_TYPE(an, mp_limb_t);
29 mp_ptr b = TALLOC_TYPE(bn, mp_limb_t);
30 lmmp_copy(a, up, an);
31 lmmp_copy(b, vp, bn);
32 while (bn > 0 || (bn == 1 && b[0] == 0)) {
33 // dst = a % b;
34 lmmp_div_(NULL, dst, a, an, b, bn);
35 lmmp_copy(a, b, bn);
36 an = bn;
37 while (dst[bn - 1] == 0 && bn > 0) {
38 --bn;
39 }
40 lmmp_copy(b, dst, bn);
41 }
42 lmmp_copy(dst, a, an);
43 TEMP_FREE;
44 return an;
45#undef an
46#undef bn
47}
