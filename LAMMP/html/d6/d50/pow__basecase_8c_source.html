17 rn -= (dst[rn - 1] == 0) ? 1 : 0

19mp_size_t lmmp_pow_basecase_(mp_ptr restrict dst, mp_size_t rn, mp_srcptr restrict base, mp_size_t n, ulong exp) {
20 lmmp_param_assert(exp >= 3);
21 lmmp_param_assert(exp % 2 == 1);
22 TEMP_DECL;
23
24#define b1 base
25#define b1n n
26 mp_ptr restrict sq = TALLOC_TYPE(rn, mp_limb_t);
27 rn = n;
28 lmmp_copy(dst, base, n);
29 lmmp_sqr_(sq, dst, rn);
30 rn <<= 1;
31 rn -= (sq[rn - 1] == 0) ? 1 : 0;
32 int lz = lmmp_leading_zeros_(exp);
33 int i = 62 - lz;
34 exp <<= lz + 1;
35/*
36 For the intermediate 0, we will skip it entirely until the next 1,
37 and then perform a multiplication. This can reduce the extra copying
38 caused by sparse 1s and improve efficiency.
39 */
40 for ( ; i > 0; ) {
41 lz = lmmp_leading_zeros_(exp);
42 if (lz == 0) {
43 mul_b(1);
44
45 lmmp_sqr_(sq, dst, rn);
46 rn <<= 1;
47 rn -= (sq[rn - 1] == 0) ? 1 : 0;
48 --i;
49 exp <<= 1;
50 } else {
51 int j = 2;
52 if (lz & 1) {
53 lmmp_copy(dst, sq, rn);
54 lmmp_sqr_(sq, dst, rn);
55 rn <<= 1;
56 rn -= (sq[rn - 1] == 0);
57 ++j;
58 for (; j <= lz; j += 2) {
59 lmmp_sqr_(dst, sq, rn);
60 rn <<= 1;
61 rn -= (dst[rn - 1] == 0);
62 lmmp_sqr_(sq, dst, rn);
63 rn <<= 1;
64 rn -= (sq[rn - 1] == 0);
65 }
66 } else {
67 for (; j <= lz; j += 2) {
68 lmmp_sqr_(dst, sq, rn);
69 rn <<= 1;
70 rn -= (dst[rn - 1] == 0);
71 lmmp_sqr_(sq, dst, rn);
72 rn <<= 1;
73 rn -= (sq[rn - 1] == 0);
74 }
75 }
76
77 i -= lz;
78 exp <<= lz;
79 }
80 }
81 lmmp_debug_assert(exp == LIMB_B_2);
82
83 mul_b(1);
84
85 TEMP_FREE;
86 return rn;
87

89#undef b1n
90#undef new_b
91}
