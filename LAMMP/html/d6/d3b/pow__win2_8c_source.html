32mp_size_t lmmp_pow_win2_(mp_ptr restrict dst, mp_size_t rn, mp_srcptr restrict base, mp_size_t n, ulong exp) {
33 lmmp_param_assert(exp > 0);
34 lmmp_param_assert(n > 0);
35 lmmp_param_assert(dst != NULL);
36 TEMP_DECL;
37
38#define b1 base
39#define b1n n
40#define new_b(_i_) mp_ptr restrict b##_i_ = TALLOC_TYPE(b##_i_##n, mp_limb_t)
41
42 mp_size_t b2n = n << 1, b3n = b2n + n;
43 new_b(2);
44 new_b(3);
45
46 lmmp_sqr_(b2, b1, b1n);
47 b2n = b1n << 1;
48 b2n -= b2[b2n - 1] == 0 ? 1 : 0;
49 lmmp_mul_(b3, b2, b2n, b1, b1n);
50 b3n = b2n + b1n;
51 b3n -= b3[b3n - 1] == 0 ? 1 : 0;
52
53 bool rsq = true;
54 mp_ptr restrict sq = BALLOC_TYPE(rn, mp_limb_t);
55 rn = 1;
56 int i = 31, j;
57 while ((exp & (0x3ull << ((i--) * 2))) == 0);
58
59 /*
60 模拟一次，让最后一次计算结果恰好是dst，避免最后一次可能的拷贝
61 */
62 for (j = i + 1; j != -1; --j) {
63 int p = (exp & (0x3ull << (j * 2))) >> (j * 2);
64 if (p != 0)
65 rsq = !rsq;
66 }
67 if (rsq) {
68 dst[0] = 1;
69 rsq = false;
70 } else {
71 sq[0] = 1;
72 rsq = true;
73 }
74
75 for (++i; i != -1; --i) {
76 int p = (exp & (0x3ull << (i * 2))) >> (i * 2);
77 if (p == 0) {
78 ;
79 } else if (p == 1) {
80 mul_b(1)
81 } else if (p == 2) {
82 mul_b(2)
83 } else {
84 mul_b(3)
85 }
86
87 if (i > 0) {
88 if (rsq) {
89 lmmp_sqr_(dst, sq, rn);
90 rn <<= 1;
91 rn -= (dst[rn - 1] == 0) ? 1 : 0;
92 lmmp_sqr_(sq, dst, rn);
93 rn <<= 1;
94 rn -= (sq[rn - 1] == 0) ? 1 : 0;
95 } else {
96 lmmp_sqr_(sq, dst, rn);
97 rn <<= 1;
98 rn -= (sq[rn - 1] == 0) ? 1 : 0;
99 lmmp_sqr_(dst, sq, rn);
100 rn <<= 1;
101 rn -= (dst[rn - 1] == 0) ? 1 : 0;
102 }
103 }
104 }
105 lmmp_debug_assert(rsq == false);
106
107 TEMP_FREE;
108 return rn;

112#undef new_b
