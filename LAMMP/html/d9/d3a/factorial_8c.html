197mp_size_t lmmp_factorial_(mp_ptr restrict dst, mp_bitcnt_t bits, mp_size_t rn, uint n) {
198 mp_size_t shw = bits / LIMB_BITS;
199 bits %= LIMB_BITS;
200 lmmp_zero(dst, shw);
201 if (n <= NPR_SHORT_LIMIT)
202 rn = lmmp_odd_nPr_ushort_(dst + shw, rn - shw, n, n);
203 else
204 rn = lmmp_odd_factorial_uint_(dst + shw, rn - shw, n);
205
206 if (bits > 0) {
207 dst[shw + rn] = lmmp_shl_(dst + shw, dst + shw, rn, bits);
208 rn += shw + 1;
209 rn -= dst[rn - 1] == 0;
210 } else {
211 rn += shw;
212 }
213 return rn;
