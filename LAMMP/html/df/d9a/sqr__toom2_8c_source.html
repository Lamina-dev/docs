20/*
21Evaluate in: -1, 0, +inf
22
23 <-s--><--n-->
24 |-a1-|--a0--|
25
26v0 = a0 ^2 # A(0)^2
27vm1 = (a0-a1)^2 # A(-1)^2
28vinf= a1 ^2 # A(inf)^2
29*/
30
31void lmmp_sqr_toom2_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na) {
32 lmmp_param_assert(na > 0);
33 lmmp_param_assert(dst != NULL);
34 lmmp_param_assert(numa!= NULL);
35 TEMP_S_DECL;
36 mp_size_t s = na >> 1, n = na - s;
37 mp_limb_t* vm1 = SALLOC_TYPE(2 * n, mp_limb_t);
38 mp_slimb_t cy, cy2;
39
40#define a0 numa
41#define a1 (numa + n)
42#define asm1 dst
43
44 if (s == n) {
45 if (lmmp_cmp_(a0, a1, n) < 0)
46 lmmp_sub_n_(asm1, a1, a0, n);
47 else
48 lmmp_sub_n_(asm1, a0, a1, n);
49 } else { // s==n-1
50 if (a0[s] == 0 && lmmp_cmp_(a0, a1, s) < 0) {
51 lmmp_sub_n_(asm1, a1, a0, s);
52 asm1[s] = 0;
53 } else
54 asm1[s] = a0[s] - lmmp_sub_n_(asm1, a0, a1, s);
55 }
56
57 lmmp_sqr_(vm1, asm1, n);
58
59#undef asm1
60#define v0 dst
61#define vinf (dst + 2 * n)
62
63 lmmp_sqr_(v0, a0, n);
64
65 lmmp_sqr_(vinf, a1, s);
66
67 cy = lmmp_add_n_(dst + 2 * n, v0 + n, vinf, n);
68 cy2 = cy + lmmp_add_n_(dst + n, dst + 2 * n, v0, n);
69 cy += lmmp_add_(dst + 2 * n, dst + 2 * n, n, vinf + n, s + s - n);
70
71 cy -= lmmp_sub_n_(dst + n, dst + n, vm1, 2 * n);
72
73 // no overflow.
74 lmmp_inc_1(dst + 2 * n, cy2);
75
76 if (cy < 0)
77 lmmp_dec(dst + 3 * n);
78 else
79 lmmp_inc_1(dst + 3 * n, cy);
80 TEMP_S_FREE;
81}
