23/*
24Evaluate in:
25 0, +1, -1, +2, -2, 1/2, +inf
26
27 <-s-><--n--><--n--><--n--><--n--><--n-->
28 |a5-|--a4--|--a3--|--a2--|--a1--|--a0--|
29 |-b1-|--b0--|
30 <-t--><--n-->
31
32 v0 = a0 * b0 # A(0)*B(0)
33 v1 = ( a0+ a1+ a2+ a3+ a4+ a5)*( b0+ b1) # A(1)*B(1) ah <= 5 bh <= 1
34 vm1 = ( a0- a1+ a2- a3+ a4- a5)*( b0- b1) # A(-1)*B(-1) |ah| <= 2 bh = 0
35 v2 = ( a0+ 2a1+4a2+8a3+16a4+32a5)*( b0+2b1) # A(2)*B(2) ah <= 62 bh <= 2
36 vm2 = ( a0- 2a1+4a2-8a3+16a4-32a5)*( b0-2b1) # A(-2)*B(-2) -41<=ah<=20 -1<=bh<=0
37 vh = (32a0+16a1+8a2+4a3+ 2a4+ a5)*(2b0+ b1) # A(1/2)*B(1/2) ah <= 62 bh <= 2
38 vinf= a5 * b1 # A(inf)*B(inf)
39*/
40
41void lmmp_mul_toom62_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na, mp_srcptr restrict numb, mp_size_t nb) {
42 lmmp_param_assert(na >= 3 * nb);
43 lmmp_param_assert(5 * nb >= na);
44
45 mp_size_t n, s, t;
46 mp_limb_t cy;
47 mp_ptr as1, asm1, as2, asm2, ash;
48 mp_ptr bs1, bsm1, bs2, bsm2, bsh;
49 mp_ptr gp;
50 enum toom7_flags aflags, bflags;
51 TEMP_S_DECL;
52
53#define a0 numa
54#define a1 (numa + n)
55#define a2 (numa + 2 * n)
56#define a3 (numa + 3 * n)
57#define a4 (numa + 4 * n)
58#define a5 (numa + 5 * n)
59#define b0 numb
60#define b1 (numb + n)
61
62 n = 1 + (na >= 3 * nb ? (na - 1) / (mp_size_t)6 : (nb - 1) >> 1);
63
64 s = na - 5 * n;
65 t = nb - n;
66
67 lmmp_debug_assert(0 < s && s <= n);
68 lmmp_debug_assert(0 < t && t <= n);
69
70 mp_ptr restrict scratch = SALLOC_TYPE(10 * n + 10, mp_limb_t);
71
72 mp_ptr restrict tmp = SALLOC_TYPE(10 * n + 10, mp_limb_t);
73 as1 = tmp;
74 asm1 = as1 + n + 1;
75 as2 = asm1 + n + 1;
76 asm2 = as2 + n + 1;
77 ash = asm2 + n + 1;
78 bs1 = ash + n + 1;
79 bsm1 = bs1 + n + 1;
80 bs2 = bsm1 + n;
81 bsm2 = bs2 + n + 1;
82 bsh = bsm2 + n + 1;
83
84 gp = dst;
85
86 /* Compute as1 and asm1. */
87 aflags = (enum toom7_flags)(toom7_w3_neg & lmmp_toom_eval_pm1_(as1, asm1, 5, numa, n, s, gp));
88
89 /* Compute as2 and asm2. */
90 aflags = (enum toom7_flags)(aflags | (toom7_w1_neg & lmmp_toom_eval_pm2_(as2, asm2, 5, numa, n, s, gp)));
91
92 /* Compute ash = 32 a0 + 16 a1 + 8 a2 + 4 a3 + 2 a4 + a5
93 = 2*(2*(2*(2*(2*a0 + a1) + a2) + a3) + a4) + a5 */
94
95 cy = lmmp_addshl1_n_(ash, a1, a0, n);
96 cy = 2 * cy + lmmp_addshl1_n_(ash, a2, ash, n);
97 cy = 2 * cy + lmmp_addshl1_n_(ash, a3, ash, n);
98 cy = 2 * cy + lmmp_addshl1_n_(ash, a4, ash, n);
99 if (s < n) {
100 mp_limb_t cy2;
101 cy2 = lmmp_addshl1_n_(ash, a5, ash, s);
102 ash[n] = 2 * cy + lmmp_shl_(ash + s, ash + s, n - s, 1);
103 lmmp_inc_1(ash + s, cy2);
104 } else
105 ash[n] = 2 * cy + lmmp_addshl1_n_(ash, a5, ash, n);
106
107 /* Compute bs1 and bsm1. */
108 if (t == n) {
109 if (lmmp_cmp_(b0, b1, n) < 0) {
110 cy = lmmp_add_n_sub_n_(bs1, bsm1, b1, b0, n);
111 bflags = toom7_w3_neg;
112 } else {
113 cy = lmmp_add_n_sub_n_(bs1, bsm1, b0, b1, n);
114 bflags = (enum toom7_flags)0;
115 }
116 bs1[n] = cy >> 1;
117 } else {
118 bs1[n] = lmmp_add_(bs1, b0, n, b1, t);
119 if (lmmp_zero_q_(b0 + t, n - t) && lmmp_cmp_(b0, b1, t) < 0) {
120 lmmp_sub_n_(bsm1, b1, b0, t);
121 lmmp_zero(bsm1 + t, n - t);
122 bflags = toom7_w3_neg;
123 } else {
124 lmmp_sub_(bsm1, b0, n, b1, t);
125 bflags = (enum toom7_flags)0;
126 }
127 }
128
129 /* Compute bs2 and bsm2. Recycling bs1 and bsm1; bs2=bs1+b1, bsm2 =
130 bsm1 - b1 */
131 lmmp_add_(bs2, bs1, n + 1, b1, t);
132 if (bflags & toom7_w3_neg) {
133 bsm2[n] = lmmp_add_(bsm2, bsm1, n, b1, t);
134 bflags = (enum toom7_flags)(bflags | toom7_w1_neg);
135 } else {
136 if (t < n) {
137 if (lmmp_zero_q_(bsm1 + t, n - t) && lmmp_cmp_(bsm1, b1, t) < 0) {
138 lmmp_sub_n_(bsm2, b1, bsm1, t);
139 lmmp_zero(bsm2 + t, n + 1 - t);
140 bflags = (enum toom7_flags)(bflags | toom7_w1_neg);
141 } else {
142 lmmp_sub_(bsm2, bsm1, n, b1, t);
143 bsm2[n] = 0;
144 }
145 } else {
146 if (lmmp_cmp_(bsm1, b1, n) < 0) {
147 lmmp_sub_n_(bsm2, b1, bsm1, n);
148 bflags = (enum toom7_flags)(bflags | toom7_w1_neg);
149 } else {
150 lmmp_sub_n_(bsm2, bsm1, b1, n);
151 }
152 bsm2[n] = 0;
153 }
154 }
155
156 /* Compute bsh, recycling bs1. bsh=bs1+b0; */
157 bsh[n] = bs1[n] + lmmp_add_n_(bsh, bs1, b0, n);
158
159 lmmp_debug_assert(as1[n] <= 5);
160 lmmp_debug_assert(bs1[n] <= 1);
161 lmmp_debug_assert(asm1[n] <= 2);
162 lmmp_debug_assert(as2[n] <= 62);
163 lmmp_debug_assert(bs2[n] <= 2);
164 lmmp_debug_assert(asm2[n] <= 41);
165 lmmp_debug_assert(bsm2[n] <= 1);
166 lmmp_debug_assert(ash[n] <= 62);
167 lmmp_debug_assert(bsh[n] <= 2);
168
169#define v0 dst /* 2n */
170#define v1 (dst + 2 * n) /* 2n+1 */
171#define vinf (dst + 6 * n) /* s+t */
172#define v2 scratch /* 2n+1 */
173#define vm2 (scratch + 2 * n + 1) /* 2n+1 */
174#define vh (scratch + 4 * n + 2) /* 2n+1 */
175#define vm1 (scratch + 6 * n + 3) /* 2n+1 */
176#define scratch_out (scratch + 8 * n + 4) /* 2n+1 */
177 /* Total scratch need: 10*n+5 */
178
179 /* Must be in allocation order, as they overwrite one limb beyond
180 * 2n+1. */
181 lmmp_mul_n_(v2, as2, bs2, n + 1); /* v2, 2n+1 limbs */
182 lmmp_mul_n_(vm2, asm2, bsm2, n + 1); /* vm2, 2n+1 limbs */
183 lmmp_mul_n_(vh, ash, bsh, n + 1); /* vh, 2n+1 limbs */
184
185 /* vm1, 2n+1 limbs */
186 lmmp_mul_n_(vm1, asm1, bsm1, n);
187 cy = 0;
188 if (asm1[n] == 1) {
189 cy = lmmp_add_n_(vm1 + n, vm1 + n, bsm1, n);
190 } else if (asm1[n] == 2) {
191 cy = lmmp_addshl1_n_(vm1 + n, vm1 + n, bsm1, n);
192 }
193 vm1[2 * n] = cy;
194
195 /* v1, 2n+1 limbs */
196 lmmp_mul_n_(v1, as1, bs1, n);
197 if (as1[n] == 1) {
198 cy = bs1[n] + lmmp_add_n_(v1 + n, v1 + n, bs1, n);
199 } else if (as1[n] == 2) {
200 cy = 2 * bs1[n] + lmmp_addshl1_n_(v1 + n, v1 + n, bs1, n);
201 } else if (as1[n] != 0) {
202 cy = as1[n] * bs1[n] + lmmp_addmul_1_(v1 + n, bs1, n, as1[n]);
203 } else
204 cy = 0;
205 if (bs1[n] != 0)
206 cy += lmmp_add_n_(v1 + n, v1 + n, as1, n);
207 v1[2 * n] = cy;
208
209 lmmp_mul_n_(v0, a0, b0, n); /* v0, 2n limbs */
210
211 /* vinf, s+t limbs */
212 if (s > t)
213 lmmp_mul_(vinf, a5, s, b1, t);
214 else
215 lmmp_mul_(vinf, b1, t, a5, s);
216
217 lmmp_toom_interp7_(dst, n, (enum toom7_flags)(aflags ^ bflags), vm2, vm1, v2, vh, s + t, scratch_out);
218
219 TEMP_S_FREE;
220}

531 else

533

576 na -= 3 * nb;
