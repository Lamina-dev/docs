42int lmmp_stack_deinit(void) {
43 if (lmmp_tmpmem_ctx.stack_begin != NULL) {
44 heap_free_func(lmmp_tmpmem_ctx.stack_begin);
45 if (lmmp_tmpmem_ctx.pool_begin != NULL)
46 heap_free_func(lmmp_tmpmem_ctx.pool_begin);
47 lmmp_tmpmem_ctx.stack_begin = NULL;
48 lmmp_tmpmem_ctx.stack_end = NULL;
49 lmmp_tmpmem_ctx.stack_top = NULL;
50 lmmp_tmpmem_ctx.pool_begin = NULL;
51 lmmp_tmpmem_ctx.pool_top = NULL;
52 lmmp_tmpmem_ctx.remain = 0; // Reset remain to 0
53 lmmp_tmpmem_ctx.capacity = 0; // Reset capacity to 0
54 return 0;
55 } else {
56 return -1;
57 }
58}

60int lmmp_stack_init(size_t size) {
61 if (lmmp_tmpmem_ctx.stack_begin != NULL) {
62 return -1;
63 } else {
64 lmmp_tmpmem_ctx.stack_begin = heap_alloc_func(LAMMP_DEFAULT_STACK_SIZE);
65 if (lmmp_tmpmem_ctx.stack_begin == NULL) {
66 lmmp_abort(LAMMP_ERROR_MEMORY_ALLOC_FAILURE, "Failed to allocate stack memory", __func__, __LINE__);
67 }
68 lmmp_tmpmem_ctx.stack_end = (mp_byte_t*)(lmmp_tmpmem_ctx.stack_begin) + LAMMP_DEFAULT_STACK_SIZE;
69 lmmp_tmpmem_ctx.stack_top = lmmp_tmpmem_ctx.stack_begin;
70 if (size > LAMMP_MAX_ALIGN) {
71 // 只有当size大于最大对齐单位时，才分配缓冲池
72 lmmp_tmpmem_ctx.pool_begin = heap_alloc_func(size);
73 if (lmmp_tmpmem_ctx.pool_begin == NULL) {
74 heap_free_func(lmmp_tmpmem_ctx.stack_begin); // Free old stack memory
75 lmmp_abort(LAMMP_ERROR_MEMORY_ALLOC_FAILURE, "Failed to allocate pool memory", __func__, __LINE__);
76 }
77 lmmp_tmpmem_ctx.pool_top = lmmp_tmpmem_ctx.pool_begin;
78 lmmp_tmpmem_ctx.remain = size;
79 lmmp_tmpmem_ctx.capacity = size;
80 }
81 return 0;
82 }
83}

85void lmmp_set_heap_allocator(const lmmp_heap_allocator_t* heap) {

115void lmmp_leak_tracker(const char* func, int line) {
116 char msg[360] = {0};
117 int offset = 0;
118 const int max_len = sizeof(msg) - 1;
119 bool t = false;
120 if (heap_alloc_count != 0) {
121 offset +=
122 snprintf(msg + offset, max_len - offset, "Heap allocations not freed: %d block(s);\n", heap_alloc_count);
123 t = true;
124 }
125 if (lmmp_tmpmem_ctx.stack_top != lmmp_tmpmem_ctx.stack_begin) {
126 offset += snprintf(msg + offset, max_len - offset,
127 "Default stack allocator is not empty. top: %p, begin: %p, end: %p;\n",
128 lmmp_tmpmem_ctx.stack_top, lmmp_tmpmem_ctx.stack_begin, lmmp_tmpmem_ctx.stack_end);
129 t = true;
130 }
131 if (lmmp_tmpmem_ctx.pool_top != lmmp_tmpmem_ctx.pool_begin) {
132 offset += snprintf(msg + offset, max_len - offset,
133 "Default pool allocator is not empty. top: %p, begin: %p; remain: %zu bytes, capacity: %zu "
134 "bytes;\n",
135 lmmp_tmpmem_ctx.pool_top, lmmp_tmpmem_ctx.pool_begin, lmmp_tmpmem_ctx.remain,
136 lmmp_tmpmem_ctx.capacity);
137 t = true;
138 }
139 if (t) {
140 lmmp_abort(LAMMP_ERROR_MEMORY_LEAK, msg, func, line);
141 }
142}
