29static inline int __lmmp_limb_bits_(mp_limb_t x) {

32 x >>= 1;

38static inline int __lmmp_limb_popcnt_(mp_limb_t x) {
39#if defined(__GNUC__) || defined(__clang__)
40 mp_limb_t count;
41#if defined(__x86_64__) && defined(USE_ASM)
42 __asm__ volatile("popcnt %1, %0" : "=r"(count) : "r"(x) : "cc");
43#else
44 count = __builtin_popcountll(x);
45#endif
46 return count;
47#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
48 return (int)__popcnt64(x);
49#else
50 int k = 0;
51 while (x) {
52 k += x & 1;
53 x >>= 1;
54 }
55 return k;
56#endif
57}

59static inline int __lmmp_leading_zeros_(mp_limb_t x) {
60 if (x == 0) return 64;
61#ifdef __GNUC__
62 return __builtin_clzll(x);
63#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
64 unsigned long index;
65 _BitScanReverse64(&index, x);
66 return 63 - (int)index;
67#else
68 int n = 0;
69 if (x <= 0x00000000FFFFFFFF) { n += 32; x <<= 32; }
70 if (x <= 0x0000FFFFFFFFFFFF) { n += 16; x <<= 16; }
71 if (x <= 0x00FFFFFFFFFFFFFF) { n += 8; x <<= 8; }
72 if (x <= 0x0FFFFFFFFFFFFFFF) { n += 4; x <<= 4; }
73 if (x <= 0x3FFFFFFFFFFFFFFF) { n += 2; x <<= 2; }
74 if (x <= 0x7FFFFFFFFFFFFFFF) { n += 1; x <<= 1; }
75 return n;
76#endif
77}

79static inline int __lmmp_tailing_zeros_(mp_limb_t x) {
80 if (x == 0) return 64;
81#ifdef __GNUC__
82 return __builtin_ctzll(x);
83#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
84 unsigned long index;
85 _BitScanForward64(&index, x);
86 return (int)index;
87#else
88 int n = 0;
89 if ((x & 0x00000000FFFFFFFF) == 0) { n += 32; x >>= 32; }
90 if ((x & 0x000000000000FFFF) == 0) { n += 16; x >>= 16; }
91 if ((x & 0x00000000000000FF) == 0) { n += 8; x >>= 8; }
92 if ((x & 0x000000000000000F) == 0) { n += 4; x >>= 4; }
93 if ((x & 0x0000000000000003) == 0) { n += 2; x >>= 2; }
94 if ((x & 0x0000000000000001) == 0) { n += 1; x >>= 1; }
95 return n;
96#endif
97}

99static inline void __lmmp_mullh_(mp_limb_t a, mp_limb_t b, mp_ptr restrict dst) {
100#if (defined(__GNUC__) || defined(__clang__)) && defined(__SIZEOF_INT128__)
101 __uint128_t prod = (__uint128_t)a * b;
102 dst[0] = (mp_limb_t)prod;
103 dst[1] = (mp_limb_t)(prod >> 64);
104#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
105 dst[0] = _umul128(a, b, dst + 1);
106#else
107 uint64_t ah = a >> 32, bh = b >> 32;
108 a = (uint32_t)a, b = (uint32_t)b;
109 uint64_t r0 = a * b, r1 = a * bh, r2 = ah * b, r3 = ah * bh;
110 r3 += (r1 >> 32) + (r2 >> 32);
111 r1 = (uint32_t)r1, r2 = (uint32_t)r2;
112 r1 += r2;
113 r1 += (r0 >> 32);
114 dst[1] = r3 + (r1 >> 32);
115 dst[0] = (r1 << 32) | (uint32_t)r0;
116#endif
117}

119static inline mp_limb_t __lmmp_mulh_(mp_limb_t a, mp_limb_t b) {
120#if (defined(__GNUC__) || defined(__clang__)) && defined(__SIZEOF_INT128__)
121 __uint128_t t = (__uint128_t)a * (__uint128_t)b;
122 return (mp_limb_t)(t >> 64);
123#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
124 return __umulh(a, b);
125#else
126 uint64_t ah = a >> 32, bh = b >> 32;
127 a = (uint32_t)a, b = (uint32_t)b;
128 uint64_t r0 = a * b, r1 = a * bh, r2 = ah * b, r3 = ah * bh;
129 r3 += (r1 >> 32) + (r2 >> 32);
130 r1 = (uint32_t)r1, r2 = (uint32_t)r2;
131 r1 += r2;
132 r1 += (r0 >> 32);
133 return r3 + (r1 >> 32);
134#endif
135}
