15mont63 只能用于小于 2^63 的数，否则会溢出导致计算结果不正确
16mont64 可以用于任意大小的数，但由于考虑了溢出的情况，所以速度理论上会慢一些（并未详细测试）
17*/

20static inline ulong mont64_reduce(u128 t, ulong m, ulong m_inv) {
21 ulong k = t[0] * m_inv;
22 u192 tmp;
23 _u128mul(tmp, k, m);
24 tmp[0] += t[0];
25 ulong c = tmp[0] < t[0];
26 tmp[1] += c;
27 c = tmp[1] == 0;
28 tmp[1] += t[1];
29 c = tmp[1] < t[1];
30 tmp[2] = c;
31 if (!c) {
32 ulong res = tmp[1] >= m ? tmp[1] - m : tmp[1];
33 return res;
34 } else {
35 _u128sub64(tmp + 1, tmp + 1, m);
36 return tmp[1];
37 }
38}

40static inline ulong mont64_R2(ulong m) {
41 u192 r = {0, 0, 1};

64static inline ulong mont63_reduce(u128 t, ulong m, ulong m_inv) {
65 ulong k = t[0] * m_inv;
66 u128 tmp;
67 _u128mul(tmp, k, m);
68 _u128add(tmp, tmp, t);
69 ulong res = tmp[1] >= m ? tmp[1] - m : tmp[1];
70 return res;
71}

73static inline ulong mont63_R2(ulong m) {
74 int shift = lmmp_leading_zeros_(m);
75 u192 r = {0, 0, 1ull << shift};
76 u128 q;
77 lmmp_div_1_s_(q, r, 3, m << shift);
78 return r[0] >> shift;
79}

98uint lmmp_powmod_uint_(uint base, ulong exp, uint mod) {
99 lmmp_param_assert(mod > base);
100 lmmp_param_assert(mod > 1);
101 ulong dst = 1;
102 ulong b = base;
103 _udiv64_t binv = _udiv64_gen(mod);
104 ulong q;
105 while (1) {
106 if (exp & 1) {
107 dst *= b;
108 q = _udiv64by64_q_preinv(dst, &binv);
109 dst -= q * mod;
110 }
111 exp >>= 1;
112 if (exp == 0)
113 break;
114 b *= b;
115 q = _udiv64by64_q_preinv(b, &binv);
116 b -= q * mod;
117 }
118 return dst;
119}

121ulong lmmp_powmod_ulong_(ulong base, ulong exp, ulong mod) {
122 if (mod <= MP_UINT_MAX)
123 return lmmp_powmod_uint_(base, exp, mod);
124 else if (mod <= MONT63_MAX) {
125 ulong R2 = mont63_R2(mod);
126 ulong m_inv = lmmp_binvert_ulong_(mod);
127 m_inv = -m_inv;
128 ulong dst = to_mont63(1, R2, mod, m_inv);
129 base = to_mont63(base, R2, mod, m_inv);
130 while (1) {
131 if (exp & 1)
132 dst = mont63_mul(dst, base, mod, m_inv);
133 exp >>= 1;
134 if (exp == 0)
135 break;
136 base = mont63_mul(base, base, mod, m_inv);
137 }
138 return from_mont63(dst, mod, m_inv);
139 } else {
140 ulong R2 = mont64_R2(mod);
141 ulong m_inv = lmmp_binvert_ulong_(mod);
142 m_inv = -m_inv;
143 ulong dst = to_mont64(1, R2, mod, m_inv);
144 base = to_mont64(base, R2, mod, m_inv);
145 while (1) {
146 if (exp & 1)
147 dst = mont64_mul(dst, base, mod, m_inv);
148 exp >>= 1;
149 if (exp == 0)
150 break;
151 base = mont64_mul(base, base, mod, m_inv);
152 }
153 return from_mont64(dst, mod, m_inv);
154 }
155}

157static inline int miller_rabin_32(ulong a, ulong t, ulong u, uint m, _udiv64_t* binv) {
158 ulong v = 1;
159 ulong base = a;
160 ulong q;
161 while (1) {
162 if (u & 1) {
163 v *= base;
164 q = _udiv64by64_q_preinv(v, binv);
165 v -= q * m;
166 }
167 u >>= 1;
168 if (u == 0)
169 break;
170 base *= base;
171 q = _udiv64by64_q_preinv(base, binv);
172 base -= q * m;
173 }
174
175 if (v == 1 || v == m - 1)
176 return 1;
177 for (ulong j = 1; j < t; ++j) {
178 v *= v;
179 q = _udiv64by64_q_preinv(v, binv);
180 v -= q * m;
181 if (v == m - 1)
182 return 1;
183 if (v == 1)
184 return 0;
185 }
186 return 0;
187}

189static inline int miller_rabin_63(ulong a, ulong t, ulong u, ulong m, ulong m_inv, ulong one, ulong m_1) {
190 ulong v = one;
191 ulong base = a;
192 while (1) {
193 if (u & 1)
194 v = mont63_mul(v, base, m, m_inv);
195 u >>= 1;
196 if (u == 0)
197 break;
198 base = mont63_mul(base, base, m, m_inv);
199 }
200 if (v == one || v == m_1)
201 return 1;
202
203 for (ulong j = 1; j < t; ++j) {
204 v = mont63_mul(v, v, m, m_inv);
205 if (v == m_1)
206 return 1;
207 if (v == one)
208 return 0;
209 }
210 return 0;
211}

249bool lmmp_is_prime_uint_(uint n) {
250 if (n < PRIME_SHORT_TABLE_N) {
251 return lmmp_is_prime_table_(n);
252 }
253 if (n % 2 == 0)
254 return false;
255 if (trial_div35711(n))
256 return false;
257 ushort bases[2];
258 bases[0] = 2;
259 bases[1] = dj_base49[((0x3AC69A35UL * n) & 0xFFFFFFFFUL) >> 21] + 3;
260 if (n % bases[0] == 0)
261 return false;
262 if (n % bases[1] == 0)
263 return false;
264
265 ulong u = n - 1, t = 0;
266 while (u % 2 == 0) u /= 2, ++t;
267
268 _udiv64_t binv = _udiv64_gen(n);
269 if (miller_rabin_32(bases[0], t, u, n, &binv))
270 if (miller_rabin_32(bases[1], t, u, n, &binv))
271 return true;
272 else
273 return false;
274 else
275 return false;
276}

278bool lmmp_is_prime_notrial_(ulong n) {
279 lmmp_param_assert(n > 1);
280 if (n < 684630005672341) {
281 ushort bases[2];
282 bases[0] = 2;
283 bases[1] = dj_base49[((0x3AC69A35UL * n) & 0xFFFFFFFFUL) >> 21] + 3;
284 if (n % bases[0] == 0)
285 return false;
286 if (n % bases[1] == 0)
287 return false;
288
289 ulong one = 1;
290 ulong m_1 = n - 1;
291 ulong m_inv = lmmp_binvert_ulong_(n);
292 m_inv = -m_inv;
293 ulong R2 = mont63_R2(n);
294 one = to_mont63(one, R2, n, m_inv);
295 m_1 = to_mont63(m_1, R2, n, m_inv);
296
297 ulong u = n - 1, t = 0;
298 while (u % 2 == 0) u /= 2, ++t;
299
300 if (miller_rabin_63(bases[0], t, u, n, m_inv, one, m_1))
301 if (miller_rabin_63(bases[1], t, u, n, m_inv, one, m_1))
302 return true;
303 else
304 return false;
305 else
306 return false;
307 } else {
308 ushort bases[3];
309 ulong bbmask = dj_base64[((0x3AC69A35UL * n) & 0xFFFFFFFFUL) >> 18];
310 bases[0] = 2;
311 bases[1] = (bbmask & 0x8000) ? 26460 : 9375;
312 bases[2] = (bbmask & 0x7FFF) + 3;
313
314 if (n % bases[0] == 0)
315 return false;
316 if (n % bases[1] == 0)
317 return false;
318 if (n % bases[2] == 0)
319 return false;
320
321 ulong one = 1;
322 ulong m_1 = n - 1;
323 ulong m_inv = lmmp_binvert_ulong_(n);
324 m_inv = -m_inv;
325
326 if (n <= MONT63_MAX) {
327 ulong R2 = mont63_R2(n);
328 one = to_mont63(one, R2, n, m_inv);
329 m_1 = to_mont63(m_1, R2, n, m_inv);
330
331 ulong u = n - 1, t = 0;
332 while (u % 2 == 0) u /= 2, ++t;
333
334 if (miller_rabin_63(bases[0], t, u, n, m_inv, one, m_1))
335 if (miller_rabin_63(bases[1], t, u, n, m_inv, one, m_1))
336 if (miller_rabin_63(bases[2], t, u, n, m_inv, one, m_1))
337 return true;
338 else
339 return false;
340 else
341 return false;
342 else
343 return false;
344 } else {
345 ulong R2 = mont64_R2(n);
346 one = to_mont64(one, R2, n, m_inv);
347 m_1 = to_mont64(m_1, R2, n, m_inv);
348
349 ulong u = n - 1, t = 0;
350 while (u % 2 == 0) u /= 2, ++t;
351
352 if (miller_rabin_64(bases[0], t, u, n, m_inv, one, m_1))
353 if (miller_rabin_64(bases[1], t, u, n, m_inv, one, m_1))
354 if (miller_rabin_64(bases[2], t, u, n, m_inv, one, m_1))
355 return true;
356 else
357 return false;
358 else
359 return false;
360 else
361 return false;
362 }
363 }

438 if (n < prime_short_table[PRIME_SHORT_TABLE_SIZE - 1]) {
439 ushort idx = lmmp_prime_cnt16_(n);

442 return MP_ULONG_MAX;

455 n += 2;
