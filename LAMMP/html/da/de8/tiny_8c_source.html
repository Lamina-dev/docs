41int lmmp_leading_zeros_(mp_limb_t x) {
42 if (x == 0) return 64;
43#ifdef __GNUC__
44 return __builtin_clzll(x);
45#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
46 unsigned long index;
47 _BitScanReverse64(&index, x);
48 return 63 - (int)index;
49#else
50 int n = 0;
51 if (x <= 0x00000000FFFFFFFF) { n += 32; x <<= 32; }
52 if (x <= 0x0000FFFFFFFFFFFF) { n += 16; x <<= 16; }
53 if (x <= 0x00FFFFFFFFFFFFFF) { n += 8; x <<= 8; }
54 if (x <= 0x0FFFFFFFFFFFFFFF) { n += 4; x <<= 4; }
55 if (x <= 0x3FFFFFFFFFFFFFFF) { n += 2; x <<= 2; }
56 if (x <= 0x7FFFFFFFFFFFFFFF) { n += 1; x <<= 1; }
57 return n;
58#endif
59}

61int lmmp_tailing_zeros_(mp_limb_t x) {
62 if (x == 0) return 64;
63#ifdef __GNUC__
64 return __builtin_ctzll(x);
65#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
66 unsigned long index;
67 _BitScanForward64(&index, x);
68 return (int)index;
69#else
70 int n = 0;
71 if ((x & 0x00000000FFFFFFFF) == 0) { n += 32; x >>= 32; }
72 if ((x & 0x000000000000FFFF) == 0) { n += 16; x >>= 16; }
73 if ((x & 0x00000000000000FF) == 0) { n += 8; x >>= 8; }
74 if ((x & 0x000000000000000F) == 0) { n += 4; x >>= 4; }
75 if ((x & 0x0000000000000003) == 0) { n += 2; x >>= 2; }
76 if ((x & 0x0000000000000001) == 0) { n += 1; x >>= 1; }
77 return n;
78#endif

81mp_limb_t lmmp_mulh_(mp_limb_t a, mp_limb_t b) {
82#if (defined(__GNUC__) || defined(__clang__)) && defined(__SIZEOF_INT128__)
83 __uint128_t t = (__uint128_t)a * (__uint128_t)b;
84 return (mp_limb_t)(t >> 64);
85#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
86 return __umulh(a, b);
87#else
88 uint64_t ah = a >> 32, bh = b >> 32;
89 a = (uint32_t)a, b = (uint32_t)b;
90 uint64_t r0 = a * b, r1 = a * bh, r2 = ah * b, r3 = ah * bh;
91 r3 += (r1 >> 32) + (r2 >> 32);
92 r1 = (uint32_t)r1, r2 = (uint32_t)r2;
93 r1 += r2;
94 r1 += (r0 >> 32);
95 return r3 + (r1 >> 32);
96#endif
97}

99void lmmp_mullh_(mp_limb_t a, mp_limb_t b, mp_ptr restrict dst) {
100#if (defined(__GNUC__) || defined(__clang__)) && defined(__SIZEOF_INT128__)
101 __uint128_t prod = (__uint128_t)a * b;
102 dst[0] = (mp_limb_t)prod;
103 dst[1] = (mp_limb_t)(prod >> 64);
104#elif defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
105 dst[0] = _umul128(a, b, dst + 1);
106#else
107 uint64_t ah = a >> 32, bh = b >> 32;
108 a = (uint32_t)a, b = (uint32_t)b;
109 uint64_t r0 = a * b, r1 = a * bh, r2 = ah * b, r3 = ah * bh;
110 r3 += (r1 >> 32) + (r2 >> 32);
111 r1 = (uint32_t)r1, r2 = (uint32_t)r2;
112 r1 += r2;
113 r1 += (r0 >> 32);
114 dst[1] = r3 + (r1 >> 32);
115 dst[0] = (r1 << 32) | (uint32_t)r0;
116#endif
117}
