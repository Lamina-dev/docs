20/*
21Evaluate in: -1, 0, +inf
22
23 <-s--><--n-->
24 |-a1-|--a0--|
25 |b1-|--b0--|
26 <-t-><--n-->
27
28v0 = a0 * b0 # A(0)*B(0)
29vm1 = (a0-a1)*(b0-b1) # A(-1)*B(-1)
30vinf= a1 * b1 # A(inf)*B(inf)
31*/
32
33void lmmp_mul_toom22_(mp_ptr restrict dst, mp_srcptr restrict numa, mp_size_t na, mp_srcptr restrict numb, mp_size_t nb) {
34 lmmp_param_assert(nb >= 5);
35 lmmp_param_assert(na >= nb);
36 lmmp_param_assert(4 * na <= 5 * nb);
37 TEMP_S_DECL;
38 mp_size_t s = na >> 1, n = na - s, t = nb - n;
39 mp_limb_t* restrict vm1 = SALLOC_TYPE(2 * n, mp_limb_t);
40 int vm1_neg = 0;
41 mp_slimb_t cy, cy2;
42
43#define a0 numa
44#define a1 (numa + n)
45#define b0 numb
46#define b1 (numb + n)
47#define asm1 dst
48#define bsm1 (dst + n)
49
50 if (s == n) {
51 if (lmmp_cmp_(a0, a1, n) < 0) {
52 lmmp_sub_n_(asm1, a1, a0, n);
53 vm1_neg = 1;
54 } else
55 lmmp_sub_n_(asm1, a0, a1, n);
56 } else { // s==n-1
57 if (a0[s] == 0 && lmmp_cmp_(a0, a1, s) < 0) {
58 lmmp_sub_n_(asm1, a1, a0, s);
59 asm1[s] = 0;
60 vm1_neg = 1;
61 } else
62 asm1[s] = a0[s] - lmmp_sub_n_(asm1, a0, a1, s);
63 }
64
65 if (t == n) {
66 if (lmmp_cmp_(b0, b1, n) < 0) {
67 lmmp_sub_n_(bsm1, b1, b0, n);
68 vm1_neg ^= 1;
69 } else
70 lmmp_sub_n_(bsm1, b0, b1, n);
71 } else {
72 if (lmmp_zero_q_(b0 + t, n - t) && lmmp_cmp_(b0, b1, t) < 0) {
73 lmmp_sub_n_(bsm1, b1, b0, t);
74 lmmp_zero(bsm1 + t, n - t);
75 vm1_neg ^= 1;
76 } else
77 lmmp_sub_(bsm1, b0, n, b1, t);
78 }
79
80 lmmp_mul_n_(vm1, asm1, bsm1, n);
81
82#undef asm1
83#undef bsm1
84#define v0 dst
85#define vinf (dst + 2 * n)
86
87 lmmp_mul_n_(v0, a0, b0, n);
88
89 lmmp_mul_(vinf, a1, s, b1, t);
90
91 cy = lmmp_add_n_(dst + 2 * n, v0 + n, vinf, n);
92 cy2 = cy + lmmp_add_n_(dst + n, dst + 2 * n, v0, n);
93 cy += lmmp_add_(dst + 2 * n, dst + 2 * n, n, vinf + n, s + t - n);
94
95 if (vm1_neg)
96 cy += lmmp_add_n_(dst + n, dst + n, vm1, 2 * n);
97 else
98 cy -= lmmp_sub_n_(dst + n, dst + n, vm1, 2 * n);
99
100 // no overflow, if s+t>n. proved.
101 lmmp_inc_1(dst + 2 * n, cy2);
102
103 if (cy < 0)
104 lmmp_dec(dst + 3 * n);
105 else
106 lmmp_inc_1(dst + 3 * n, cy);
107 TEMP_S_FREE;
108}
